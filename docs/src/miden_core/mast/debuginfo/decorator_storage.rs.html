17/// A two-level compressed sparse row (CSR) representation for indexing decorator IDs per
18/// operation per node.
19///
20/// This structure provides efficient access to decorator IDs in a hierarchical manner:
21/// 1. First level: Node -> Operations (operations belong to nodes)
22/// 2. Second level: Operation -> Decorator IDs (decorator IDs belong to operations)
23///
24/// The data layout follows CSR format at both levels:
25///
26/// - `decorator_ids` contains all the DecoratorId values in a single flat array. These are the
27///   actual decorator identifiers that need to be accessed. We store them contiguously to minimize
28///   memory overhead and improve cache locality when iterating.
29///
30/// - `op_indptr_for_dec_ids` stores pointer indices that map operations to their position within
31///   the `decorator_ids` array. For each operation, it contains the start index where that
32///   operation's decorator IDs begin in the flat storage.
33///
34/// - `node_indptr_for_op_idx` stores pointer indices that map nodes to their position within the
35///   `op_indptr_for_dec_ids` array. For each node, it contains the start index where that node's
36///   operation pointers begin.
37///
38/// Together, these three arrays form a two-level index structure that allows efficient
39/// lookup of decorator IDs for any operation in any node, while minimizing memory usage
40/// for sparse decorator data.
41///
42/// # Example
43///
44/// Consider this COO (Coordinate format) representation:
45/// ```text
46/// Node 0, Op 0: [decorator_id_0, decorator_id_1]
47/// Node 0, Op 1: [decorator_id_2]
48/// Node 1, Op 0: [decorator_id_3, decorator_id_4, decorator_id_5]
49/// ```
50///
51/// This would be stored as:
52/// ```text
53/// decorator_ids:         [0, 1, 2, 3, 4, 5]
54/// op_indptr_for_dec_ids: [0, 2, 3, 6]  // Node 0: ops [0,2], [2,3]; Node 1: ops [3,6]
55/// node_indptr_for_op_idx: [0, 2, 3]   // Node 0: [0,2], Node 1: [2,3]
56/// ```
57///
58/// See the unit test `test_csr_and_coo_produce_same_elements` for a comprehensive example
59/// demonstrating how this encoding works and verifying round-trip conversion from COO to CSR.
60#[derive(Debug, Clone, PartialEq, Eq)]
61#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
62#[cfg_attr(all(feature = "arbitrary", test), miden_test_serde_macros::serde_test)]
63pub struct OpToDecoratorIds {
64    /// All the decorator IDs per operation per node, in a CSR relationship with
65    /// node_indptr_for_op_idx and op_indptr_for_dec_ids
66    decorator_ids: Vec<DecoratorId>,
67    /// For the node whose operation indices are in
68    /// `op_indptr_for_dec_ids[node_start..node_end]`,
69    /// the indices of its i-th operation are at:
70    /// ```text
71    /// decorator_ids[op_indptr_for_dec_ids[node_start + i]..
72    ///               op_indptr_for_dec_ids[node_start + i + 1]]
73    /// ```
74    op_indptr_for_dec_ids: Vec<usize>,
75    /// The decorated operation indices for the n-th node are at
76    /// `op_indptr_for_dec_ids[node_indptr_for_op_idx[n]..node_indptr_for_op_idx[n+1]]`
77    node_indptr_for_op_idx: IndexVec<MastNodeId, usize>,
78}
79
80/// Error type for decorator index mapping operations
81#[derive(Debug, PartialEq, Eq, thiserror::Error)]
82pub enum DecoratorIndexError {
83    /// Node index out of bounds
84    #[error("Invalid node index {0:?}")]
85    NodeIndex(MastNodeId),
86    /// Operation index out of bounds for the given node
87    #[error("Invalid operation index {operation} for node {node:?}")]
88    OperationIndex { node: MastNodeId, operation: usize },
89    /// Invalid internal data structure (corrupted pointers)
90    #[error("Invalid internal data structure in OpToDecoratorIds")]
91    InternalStructure,
92}

118    /// Create a OpToDecoratorIds from raw CSR components.
119    ///
120    /// This is useful for deserialization or testing purposes where you need to reconstruct
121    /// the data structure from its raw components.
122    ///
123    /// # Arguments
124    /// * `decorator_ids` - Flat storage of all decorator IDs. These are the actual decorator
125    ///   identifiers that will be accessed through the index structure.
126    /// * `op_indptr_for_dec_ids` - Pointer indices for operations within decorator_ids. These must
127    ///   be monotonically increasing and within bounds of `decorator_ids`. The slice must not be
128    ///   empty, and the first element must be 0.
129    /// * `node_indptr_for_op_idx` - Pointer indices for nodes within op_indptr_for_dec_ids. These
130    ///   must be monotonically increasing and within bounds of `op_indptr_for_dec_ids`. The slice
131    ///   must not be empty, and the first element must be 0.
132    ///
133    /// # Valid Edge Cases
134    /// - All three vectors empty (no nodes, no decorators)
135    /// - Empty decorator vectors with node pointers all zero (nodes exist but have no decorators)
136    ///
137    /// # Validation Rules
138    /// For non-empty structures:
139    /// - Pointer arrays must start at zero
140    /// - Pointers must be monotonically non-decreasing
141    /// - Last value in `op_indptr_for_dec_ids` must be <= `decorator_ids.len()`
142    /// - Last value in `node_indptr_for_op_idx` must be < `op_indptr_for_dec_ids.len()`
143    pub(super) fn from_components(
144        decorator_ids: Vec<DecoratorId>,
145        op_indptr_for_dec_ids: Vec<usize>,
146        node_indptr_for_op_idx: IndexVec<MastNodeId, usize>,
147    ) -> Result<Self, DecoratorIndexError> {
148        // Completely empty structures are valid (no nodes, no decorators)
149        if decorator_ids.is_empty()
150            && op_indptr_for_dec_ids.is_empty()
151            && node_indptr_for_op_idx.is_empty()
152        {
153            return Ok(Self {
154                decorator_ids,
155                op_indptr_for_dec_ids,
156                node_indptr_for_op_idx,
157            });
158        }
159
160        // Nodes with no decorators are valid
161        if decorator_ids.is_empty() && op_indptr_for_dec_ids.is_empty() {
162            // All node pointers must be 0
163            if node_indptr_for_op_idx.iter().all(|&ptr| ptr == 0) {
164                return Ok(Self {
165                    decorator_ids,
166                    op_indptr_for_dec_ids,
167                    node_indptr_for_op_idx,
168                });
169            } else {
170                return Err(DecoratorIndexError::InternalStructure);
171            }
172        }
173
174        // Validate the structure
175        if op_indptr_for_dec_ids.is_empty() {
176            return Err(DecoratorIndexError::InternalStructure);
177        }
178
179        // Check that op_indptr_for_dec_ids starts at 0
180        if op_indptr_for_dec_ids[0] != 0 {
181            return Err(DecoratorIndexError::InternalStructure);
182        }
183
184        // Check that the last operation pointer doesn't exceed decorator IDs length
185        let Some(&last_op_ptr) = op_indptr_for_dec_ids.last() else {
186            return Err(DecoratorIndexError::InternalStructure);
187        };
188        if last_op_ptr > decorator_ids.len() {
189            return Err(DecoratorIndexError::InternalStructure);
190        }
191
192        // Check that node pointers are within bounds of operation pointers
193        if node_indptr_for_op_idx.is_empty() {
194            return Err(DecoratorIndexError::InternalStructure);
195        }
196
197        let node_slice = node_indptr_for_op_idx.as_slice();
198
199        // Check that node_indptr_for_op_idx starts at 0
200        if node_slice[0] != 0 {
201            return Err(DecoratorIndexError::InternalStructure);
202        }
203
204        let Some(&last_node_ptr) = node_slice.last() else {
205            return Err(DecoratorIndexError::InternalStructure);
206        };
207        // Node pointers must be valid indices into op_indptr
208        if last_node_ptr > op_indptr_for_dec_ids.len() - 1 {
209            return Err(DecoratorIndexError::InternalStructure);
210        }
211
212        // Ensure monotonicity of pointers
213        for window in op_indptr_for_dec_ids.windows(2) {
214            if window[0] > window[1] {
215                return Err(DecoratorIndexError::InternalStructure);
216            }
217        }
218
219        for window in node_slice.windows(2) {
220            if window[0] > window[1] {
221                return Err(DecoratorIndexError::InternalStructure);
222            }
223        }
224
225        Ok(Self {
226            decorator_ids,
227            op_indptr_for_dec_ids,
228            node_indptr_for_op_idx,
229        })
230    }
231
232    /// Validate CSR structure integrity.
233    ///
234    /// Checks:
235    /// - All decorator IDs are valid (< decorator_count)
236    /// - op_indptr_for_dec_ids is monotonic, starts at 0, ends at decorator_ids.len()
237    /// - node_indptr_for_op_idx is monotonic, starts at 0, ends <= op_indptr_for_dec_ids.len()-1
238    pub(super) fn validate_csr(&self, decorator_count: usize) -> Result<(), String> {
239        // Completely empty structures are valid (no nodes, no decorators)
240        if self.decorator_ids.is_empty()
241            && self.op_indptr_for_dec_ids.is_empty()
242            && self.node_indptr_for_op_idx.is_empty()
243        {
244            return Ok(());
245        }
246
247        // Nodes with no decorators are valid
248        if self.decorator_ids.is_empty() && self.op_indptr_for_dec_ids.is_empty() {
249            // All node pointers must be 0
250            if !self.node_indptr_for_op_idx.iter().all(|&ptr| ptr == 0) {
251                return Err("node pointers must all be 0 when there are no decorators".to_string());
252            }
253            return Ok(());
254        }
255
256        // Validate all decorator IDs
257        for &dec_id in &self.decorator_ids {
258            if dec_id.to_usize() >= decorator_count {
259                return Err(format!(
260                    "Invalid decorator ID {}: exceeds decorator count {}",
261                    dec_id.to_usize(),
262                    decorator_count
263                ));
264            }
265        }
266
267        // Validate op_indptr_for_dec_ids
268        if self.op_indptr_for_dec_ids.is_empty() {
269            return Err("op_indptr_for_dec_ids cannot be empty".to_string());
270        }
271
272        if self.op_indptr_for_dec_ids[0] != 0 {
273            return Err("op_indptr_for_dec_ids must start at 0".to_string());
274        }
275
276        for window in self.op_indptr_for_dec_ids.windows(2) {
277            if window[0] > window[1] {
278                return Err(format!(
279                    "op_indptr_for_dec_ids not monotonic: {} > {}",
280                    window[0], window[1]
281                ));
282            }
283        }
284
285        if *self.op_indptr_for_dec_ids.last().unwrap() != self.decorator_ids.len() {
286            return Err(format!(
287                "op_indptr_for_dec_ids end {} doesn't match decorator_ids length {}",
288                self.op_indptr_for_dec_ids.last().unwrap(),
289                self.decorator_ids.len()
290            ));
291        }
292
293        // Validate node_indptr_for_op_idx
294        let node_slice = self.node_indptr_for_op_idx.as_slice();
295        if node_slice.is_empty() {
296            return Err("node_indptr_for_op_idx cannot be empty".to_string());
297        }
298
299        if node_slice[0] != 0 {
300            return Err("node_indptr_for_op_idx must start at 0".to_string());
301        }
302
303        for window in node_slice.windows(2) {
304            if window[0] > window[1] {
305                return Err(format!(
306                    "node_indptr_for_op_idx not monotonic: {} > {}",
307                    window[0], window[1]
308                ));
309            }
310        }
311
312        // Node pointers must be valid indices into op_indptr
313        let max_node_ptr = self.op_indptr_for_dec_ids.len() - 1;
314        if *node_slice.last().unwrap() > max_node_ptr {
315            return Err(format!(
316                "node_indptr_for_op_idx end {} exceeds op_indptr bounds {}",
317                node_slice.last().unwrap(),
318                max_node_ptr
319            ));
320        }
321
322        Ok(())
323    }

336    /// Deserialize OpToDecoratorIds from the source reader.
337    pub(super) fn read_from<R: ByteReader>(
338        source: &mut R,
339        decorator_count: usize,
340    ) -> Result<Self, DeserializationError> {
341        let decorator_ids: Vec<DecoratorId> = Deserializable::read_from(source)?;
342        let op_indptr_for_dec_ids: Vec<usize> = Deserializable::read_from(source)?;
343        let node_indptr_for_op_idx: IndexVec<MastNodeId, usize> =
344            Deserializable::read_from(source)?;
345
346        let result =
347            Self::from_components(decorator_ids, op_indptr_for_dec_ids, node_indptr_for_op_idx)
348                .map_err(|e| DeserializationError::InvalidValue(e.to_string()))?;
349
350        result.validate_csr(decorator_count).map_err(|e| {
351            DeserializationError::InvalidValue(format!("OpToDecoratorIds validation failed: {e}"))
352        })?;
353
354        Ok(result)
355    }

371    /// Add decorator information for a node incrementally.
372    ///
373    /// This method allows building up the OpToDecoratorIds structure by adding
374    /// decorator IDs for nodes in sequential order only.
375    ///
376    /// # Arguments
377    /// * `node` - The node ID to add decorator IDs for. Must be the next sequential node.
378    /// * `decorators_info` - Vector of (operation_index, decorator_id) tuples. The operation
379    ///   indices should be sorted (as guaranteed by validate_decorators). Operations not present in
380    ///   this vector will have no decorator IDs.
381    ///
382    /// # Returns
383    /// Ok(()) if successful, Err(DecoratorIndexError) if the node is not the next sequential
384    /// node.
385    ///
386    /// # Behavior
387    /// - Can only add decorator IDs for the next sequential node ID
388    /// - Automatically creates empty operations for gaps in operation indices
389    /// - Maintains the two-level CSR structure invariant
390    pub fn add_decorator_info_for_node(
391        &mut self,
392        node: MastNodeId,
393        decorators_info: Vec<(usize, DecoratorId)>,
394    ) -> Result<(), DecoratorIndexError> {
395        // Enforce sequential node ids
396        let expected = MastNodeId::new_unchecked(self.num_nodes() as u32);
397        if node < expected {
398            return Err(DecoratorIndexError::NodeIndex(node));
399        }
400        // Create empty nodes for gaps in node indices
401        for idx in expected.0..node.0 {
402            self.add_decorator_info_for_node(MastNodeId::new_unchecked(idx), vec![])?;
403        }
404
405        // Start of this node's operations is the current length (do NOT reuse previous sentinel)
406        let op_start = self.op_indptr_for_dec_ids.len();
407
408        // Maintain node CSR: node_indptr[i] = start index for node i
409        if self.node_indptr_for_op_idx.is_empty() {
410            self.node_indptr_for_op_idx
411                .push(op_start)
412                .map_err(|_| DecoratorIndexError::OperationIndex { node, operation: op_start })?;
413        } else {
414            // Overwrite the previous "end" slot to become this node's start
415            let last = MastNodeId::new_unchecked((self.node_indptr_for_op_idx.len() - 1) as u32);
416            self.node_indptr_for_op_idx[last] = op_start;
417        }
418
419        if decorators_info.is_empty() {
420            // Empty node needs sentinel if it follows decorated nodes
421            // CSR requires all node_indptr values to be valid op_indptr indices.
422            // If op_start == op_indptr.len(), add a sentinel so the pointer stays in bounds.
423            if op_start == self.op_indptr_for_dec_ids.len()
424                && !self.op_indptr_for_dec_ids.is_empty()
425            {
426                self.op_indptr_for_dec_ids.push(self.decorator_ids.len());
427            }
428
429            self.node_indptr_for_op_idx
430                .push(op_start)
431                .map_err(|_| DecoratorIndexError::OperationIndex { node, operation: op_start })?;
432        } else {
433            // Build op->decorator CSR for this node
434            let max_op_idx = decorators_info.last().unwrap().0; // input is sorted by op index
435            let mut it = decorators_info.into_iter().peekable();
436
437            for op in 0..=max_op_idx {
438                // pointer to start of decorator IDs for op
439                self.op_indptr_for_dec_ids.push(self.decorator_ids.len());
440                while it.peek().is_some_and(|(i, _)| *i == op) {
441                    self.decorator_ids.push(it.next().unwrap().1);
442                }
443            }
444            // final sentinel for this node
445            self.op_indptr_for_dec_ids.push(self.decorator_ids.len());
446
447            // Push end pointer for this node (index of last op pointer, which is the sentinel)
448            // This is len()-1 because we just pushed the sentinel above
449            let end_ops = self.op_indptr_for_dec_ids.len() - 1;
450            self.node_indptr_for_op_idx
451                .push(end_ops)
452                .map_err(|_| DecoratorIndexError::OperationIndex { node, operation: end_ops })?;
453        }
454
455        Ok(())
456    }

474    /// Get all decorator IDs for a specific operation within a node.
475    ///
476    /// # Arguments
477    /// * `node` - The node ID
478    /// * `operation` - The operation index within the node
479    ///
480    /// # Returns
481    /// A slice of decorator IDs for the operation, or an error if indices are invalid.
482    pub fn decorator_ids_for_operation(
483        &self,
484        node: MastNodeId,
485        operation: usize,
486    ) -> Result<&[DecoratorId], DecoratorIndexError> {
487        let op_range = self.operation_range_for_node(node)?;
488        // that operation does not have listed decorator indices
489        if operation >= op_range.len() {
490            return Ok(&[]);
491        }
492
493        let op_start_idx = op_range.start + operation;
494        if op_start_idx + 1 >= self.op_indptr_for_dec_ids.len() {
495            return Err(DecoratorIndexError::InternalStructure);
496        }
497
498        let dec_start = self.op_indptr_for_dec_ids[op_start_idx];
499        let dec_end = self.op_indptr_for_dec_ids[op_start_idx + 1];
500
501        if dec_start > dec_end || dec_end > self.decorator_ids.len() {
502            return Err(DecoratorIndexError::InternalStructure);
503        }
504
505        Ok(&self.decorator_ids[dec_start..dec_end])
506    }

508    /// Get an iterator over all operations and their decorator IDs for a given node.
509    ///
510    /// # Arguments
511    /// * `node` - The node ID
512    ///
513    /// # Returns
514    /// An iterator yielding (operation_index, decorator_ids_slice) tuples, or an error if the node
515    /// is invalid.
516    pub fn decorator_ids_for_node(
517        &self,
518        node: MastNodeId,
519    ) -> Result<impl Iterator<Item = (usize, &[DecoratorId])>, DecoratorIndexError> {
520        let op_range = self.operation_range_for_node(node)?;
521        let num_ops = op_range.len();
522
523        Ok((0..num_ops).map(move |op_idx| {
524            let op_start_idx = op_range.start + op_idx;
525            let dec_start = self.op_indptr_for_dec_ids[op_start_idx];
526            let dec_end = self.op_indptr_for_dec_ids[op_start_idx + 1];
527            (op_idx, &self.decorator_ids[dec_start..dec_end])
528        }))
529    }

562    /// Get the range of operation indices for a given node.
563    ///
564    /// # Arguments
565    /// * `node` - The node ID
566    ///
567    /// # Returns
568    /// A range representing the start and end (exclusive) operation indices for the node.
569    pub fn operation_range_for_node(
570        &self,
571        node: MastNodeId,
572    ) -> Result<core::ops::Range<usize>, DecoratorIndexError> {
573        let node_slice = self.node_indptr_for_op_idx.as_slice();
574        let node_idx = node.to_usize();
575
576        if node_idx + 1 >= node_slice.len() {
577            return Err(DecoratorIndexError::NodeIndex(node));
578        }
579
580        let start = node_slice[node_idx];
581        let end = node_slice[node_idx + 1];
582
583        if start > end || end > self.op_indptr_for_dec_ids.len() {
584            return Err(DecoratorIndexError::InternalStructure);
585        }
586
587        Ok(start..end)
588    }
589}

591impl Default for OpToDecoratorIds {
592    fn default() -> Self {
593        Self::new()
594    }
595}

597/// Immutable view over all `(op_idx, DecoratorId)` pairs for a node.
598/// Uses the two-level CSR encoded by `node_indptr_for_op_idx` and `op_indptr_for_dec_idx`.
599pub struct DecoratedLinks<'a> {
600    // Absolute op-pointer index range for this node: [start_op .. end_op)
601    start_op: usize,
602    end_op: usize,
603
604    // CSR arrays (borrowed; not owned)
605    op_indptr_for_dec_idx: &'a [usize], // len = total_ops + 1
606    decorator_indices: &'a [DecoratorId],
607}
608
609impl<'a> DecoratedLinks<'a> {
610    fn new(
611        start_op: usize,
612        end_op: usize,
613        op_indptr_for_dec_idx: &'a [usize],
614        decorator_indices: &'a [DecoratorId],
615    ) -> Self {
616        Self {
617            start_op,
618            end_op,
619            op_indptr_for_dec_idx,
620            decorator_indices,
621        }
622    }
623
624    /// Total number of `(op_idx, DecoratorId)` pairs in this view (exact).
625    #[inline]
626    pub fn len_pairs(&self) -> usize {
627        let s = self.op_indptr_for_dec_idx[self.start_op];
628        let e = self.op_indptr_for_dec_idx[self.end_op];
629        e - s
630    }
631
632    #[inline]
633    pub fn is_empty(&self) -> bool {
634        self.len_pairs() == 0
635    }
636}

638/// The concrete, zero-alloc iterator over `(relative_op_idx, DecoratorId)`.
639pub struct DecoratedLinksIter<'a> {
640    // absolute op-pointer indices (into op_indptr_for_dec_idx)
641    cur_op: usize,
642    end_op: usize,
643    base_op: usize, // for relative index = cur_op - base_op
644
645    // inner slice [inner_i .. inner_end) indexes into decorator_indices
646    inner_i: usize,
647    inner_end: usize,
648
649    // borrowed CSR arrays
650    op_indptr_for_dec_idx: &'a [usize],
651    decorator_indices: &'a [DecoratorId],
652
653    // exact count of remaining pairs
654    remaining: usize,
655}
656
657impl<'a> IntoIterator for DecoratedLinks<'a> {
658    type Item = DecoratedOpLink;
659    type IntoIter = DecoratedLinksIter<'a>;
660
661    fn into_iter(self) -> Self::IntoIter {
662        // Precompute the exact number of pairs in the node's range.
663        let remaining = {
664            // Add bounds check to prevent panic on empty storage
665            if self.start_op >= self.op_indptr_for_dec_idx.len()
666                || self.end_op > self.op_indptr_for_dec_idx.len()  // Note: end_op can be equal to len
667                || self.start_op > self.end_op
668            // Invalid range
669            {
670                0
671            } else {
672                // For valid ranges, compute the actual number of decorator pairs
673                let mut total = 0;
674                for op_idx in self.start_op..self.end_op {
675                    let start_idx = self.op_indptr_for_dec_idx[op_idx];
676                    let end_idx = if op_idx + 1 < self.op_indptr_for_dec_idx.len() {
677                        self.op_indptr_for_dec_idx[op_idx + 1]
678                    } else {
679                        self.op_indptr_for_dec_idx.len()
680                    };
681                    total += end_idx - start_idx;
682                }
683                total
684            }
685        };
686
687        // Initialize inner range to the first op (if any).
688        let (inner_i, inner_end) = if self.start_op < self.end_op
689            && self.start_op + 1 < self.op_indptr_for_dec_idx.len()
690        {
691            let s0 = self.op_indptr_for_dec_idx[self.start_op];
692            let e0 = self.op_indptr_for_dec_idx[self.start_op + 1];
693            (s0, e0)
694        } else {
695            (0, 0)
696        };
697
698        DecoratedLinksIter {
699            cur_op: self.start_op,
700            end_op: self.end_op,
701            base_op: self.start_op,
702            inner_i,
703            inner_end,
704            op_indptr_for_dec_idx: self.op_indptr_for_dec_idx,
705            decorator_indices: self.decorator_indices,
706            remaining,
707        }
708    }
709}
710
711impl<'a> DecoratedLinksIter<'a> {
712    #[inline]
713    fn advance_outer(&mut self) -> bool {
714        self.cur_op += 1;
715        if self.cur_op >= self.end_op {
716            return false;
717        }
718        // Bounds check: ensure cur_op and cur_op+1 are within the pointer array
719        if self.cur_op + 1 >= self.op_indptr_for_dec_idx.len() {
720            return false;
721        }
722        let s = self.op_indptr_for_dec_idx[self.cur_op];
723        let e = self.op_indptr_for_dec_idx[self.cur_op + 1];
724        self.inner_i = s;
725        self.inner_end = e;
726        true
727    }
728}
729
730impl<'a> Iterator for DecoratedLinksIter<'a> {
731    type Item = DecoratedOpLink;
732
733    #[inline]
734    fn next(&mut self) -> Option<Self::Item> {
735        while self.cur_op < self.end_op {
736            // Emit from current op's decorator slice if non-empty
737            if self.inner_i < self.inner_end {
738                let rel_op = self.cur_op - self.base_op; // relative op index within the node
739                let id = self.decorator_indices[self.inner_i];
740                self.inner_i += 1;
741                self.remaining -= 1;
742                return Some((rel_op, id));
743            }
744            // Move to next operation (which might be empty as well)
745            if !self.advance_outer() {
746                break;
747            }
748        }
749        None
750    }
751
752    #[inline]
753    fn size_hint(&self) -> (usize, Option<usize>) {
754        (self.remaining, Some(self.remaining))
755    }
756}
757
758impl<'a> ExactSizeIterator for DecoratedLinksIter<'a> {
759    #[inline]
760    fn len(&self) -> usize {
761        self.remaining
762    }
763}

765#[cfg(feature = "arbitrary")]
766impl Arbitrary for OpToDecoratorIds {
767    type Parameters = proptest::collection::SizeRange;
768    type Strategy = BoxedStrategy<Self>;
769
770    fn arbitrary_with(size: Self::Parameters) -> Self::Strategy {
771        use proptest::{collection::vec, prelude::*};
772
773        // Generate small, controlled structures to avoid infinite loops
774        // Keep node and operation counts small and bounded, always at least 1
775        (1usize..40, 1usize..=72) // (max_nodes, max_ops_per_node)
776            .prop_flat_map(move |(max_nodes, max_ops_per_node)| {
777                vec(
778                    // Generate (node_id, op_id, decorator_id) with bounded values
779                    (0..max_nodes, 0..max_ops_per_node, any::<u32>()),
780                    size.clone(), // Limit total entries to size
781                )
782                .prop_map(move |coo_data| {
783                    // Build the OpToDecoratorIds incrementally
784                    let mut mapping = OpToDecoratorIds::new();
785
786                    // Group by node_id, then by op_id to maintain sorted order
787                    use alloc::collections::BTreeMap;
788                    let mut node_to_ops: BTreeMap<u32, BTreeMap<u32, Vec<u32>>> = BTreeMap::new();
789
790                    for (node_id, op_id, decorator_id) in coo_data {
791                        node_to_ops
792                            .entry(node_id as u32)
793                            .or_default()
794                            .entry(op_id as u32)
795                            .or_default()
796                            .push(decorator_id);
797                    }
798
799                    // Add nodes in order to satisfy sequential constraint
800                    for (node_id, ops_map) in node_to_ops {
801                        let mut decorators_info: Vec<(usize, DecoratorId)> = Vec::new();
802                        for (op_id, decorator_ids) in ops_map {
803                            for decorator_id in decorator_ids {
804                                decorators_info.push((op_id as usize, DecoratorId(decorator_id)));
805                            }
806                        }
807                        // Sort by operation index to meet add_decorator_info_for_node requirements
808                        decorators_info.sort_by_key(|(op_idx, _)| *op_idx);
809
810                        mapping
811                            .add_decorator_info_for_node(
812                                MastNodeId::new_unchecked(node_id),
813                                decorators_info,
814                            )
815                            .unwrap();
816                    }
817
818                    mapping
819                })
820            })
821            .boxed()
822    }
823}
