1//! CSR storage for mapping (NodeId, OpIdx) -> AsmOpId.
2//!
3//! This module stores AssemblyOp mappings using a sparse CSR format. Unlike the previous design
4//! that stored `Option<AsmOpId>` per operation slot, this stores only the operations that actually
5//! have an AssemblyOp, along with their operation index.
6//!
7//! # Data Layout
8//!
9//! For each node, we store a list of `(op_idx, asm_op_id)` pairs representing which operations
10//! have an AssemblyOp. The pairs are sorted by `op_idx` within each node.
11//!
12//! # Example
13//!
14//! ```text
15//! Node 0: Op 2 -> asm_op_0
16//! Node 1: Op 0 -> asm_op_1, Op 2 -> asm_op_2
17//! ```
18//!
19//! This would be stored as:
20//! ```text
21//! data: [(2, asm_op_0), (0, asm_op_1), (2, asm_op_2)]
22//! indptr: [0, 1, 3]  // Node 0: [0,1), Node 1: [1,3)
23//! ```
24

36// OP TO ASMOP ID
37// ================================================================================================

39/// CSR storage mapping (NodeId, OpIdx) -> AsmOpId.
40///
41/// Unlike [`OpToDecoratorIds`](super::OpToDecoratorIds), each operation has at most one
42/// AssemblyOp. We store only the operations that have an AssemblyOp, using sparse storage.
43///
44/// This structure provides efficient lookup of AssemblyOps by node and operation index, which is
45/// needed for error context reporting and debugging tools.
46#[derive(Debug, Clone, PartialEq, Eq)]
47#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
48pub struct OpToAsmOpId {
49    /// CSR storage: each row (node) contains `(op_idx, asm_op_id)` pairs.
50    /// Only operations with an AssemblyOp are stored (sparse representation).
51    inner: CsrMatrix<MastNodeId, (usize, AsmOpId)>,
52}
53
54impl Default for OpToAsmOpId {
55    fn default() -> Self {
56        Self::new()
57    }
58}
59
60impl OpToAsmOpId {
61    /// Creates a new empty [`OpToAsmOpId`].
62    pub fn new() -> Self {
63        Self { inner: CsrMatrix::new() }
64    }
65
66    /// Creates an [`OpToAsmOpId`] with the specified capacity.
67    pub fn with_capacity(nodes_capacity: usize, operations_capacity: usize) -> Self {
68        Self {
69            inner: CsrMatrix::with_capacity(nodes_capacity, operations_capacity),
70        }
71    }
72
73    /// Returns `true` if this storage contains no nodes.
74    pub fn is_empty(&self) -> bool {
75        self.inner.is_empty()
76    }
77
78    /// Returns the number of nodes in this storage.
79    pub fn num_nodes(&self) -> usize {
80        self.inner.num_rows()
81    }
82
83    /// Returns the total number of (op_idx, asm_op_id) entries across all nodes.
84    ///
85    /// Note: This is the number of operations that have an AssemblyOp, not the total number of
86    /// operations.
87    pub fn num_operations(&self) -> usize {
88        self.inner.num_elements()
89    }
90
91    /// Registers AssemblyOps for a node's operations.
92    ///
93    /// `asm_ops` is a list of `(op_idx, asm_op_id)` pairs. The `op_idx` values must be strictly
94    /// increasing. Operations not listed will have no AsmOpId (sparse storage).
95    ///
96    /// Nodes must be added in sequential order starting from 0. If a node is skipped, empty
97    /// placeholder nodes are automatically created.
98    ///
99    /// # Arguments
100    ///
101    /// * `node_id` - The node to register operations for. Must be >= current node count.
102    /// * `num_operations` - Total number of operations in this node (used for bounds checking).
103    /// * `asm_ops` - List of (operation_index, AsmOpId) pairs, sorted by operation_index.
104    ///
105    /// # Errors
106    ///
107    /// Returns an error if:
108    /// - `node_id` is less than the current node count (already added)
109    /// - Operation indices are not strictly increasing
110    /// - An operation index exceeds `num_operations`
111    pub fn add_asm_ops_for_node(
112        &mut self,
113        node_id: MastNodeId,
114        num_operations: usize,
115        asm_ops: Vec<(usize, AsmOpId)>,
116    ) -> Result<(), AsmOpIndexError> {
117        let expected_node = self.num_nodes() as u32;
118        let node_idx = u32::from(node_id);
119
120        // Check if trying to add a node that was already added
121        if node_idx < expected_node {
122            return Err(AsmOpIndexError::NodeIndex(node_id));
123        }
124
125        // Create empty nodes for any gaps
126        for _ in expected_node..node_idx {
127            self.inner.push_empty_row().map_err(|_| AsmOpIndexError::InternalStructure)?;
128        }
129
130        // Verify strictly increasing operation indices
131        for window in asm_ops.windows(2) {
132            if window[0].0 >= window[1].0 {
133                return Err(AsmOpIndexError::NonIncreasingOpIndices);
134            }
135        }
136
137        // Verify all indices are within bounds
138        if let Some((max_idx, _)) = asm_ops.last()
139            && *max_idx >= num_operations
140        {
141            return Err(AsmOpIndexError::OpIndexOutOfBounds(*max_idx, num_operations));
142        }
143
144        self.inner.push_row(asm_ops).map_err(|_| AsmOpIndexError::InternalStructure)?;
145
146        Ok(())
147    }
148
149    /// Returns the AsmOpId for a specific operation within a node, if any.
150    ///
151    /// If the operation doesn't have a direct AssemblyOp, this performs a backward search to find
152    /// the most recent AssemblyOp (needed for multi-cycle instructions like `assertz` where only
153    /// the first operation has an AssemblyOp).
154    ///
155    /// # Arguments
156    ///
157    /// * `node_id` - The node to query.
158    /// * `op_idx` - The operation index within the node.
159    ///
160    /// # Returns
161    ///
162    /// - `Some(asm_op_id)` if the operation has an associated AssemblyOp (direct or via backward
163    ///   search)
164    /// - `None` if the node doesn't exist or no AssemblyOp is found
165    pub fn asm_op_id_for_operation(&self, node_id: MastNodeId, op_idx: usize) -> Option<AsmOpId> {
166        let entries = self.inner.row(node_id)?;
167
168        // Binary search for the largest op_idx <= target
169        // We're looking for the entry whose op_idx is closest to (but not greater than) op_idx
170        match entries.binary_search_by_key(&op_idx, |(idx, _)| *idx) {
171            Ok(i) => Some(entries[i].1),
172            Err(i) if i > 0 => Some(entries[i - 1].1),
173            Err(_) => None,
174        }
175    }
176
177    /// Returns the first AsmOpId for a node, if any operations have one.
178    ///
179    /// This is useful for getting context about a node when no specific operation
180    /// index is available.
181    pub fn first_asm_op_for_node(&self, node_id: MastNodeId) -> Option<AsmOpId> {
182        let entries = self.inner.row(node_id)?;
183        entries.first().map(|(_, id)| *id)
184    }
185
186    /// Validates the CSR structure integrity.
187    ///
188    /// # Arguments
189    ///
190    /// * `asm_op_count` - The total number of AssemblyOps that should be valid.
191    ///
192    /// # Returns
193    ///
194    /// `Ok(())` if the structure is valid, otherwise an error message.
195    pub(super) fn validate_csr(&self, asm_op_count: usize) -> Result<(), String> {
196        self.inner
197            .validate_with(|(_op_idx, asm_op_id)| (u32::from(*asm_op_id) as usize) < asm_op_count)
198            .map_err(|e| format_validation_error(e, asm_op_count))
199    }
200
201    /// Creates a new [`OpToAsmOpId`] with remapped node IDs.
202    ///
203    /// This is used when nodes are removed from a MastForest and the remaining nodes are
204    /// renumbered. The remapping maps old node IDs to new node IDs.
205    ///
206    /// Nodes that are not in the remapping are considered removed and their asm_op data is
207    /// discarded.
208    pub fn remap_nodes(&self, remapping: &BTreeMap<MastNodeId, MastNodeId>) -> Self {
209        if self.is_empty() {
210            return Self::new();
211        }
212        if remapping.is_empty() {
213            // No remapping means no nodes were removed/reordered, keep current storage
214            return self.clone();
215        }
216
217        // Find the max new node ID to determine the size of the new structure
218        let max_new_id = remapping.values().map(|id| u32::from(*id)).max().unwrap_or(0) as usize;
219        let num_new_nodes = max_new_id + 1;
220
221        // Collect the data for each new node ID
222        let mut new_node_data: BTreeMap<usize, Vec<(usize, AsmOpId)>> = BTreeMap::new();
223
224        for (old_id, new_id) in remapping {
225            let new_idx = u32::from(*new_id) as usize;
226
227            if let Some(entries) = self.inner.row(*old_id)
228                && !entries.is_empty()
229            {
230                new_node_data.insert(new_idx, entries.to_vec());
231            }
232        }
233
234        // Build the new CSR structure
235        let mut new_inner = CsrMatrix::with_capacity(num_new_nodes, self.inner.num_elements());
236
237        for new_idx in 0..num_new_nodes {
238            if let Some(data) = new_node_data.get(&new_idx) {
239                new_inner.push_row(data.iter().copied()).expect("node count should fit in u32");
240            } else {
241                new_inner.push_empty_row().expect("node count should fit in u32");
242            }
243        }
244
245        Self { inner: new_inner }
246    }
247
248    /// Serializes this [`OpToAsmOpId`] into the target writer.
249    pub(super) fn write_into<W: ByteWriter>(&self, target: &mut W) {
250        self.inner.write_into(target);
251    }
252
253    /// Deserializes an [`OpToAsmOpId`] from the source reader.
254    pub(super) fn read_from<R: ByteReader>(
255        source: &mut R,
256        asm_op_count: usize,
257    ) -> Result<Self, DeserializationError> {
258        let inner: CsrMatrix<MastNodeId, (usize, AsmOpId)> = Deserializable::read_from(source)?;
259
260        let result = Self { inner };
261
262        result.validate_csr(asm_op_count).map_err(|e| {
263            DeserializationError::InvalidValue(format!("OpToAsmOpId validation failed: {}", e))
264        })?;
265
266        Ok(result)
267    }
268}

270// ASMOP INDEX ERROR
271// ================================================================================================
272
273/// Error type for AsmOp index mapping operations.
274#[derive(Debug, PartialEq, Eq, thiserror::Error)]
275pub enum AsmOpIndexError {
276    /// Node index is invalid (either out of sequence or already added).
277    #[error("Invalid node index {0:?}")]
278    NodeIndex(MastNodeId),
279    /// Operation indices must be strictly increasing within the input.
280    #[error("Operation indices must be strictly increasing")]
281    NonIncreasingOpIndices,
282    /// Operation index is out of bounds for the node's operation count.
283    #[error("Operation index {0} exceeds node's operation count {1}")]
284    OpIndexOutOfBounds(usize, usize),
285    /// Internal CSR structure is corrupted.
286    #[error("Internal CSR structure error")]
287    InternalStructure,
288}
289
290/// Format a CsrValidationError into a human-readable string.
291fn format_validation_error(error: CsrValidationError, asm_op_count: usize) -> String {
292    match error {
293        CsrValidationError::IndptrStartNotZero(val) => format!("indptr must start at 0, got {val}"),
294        CsrValidationError::IndptrNotMonotonic { index, prev, curr } => {
295            format!("indptr not monotonic at index {index}: {prev} > {curr}")
296        },
297        CsrValidationError::IndptrDataMismatch { indptr_end, data_len } => {
298            format!("indptr ends at {indptr_end}, but data.len() is {data_len}")
299        },
300        CsrValidationError::InvalidData { row, position } => format!(
301            "Invalid AsmOpId at row {row}, position {position}: exceeds asm_op count {asm_op_count}"
302        ),
303    }
304}
