18/// Abstract definition of a processor's components: system, stack, advice provider, memory, hasher,
19/// and operation helper register computation.
20pub(crate) trait Processor: Sized {
21    type System: SystemInterface;
22    type Stack: StackInterface;
23    type AdviceProvider: AdviceProviderInterface;
24    type Memory: MemoryInterface;
25    type Hasher: HasherInterface;
26
27    /// Returns a reference to the internal stack.
28    fn stack(&self) -> &Self::Stack;
29
30    /// Returns a mutable reference to the internal stack.
31    fn stack_mut(&mut self) -> &mut Self::Stack;
32
33    /// Returns a reference to the internal system.
34    fn system(&self) -> &Self::System;
35
36    /// Returns a mutable reference to the internal system.
37    fn system_mut(&mut self) -> &mut Self::System;
38
39    /// Returns a reference to the internal advice provider.
40    fn advice_provider(&self) -> &Self::AdviceProvider;
41
42    /// Returns a mutable reference to the internal advice provider.
43    fn advice_provider_mut(&mut self) -> &mut Self::AdviceProvider;
44
45    /// Returns a mutable reference to the internal memory subsystem.
46    fn memory_mut(&mut self) -> &mut Self::Memory;
47
48    /// Returns a mutable reference to the internal hasher subsystem.
49    fn hasher(&mut self) -> &mut Self::Hasher;
50
51    /// Saves the current execution context and truncates the stack to 16 elements in preparation to
52    /// start a new execution context.
53    fn save_context_and_truncate_stack(&mut self);
54
55    /// Restores the execution context to the state it was in before the last `call`, `syscall` or
56    /// `dyncall. This includes restoring the overflow stack and the system parameters.
57    fn restore_context(&mut self) -> Result<(), OperationError>;
58
59    /// Returns the current precompile transcript state (sponge capacity).
60    ///
61    /// Used by `log_precompile` to thread the transcript across invocations.
62    fn precompile_transcript_state(&self) -> PrecompileTranscriptState;
63
64    /// Sets the precompile transcript state (sponge capacity) to a new value.
65    ///
66    /// Called by `log_precompile` after recording a new commitment.
67    fn set_precompile_transcript_state(&mut self, state: PrecompileTranscriptState);
68
69    /// Executes the decorators that should be executed before entering a node.
70    fn execute_before_enter_decorators(
71        &self,
72        node_id: MastNodeId,
73        current_forest: &MastForest,
74        host: &mut impl BaseHost,
75    ) -> ControlFlow<BreakReason>;
76
77    /// Executes the decorators that should be executed after exiting a node.
78    fn execute_after_exit_decorators(
79        &self,
80        node_id: MastNodeId,
81        current_forest: &MastForest,
82        host: &mut impl BaseHost,
83    ) -> ControlFlow<BreakReason>;
84
85    /// Executes any decorator in a basic block that is to be executed before the operation at the
86    /// given index in the block.
87    fn execute_decorators_for_op(
88        &self,
89        node_id: MastNodeId,
90        op_idx_in_block: usize,
91        current_forest: &MastForest,
92        host: &mut impl BaseHost,
93    ) -> ControlFlow<BreakReason>;
94
95    /// Executes any decorator in a basic block that is to be executed after all operations in the
96    /// block. This only differs from `execute_after_exit_decorators` in that these decorators are
97    /// stored in the basic block node itself.
98    fn execute_end_of_block_decorators(
99        &self,
100        basic_block_node: &BasicBlockNode,
101        node_id: MastNodeId,
102        current_forest: &Arc<MastForest>,
103        host: &mut impl BaseHost,
104    ) -> ControlFlow<BreakReason>;
105}

110/// Trait representing the system state of the processor.
111pub(crate) trait SystemInterface {
112    // ACCESSORS
113    // --------------------------------------------------------------------------------------------
114
115    /// Returns the value of the CALLER_HASH register, which is the hash of the procedure that
116    /// initiated the most recent SYSCALL, or ZERO if not in a syscall context.
117    fn caller_hash(&self) -> Word;
118
119    /// Returns the current clock cycle.
120    fn clock(&self) -> RowIndex;
121
122    /// Returns the current context ID.
123    fn ctx(&self) -> ContextId;
124
125    // MUTATORS
126    // --------------------------------------------------------------------------------------------
127
128    /// Sets the CALLER_HASH register to the provided value.
129    fn set_caller_hash(&mut self, caller_hash: Word);
130
131    /// Sets the current context ID to the provided value.
132    fn set_ctx(&mut self, ctx: ContextId);
133
134    // Increments the clock by 1.
135    fn increment_clock(&mut self);
136}

138// STACK INTERFACE
139// ================================================================================================
140
141/// We model the stack as a slice of `Felt` values, where the top of the stack is at the last index
142/// of the slice. The stack is mutable, and the processor can manipulate it directly. A "stack top
143/// pointer" tracks the current top of the stack, which we define to be the top 16 elements. Indices
144/// are always taken relative to the top element of the stack, meaning that `stack_get(0)` returns
145/// the top element, `stack_get(1)` returns the second element from the top, and so on. The stack is
146/// always at least 16 elements deep.
147pub(crate) trait StackInterface {
148    /// Returns the top 16 elements of the stack, such that the top of the stack is at the last
149    /// index of the returned slice.
150    fn top(&self) -> &[Felt];
151
152    /// Returns the element on the stack at index `idx`.
153    ///
154    /// `idx` must be less than or equal to 15.
155    fn get(&self, idx: usize) -> Felt;
156
157    /// Mutable variant of `stack_get()`.
158    ///
159    /// `idx` must be less than or equal to 15.
160    fn get_mut(&mut self, idx: usize) -> &mut Felt;
161
162    /// Returns the word on the stack starting at index `start_idx` in "stack order".
163    ///
164    /// That is, for `start_idx=0` the top element of the stack will be at the last position in the
165    /// word.
166    ///
167    /// For example, if the stack looks like this:
168    ///
169    /// top                                                       bottom
170    /// v                                                           v
171    /// a | b | c | d | e | f | g | h | i | j | k | l | m | n | o | p
172    ///
173    /// Then
174    /// - `stack_get_word(0)` returns `[a, b, c, d]`,
175    /// - `stack_get_word(1)` returns `[b, c, d, e]`,
176    /// - etc.
177    ///
178    /// Word\[0\] corresponds to the top of stack.
179    ///
180    /// `start_idx` must be less than or equal to 12.
181    fn get_word(&self, start_idx: usize) -> Word;
182
183    /// Returns two words (8 elements) from the stack starting at index `start_idx`.
184    ///
185    /// This is equivalent to reading 8 consecutive elements from the stack, or to reading the words
186    /// as `Self::get_word(0)` and `Self::get_word(4)`.
187    ///
188    /// For example, if the stack looks like this:
189    ///
190    /// top                                                       bottom
191    /// v                                                           v
192    /// a | b | c | d | e | f | g | h | i | j | k | l | m | n | o | p
193    ///
194    /// Then `get_double_word(0)` returns `[a, b, c, d, e, f, g, h]`.
195    ///
196    /// `start_idx` must be less than or equal to 8.
197    fn get_double_word(&self, start_idx: usize) -> [Felt; 8] {
198        core::array::from_fn(|i| self.get(start_idx + i))
199    }
200
201    /// Returns the number of elements on the stack in the current context.
202    fn depth(&self) -> u32;
203
204    /// Writes an element to the stack at the given index.
205    ///
206    /// `idx` must be less than or equal to 15.
207    fn set(&mut self, idx: usize, element: Felt);
208
209    /// Writes a word to the stack starting at the given index.
210    ///
211    /// Word\[0\] goes to stack position start_idx (top), word\[1\] to start_idx+1, etc.
212    ///
213    /// `start_idx` must be less than or equal to 12.
214    fn set_word(&mut self, start_idx: usize, word: &Word);
215
216    /// Swaps the elements at the given indices on the stack.
217    ///
218    /// `idx1` and `idx2` must be less than or equal to 15.
219    fn swap(&mut self, idx1: usize, idx2: usize);
220
221    /// Swaps the nth word from the top of the stack with the top word of the stack.
222    ///
223    /// Valid values of `n` are 1, 2, and 3.
224    fn swapw_nth(&mut self, n: usize);
225
226    /// Rotates the top `n` elements of the stack to the left by 1.
227    ///
228    /// For example, if the stack is [a, b, c, d], with `d` at the top, then `rotate_left(3)` will
229    /// result in the top 3 elements being rotated left: [a, c, d, b].
230    ///
231    /// This operation is useful for implementing the `movup` instructions.
232    ///
233    /// The stack size doesn't change.
234    ///
235    /// Note: This method doesn't use the `get()` and `write()` methods because it is
236    /// more efficient to directly manipulate the stack array (~10% performance difference).
237    fn rotate_left(&mut self, n: usize);
238
239    /// Rotates the top `n` elements of the stack to the right by 1.
240    ///
241    /// Analogous to `rotate_left`, but in the opposite direction.
242    ///
243    /// Note: This method doesn't use the `get()` and `write()` methods because it is
244    /// more efficient to directly manipulate the stack array (~10% performance difference).
245    fn rotate_right(&mut self, n: usize);
246
247    /// Increments the stack top pointer by one, announcing the intent to add a new element to the
248    /// stack. That is, the stack size is incremented, but the element is not written yet.
249    ///
250    /// This can be understood as pushing a `None` on top of the stack, such that a subsequent call
251    /// to `write(0)` or `write_word(0)` will write an element to that new position.
252    ///
253    /// It is guaranteed that any operation that calls `increment_size()` will subsequently
254    /// call `write(0)` or `write_word(0)` to write an element to that position on the
255    /// stack.
256    fn increment_size(&mut self) -> Result<(), ExecutionError>;
257
258    /// Decrements the stack size by one, removing the top element from the stack.
259    ///
260    /// Concretely, this decrements the stack top pointer by one (removing the top element), and
261    /// pushes a `ZERO` at the bottom of the stack if the stack size is already at 16 elements
262    /// (since the stack size can never be less than 16).
263    fn decrement_size(&mut self) -> Result<(), OperationError>;
264}

269/// Trait representing an advice provider for the processor.
270pub(crate) trait AdviceProviderInterface {
271    /// Pops an element from the advice stack and returns it.
272    fn pop_stack(&mut self) -> Result<Felt, AdviceError>;
273
274    /// Pops a word (4 elements) from the advice stack and returns it.
275    ///
276    /// Note: a word is popped off the stack element-by-element. For example, a `[d, c, b, a, ...]`
277    /// stack (i.e., `d` is at the top of the stack) will yield `[d, c, b, a]`.
278    ///
279    /// # Errors
280    /// Returns an error if the advice stack does not contain a full word.
281    fn pop_stack_word(&mut self) -> Result<Word, AdviceError>;
282
283    /// Pops a double word (8 elements) from the advice stack and returns them.
284    ///
285    /// Note: words are popped off the stack element-by-element. For example, a
286    /// `[h, g, f, e, d, c, b, a, ...]` stack (i.e., `h` is at the top of the stack) will yield
287    /// two words: `[h, g, f,e ], [d, c, b, a]`.
288    fn pop_stack_dword(&mut self) -> Result<[Word; 2], AdviceError>;
289
290    /// Returns a path to a node at the specified depth and index in a Merkle tree with the
291    /// specified root.
292    ///
293    /// The returned Merkle path is behind an `Option` to support environments where there is no
294    /// advice provider. The Merkle path is guaranteed to be provided to [HasherInterface] methods
295    /// and otherwise ignored, and therefore `None` can be returned when combined with a
296    /// [HasherInterface] implementation that ignores the Merkle path.
297    fn get_merkle_path(
298        &self,
299        root: Word,
300        depth: Felt,
301        index: Felt,
302    ) -> Result<Option<MerklePath>, AdviceError>;
303
304    /// Updates a node at the specified depth and index in a Merkle tree with the specified root;
305    /// returns the Merkle path from the updated node to the new root.
306    ///
307    /// The returned Merkle path is behind an `Option` to support environments where there is no
308    /// advice provider. The Merkle path is guaranteed to be provided to [HasherInterface] methods
309    /// and otherwise ignored, and therefore `None` can be returned when combined with a
310    /// [HasherInterface] implementation that ignores the Merkle path.
311    fn update_merkle_node(
312        &mut self,
313        root: Word,
314        depth: Felt,
315        index: Felt,
316        value: Word,
317    ) -> Result<Option<MerklePath>, AdviceError>;
318}

353/// Trait representing the hasher subsystem of the processor.
354pub(crate) trait HasherInterface {
355    /// Applies a single permutation of the hash function to the provided state and records the
356    /// execution trace of this computation.
357    ///
358    /// The returned tuple contains the hasher state after the permutation and the row address of
359    /// the execution trace at which the permutation started.
360    ///
361    /// The address is only needed for operation helpers in trace generation, and thus an
362    /// implementation might choose to return a default/invalid address if it is not needed.
363    fn permute(&mut self, state: HasherState) -> Result<(Felt, HasherState), OperationError>;
364
365    /// Verifies that the `claimed_root` is indeed the root of a Merkle tree containing `value` at
366    /// the specified `index`.
367    ///
368    /// Returns the address of the computation, defined as the row index of the Hasher chiplet trace
369    /// at which the computation started.
370    ///
371    /// The Merkle path is guaranteed to be provided by [AdviceProviderInterface::get_merkle_path],
372    /// and thus an implementation might choose to return `None` from `get_merkle_path`, and ignore
373    /// it in this method.
374    ///
375    /// Additionally, the address is only needed for operation helpers in trace generation, and thus
376    /// an implementation might choose to return a default/invalid address if it is not needed.
377    fn verify_merkle_root(
378        &mut self,
379        claimed_root: Word,
380        value: Word,
381        path: Option<&MerklePath>,
382        index: Felt,
383        on_err: impl FnOnce() -> OperationError,
384    ) -> Result<Felt, OperationError>;
385
386    /// Verifies that the `claimed_old_root` is indeed the root of a Merkle tree containing
387    /// `old_value` at the specified `index`, and computes a new Merkle root after updating the node
388    /// at `index` to `new_value`.
389    ///
390    /// Returns the new root and the address of the computation, defined as the row index of the
391    /// Hasher chiplet trace at which the computation started.
392    ///
393    /// The Merkle path is guaranteed to be provided by [AdviceProviderInterface::get_merkle_path],
394    /// and thus an implementation might choose to return `None` from `get_merkle_path`, and ignore
395    /// it in this method.
396    ///
397    /// Additionally, the address is only needed for operation helpers in trace generation, and thus
398    /// an implementation might choose to return a default/invalid address if it is not needed.
399    fn update_merkle_root(
400        &mut self,
401        claimed_old_root: Word,
402        old_value: Word,
403        new_value: Word,
404        path: Option<&MerklePath>,
405        index: Felt,
406        on_err: impl FnOnce() -> OperationError,
407    ) -> Result<(Felt, Word), OperationError>;
408}
