15/// Performs FRI layer folding by a factor of 4 for FRI protocol executed in a degree 2
16/// extension of the base field. Additionally, performs several computations which simplify
17/// FRI verification procedure.
18///
19/// Specifically:
20/// - Folds 4 query values (v0, v1), (v2, v3), (v4, v5), (v6, v7) into a single value (ne0, ne1).
21/// - Computes new value of the domain generator power: poe' = poe^4.
22/// - Increments layer pointer (cptr) by 8.
23/// - Checks that the previous folding was done correctly.
24/// - Shifts the stack to the left to move an item from the overflow table to stack position 15.
25///
26/// Stack transition for this operation looks as follows:
27///
28/// Input:
29/// [v0, v1, v2, v3, v4, v5, v6, v7, f_pos, d_seg, poe, pe1, pe0, a1, a0, cptr, ...]
30///
31/// Output:
32/// [t1, t0, s1, s0, df3, df2, df1, df0, poe^2, f_tau, cptr+2, poe^4, f_pos, ne1, ne0, eptr,
33/// ...]
34///
35/// In the above, eptr is moved from the stack overflow table and is expected to be the address
36/// of the final FRI layer.
37///
38/// To keep the degree of the constraints low, a number of intermediate values are used.
39/// Specifically, the operation relies on all 6 helper registers, and also uses the first 10
40/// elements of the stack at the next state for degree reduction purposes. Thus, once the
41/// operation has been executed, the top 10 elements of the stack can be considered to be
42/// "garbage".
43#[inline(always)]
44pub(super) fn op_fri_ext2fold4<P>(
45    processor: &mut P,
46) -> Result<OperationHelperRegisters, OperationError>
47where
48    P: Processor,
49{
50    // --- read all relevant variables from the stack ---------------------
51    let query_values = get_query_values(processor);
52    let folded_pos = processor.stack().get(8);
53    // the segment identifier of the position in the source domain
54    let domain_segment = processor.stack().get(9).as_canonical_u64();
55    // the power of the domain generator which can be used to determine current domain value x
56    let poe = processor.stack().get(10);
57    if poe.is_zero() {
58        return Err(OperationError::FriError("domain size was 0".into()));
59    }
60    // the result of the previous layer folding
61    let prev_value = {
62        let pe1 = processor.stack().get(11);
63        let pe0 = processor.stack().get(12);
64        QuadFelt::from_basis_coefficients_fn(|i: usize| [pe0, pe1][i])
65    };
66    // the verifier challenge for the current layer
67    let alpha = {
68        let a1 = processor.stack().get(13);
69        let a0 = processor.stack().get(14);
70        QuadFelt::from_basis_coefficients_fn(|i: usize| [a0, a1][i])
71    };
72    // the memory address of the current layer
73    let layer_ptr = processor.stack().get(15);
74
75    // --- make sure the previous folding was done correctly --------------
76    if domain_segment > 3 {
77        return Err(OperationError::FriError(format!(
78            "domain segment value cannot exceed 3, but was {domain_segment}"
79        )));
80    }
81
82    let d_seg = domain_segment as usize;
83    if query_values[d_seg] != prev_value {
84        return Err(OperationError::FriError(format!(
85            "degree-respecting projection is inconsistent: expected {} but was {}",
86            prev_value, query_values[d_seg]
87        )));
88    }
89
90    // --- fold query values ----------------------------------------------
91    let f_tau = get_tau_factor(d_seg);
92    let x = poe * f_tau;
93    let x_inv = x.inverse();
94
95    let (ev, es) = compute_evaluation_points(alpha, x_inv);
96    let (folded_value, tmp0, tmp1) = fold4(query_values, ev, es);
97
98    // --- write the relevant values into the next state of the stack -----
99    let tmp0 = tmp0.as_basis_coefficients_slice();
100    let tmp1 = tmp1.as_basis_coefficients_slice();
101    let ds = get_domain_segment_flags(d_seg);
102    let folded_value = folded_value.as_basis_coefficients_slice();
103
104    let poe2 = poe * poe;
105    let poe4 = poe2 * poe2;
106
107    processor.stack_mut().decrement_size()?;
108
109    processor.stack_mut().set(0, tmp0[1]);
110    processor.stack_mut().set(1, tmp0[0]);
111    processor.stack_mut().set(2, tmp1[1]);
112    processor.stack_mut().set(3, tmp1[0]);
113    processor.stack_mut().set_word(4, &ds.into());
114    processor.stack_mut().set(8, poe2);
115    processor.stack_mut().set(9, f_tau);
116    processor.stack_mut().set(10, layer_ptr + EIGHT);
117    processor.stack_mut().set(11, poe4);
118    processor.stack_mut().set(12, folded_pos);
119    processor.stack_mut().set(13, folded_value[1]);
120    processor.stack_mut().set(14, folded_value[0]);
121
122    Ok(OperationHelperRegisters::FriExt2Fold4 { ev, es, x, x_inv })
123}

128/// Returns 4 query values in the source domain. These values are to be folded into a single
129/// value in the folded domain.
130///
131/// Stack layout: positions 0-7 contain coefficients for 4 QuadFelt elements.
132/// QuadFelts are constructed as:
133/// - query_values[0] = (v0, v1), query_values[1] = (v2, v3)
134/// - query_values[2] = (v4, v5), query_values[3] = (v6, v7)
135#[inline(always)]
136fn get_query_values<P: Processor>(processor: &mut P) -> [QuadFelt; 4] {
137    let [v0, v1, v2, v3]: [Felt; 4] = processor.stack().get_word(0).into();
138    let [v4, v5, v6, v7]: [Felt; 4] = processor.stack().get_word(4).into();
139
140    [
141        QuadFelt::from_basis_coefficients_fn(|i: usize| [v0, v1][i]),
142        QuadFelt::from_basis_coefficients_fn(|i: usize| [v2, v3][i]),
143        QuadFelt::from_basis_coefficients_fn(|i: usize| [v4, v5][i]),
144        QuadFelt::from_basis_coefficients_fn(|i: usize| [v6, v7][i]),
145    ]
146}

151const EIGHT: Felt = Felt::new(8);
152const TWO_INV: Felt = Felt::new(9223372034707292161);
153
154// Pre-computed powers of 1/tau, where tau is the generator of multiplicative subgroup of size 4
155// (i.e., tau is the 4th root of unity). Correctness of these constants is checked in the test at
156// the end of this module.
157const TAU_INV: Felt = Felt::new(18446462594437873665); // tau^{-1}
158const TAU2_INV: Felt = Felt::new(18446744069414584320); // tau^{-2}
159const TAU3_INV: Felt = Felt::new(281474976710656); // tau^{-3}
160
161/// Determines tau factor (needed to compute x value) for the specified domain segment.
162fn get_tau_factor(domain_segment: usize) -> Felt {
163    match domain_segment {
164        0 => ONE,
165        1 => TAU_INV,
166        2 => TAU2_INV,
167        3 => TAU3_INV,
168        _ => panic!("invalid domain segment {domain_segment}"),
169    }
170}
171
172/// Determines a set of binary flags needed to describe the specified domain segment.
173fn get_domain_segment_flags(domain_segment: usize) -> [Felt; 4] {
174    match domain_segment {
175        0 => [ONE, ZERO, ZERO, ZERO],
176        1 => [ZERO, ONE, ZERO, ZERO],
177        2 => [ZERO, ZERO, ONE, ZERO],
178        3 => [ZERO, ZERO, ZERO, ONE],
179        _ => panic!("invalid domain segment {domain_segment}"),
180    }
181}

183/// Computes 2 evaluation points needed for [fold4] function.
184fn compute_evaluation_points(alpha: QuadFelt, x_inv: Felt) -> (QuadFelt, QuadFelt) {
185    let ev = alpha * x_inv;
186    let es = ev * ev;
187    (ev, es)
188}
189
190/// Performs folding by a factor of 4. ev and es are values computed based on x and
191/// verifier challenge alpha as follows:
192/// - ev = alpha / x
193/// - es = (alpha / x)^2
194fn fold4(values: [QuadFelt; 4], ev: QuadFelt, es: QuadFelt) -> (QuadFelt, QuadFelt, QuadFelt) {
195    let tmp0 = fold2(values[0], values[2], ev);
196    let tmp1 = fold2(values[1], values[3], ev * TAU_INV);
197    let folded_value = fold2(tmp0, tmp1, es);
198    (folded_value, tmp0, tmp1)
199}
200
201/// Performs folding by a factor of 2. ep is a value computed based on x and verifier challenge
202/// alpha.
203fn fold2(f_x: QuadFelt, f_neg_x: QuadFelt, ep: QuadFelt) -> QuadFelt {
204    (f_x + f_neg_x + ((f_x - f_neg_x) * ep)) * TWO_INV
205}
