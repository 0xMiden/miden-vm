15/// Performs FRI layer folding by a factor of 4 for FRI protocol executed in a degree 2
16/// extension of the base field. Additionally, performs several computations which simplify
17/// FRI verification procedure.
18///
19/// Specifically:
20/// - Folds 4 query values (v0, v1), (v2, v3), (v4, v5), (v6, v7) into a single value (ne0, ne1).
21/// - Computes new value of the domain generator power: poe' = poe^4.
22/// - Increments layer pointer (cptr) by 8.
23/// - Checks that the previous folding was done correctly.
24/// - Shifts the stack to the left to move an item from the overflow table to stack position 15.
25///
26/// Bit-reversal handling:
27/// - Query values exist on the stack in bit-reversed order.
28/// - p on the stack is a bit-reversed tree index. The instruction internally computes d_seg = p & 3
29///   (bit-reversed coset index) for the consistency check and tau factor. f_pos is provided
30///   separately on the stack and the AIR constrains f_pos = p >> 2 via p = 4*f_pos + d_seg.
31///
32/// Stack transition for this operation looks as follows:
33///
34/// Input:
35/// [v0, v1, v2, v3, v4, v5, v6, v7, f_pos, p, poe, pe0, pe1, a0, a1, cptr, ...]
36///
37/// Output:
38/// [t0, t1, s0, s1, df3, df2, df1, df0, poe^2, f_tau, cptr+8, poe^4, f_pos, ne0, ne1, eptr, ...]
39///
40/// In the above, eptr is moved from the stack overflow table and is expected to be the address
41/// of the final FRI layer.
42///
43/// To keep the degree of the constraints low, a number of intermediate values are used.
44/// Specifically, the operation relies on all 6 helper registers, and also uses the first 10
45/// elements of the stack at the next state for degree reduction purposes. Thus, once the
46/// operation has been executed, the top 10 elements of the stack can be considered to be
47/// "garbage".
48#[inline(always)]
49pub(super) fn op_fri_ext2fold4<P>(
50    processor: &mut P,
51) -> Result<OperationHelperRegisters, OperationError>
52where
53    P: Processor,
54{
55    // --- read all relevant variables from the stack ---------------------
56    let query_values = get_query_values(processor);
57    // Reorder from bit-reversed to natural for fold4.
58    let query_values_reordered = reorder_bitrev4(query_values);
59    // p is the bit-reversed tree index; compute f_pos and d_seg from it
60    let p = processor.stack().get(9).as_canonical_u64();
61    let domain_segment = p & 3;
62    let folded_pos = Felt::new(p >> 2);
63    // the power of the domain generator which can be used to determine current domain value x
64    let poe = processor.stack().get(10);
65    if poe.is_zero() {
66        return Err(OperationError::FriError("domain size was 0".into()));
67    }
68    // the result of the previous layer folding
69    let prev_value = {
70        let pe0 = processor.stack().get(11);
71        let pe1 = processor.stack().get(12);
72        QuadFelt::from_basis_coefficients_fn(|i: usize| [pe0, pe1][i])
73    };
74    // the verifier challenge for the current layer
75    let alpha = {
76        let a0 = processor.stack().get(13);
77        let a1 = processor.stack().get(14);
78        QuadFelt::from_basis_coefficients_fn(|i: usize| [a0, a1][i])
79    };
80    // the memory address of the current layer
81    let layer_ptr = processor.stack().get(15);
82
83    // --- make sure the previous folding was done correctly --------------
84    if domain_segment > 3 {
85        return Err(OperationError::FriError(format!(
86            "domain segment value cannot exceed 3, but was {domain_segment}"
87        )));
88    }
89
90    // Consistency check: query_values[d_seg] == prev_value.
91    // Both query_values and d_seg are bit-reversed, so indexing works out correctly.
92    let d_seg = domain_segment as usize;
93    if query_values[d_seg] != prev_value {
94        return Err(OperationError::FriError(format!(
95            "degree-respecting projection is inconsistent at d_seg={d_seg} poe={} fpos={}: expected {} but was {}; all values: [0]={} [1]={} [2]={} [3]={}",
96            poe.as_canonical_u64(),
97            folded_pos.as_canonical_u64(),
98            prev_value,
99            query_values[d_seg],
100            query_values[0],
101            query_values[1],
102            query_values[2],
103            query_values[3]
104        )));
105    }
106
107    // --- fold query values ----------------------------------------------
108    // Bit-reverse d_seg to get natural coset index for tau factor and domain segment flags.
109    let d_seg_rev = bit_reverse_segment(d_seg);
110    let f_tau = get_tau_factor(d_seg_rev);
111    let x = poe * f_tau;
112    let x_inv = x.inverse();
113
114    let (ev, es) = compute_evaluation_points(alpha, x_inv);
115    let (folded_value, tmp0, tmp1) = fold4(query_values_reordered, ev, es);
116
117    // --- write the relevant values into the next state of the stack -----
118    let tmp0 = tmp0.as_basis_coefficients_slice();
119    let tmp1 = tmp1.as_basis_coefficients_slice();
120    let ds = get_domain_segment_flags(d_seg_rev);
121    let folded_value = folded_value.as_basis_coefficients_slice();
122
123    let poe2 = poe * poe;
124    let poe4 = poe2 * poe2;
125
126    processor.stack_mut().decrement_size()?;
127
128    processor.stack_mut().set(0, tmp0[0]);
129    processor.stack_mut().set(1, tmp0[1]);
130    processor.stack_mut().set(2, tmp1[0]);
131    processor.stack_mut().set(3, tmp1[1]);
132    processor.stack_mut().set_word(4, &ds.into());
133    processor.stack_mut().set(8, poe2);
134    processor.stack_mut().set(9, f_tau);
135    processor.stack_mut().set(10, layer_ptr + EIGHT);
136    processor.stack_mut().set(11, poe4);
137    processor.stack_mut().set(12, folded_pos);
138    processor.stack_mut().set(13, folded_value[0]);
139    processor.stack_mut().set(14, folded_value[1]);
140
141    Ok(OperationHelperRegisters::FriExt2Fold4 { ev, es, x, x_inv })
142}

147/// Returns 4 query values in the source domain. These values are to be folded into a single
148/// value in the folded domain.
149///
150/// Stack layout: positions 0-7 contain coefficients for 4 QuadFelt elements.
151/// QuadFelts are constructed as:
152/// - query_values[0] = (v0, v1), query_values[1] = (v2, v3)
153/// - query_values[2] = (v4, v5), query_values[3] = (v6, v7)
154#[inline(always)]
155fn get_query_values<P: Processor>(processor: &mut P) -> [QuadFelt; 4] {
156    let [v0, v1, v2, v3]: [Felt; 4] = processor.stack().get_word(0).into();
157    let [v4, v5, v6, v7]: [Felt; 4] = processor.stack().get_word(4).into();
158
159    [
160        QuadFelt::from_basis_coefficients_fn(|i: usize| [v0, v1][i]),
161        QuadFelt::from_basis_coefficients_fn(|i: usize| [v2, v3][i]),
162        QuadFelt::from_basis_coefficients_fn(|i: usize| [v4, v5][i]),
163        QuadFelt::from_basis_coefficients_fn(|i: usize| [v6, v7][i]),
164    ]
165}

167/// Bit-reverses a 2-bit segment index: 0->0, 1->2, 2->1, 3->3.
168#[inline(always)]
169fn bit_reverse_segment(domain_segment: usize) -> usize {
170    match domain_segment {
171        0 => 0,
172        1 => 2,
173        2 => 1,
174        3 => 3,
175        _ => panic!("invalid domain segment {domain_segment}"),
176    }
177}
178
179/// Reorders 4 evals from bit-reversed order to natural order.
180#[inline(always)]
181fn reorder_bitrev4(values: [QuadFelt; 4]) -> [QuadFelt; 4] {
182    [values[0], values[2], values[1], values[3]]
183}

188const EIGHT: Felt = Felt::new(8);
189const TWO_INV: Felt = Felt::new(9223372034707292161);
190
191// Pre-computed powers of 1/tau, where tau is the generator of multiplicative subgroup of size 4
192// (i.e., tau is the 4th root of unity). Correctness of these constants is checked in the test at
193// the end of this module.
194const TAU_INV: Felt = Felt::new(18446462594437873665); // tau^{-1}
195const TAU2_INV: Felt = Felt::new(18446744069414584320); // tau^{-2}
196const TAU3_INV: Felt = Felt::new(281474976710656); // tau^{-3}
197
198/// Determines tau factor (needed to compute x value) for the specified domain segment.
199fn get_tau_factor(domain_segment: usize) -> Felt {
200    match domain_segment {
201        0 => ONE,
202        1 => TAU_INV,
203        2 => TAU2_INV,
204        3 => TAU3_INV,
205        _ => panic!("invalid domain segment {domain_segment}"),
206    }
207}
208
209/// Determines a set of binary flags needed to describe the specified domain segment.
210fn get_domain_segment_flags(domain_segment: usize) -> [Felt; 4] {
211    match domain_segment {
212        0 => [ONE, ZERO, ZERO, ZERO],
213        1 => [ZERO, ONE, ZERO, ZERO],
214        2 => [ZERO, ZERO, ONE, ZERO],
215        3 => [ZERO, ZERO, ZERO, ONE],
216        _ => panic!("invalid domain segment {domain_segment}"),
217    }
218}

220/// Computes 2 evaluation points needed for [fold4] function.
221fn compute_evaluation_points(alpha: QuadFelt, x_inv: Felt) -> (QuadFelt, QuadFelt) {
222    let ev = alpha * x_inv;
223    let es = ev * ev;
224    (ev, es)
225}
226
227/// Performs folding by a factor of 4. ev and es are values computed based on x and
228/// verifier challenge alpha as follows:
229/// - ev = alpha / x
230/// - es = (alpha / x)^2
231fn fold4(values: [QuadFelt; 4], ev: QuadFelt, es: QuadFelt) -> (QuadFelt, QuadFelt, QuadFelt) {
232    let tmp0 = fold2(values[0], values[2], ev);
233    let tmp1 = fold2(values[1], values[3], ev * TAU_INV);
234    let folded_value = fold2(tmp0, tmp1, es);
235    (folded_value, tmp0, tmp1)
236}
237
238/// Performs folding by a factor of 2. ep is a value computed based on x and verifier challenge
239/// alpha.
240fn fold2(f_x: QuadFelt, f_neg_x: QuadFelt, ep: QuadFelt) -> QuadFelt {
241    (f_x + f_neg_x + ((f_x - f_neg_x) * ep)) * TWO_INV
