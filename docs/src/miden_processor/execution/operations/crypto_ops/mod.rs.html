27/// Performs a hash permutation operation.
28/// Applies Poseidon2 permutation to the top 12 elements of the stack.
29///
30/// Stack layout:
31/// ```text
32/// stack[0..4]   = R1 word (rate word 1)      → state[0..4]
33/// stack[4..8]   = R2 word (rate word 2)      → state[4..8]
34/// stack[8..12]  = CAP word (capacity)        → state[8..12]
35/// ```
36///
37/// The top of the stack (`get(0)`) maps to `state[0]`, giving the sponge state
38/// `[R1, R2, CAP]` where R1[0] is at the top of the stack.
39#[inline(always)]
40pub(super) fn op_hperm<P: Processor, T: Tracer>(
41    processor: &mut P,
42    tracer: &mut T,
43) -> Result<OperationHelperRegisters, OperationError> {
44    // Build sponge state from stack: state[i] = stack.get(i)
45    // Read first 8 elements using get_double_word, then remaining 4 elements
46    let double_word: [Felt; 8] = processor.stack().get_double_word(0);
47    let word: Word = processor.stack().get_word(8);
48    let input_state: [Felt; STATE_WIDTH] = [
49        double_word[0],
50        double_word[1],
51        double_word[2],
52        double_word[3],
53        double_word[4],
54        double_word[5],
55        double_word[6],
56        double_word[7],
57        word[0],
58        word[1],
59        word[2],
60        word[3],
61    ];
62
63    // Apply Poseidon2 permutation
64    let (addr, output_state) = processor.hasher().permute(input_state)?;
65
66    // Write result back to stack (state[0] at top).
67    let r0: Word = output_state[STATE_RATE_0_RANGE].try_into().expect("r0 slice has length 4");
68    let r1: Word = output_state[STATE_RATE_1_RANGE].try_into().expect("r1 slice has length 4");
69    let cap: Word = output_state[STATE_CAP_RANGE].try_into().expect("cap slice has length 4");
70    processor.stack_mut().set_word(0, &r0);
71    processor.stack_mut().set_word(4, &r1);
72    processor.stack_mut().set_word(8, &cap);
73
74    tracer.record_hasher_permute(input_state, output_state);
75    Ok(OperationHelperRegisters::HPerm { addr })
76}

78/// Verifies that a Merkle path from the specified node resolves to the specified root. The
79/// stack is expected to be arranged as follows (from the top):
80/// - value of the node, 4 elements.
81/// - depth of the node, 1 element; this is expected to be the depth of the Merkle tree
82/// - index of the node, 1 element.
83/// - root of the tree, 4 elements.
84///
85/// To perform the operation we do the following:
86/// 1. Look up the Merkle path in the advice provider for the specified tree root.
87/// 2. Use the hasher to compute the root of the Merkle path for the specified node.
88/// 3. Verify that the computed root is equal to the root provided via the stack.
89/// 4. Copy the stack state over to the next clock cycle with no changes.
90///
91/// # Errors
92/// Returns an error if:
93/// - Merkle tree for the specified root cannot be found in the advice provider.
94/// - The specified depth is either zero or greater than the depth of the Merkle tree identified by
95///   the specified root.
96/// - Path to the node at the specified depth and index is not known to the advice provider.
97#[inline(always)]
98pub(super) fn op_mpverify<P: Processor, T: Tracer>(
99    processor: &mut P,
100    err_code: Felt,
101    program: &MastForest,
102    tracer: &mut T,
103) -> Result<OperationHelperRegisters, CryptoError> {
104    // read node value, depth, index and root value from the stack
105    let node = processor.stack().get_word(0);
106    let depth = processor.stack().get(4);
107    let index = processor.stack().get(5);
108    let root = processor.stack().get_word(6);
109
110    // get a Merkle path from the advice provider for the specified root and node index
111    let path = processor.advice_provider().get_merkle_path(root, depth, index)?;
112
113    tracer.record_hasher_build_merkle_root(node, path.as_ref(), index, root);
114
115    // verify the path
116    let addr = processor.hasher().verify_merkle_root(root, node, path.as_ref(), index, || {
117        // If the hasher doesn't compute the same root (using the same path),
118        // then it means that `node` is not the value currently in the tree at `index`
119        let err_msg = program.resolve_error_message(err_code);
120        OperationError::MerklePathVerificationFailed {
121            inner: Box::new(MerklePathVerificationFailedInner {
122                value: node,
123                index,
124                root,
125                err_code,
126                err_msg,
127            }),
128        }
129    })?;
130
131    Ok(OperationHelperRegisters::MerklePath { addr })
132}

134/// Computes a new root of a Merkle tree where a node at the specified index is updated to
135/// the specified value. The stack is expected to be arranged as follows (from the top):
136/// - old value of the node, 4 elements.
137/// - depth of the node, 1 element; this is expected to be the depth of the Merkle tree.
138/// - index of the node, 1 element.
139/// - current root of the tree, 4 elements.
140/// - new value of the node, 4 elements.
141///
142/// To perform the operation we do the following:
143/// 1. Update the node at the specified index in the Merkle tree with the specified root, and get
144///    the Merkle path to it.
145/// 2. Use the hasher to update the root of the Merkle path for the specified node. For this we need
146///    to provide the old and the new node value.
147/// 3. Verify that the computed old root is equal to the input root provided via the stack.
148/// 4. Replace the old node value with the computed new root.
149///
150/// The Merkle path for the node is expected to be provided by the prover non-deterministically
151/// (via the advice provider). At the end of the operation, the old node value is replaced with
152/// the new roots value computed based on the provided path. Everything else on the stack
153/// remains the same.
154///
155/// The original Merkle tree is cloned before the update is performed, and thus, after the
156/// operation, the advice provider will keep track of both the old and the new trees.
157///
158/// # Errors
159/// Returns an error if:
160/// - Merkle tree for the specified root cannot be found in the advice provider.
161/// - The specified depth is either zero or greater than the depth of the Merkle tree identified by
162///   the specified root.
163/// - Path to the node at the specified depth and index is not known to the advice provider.
164#[inline(always)]
165pub(super) fn op_mrupdate<P: Processor, T: Tracer>(
166    processor: &mut P,
167    tracer: &mut T,
168) -> Result<OperationHelperRegisters, CryptoError> {
169    // read old node value, depth, index, tree root and new node values from the stack
170    let old_value = processor.stack().get_word(0);
171    let depth = processor.stack().get(4);
172    let index = processor.stack().get(5);
173    let claimed_old_root = processor.stack().get_word(6);
174    let new_value = processor.stack().get_word(10);
175
176    // update the node at the specified index in the Merkle tree specified by the old root, and
177    // get a Merkle path to it. The length of the returned path is expected to match the
178    // specified depth. If the new node is the root of a tree, this instruction will append the
179    // whole sub-tree to this node.
180    let path = processor.advice_provider_mut().update_merkle_node(
181        claimed_old_root,
182        depth,
183        index,
184        new_value,
185    )?;
186
187    if let Some(path) = &path
188        && path.len() != depth.as_canonical_u64() as usize
189    {
190        return Err(OperationError::InvalidMerklePathLength { path_len: path.len(), depth }.into());
191    }
192
193    let (addr, new_root) = processor.hasher().update_merkle_root(
194        claimed_old_root,
195        old_value,
196        new_value,
197        path.as_ref(),
198        index,
199        || OperationError::MerklePathVerificationFailed {
200            inner: Box::new(MerklePathVerificationFailedInner {
201                value: old_value,
202                index,
203                root: claimed_old_root,
204                err_code: ZERO,
205                err_msg: None,
206            }),
207        },
208    )?;
209    tracer.record_hasher_update_merkle_root(
210        old_value,
211        new_value,
212        path.as_ref(),
213        index,
214        claimed_old_root,
215        new_root,
216    );
217
218    // Replace the old node value with computed new root.
219    processor.stack_mut().set_word(0, &new_root);
220
221    Ok(OperationHelperRegisters::MerklePath { addr })
222}

227/// Performs 8 steps of the Horner evaluation method on a polynomial with coefficients over
228/// the base field using a 3-level computation to reduce constraint degree.
229///
230/// The computation processes 8 base field coefficients from the stack using Horner's method.
231/// If we denote the values at stack positions 0..7 as `s[0]..s[7]`, the computation is:
232///
233/// - Level 1: tmp0 = (acc * α + s[0]) * α + s[1]
234/// - Level 2: tmp1 = ((tmp0 * α + s[2]) * α + s[3]) * α + s[4]
235/// - Level 3: acc' = ((tmp1 * α + s[5]) * α + s[6]) * α + s[7]
236///
237/// This evaluates the polynomial:
238///
239/// P(X) := s[0] * X^7 + s[1] * X^6 + s[2] * X^5 + s[3] * X^4 + s[4] * X^3 + s[5] * X^2 + s[6] * X +
240/// s[7]
241///
242/// where s[0] is the highest-degree coefficient and s[7] is the constant term.
243///
244/// The instruction can be used to compute the evaluation of polynomials of arbitrary degree
245/// by repeated invocations interleaved with any operation that loads the next batch of 8
246/// coefficients on the top of the operand stack, i.e., `mem_stream` or `adv_pipe`.
247///
248/// The stack transition of the instruction can be visualized as follows:
249///
250/// Input:
251///
252/// +------+------+------+------+------+------+------+------+---+---+---+---+---+----------+------+------+
253/// | s[0] | s[1] | s[2] | s[3] | s[4] | s[5] | s[6] | s[7] | - | - | - | - | - |alpha_addr| acc1 | acc0 |
254/// +------+------+------+------+------+------+------+------+---+---+---+---+---+----------+------+------+
255///   (X^7)  (X^6)  (X^5)  (X^4)  (X^3)  (X^2)  (X^1)  (X^0)
256///
257/// Output:
258///
259/// +------+------+------+------+------+------+------+------+---+---+---+---+---+----------+-------+-------+
260/// | s[0] | s[1] | s[2] | s[3] | s[4] | s[5] | s[6] | s[7] | - | - | - | - | - |alpha_addr| acc1' | acc0' |
261/// +------+------+------+------+------+------+------+------+---+---+---+---+---+----------+-------+-------+
262///
263/// Here:
264///
265/// 1. s[i] for i in 0..=7 is the coefficient at stack position i. s[0] is the highest-degree
266///    coefficient (X^7) and s[7] is the constant term (X^0).
267/// 2. (acc0, acc1) is a quadratic extension field element accumulating the Horner evaluation.
268///    (acc0', acc1') is the updated accumulator after processing this batch.
269/// 3. alpha_addr is the memory address of the evaluation point α = (α₀, α₁). The operation reads α₀
270///    from alpha_addr and α₁ from alpha_addr + 1.
271///
272/// The instruction uses helper registers to store intermediate values:
273/// - h₀, h₁: evaluation point α = (α₀, α₁)
274/// - h₂, h₃: Level 2 intermediate result tmp1
275/// - h₄, h₅: Level 1 intermediate result tmp0
276#[inline(always)]
277pub(super) fn op_horner_eval_base<P: Processor, T: Tracer>(
278    processor: &mut P,
279    tracer: &mut T,
280) -> Result<OperationHelperRegisters, crate::MemoryError> {
281    // Stack positions: low coefficient closer to top (lower index)
282    const ALPHA_ADDR_INDEX: usize = 13;
283    const ACC_LOW_INDEX: usize = 14;
284    const ACC_HIGH_INDEX: usize = 15;
285
286    let clk = processor.system().clock();
287    let ctx = processor.system().ctx();
288
289    // Read the evaluation point alpha from memory
290    let alpha = {
291        let addr = processor.stack().get(ALPHA_ADDR_INDEX);
292        let eval_point_0 = processor.memory_mut().read_element(ctx, addr)?;
293        let eval_point_1 = processor.memory_mut().read_element(ctx, addr + ONE)?;
294
295        tracer.record_memory_read_element_pair(
296            eval_point_0,
297            addr,
298            eval_point_1,
299            addr + ONE,
300            ctx,
301            clk,
302        );
303
304        QuadFelt::from_basis_coefficients_fn(|i: usize| [eval_point_0, eval_point_1][i])
305    };
306
307    // Read the coefficients from the stack (top 8 elements)
308    let coef: [Felt; 8] = processor.stack().get_double_word(0);
309
310    let c0 = QuadFelt::from(coef[0]);
311    let c1 = QuadFelt::from(coef[1]);
312    let c2 = QuadFelt::from(coef[2]);
313    let c3 = QuadFelt::from(coef[3]);
314    let c4 = QuadFelt::from(coef[4]);
315    let c5 = QuadFelt::from(coef[5]);
316    let c6 = QuadFelt::from(coef[6]);
317    let c7 = QuadFelt::from(coef[7]);
318
319    // Read the current accumulator (LE: low at lower index)
320    let acc_low = processor.stack().get(ACC_LOW_INDEX);
321    let acc_high = processor.stack().get(ACC_HIGH_INDEX);
322    let acc = QuadFelt::from_basis_coefficients_fn(|i: usize| [acc_low, acc_high][i]);
323
324    // Level 1: tmp0 = (acc * α + c₀) * α + c₁
325    let tmp0 = (acc * alpha + c0) * alpha + c1;
326
327    // Level 2: tmp1 = ((tmp0 * α + c₂) * α + c₃) * α + c₄
328    let tmp1 = ((tmp0 * alpha + c2) * alpha + c3) * alpha + c4;
329
330    // Level 3: acc' = ((tmp1 * α + c₅) * α + c₆) * α + c₇
331    let acc_new = ((tmp1 * alpha + c5) * alpha + c6) * alpha + c7;
332
333    // Update the accumulator values on the stack (LE: low at lower index)
334    let acc_new_base_elements = acc_new.as_basis_coefficients_slice();
335    processor.stack_mut().set(ACC_HIGH_INDEX, acc_new_base_elements[1]);
336    processor.stack_mut().set(ACC_LOW_INDEX, acc_new_base_elements[0]);
337
338    // Return the user operation helpers
339    Ok(OperationHelperRegisters::HornerEvalBase { alpha, tmp0, tmp1 })
340}

342/// Performs 4 steps of the Horner evaluation method on a polynomial with coefficients over
343/// the quadratic extension field.
344///
345/// The computation processes 4 extension field coefficients from the stack using Horner's method.
346/// If we denote the QuadFelt values at stack positions (0,1), (2,3), (4,5), (6,7) as
347/// `s[0]..s[3]`, the computation is:
348///
349/// - Level 1: acc_tmp = (acc * α + s[0]) * α + s[1]
350/// - Level 2: acc' = ((acc_tmp * α + s[2]) * α + s[3]
351///
352/// This evaluates the polynomial:
353///
354/// P(X) := s[0] * X^3 + s[1] * X^2 + s[2] * X + s[3]
355///
356/// where s[0] is the highest-degree coefficient and s[3] is the constant term.
357///
358/// The instruction can be used to compute the evaluation of polynomials of arbitrary degree
359/// by repeated invocations interleaved with any operation that loads the next batch of 4
360/// coefficients on the top of the operand stack, i.e., `mem_stream` or `adv_pipe`.
361///
362/// The stack transition of the instruction can be visualized as follows:
363///
364/// Input:
365///
366/// +-------+-------+-------+-------+-------+-------+-------+-------+---+---+---+---+---+----------+------+------+
367/// | s0_lo | s0_hi | s1_lo | s1_hi | s2_lo | s2_hi | s3_lo | s3_hi | - | - | - | - | - |alpha_addr| acc0 | acc1 |
368/// +-------+-------+-------+-------+-------+-------+-------+-------+---+---+---+---+---+----------+------+------+
369///   (X^3)           (X^2)           (X^1)           (X^0)
370///
371/// Output:
372///
373/// +-------+-------+-------+-------+-------+-------+-------+-------+---+---+---+---+---+----------+-------+-------+
374/// | s0_lo | s0_hi | s1_lo | s1_hi | s2_lo | s2_hi | s3_lo | s3_hi | - | - | - | - | - |alpha_addr| acc0' | acc1' |
375/// +-------+-------+-------+-------+-------+-------+-------+-------+---+---+---+---+---+----------+-------+-------+
376///
377/// Here:
378///
379/// 1. s[i] = (si_lo, si_hi) for i in 0..=3 is the extension field coefficient at stack position
380///    2*i. s[0] is the highest-degree coefficient (X^3) and s[3] is the constant term (X^0).
381/// 2. (acc0, acc1) is a quadratic extension field element accumulating the Horner evaluation.
382///    (acc0', acc1') is the updated accumulator after processing this batch.
383/// 3. alpha_addr is the memory address of the evaluation point α = (α₀, α₁).
384///
385/// The instruction uses helper registers to hold α and the intermediate value acc_tmp.
386#[inline(always)]
387pub(super) fn op_horner_eval_ext<P: Processor, T: Tracer>(
388    processor: &mut P,
389    tracer: &mut T,
390) -> Result<OperationHelperRegisters, crate::MemoryError> {
391    // Stack positions: low coefficient closer to top (lower index)
392    const ALPHA_ADDR_INDEX: usize = 13;
393    const ACC_LOW_INDEX: usize = 14;
394    const ACC_HIGH_INDEX: usize = 15;
395
396    let clk = processor.system().clock();
397    let ctx = processor.system().ctx();
398
399    // Read the coefficients from the stack as extension field elements (4 QuadFelt elements)
400    // Stack layout: [s0_lo, s0_hi, s1_lo, s1_hi, s2_lo, s2_hi, s3_lo, s3_hi, ...]
401    // s[0] at stack[0,1] is highest degree (X^3), s[3] at stack[6,7] is constant (X^0)
402    let coef: [QuadFelt; 4] = core::array::from_fn(|j| {
403        let lo = processor.stack().get(2 * j);
404        let hi = processor.stack().get(2 * j + 1);
405        QuadFelt::from_basis_coefficients_fn(|i: usize| [lo, hi][i])
406    });
407
408    // Read the evaluation point alpha from memory
409    let (alpha, k0, k1) = {
410        let addr = processor.stack().get(ALPHA_ADDR_INDEX);
411        let word = processor.memory_mut().read_word(ctx, addr, clk)?;
412        tracer.record_memory_read_word(
413            word,
414            addr,
415            processor.system().ctx(),
416            processor.system().clock(),
417        );
418
419        (
420            QuadFelt::from_basis_coefficients_fn(|i: usize| [word[0], word[1]][i]),
421            word[2],
422            word[3],
423        )
424    };
425
426    // Read the current accumulator (LE: low at lower index)
427    let acc_low = processor.stack().get(ACC_LOW_INDEX);
428    let acc_high = processor.stack().get(ACC_HIGH_INDEX);
429    let acc_old = QuadFelt::from_basis_coefficients_fn(|i: usize| [acc_low, acc_high][i]);
430
431    // Compute the temporary accumulator (first 2 coefficients from stack)
432    // Process coef[0], coef[1] (highest degree coefficients)
433    let acc_tmp = coef.iter().take(2).fold(acc_old, |acc, coef| *coef + alpha * acc);
434
435    // Compute the final accumulator (remaining 2 coefficients)
436    // Process coef[2], coef[3] (lower degree coefficients)
437    let acc_new = coef.iter().skip(2).fold(acc_tmp, |acc, coef| *coef + alpha * acc);
438
439    // Update the accumulator values on the stack (LE: low at lower index)
440    let acc_new_base_elements = acc_new.as_basis_coefficients_slice();
441    processor.stack_mut().set(ACC_HIGH_INDEX, acc_new_base_elements[1]);
442    processor.stack_mut().set(ACC_LOW_INDEX, acc_new_base_elements[0]);
443
444    // Return the user operation helpers
445    Ok(OperationHelperRegisters::HornerEvalExt { alpha, k0, k1, acc_tmp })
446}

448// LOG PRECOMPILE OPERATION
449// ================================================================================================
450
451/// Logs a precompile event by absorbing `TAG` and `COMM` into the precompile sponge
452/// capacity.
453///
454/// Stack transition:
455/// `[COMM, TAG, PAD, ...] -> [R0, R1, CAP_NEXT, ...]`
456///
457/// Where:
458/// - The hasher computes: `[R0, R1, CAP_NEXT] = Poseidon2([COMM, TAG, CAP_PREV])`
459/// - `CAP_PREV` is the previous sponge capacity provided non-deterministically via helper
460///   registers.
461/// - Stack elements are in LSB-first order (structural order).
462#[inline(always)]
463pub(super) fn op_log_precompile<P: Processor, T: Tracer>(
464    processor: &mut P,
465    tracer: &mut T,
466) -> Result<OperationHelperRegisters, OperationError> {
467    // Read COMM and TAG from the stack
468    let comm: Word = processor.stack().get_word(0);
469    let tag: Word = processor.stack().get_word(4);
470
471    // Get the current precompile sponge capacity
472    let cap_prev = processor.precompile_transcript_state();
473
474    // Build the full 12-element hasher state for Poseidon2 permutation
475    // State layout: [RATE0 = COMM, RATE1 = TAG, CAPACITY = CAP_PREV]
476    let mut hasher_state: [Felt; STATE_WIDTH] = [ZERO; 12];
477    hasher_state[STATE_RATE_0_RANGE].copy_from_slice(comm.as_slice());
478    hasher_state[STATE_RATE_1_RANGE].copy_from_slice(tag.as_slice());
479    hasher_state[STATE_CAP_RANGE].copy_from_slice(cap_prev.as_slice());
480
481    // Perform the Poseidon2 permutation
482    let (addr, output_state) = processor.hasher().permute(hasher_state)?;
483
484    // Extract R0, R1 and CAP_NEXT from the output state
485    let r0: Word = output_state[STATE_RATE_0_RANGE.clone()]
486        .try_into()
487        .expect("r0 slice has length 4");
488    let r1: Word = output_state[STATE_RATE_1_RANGE.clone()]
489        .try_into()
490        .expect("r1 slice has length 4");
491    let cap_next: Word = output_state[STATE_CAP_RANGE.clone()]
492        .try_into()
493        .expect("cap_next slice has length 4");
494
495    // Update the processor's precompile sponge capacity
496    processor.set_precompile_transcript_state(cap_next);
497
498    // Write the output to the stack (top 12 elements): [R0, R1, CAP_NEXT, ...].
499    processor.stack_mut().set_word(0, &r0);
500    processor.stack_mut().set_word(4, &r1);
501    processor.stack_mut().set_word(8, &cap_next);
502
503    // Record the hasher permutation for trace generation
504    tracer.record_hasher_permute(hasher_state, output_state);
505
506    // Return helper registers containing the hasher address and CAP_PREV
507    Ok(OperationHelperRegisters::LogPrecompile { addr, cap_prev })
508}

510// STREAM CIPHER OPERATION
511// ================================================================================================
512
513/// Encrypts data from source memory to destination memory using Poseidon2 sponge keystream.
514///
515/// This operation performs AEAD encryption by:
516/// 1. Loading 8 elements (2 words) from source memory at stack[12]
517/// 2. Adding each element to the corresponding rate element (stack[0..7])
518/// 3. Writing the resulting ciphertext to destination memory at stack[13]
519/// 4. Updating stack[0..7] with the ciphertext (becomes new rate for next hperm)
520/// 5. Preserving capacity (stack[8..11])
521/// 6. Incrementing both source and destination pointers by 8
522///
523/// Stack transition:
524/// [rate(8), cap(4), src_ptr, dst_ptr, ...] -> [ciphertext(8), cap(4), src_ptr+8, dst_ptr+8,
525/// ...]
526#[inline(always)]
527pub(super) fn op_crypto_stream<P: Processor, T: Tracer>(
528    processor: &mut P,
529    tracer: &mut T,
530) -> Result<OperationHelperRegisters, crate::MemoryError> {
531    // Stack layout: [rate(8), capacity(4), src_ptr, dst_ptr, ...]
532    const SRC_PTR_IDX: usize = 12;
533    const DST_PTR_IDX: usize = 13;
534
535    let ctx = processor.system().ctx();
536    let clk = processor.system().clock();
537
538    // Get source and destination pointers
539    let src_addr = processor.stack().get(SRC_PTR_IDX);
540    let dst_addr = processor.stack().get(DST_PTR_IDX);
541
542    // Validate address ranges and check for overlap using half-open intervals.
543    validate_dual_word_stream_addrs(src_addr, dst_addr, ctx, clk)?;
544
545    // Load plaintext from source memory (2 words = 8 elements)
546    let src_addr_word2 = src_addr + WORD_SIZE_FELT;
547    let plaintext_word1 = processor.memory_mut().read_word(ctx, src_addr, clk)?;
548    let plaintext_word2 = processor.memory_mut().read_word(ctx, src_addr_word2, clk)?;
549
550    // Get rate (keystream) from stack[0..7]
551    let rate: [Felt; 8] = processor.stack().get_double_word(0);
552
553    // Encrypt: ciphertext = plaintext + rate (element-wise addition in field)
554    let ciphertext_word1 = [
555        plaintext_word1[0] + rate[0],
556        plaintext_word1[1] + rate[1],
557        plaintext_word1[2] + rate[2],
558        plaintext_word1[3] + rate[3],
559    ]
560    .into();
561    let ciphertext_word2 = [
562        plaintext_word2[0] + rate[4],
563        plaintext_word2[1] + rate[5],
564        plaintext_word2[2] + rate[6],
565        plaintext_word2[3] + rate[7],
566    ]
567    .into();
568
569    // Write ciphertext to destination memory
570    let dst_addr_word2 = dst_addr + WORD_SIZE_FELT;
571    processor.memory_mut().write_word(ctx, dst_addr, clk, ciphertext_word1)?;
572    processor.memory_mut().write_word(ctx, dst_addr_word2, clk, ciphertext_word2)?;
573
574    tracer.record_crypto_stream(
575        [plaintext_word1, plaintext_word2],
576        src_addr,
577        [ciphertext_word1, ciphertext_word2],
578        dst_addr,
579        ctx,
580        clk,
581    );
582
583    // Update stack[0..7] with ciphertext (becomes new rate for next hperm)
584    processor.stack_mut().set_word(0, &ciphertext_word1);
585    processor.stack_mut().set_word(4, &ciphertext_word2);
586
587    // Increment pointers by 8 (2 words)
588    processor.stack_mut().set(SRC_PTR_IDX, src_addr + DOUBLE_WORD_SIZE);
589    processor.stack_mut().set(DST_PTR_IDX, dst_addr + DOUBLE_WORD_SIZE);
590
591    Ok(OperationHelperRegisters::Empty)
592}

597/// Validates that two 2-word (8-element) memory ranges starting at `src_addr` and `dst_addr`
598/// are within u32 bounds and do not overlap in the same cycle.
599///
600/// Uses half-open intervals: [addr, addr+8). If ranges overlap, returns an IllegalMemoryAccess
601/// error pointing at the first destination word that would be written.
602#[inline(always)]
603fn validate_dual_word_stream_addrs(
604    src_addr: Felt,
605    dst_addr: Felt,
606    ctx: ContextId,
607    clk: RowIndex,
608) -> Result<(), MemoryError> {
609    // Convert to u32 and check end-exclusive bounds
610    let src_addr_u64 = src_addr.as_canonical_u64();
611    let dst_addr_u64 = dst_addr.as_canonical_u64();
612
613    let src_addr_u32 = u32::try_from(src_addr_u64)
614        .map_err(|_| MemoryError::AddressOutOfBounds { addr: src_addr_u64 })?;
615    let src_end = src_addr_u32
616        .checked_add(8)
617        .ok_or(MemoryError::AddressOutOfBounds { addr: src_addr_u64 })?;
618
619    let dst_addr_u32 = u32::try_from(dst_addr_u64)
620        .map_err(|_| MemoryError::AddressOutOfBounds { addr: dst_addr_u64 })?;
621    let dst_end = dst_addr_u32
622        .checked_add(8)
623        .ok_or(MemoryError::AddressOutOfBounds { addr: dst_addr_u64 })?;
624
625    // Check for overlap between [src, src+8) and [dst, dst+8)
626    if src_addr_u32 < dst_end && dst_addr_u32 < src_end {
627        let dst_word2 = dst_addr_u32 + 4; // safe since dst_end computed above
628        // We write dst first, then dst+4. Use the first that overlaps.
629        let overlap_first = (dst_addr_u32 >= src_addr_u32) && (dst_addr_u32 < src_end);
630        let offending_addr = if overlap_first { dst_addr_u32 } else { dst_word2 };
631        return Err(MemoryError::IllegalMemoryAccess {
632            ctx,
633            addr: offending_addr,
634            clk: Felt::from(clk),
635        });
636    }
637
638    Ok(())
639}
