12/// Checks that the evaluation of an arithmetic circuit is equal to zero.
13///
14/// The inputs are composed of:
15///
16/// 1. a pointer to the memory region containing the arithmetic circuit description, which itself is
17///    arranged as:
18///
19///    a. `Read` section:
20///       1. Inputs to the circuit which are elements in the quadratic extension field,
21///       2. Constants of the circuit which are elements in the quadratic extension field,
22///
23///    b. `Eval` section, which contains the encodings of the evaluation gates of the circuit,
24///    where each gate is encoded as a single base field element.
25/// 2. the number of quadratic extension field elements read in the `READ` section,
26/// 3. the number of field elements, one base field element per gate, in the `EVAL` section,
27///
28/// Stack transition:
29/// [ptr, num_read, num_eval, ...] -> [ptr, num_read, num_eval, ...]
30#[inline(always)]
31pub(super) fn op_eval_circuit<P, T>(processor: &mut P, tracer: &mut T) -> Result<(), AceEvalError>
32where
33    P: Processor,
34    T: Tracer<Processor = P>,
35{
36    let num_eval = processor.stack().get(2);
37    let num_read = processor.stack().get(1);
38    let ptr = processor.stack().get(0);
39    let ctx = processor.system().ctx();
40    let clk = processor.system().clock();
41
42    let circuit_evaluation =
43        eval_circuit_impl(ctx, ptr, clk, num_read, num_eval, processor.memory_mut())?;
44    tracer.record_circuit_evaluation(circuit_evaluation);
45
46    Ok(())
47}

49/// Evaluates an arithmetic circuit encoded in memory starting at `ptr`.
50///
51/// This reads `num_vars` quadratic extension field elements from memory (the READ section), then
52/// reads `num_eval` base field element gate encodings (the EVAL section), evaluating each gate
53/// in sequence. Returns the resulting [`CircuitEvaluation`] if the circuit evaluates to zero.
54pub(crate) fn eval_circuit_impl(
55    ctx: ContextId,
56    ptr: Felt,
57    clk: RowIndex,
58    num_vars: Felt,
59    num_eval: Felt,
60    mem: &mut impl MemoryInterface,
61) -> Result<CircuitEvaluation, AceEvalError> {
62    let num_vars = num_vars.as_canonical_u64();
63    let num_eval = num_eval.as_canonical_u64();
64
65    let num_wires = num_vars.saturating_add(num_eval);
66    if num_wires > MAX_NUM_ACE_WIRES as u64 {
67        const {
68            // If this fails, update the error message below
69            assert!(MAX_NUM_ACE_WIRES == (1_u32 << 30) - 1);
70        }
71        return Err(
72            AceError(format!("num of wires must be less than 2^30 but was {num_wires}")).into()
73        );
74    }
75
76    // Ensure vars and instructions are word-aligned and non-empty. Note that variables are
77    // quadratic extension field elements while instructions are encoded as base field elements.
78    // Hence we can pack 2 variables and 4 instructions per word.
79    if !num_vars.is_multiple_of(2) || num_vars == 0 {
80        return Err(AceError(format!(
81            "num of variables should be word aligned and non-zero but was {num_vars}"
82        ))
83        .into());
84    }
85    if !num_eval.is_multiple_of(4) || num_eval == 0 {
86        return Err(AceError(format!(
87            "num of evaluation gates should be word aligned and non-zero but was {num_eval}"
88        ))
89        .into());
90    }
91
92    // Ensure instructions are word-aligned and non-empty
93    let num_read_rows = (num_vars / 2) as u32;
94    let num_eval_rows = num_eval as u32;
95
96    let mut evaluation_context = CircuitEvaluation::new(ctx, clk, num_read_rows, num_eval_rows);
97
98    let mut ptr = ptr;
99    // perform READ operations
100    for _ in 0..num_read_rows {
101        let word = mem.read_word(ctx, ptr, clk)?;
102        evaluation_context.do_read(ptr, word);
103        ptr += PTR_OFFSET_WORD;
104    }
105    // perform EVAL operations
106    for _ in 0..num_eval_rows {
107        let instruction = mem.read_element(ctx, ptr)?;
108        evaluation_context.do_eval(ptr, instruction)?;
109        ptr += PTR_OFFSET_ELEM;
110    }
111
112    // Ensure the circuit evaluated to zero.
113    if evaluation_context.output_value().is_none_or(|eval| eval != QuadFelt::ZERO) {
114        return Err(AceError("circuit does not evaluate to zero".into()).into());
115    }
116
117    Ok(evaluation_context)
118}
