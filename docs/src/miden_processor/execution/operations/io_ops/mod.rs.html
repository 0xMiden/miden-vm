17/// Pops an element from the advice stack and pushes it onto the operand stack.
18///
19/// # Errors
20/// Returns an error if the advice stack is empty.
21#[inline(always)]
22pub(super) fn op_advpop<P, T>(
23    processor: &mut P,
24    tracer: &mut T,
25) -> Result<OperationHelperRegisters, IoError>
26where
27    P: Processor,
28    T: Tracer<Processor = P>,
29{
30    let value = processor.advice_provider_mut().pop_stack()?;
31    tracer.record_advice_pop_stack(value);
32
33    processor.stack_mut().increment_size()?;
34    processor.stack_mut().set(0, value);
35
36    Ok(OperationHelperRegisters::Empty)
37}

39/// Pops a word (4 elements) from the advice stack and overwrites the top word on the operand
40/// stack with it.
41///
42/// # Errors
43/// Returns an error if the advice stack contains fewer than four elements.
44#[inline(always)]
45pub(super) fn op_advpopw<P, T>(
46    processor: &mut P,
47    tracer: &mut T,
48) -> Result<OperationHelperRegisters, IoError>
49where
50    P: Processor,
51    T: Tracer<Processor = P>,
52{
53    let word = processor.advice_provider_mut().pop_stack_word()?;
54    tracer.record_advice_pop_stack_word(word);
55
56    // Set word on stack (word[0] at top).
57    processor.stack_mut().set_word(0, &word);
58
59    Ok(OperationHelperRegisters::Empty)
60}

62/// Loads a word (4 elements) starting at the specified memory address onto the stack.
63///
64/// The operation works as follows:
65/// - The memory address is popped off the stack.
66/// - A word is retrieved from memory starting at the specified address, which must be aligned to a
67///   word boundary. The memory is always initialized to ZEROs, and thus, for any of the four
68///   addresses which were not previously been written to, four ZERO elements are returned.
69/// - The top four elements of the stack are overwritten with values retrieved from memory.
70///
71/// Thus, the net result of the operation is that the stack is shifted left by one item.
72///
73/// # Errors
74/// - Returns an error if the address is not aligned to a word boundary.
75#[inline(always)]
76pub(super) fn op_mloadw<P, T>(
77    processor: &mut P,
78    tracer: &mut T,
79) -> Result<OperationHelperRegisters, IoError>
80where
81    P: Processor,
82    T: Tracer<Processor = P>,
83{
84    let addr = processor.stack().get(0);
85    let ctx = processor.system().ctx();
86    let clk = processor.system().clock();
87
88    processor.stack_mut().decrement_size()?;
89
90    let word = processor.memory_mut().read_word(ctx, addr, clk)?;
91    tracer.record_memory_read_word(
92        word,
93        addr,
94        processor.system().ctx(),
95        processor.system().clock(),
96    );
97
98    // Set word on stack (word[0] at top).
99    processor.stack_mut().set_word(0, &word);
100
101    Ok(OperationHelperRegisters::Empty)
102}

104/// Stores a word (4 elements) from the stack into the specified memory address.
105///
106/// The operation works as follows:
107/// - The memory address is popped off the stack.
108/// - The top four stack items are saved starting at the specified memory address, which must be
109///   aligned on a word boundary. The items are not removed from the stack.
110///
111/// Thus, the net result of the operation is that the stack is shifted left by one item.
112///
113/// # Errors
114/// - Returns an error if the address is not aligned to a word boundary.
115#[inline(always)]
116pub(super) fn op_mstorew<P, T>(
117    processor: &mut P,
118    tracer: &mut T,
119) -> Result<OperationHelperRegisters, IoError>
120where
121    P: Processor,
122    T: Tracer<Processor = P>,
123{
124    let addr = processor.stack().get(0);
125    // Address is at position 0, so word starts at position 1
126    let word = [
127        processor.stack().get(1),
128        processor.stack().get(2),
129        processor.stack().get(3),
130        processor.stack().get(4),
131    ]
132    .into();
133    let ctx = processor.system().ctx();
134    let clk = processor.system().clock();
135
136    processor.stack_mut().decrement_size()?;
137
138    processor.memory_mut().write_word(ctx, addr, clk, word)?;
139    tracer.record_memory_write_word(
140        word,
141        addr,
142        processor.system().ctx(),
143        processor.system().clock(),
144    );
145
146    Ok(OperationHelperRegisters::Empty)
147}

149/// Loads the element from the specified memory address onto the stack.
150///
151/// The operation works as follows:
152/// - The memory address is popped off the stack.
153/// - The element is retrieved from memory at the specified address. The memory is always
154///   initialized to ZEROs, and thus, if the specified address has never been written to, the ZERO
155///   element is returned.
156/// - The element retrieved from memory is pushed to the top of the stack.
157#[inline(always)]
158pub(super) fn op_mload<P: Processor, T: Tracer>(
159    processor: &mut P,
160    tracer: &mut T,
161) -> Result<OperationHelperRegisters, IoError> {
162    let ctx = processor.system().ctx();
163    let addr = processor.stack().get(0);
164
165    let element = processor.memory_mut().read_element(ctx, addr)?;
166    tracer.record_memory_read_element(
167        element,
168        addr,
169        processor.system().ctx(),
170        processor.system().clock(),
171    );
172
173    processor.stack_mut().set(0, element);
174
175    Ok(OperationHelperRegisters::Empty)
176}

178/// Stores an element from the stack into the first slot at the specified memory address.
179///
180/// The operation works as follows:
181/// - The memory address is popped off the stack.
182/// - The top stack element is saved at the specified memory address. The element is not removed
183///   from the stack.
184///
185/// Thus, the net result of the operation is that the stack is shifted left by one item.
186#[inline(always)]
187pub(super) fn op_mstore<P, T>(
188    processor: &mut P,
189    tracer: &mut T,
190) -> Result<OperationHelperRegisters, IoError>
191where
192    P: Processor,
193    T: Tracer<Processor = P>,
194{
195    let addr = processor.stack().get(0);
196    let value = processor.stack().get(1);
197    let ctx = processor.system().ctx();
198
199    processor.stack_mut().decrement_size()?;
200
201    processor.memory_mut().write_element(ctx, addr, value)?;
202    tracer.record_memory_write_element(
203        value,
204        addr,
205        processor.system().ctx(),
206        processor.system().clock(),
207    );
208
209    Ok(OperationHelperRegisters::Empty)
210}

212/// Loads two words from memory and replaces the top 8 elements of the stack with their
213/// contents.
214///
215/// The operation works as follows:
216/// - The memory address of the first word is retrieved from 13th stack element (position 12).
217/// - Two consecutive words, starting at this address, are loaded from memory.
218/// - Elements of these words are written to the top 8 elements of the stack (element-wise, in stack
219///   order).
220/// - Memory address (in position 12) is incremented by 8.
221/// - All other stack elements remain the same.
222///
223/// # Errors
224/// - Returns an error if the address is not aligned to a word boundary.
225#[inline(always)]
226pub(super) fn op_mstream<P: Processor, T: Tracer>(
227    processor: &mut P,
228    tracer: &mut T,
229) -> Result<OperationHelperRegisters, IoError> {
230    // The stack index where the memory address to load the words from is stored.
231    const MEM_ADDR_STACK_IDX: usize = 12;
232
233    let ctx = processor.system().ctx();
234    let clk = processor.system().clock();
235
236    // load two words from memory
237    let addr_first_word = processor.stack().get(MEM_ADDR_STACK_IDX);
238    let words = {
239        let addr_second_word = addr_first_word + WORD_SIZE_FELT;
240
241        let first_word = processor.memory_mut().read_word(ctx, addr_first_word, clk)?;
242        let second_word = processor.memory_mut().read_word(ctx, addr_second_word, clk)?;
243
244        tracer.record_memory_read_dword([first_word, second_word], addr_first_word, ctx, clk);
245
246        [first_word, second_word]
247    };
248
249    // Replace the stack elements with the elements from memory (in stack order). The word at
250    // address `addr` is at the top of the stack.
251    processor.stack_mut().set_word(0, &words[0]);
252    processor.stack_mut().set_word(4, &words[1]);
253
254    // increment the address by 8 (2 words)
255    processor
256        .stack_mut()
257        .set(MEM_ADDR_STACK_IDX, addr_first_word + DOUBLE_WORD_SIZE);
258
259    Ok(OperationHelperRegisters::Empty)
260}

262/// Moves 8 elements from the advice stack to the memory, via the operand stack.
263///
264/// The operation works as follows:
265/// - Two words are popped from the top of the advice stack.
266/// - The destination memory address for the first word is retrieved from the 13th stack element
267///   (position 12).
268/// - The two words are written to memory consecutively, starting at this address.
269/// - These words replace the top 8 elements of the stack (element-wise, in stack order).
270/// - Memory address (in position 12) is incremented by 8.
271/// - All other stack elements remain the same.
272///
273/// # Errors
274/// - Returns an error if the address is not aligned to a word boundary.
275#[inline(always)]
276pub(super) fn op_pipe<P: Processor, T: Tracer>(
277    processor: &mut P,
278    tracer: &mut T,
279) -> Result<OperationHelperRegisters, IoError> {
280    /// WORD_SIZE, but as a `Felt`.
281    const WORD_SIZE_FELT: Felt = Felt::new(4);
282    /// The size of a double-word.
283    const DOUBLE_WORD_SIZE: Felt = Felt::new(8);
284
285    // The stack index where the memory address to load the words from is stored.
286    const MEM_ADDR_STACK_IDX: usize = 12;
287
288    let clk = processor.system().clock();
289    let ctx = processor.system().ctx();
290    let addr_first_word = processor.stack().get(MEM_ADDR_STACK_IDX);
291    let addr_second_word = addr_first_word + WORD_SIZE_FELT;
292
293    // pop two words from the advice stack
294    let words = processor.advice_provider_mut().pop_stack_dword()?;
295
296    // write the words to memory
297    processor.memory_mut().write_word(ctx, addr_first_word, clk, words[0])?;
298    processor.memory_mut().write_word(ctx, addr_second_word, clk, words[1])?;
299
300    tracer.record_pipe(words, addr_first_word, ctx, clk);
301
302    // Replace the elements on the stack with the word elements (in stack order).
303    // words[0] goes to top positions (0-3), words[1] goes to positions (4-7).
304    processor.stack_mut().set_word(0, &words[0]);
305    processor.stack_mut().set_word(4, &words[1]);
306
307    // increment the address by 8 (2 words)
308    processor
309        .stack_mut()
310        .set(MEM_ADDR_STACK_IDX, addr_first_word + DOUBLE_WORD_SIZE);
311
312    Ok(OperationHelperRegisters::Empty)
313}
