23macro_rules! require_u32_operands {
24    ($processor:expr, [$($idx:expr),*]) => {{
25        let mut invalid_values = Vec::new();
26
27        paste!{
28            $(
29                let [<operand_ $idx>] = $processor.stack().get($idx);
30                if [<operand_ $idx>].as_canonical_u64() > U32_MAX {
31                    invalid_values.push([<operand_ $idx>]);
32                }
33            )*
34
35            if !invalid_values.is_empty() {
36                return Err(OperationError::NotU32Values { values: invalid_values });
37            }
38            // Return tuple of operands based on indices
39            ($([<operand_ $idx>]),*)
40        }
41    }};
42}

47/// Removes and splits the top element of the stack into two 32-bit values, and pushes them onto
48/// the stack.
49///
50/// Input: [value, ...] where value is a field element
51/// Output: [lo, hi, ...] where lo is on top (primary result is the u32 value)
52#[inline(always)]
53pub(super) fn op_u32split<P, T>(
54    processor: &mut P,
55    tracer: &mut T,
56) -> Result<OperationHelperRegisters, ExecutionError>
57where
58    P: Processor,
59    T: Tracer<Processor = P>,
60{
61    let (top_hi, top_lo) = {
62        let top = processor.stack().get(0);
63        split_element(top)
64    };
65    tracer.record_u32_range_checks(processor.system().clock(), top_lo, top_hi);
66
67    processor.stack_mut().increment_size()?;
68    processor.stack_mut().set(0, top_lo);
69
70    processor.stack_mut().set(1, top_hi);
71
72    Ok(OperationHelperRegisters::U32Split { lo: top_lo, hi: top_hi })
73}

75/// Adds the top two elements of the stack and pushes the result onto the stack.
76///
77/// Input: [a, b, ...] where a is on top
78/// Output: [sum, carry, ...] where sum is on top
79#[inline(always)]
80pub(super) fn op_u32add<P: Processor, T: Tracer>(
81    processor: &mut P,
82    tracer: &mut T,
83) -> Result<OperationHelperRegisters, OperationError> {
84    let (carry, sum) = {
85        let (a, b) = require_u32_operands!(processor, [0, 1]);
86
87        let result = Felt::new(a.as_canonical_u64() + b.as_canonical_u64());
88        split_element(result)
89    };
90    tracer.record_u32_range_checks(processor.system().clock(), sum, carry);
91
92    processor.stack_mut().set(0, sum);
93    processor.stack_mut().set(1, carry);
94
95    Ok(OperationHelperRegisters::U32Add { sum, carry })
96}

98/// Pops three elements off the stack, adds them, splits the result into low and high 32-bit
99/// values, and pushes these values back onto the stack.
100///
101/// Input: [a, b, c, ...] where a is on top
102/// Output: [sum, carry, ...] where sum is on top
103///
104/// The size of the stack is decremented by 1.
105#[inline(always)]
106pub(super) fn op_u32add3<P, T>(
107    processor: &mut P,
108    tracer: &mut T,
109) -> Result<OperationHelperRegisters, OperationError>
110where
111    P: Processor,
112    T: Tracer<Processor = P>,
113{
114    let (carry, sum) = {
115        let (a, b, c) = require_u32_operands!(processor, [0, 1, 2]);
116
117        let result = Felt::new(a.as_canonical_u64() + b.as_canonical_u64() + c.as_canonical_u64());
118        split_element(result)
119    };
120    tracer.record_u32_range_checks(processor.system().clock(), sum, carry);
121
122    // write sum to the new top of the stack, and carry after
123    processor.stack_mut().decrement_size()?;
124    processor.stack_mut().set(0, sum);
125    processor.stack_mut().set(1, carry);
126
127    Ok(OperationHelperRegisters::U32Add3 { sum, carry })
128}

130/// Pops two elements off the stack, subtracts the top element from the second element, and
131/// pushes the result as well as a flag indicating whether there was underflow back onto the
132/// stack.
133///
134/// Input: [b, a, ...] where b (subtrahend) is on top
135/// Output: [borrow, diff, ...] where borrow is on top, computes a - b
136#[inline(always)]
137pub(super) fn op_u32sub<P: Processor, T: Tracer>(
138    processor: &mut P,
139    tracer: &mut T,
140) -> Result<OperationHelperRegisters, OperationError> {
141    let (b, a) = require_u32_operands!(processor, [0, 1]);
142
143    let result = a.as_canonical_u64().wrapping_sub(b.as_canonical_u64());
144    let borrow = Felt::new(result >> 63);
145    let diff = Felt::new(result & u32::MAX as u64);
146
147    tracer.record_u32_range_checks(processor.system().clock(), diff, ZERO);
148
149    processor.stack_mut().set(0, borrow);
150    processor.stack_mut().set(1, diff);
151
152    Ok(OperationHelperRegisters::U32Sub { second_new: diff })
153}

155/// Pops two elements off the stack, multiplies them, splits the result into low and high
156/// 32-bit values, and pushes these values back onto the stack.
157///
158/// Input: [a, b, ...] where a is on top
159/// Output: [lo, hi, ...] where lo is on top
160#[inline(always)]
161pub(super) fn op_u32mul<P: Processor, T: Tracer>(
162    processor: &mut P,
163    tracer: &mut T,
164) -> Result<OperationHelperRegisters, OperationError> {
165    let (a, b) = require_u32_operands!(processor, [0, 1]);
166
167    let result = Felt::new(a.as_canonical_u64() * b.as_canonical_u64());
168    let (hi, lo) = split_element(result);
169    tracer.record_u32_range_checks(processor.system().clock(), lo, hi);
170
171    processor.stack_mut().set(0, lo);
172    processor.stack_mut().set(1, hi);
173
174    Ok(OperationHelperRegisters::U32Mul { lo, hi })
175}

177/// Pops three elements off the stack, multiplies the first two and adds the third element to
178/// the result, splits the result into low and high 32-bit values, and pushes these values
179/// back onto the stack.
180///
181/// Input: [a, b, c, ...] where a is on top
182/// Output: [lo, hi, ...] where lo is on top, computes a * b + c

206/// Pops two elements off the stack, divides the second element by the top element, and pushes
207/// the remainder and the quotient back onto the stack.
208///
209/// Input: [b, a, ...] where b (divisor) is on top, a (dividend) is below
210/// Output: [remainder, quotient, ...] where remainder is on top, computes a / b
211///
212/// # Errors
213/// Returns an error if the divisor is ZERO.
214#[inline(always)]
215pub(super) fn op_u32div<P: Processor, T: Tracer>(
216    processor: &mut P,
217    tracer: &mut T,
218) -> Result<OperationHelperRegisters, OperationError> {
219    let (denominator, numerator) = {
220        let (b, a) = require_u32_operands!(processor, [0, 1]);
221
222        // b is divisor (top element), a is dividend (second element)
223        (b.as_canonical_u64(), a.as_canonical_u64())
224    };
225
226    if denominator == 0 {
227        return Err(OperationError::DivideByZero);
228    }
229
230    // a/b = q + r/b for some q>=0 and 0<=r<b
231    let quotient = numerator / denominator;
232    let remainder = numerator - quotient * denominator;
233
234    // remainder is placed on top of the stack, followed by quotient
235    processor.stack_mut().set(0, Felt::new(remainder));
236    processor.stack_mut().set(1, Felt::new(quotient));
237
238    // These range checks help enforce that quotient <= numerator.
239    let lo = Felt::new(numerator - quotient);
240    // These range checks help enforce that remainder < denominator.
241    let hi = Felt::new(denominator - remainder - 1);
242
243    tracer.record_u32_range_checks(processor.system().clock(), lo, hi);
244    Ok(OperationHelperRegisters::U32Div { lo, hi })
245}

247/// Pops two elements off the stack, computes their bitwise AND, and pushes the result back
248/// onto the stack.
249///
250/// Input: [a, b, ...] where a is on top
251/// Output: [result, ...] where result = a AND b
252#[inline(always)]
253pub(super) fn op_u32and<P, T>(
254    processor: &mut P,
255    tracer: &mut T,
256) -> Result<OperationHelperRegisters, OperationError>
257where
258    P: Processor,
259    T: Tracer<Processor = P>,
260{
261    let (a, b) = require_u32_operands!(processor, [0, 1]);
262    tracer.record_u32and(a, b);
263
264    let result = a.as_canonical_u64() & b.as_canonical_u64();
265
266    // Update stack
267    processor.stack_mut().decrement_size()?;
268    processor.stack_mut().set(0, Felt::new(result));
269    Ok(OperationHelperRegisters::Empty)
270}

297/// Pops top two element off the stack, splits them into low and high 32-bit values, checks if
298/// the high values are equal to 0; if they are, puts the original elements back onto the
299/// stack; if they are not, returns an error.
300#[inline(always)]
301pub(super) fn op_u32assert2<P: Processor, T: Tracer>(
302    processor: &mut P,
303    _err_code: Felt,
304    tracer: &mut T,
305) -> Result<OperationHelperRegisters, OperationError> {
306    let (first, second) = require_u32_operands!(processor, [0, 1]);
307
308    tracer.record_u32_range_checks(processor.system().clock(), first, second);
309
310    // Stack remains unchanged for assert operations
311
312    Ok(OperationHelperRegisters::U32Assert2 { first, second })
313}

318/// Splits an element into two field elements containing 32-bit integer values
319#[inline(always)]
320fn split_element(value: Felt) -> (Felt, Felt) {
321    let value = value.as_canonical_u64();
322    let lo = (value as u32) as u64;
323    let hi = value >> 32;
324    (Felt::new(hi), Felt::new(lo))
