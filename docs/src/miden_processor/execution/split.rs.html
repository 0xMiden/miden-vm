17/// Executes a Split node from the start.
18#[inline(always)]
19pub(super) fn start_split_node<P, H, S, T>(
20    state: &mut ExecutionState<'_, P, H, S, T>,
21    split_node: &SplitNode,
22    node_id: MastNodeId,
23    current_forest: &Arc<MastForest>,
24) -> ControlFlow<BreakReason>
25where
26    P: Processor,
27    H: BaseHost,
28    S: Stopper<Processor = P>,
29    T: Tracer<Processor = P>,
30{
31    state.tracer.start_clock_cycle(
32        state.processor,
33        Continuation::StartNode(node_id),
34        state.continuation_stack,
35        current_forest,
36    );
37
38    // Execute decorators that should be executed before entering the node
39    state
40        .processor
41        .execute_before_enter_decorators(node_id, current_forest, state.host)?;
42
43    let condition = state.processor.stack().get(0);
44
45    // drop the condition from the stack
46    if let Err(err) = state.processor.stack_mut().decrement_size().map_exec_err(
47        current_forest,
48        node_id,
49        state.host,
50    ) {
51        return ControlFlow::Break(BreakReason::Err(err));
52    }
53
54    // execute the appropriate branch
55    state.continuation_stack.push_finish_split(node_id);
56    if condition == ONE {
57        state.continuation_stack.push_start_node(split_node.on_true());
58    } else if condition == ZERO {
59        state.continuation_stack.push_start_node(split_node.on_false());
60    } else {
61        let err = OperationError::NotBinaryValueIf { value: condition };
62        return ControlFlow::Break(BreakReason::Err(err.with_context(
63            current_forest,
64            node_id,
65            state.host,
66        )));
67    };
68
69    // Finalize the clock cycle corresponding to the SPLIT operation.
70    finalize_clock_cycle(
71        state.processor,
72        state.tracer,
73        state.stopper,
74        state.continuation_stack,
75        current_forest,
76    )
77}

79/// Executes the finish phase of a Split node.
80#[inline(always)]
81pub(super) fn finish_split_node<P, H, S, T>(
82    state: &mut ExecutionState<'_, P, H, S, T>,
83    node_id: MastNodeId,
84    current_forest: &Arc<MastForest>,
85) -> ControlFlow<BreakReason>
86where
87    P: Processor,
88    H: BaseHost,
89    S: Stopper<Processor = P>,
90    T: Tracer<Processor = P>,
91{
92    state.tracer.start_clock_cycle(
93        state.processor,
94        Continuation::FinishSplit(node_id),
95        state.continuation_stack,
96        current_forest,
97    );
98
99    // Finalize the clock cycle corresponding to the END operation.
100    finalize_clock_cycle_with_continuation(
101        state.processor,
102        state.tracer,
103        state.stopper,
104        state.continuation_stack,
105        || Some(Continuation::AfterExitDecorators(node_id)),
106        current_forest,
107    )?;
108
109    state
110        .processor
111        .execute_after_exit_decorators(node_id, current_forest, state.host)
112}
