14// EXTERNAL NODE PROCESSING

17/// Executes an External node.
18#[inline(always)]
19pub(super) fn execute_external_node(
20    processor: &mut impl Processor,
21    external_node_id: MastNodeId,
22    current_forest: &mut Arc<MastForest>,
23    host: &mut impl Host,
24) -> ControlFlow<InternalBreakReason> {
25    // Execute decorators that should be executed before entering the node
26    processor
27        .execute_before_enter_decorators(external_node_id, current_forest, host)
28        .map_break(InternalBreakReason::from)?;
29
30    // This is a sans-IO point: we cannot proceed with loading the MAST forest, since some
31    // processors need this to be done asynchronously. Thus, we break here and make the implementing
32    // processor handle the loading in the outer execution loop. When done, the processor *must*
33    // call `finish_load_mast_forest_from_external()` below for execution to proceed properly.
34    let external_node = current_forest[external_node_id].unwrap_external();
35    ControlFlow::Break(InternalBreakReason::LoadMastForestFromExternal {
36        external_node_id,
37        procedure_hash: external_node.digest(),
38    })
39}
40
41/// Function to be called after [`InternalBreakReason::LoadMastForestFromExternal`] is handled. See
42/// the documentation of that enum variant for more details.
43pub fn finish_load_mast_forest_from_external(
44    resolved_node_id: MastNodeId,
45    new_mast_forest: Arc<MastForest>,
46    external_node_id: MastNodeId,
47    current_forest: &mut Arc<MastForest>,
48    continuation_stack: &mut ContinuationStack,
49    host: &mut impl Host,
50    tracer: &mut impl Tracer,
51) -> ControlFlow<BreakReason> {
52    let external_node = current_forest[external_node_id].unwrap_external();
53    // if the node that we got by looking up an external reference is also an External
54    // node, we are about to enter into an infinite loop - so, return an error
55    if new_mast_forest[resolved_node_id].is_external() {
56        return ControlFlow::Break(BreakReason::Err(
57            OperationError::CircularExternalNode(external_node.digest()).with_context(
58                current_forest,
59                external_node_id,
60                host,
61            ),
62        ));
63    }
64
65    tracer.record_mast_forest_resolution(resolved_node_id, &new_mast_forest);
66
67    // Push a continuation to execute after_exit decorators when we return from the external
68    // forest
69    continuation_stack.push_finish_external(external_node_id);
70
71    // Push current forest to the continuation stack so that we can return to it
72    continuation_stack.push_enter_forest(current_forest.clone());
73
74    // Push the root node of the external MAST forest onto the continuation stack.
75    continuation_stack.push_start_node(resolved_node_id);
76
77    // Update the current forest to the new MAST forest.
78    *current_forest = new_mast_forest;
79
80    // Note that executing an External node does not end the clock cycle, so we do not finalize the
81    // clock cycle here.
82    ControlFlow::Continue(())
83}
