20/// Execute the given basic block node.
21#[inline(always)]
22pub(super) fn execute_basic_block_node_from_start<P, H, S, T>(
23    state: &mut ExecutionState<'_, P, H, S, T>,
24    basic_block_node: &BasicBlockNode,
25    node_id: MastNodeId,
26    current_forest: &Arc<MastForest>,
27) -> ControlFlow<InternalBreakReason>
28where
29    P: Processor,
30    H: BaseHost,
31    S: Stopper<Processor = P>,
32    T: Tracer<Processor = P>,
33{
34    state.tracer.start_clock_cycle(
35        state.processor,
36        Continuation::StartNode(node_id),
37        state.continuation_stack,
38        current_forest,
39    );
40
41    // Execute decorators that should be executed before entering the node
42    state
43        .processor
44        .execute_before_enter_decorators(node_id, current_forest, state.host)
45        .map_break(InternalBreakReason::from)?;
46
47    // Finalize the clock cycle corresponding to the SPAN operation.
48    finalize_clock_cycle_with_continuation(
49        state.processor,
50        state.tracer,
51        state.stopper,
52        state.continuation_stack,
53        || {
54            Some(Continuation::ResumeBasicBlock {
55                node_id,
56                batch_index: 0,
57                op_idx_in_batch: 0,
58            })
59        },
60        current_forest,
61    )
62    .map_break(InternalBreakReason::from)?;
63
64    // Execute the first batch separately, since `execute_basic_block_node_from_batch` executes
65    // starting from the RESPAN preceding the batch (and there is no such RESPAN before the first
66    // batch).
67    if !basic_block_node.op_batches().is_empty() {
68        execute_op_batch(state, basic_block_node, 0, 0, 0, current_forest)?;
69    }
70
71    // Execute the rest of the batches.
72    execute_basic_block_node_from_batch(state, basic_block_node, node_id, 1, current_forest)
73}

75/// Executes the give basic block node starting from the specified operation index within the
76/// specified batch.
77#[inline(always)]
78pub(super) fn execute_basic_block_node_from_op_idx<P, H, S, T>(
79    state: &mut ExecutionState<'_, P, H, S, T>,
80    basic_block_node: &BasicBlockNode,
81    node_id: MastNodeId,
82    start_batch_index: usize,
83    start_op_idx_in_batch: usize,
84    current_forest: &Arc<MastForest>,
85) -> ControlFlow<InternalBreakReason>
86where
87    P: Processor,
88    H: BaseHost,
89    S: Stopper<Processor = P>,
90    T: Tracer<Processor = P>,
91{
92    let batch_offset_in_block = basic_block_node
93        .op_batches()
94        .iter()
95        .take(start_batch_index)
96        .map(|batch| batch.ops().len())
97        .sum();
98
99    // Finish executing the specified batch from the given op index
100    execute_op_batch(
101        state,
102        basic_block_node,
103        start_batch_index,
104        start_op_idx_in_batch,
105        batch_offset_in_block,
106        current_forest,
107    )?;
108
109    // Execute the rest of the batches
110    execute_basic_block_node_from_batch(
111        state,
112        basic_block_node,
113        node_id,
114        start_batch_index + 1,
115        current_forest,
116    )
117}

119/// Executes the give basic block node starting from the RESPAN preceding the specified batch.
120#[inline(always)]
121pub(super) fn execute_basic_block_node_from_batch<P, H, S, T>(
122    state: &mut ExecutionState<'_, P, H, S, T>,
123    basic_block_node: &BasicBlockNode,
124    node_id: MastNodeId,
125    start_batch_index: usize,
126    current_forest: &Arc<MastForest>,
127) -> ControlFlow<InternalBreakReason>
128where
129    P: Processor,
130    H: BaseHost,
131    S: Stopper<Processor = P>,
132    T: Tracer<Processor = P>,
133{
134    let mut batch_offset_in_block = basic_block_node
135        .op_batches()
136        .iter()
137        .take(start_batch_index)
138        .map(|batch| batch.ops().len())
139        .sum();
140
141    for (batch_index, op_batch) in
142        basic_block_node.op_batches().iter().enumerate().skip(start_batch_index)
143    {
144        {
145            // Start clock cycle corresponding to the RESPAN operation before the batch.
146            state.tracer.start_clock_cycle(
147                state.processor,
148                Continuation::Respan { node_id, batch_index },
149                state.continuation_stack,
150                current_forest,
151            );
152
153            // Finalize the clock cycle corresponding to the RESPAN operation.
154            //
155            // Note: in the continuation closure, the continuation encodes resuming from the start
156            // of the batch *after* the RESPAN operation. This is because the continuation encodes
157            // what happens *after* the clock is incremented. For example, if we were to put a
158            // `Continuation::Respan` here instead, and execution was stopped after this RESPAN,
159            // then the next call to `Processor::execute_impl()` would re-execute the RESPAN.
160            finalize_clock_cycle_with_continuation(
161                state.processor,
162                state.tracer,
163                state.stopper,
164                state.continuation_stack,
165                || {
166                    Some(Continuation::ResumeBasicBlock {
167                        node_id,
168                        batch_index,
169                        op_idx_in_batch: 0,
170                    })
171                },
172                current_forest,
173            )
174            .map_break(InternalBreakReason::from)?;
175        }
176
177        // Execute the batch.
178        execute_op_batch(
179            state,
180            basic_block_node,
181            batch_index,
182            0,
183            batch_offset_in_block,
184            current_forest,
185        )?;
186        batch_offset_in_block += op_batch.ops().len();
187    }
188
189    finish_basic_block(state, basic_block_node, node_id, current_forest)
190        .map_break(InternalBreakReason::from)
191}

193/// Execute the finish phase of a basic block node.
194#[inline(always)]
195pub(super) fn finish_basic_block<P, H, S, T>(
196    state: &mut ExecutionState<'_, P, H, S, T>,
197    basic_block_node: &BasicBlockNode,
198    node_id: MastNodeId,
199    current_forest: &Arc<MastForest>,
200) -> ControlFlow<BreakReason>
201where
202    P: Processor,
203    H: BaseHost,
204    S: Stopper<Processor = P>,
205    T: Tracer<Processor = P>,
206{
207    state.tracer.start_clock_cycle(
208        state.processor,
209        Continuation::FinishBasicBlock(node_id),
210        state.continuation_stack,
211        current_forest,
212    );
213
214    // Finalize the clock cycle corresponding to the END operation.
215    finalize_clock_cycle_with_continuation(
216        state.processor,
217        state.tracer,
218        state.stopper,
219        state.continuation_stack,
220        || Some(Continuation::AfterExitDecoratorsBasicBlock(node_id)),
221        current_forest,
222    )?;
223
224    state.processor.execute_end_of_block_decorators(
225        basic_block_node,
226        node_id,
227        current_forest,
228        state.host,
229    )?;
230    state
231        .processor
232        .execute_after_exit_decorators(node_id, current_forest, state.host)
233}

238/// Executes a single operation batch within a basic block node, starting from the operation
239/// index `start_op_idx`.
240#[inline(always)]
241fn execute_op_batch<P, H, S, T>(
242    state: &mut ExecutionState<'_, P, H, S, T>,
243    basic_block: &BasicBlockNode,
244    batch_index: usize,
245    start_op_idx: usize,
246    batch_offset_in_block: usize,
247    current_forest: &Arc<MastForest>,
248) -> ControlFlow<InternalBreakReason>
249where
250    P: Processor,
251    H: BaseHost,
252    S: Stopper<Processor = P>,
253    T: Tracer<Processor = P>,
254{
255    let batch = &basic_block.op_batches()[batch_index];
256
257    // Get the node ID once since it doesn't change within the loop
258    let node_id = basic_block
259        .linked_id()
260        .expect("basic block node should be linked when executing operations");
261
262    // Execute operations in the batch one by one
263    for (op_idx_in_batch, op) in batch.ops().iter().enumerate().skip(start_op_idx) {
264        let op_idx_in_block = batch_offset_in_block + op_idx_in_batch;
265
266        state.tracer.start_clock_cycle(
267            state.processor,
268            Continuation::ResumeBasicBlock { node_id, batch_index, op_idx_in_batch },
269            state.continuation_stack,
270            current_forest,
271        );
272
273        state
274            .processor
275            .execute_decorators_for_op(node_id, op_idx_in_block, current_forest, state.host)
276            .map_break(InternalBreakReason::from)?;
277
278        // Execute the operation.
279        let operation_helpers = match op {
280            Operation::Emit => {
281                // This is a sans-IO point: we cannot proceed with handling the Emit operation,
282                // since some processors need this to be done asynchronously. Thus, we break
283                // here and make the implementing processor handle the loading in the outer
284                // execution loop. When done, the processor *must* call
285                // `finish_emit_op_execution()` below for execution to proceed properly.
286                return ControlFlow::Break(InternalBreakReason::Emit {
287                    basic_block_node_id: node_id,
288                    op_idx: op_idx_in_block,
289                    continuation: get_continuation_after_executing_operation(
290                        basic_block,
291                        node_id,
292                        batch_index,
293                        op_idx_in_batch,
294                    ),
295                });
296            },
297            _ => {
298                // If the operation is not an Emit, we execute it normally.
299                match execute_op(
300                    state.processor,
301                    op,
302                    op_idx_in_block,
303                    current_forest,
304                    node_id,
305                    state.host,
306                    state.tracer,
307                ) {
308                    Ok(operation_helpers) => operation_helpers,
309                    Err(err) => {
310                        return ControlFlow::Break(BreakReason::Err(err).into());
311                    },
312                }
313            },
314        };
315
316        // Finalize the clock cycle corresponding to the operation.
317        finalize_clock_cycle_with_continuation_and_op_helpers(
318            state.processor,
319            state.tracer,
320            state.stopper,
321            state.continuation_stack,
322            || {
323                Some(get_continuation_after_executing_operation(
324                    basic_block,
325                    node_id,
326                    batch_index,
327                    op_idx_in_batch,
328                ))
329            },
330            operation_helpers,
331            current_forest,
332        )
333        .map_break(InternalBreakReason::from)?;
334    }
335
336    ControlFlow::Continue(())
337}

339/// Given the current operation being executed within a basic block, returns the appropriate
340/// continuation to add to the continuation stack if execution is stopped right after execution the
341/// operation (node_id, batch_index, op_idx_in_batch).
342///
343/// That is, `op_idx_in_batch` is the index of the operation that was just executed within the batch
344/// `batch_index` of the basic block `basic_block_node`.
345#[inline(always)]
346fn get_continuation_after_executing_operation(
347    basic_block_node: &BasicBlockNode,
348    node_id: MastNodeId,
349    batch_index: usize,
350    op_idx_in_batch: usize,
351) -> Continuation {
352    let last_op_idx_in_batch = basic_block_node.op_batches()[batch_index].ops().len() - 1;
353    let last_batch_idx_in_block = basic_block_node.num_op_batches() - 1;
354
355    if op_idx_in_batch < last_op_idx_in_batch {
356        // The operation that just executed was not the last one in the batch, so continue within
357        // the same batch at the following operation
358        Continuation::ResumeBasicBlock {
359            node_id,
360            batch_index,
361            op_idx_in_batch: op_idx_in_batch + 1,
362        }
363    } else if batch_index < last_batch_idx_in_block {
364        // The operation that just executed was the last one in the batch, but there are more
365        // batches to execute in this basic block, so continue at the RESPAN before the next batch
366        Continuation::Respan { node_id, batch_index: batch_index + 1 }
367    } else {
368        // The operation that just executed was the last one in the last batch, so finish the basic
369        // block
370        Continuation::FinishBasicBlock(node_id)
371    }
372}

377/// Function to be called after [`InternalBreakReason::Emit`] is handled. See the documentation of
378/// that enum variant for more details.
379pub fn finish_emit_op_execution<P, S, T>(
380    post_emit_continuation: Continuation,
381    processor: &mut P,
382    continuation_stack: &mut ContinuationStack,
383    current_forest: &Arc<MastForest>,
384    tracer: &mut T,
385    stopper: &S,
386) -> ControlFlow<BreakReason>
387where
388    P: Processor,
389    S: Stopper<Processor = P>,
390    T: Tracer<Processor = P>,
391{
392    // When we enter here, the `continuation_stack` top contains the continuation to execute *after*
393    // the basic block that contained the `Emit` operation (i.e. after all operations are executed,
394    // and the finish phase of the basic block is complete). Hence, we need to add the
395    // `post_emit_continuation` on top of the continuation stack so that execution resumes at the
396    // operation right after the `Emit`.
397    //
398    // However, if the `stopper` stops execution in `finalize_clock_cycle_with_continuation()`, the
399    // stopper will already include the `post_emit_continuation` in the break reason (which the
400    // processor will then push onto the continuation stack). Hence, in this case, we do not need to
401    // push the `post_emit_continuation` ourselves. In other words, *only if* the
402    // `finalize_clock_cycle_with_continuation()` completes successfully do we need to push the
403    // `post_emit_continuation` ourselves.
404
405    finalize_clock_cycle_with_continuation(
406        processor,
407        tracer,
408        stopper,
409        continuation_stack,
410        {
411            let post_emit_continuation = post_emit_continuation.clone();
412            || Some(post_emit_continuation)
413        },
414        current_forest,
415    )?;
416
417    continuation_stack.push_continuation(post_emit_continuation);
418
419    ControlFlow::Continue(())
