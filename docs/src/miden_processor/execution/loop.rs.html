17/// Executes a Loop node from the start.
18#[inline(always)]
19pub(super) fn start_loop_node<P, H, S, T>(
20    state: &mut ExecutionState<'_, P, H, S, T>,
21    loop_node: &LoopNode,
22    current_node_id: MastNodeId,
23    current_forest: &Arc<MastForest>,
24) -> ControlFlow<BreakReason>
25where
26    P: Processor,
27    H: BaseHost,
28    S: Stopper<Processor = P>,
29    T: Tracer<Processor = P>,
30{
31    state.tracer.start_clock_cycle(
32        state.processor,
33        Continuation::StartNode(current_node_id),
34        state.continuation_stack,
35        current_forest,
36    );
37
38    // Execute decorators that should be executed before entering the node
39    state
40        .processor
41        .execute_before_enter_decorators(current_node_id, current_forest, state.host)?;
42
43    let condition = state.processor.stack().get(0);
44
45    // drop the condition from the stack
46    if let Err(err) = state.processor.stack_mut().decrement_size().map_exec_err(
47        current_forest,
48        current_node_id,
49        state.host,
50    ) {
51        return ControlFlow::Break(BreakReason::Err(err));
52    }
53
54    // execute the loop body as long as the condition is true
55    if condition == ONE {
56        // Push the loop to check condition again after body executes.
57        //
58        // WARNING: if we eventually push another continuation in between the `FinishLoop` and the
59        // `StartNode` continuations, then the logic in `ExecutionTracer::start_clock_cycle()` that
60        // computes the value for the `is_loop_body` flag will be incorrect and needs to be
61        // adjusted.
62        state.continuation_stack.push_finish_loop_entered(current_node_id);
63        state.continuation_stack.push_start_node(loop_node.body());
64
65        // Finalize the clock cycle corresponding to the LOOP operation.
66        finalize_clock_cycle(
67            state.processor,
68            state.tracer,
69            state.stopper,
70            state.continuation_stack,
71            current_forest,
72        )
73    } else if condition == ZERO {
74        // Start and exit the loop immediately - corresponding to adding a LOOP and END row
75        // immediately since there is no body to execute.
76
77        // Finalize the clock cycle corresponding to the LOOP operation.
78        finalize_clock_cycle_with_continuation(
79            state.processor,
80            state.tracer,
81            state.stopper,
82            state.continuation_stack,
83            || {
84                Some(Continuation::FinishLoop {
85                    node_id: current_node_id,
86                    was_entered: false,
87                })
88            },
89            current_forest,
90        )?;
91
92        finish_loop_node(state, false, current_node_id, current_forest)
93    } else {
94        let err = OperationError::NotBinaryValueLoop { value: condition };
95        ControlFlow::Break(BreakReason::Err(err.with_context(
96            current_forest,
97            current_node_id,
98            state.host,
99        )))
100    }
101}

103/// Executes the finish phase of a Loop node.
104///
105/// This function is called either after the loop body has executed (in which case
106/// `loop_was_entered` is true), or when the loop condition was found to be ZERO at the start of
107/// the loop (in which case `loop_was_entered` is false).
108#[inline(always)]
109pub(super) fn finish_loop_node<P, H, S, T>(
110    state: &mut ExecutionState<'_, P, H, S, T>,
111    loop_was_entered: bool,
112    current_node_id: MastNodeId,
113    current_forest: &Arc<MastForest>,
114) -> ControlFlow<BreakReason>
115where
116    P: Processor,
117    H: BaseHost,
118    S: Stopper<Processor = P>,
119    T: Tracer<Processor = P>,
120{
121    // This happens after loop body execution or when the loop condition was ZERO at the start.
122    // Check condition again to see if we should continue looping. If the loop was never entered, we
123    // know the condition is ZERO.
124    let condition = if loop_was_entered {
125        state.processor.stack().get(0)
126    } else {
127        ZERO
128    };
129    let loop_node = current_forest[current_node_id].unwrap_loop();
130
131    if condition == ONE {
132        // Start the clock cycle corresponding to the REPEAT operation, before re-entering the loop
133        // body.
134        state.tracer.start_clock_cycle(
135            state.processor,
136            Continuation::FinishLoop {
137                node_id: current_node_id,
138                was_entered: true,
139            },
140            state.continuation_stack,
141            current_forest,
142        );
143
144        // Drop the condition from the stack (we know the loop was entered since condition is
145        // ONE).
146        if let Err(err) = state.processor.stack_mut().decrement_size().map_exec_err(
147            current_forest,
148            current_node_id,
149            state.host,
150        ) {
151            return ControlFlow::Break(BreakReason::Err(err));
152        }
153
154        state.continuation_stack.push_finish_loop_entered(current_node_id);
155        state.continuation_stack.push_start_node(loop_node.body());
156
157        // Finalize the clock cycle corresponding to the REPEAT operation.
158        finalize_clock_cycle(
159            state.processor,
160            state.tracer,
161            state.stopper,
162            state.continuation_stack,
163            current_forest,
164        )
165    } else if condition == ZERO {
166        // Exit the loop - start the clock cycle corresponding to the END operation.
167        state.tracer.start_clock_cycle(
168            state.processor,
169            Continuation::FinishLoop {
170                node_id: current_node_id,
171                was_entered: loop_was_entered,
172            },
173            state.continuation_stack,
174            current_forest,
175        );
176
177        // The END operation only drops the condition from the stack if the loop was entered. This
178        // is because if the loop was never entered, then the condition will have already been
179        // dropped by the LOOP operation. Compare this with when the loop body *is* entered, then
180        // the loop body is responsible for pushing the condition back onto the stack, and therefore
181        // the END instruction must drop it.
182        if loop_was_entered
183            && let Err(err) = state.processor.stack_mut().decrement_size().map_exec_err(
184                current_forest,
185                current_node_id,
186                state.host,
187            )
188        {
189            return ControlFlow::Break(BreakReason::Err(err));
190        }
191
192        // Finalize the clock cycle corresponding to the END operation.
193        finalize_clock_cycle_with_continuation(
194            state.processor,
195            state.tracer,
196            state.stopper,
197            state.continuation_stack,
198            || Some(Continuation::AfterExitDecorators(current_node_id)),
199            current_forest,
200        )?;
201
202        state
203            .processor
204            .execute_after_exit_decorators(current_node_id, current_forest, state.host)
205    } else {
206        let err = OperationError::NotBinaryValueLoop { value: condition };
207        ControlFlow::Break(BreakReason::Err(err.with_context(
208            current_forest,
209            current_node_id,
210            state.host,
211        )))
212    }
