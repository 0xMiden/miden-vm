9// AUXILIARY TRACE BUILDER
10// ================================================================================================
11
12/// Describes how to construct execution traces of stack-related auxiliary trace segment columns
13/// (used in multiset checks).
14#[derive(Debug, Clone)]
15pub struct AuxTraceBuilder;
16
17impl AuxTraceBuilder {
18    /// Builds and returns stack auxiliary trace columns. Currently this consists of a single
19    /// column p1 describing states of the stack overflow table.
20    pub fn build_aux_columns<E: ExtensionField<Felt>>(
21        &self,
22        main_trace: &MainTrace,
23        challenges: &Challenges<E>,
24    ) -> Vec<Vec<E>> {
25        let p1 = self.build_aux_column(main_trace, challenges);
26
27        debug_assert_eq!(*p1.last().unwrap(), E::ONE);
28        vec![p1]
29    }
30}
31
32impl<E: ExtensionField<Felt>> AuxColumnBuilder<E> for AuxTraceBuilder {
33    /// Removes a row from the stack overflow table.
34    fn get_requests_at(
35        &self,
36        main_trace: &MainTrace,
37        challenges: &Challenges<E>,
38        i: RowIndex,
39        _debugger: &mut BusDebugger<E>,
40    ) -> E {
41        let is_left_shift = main_trace.is_left_shift(i);
42        let is_dyncall = main_trace.get_op_code(i) == Felt::from_u8(opcodes::DYNCALL);
43        let is_non_empty_overflow = main_trace.is_non_empty_overflow(i);
44
45        if is_left_shift && is_non_empty_overflow {
46            let b1 = main_trace.parent_overflow_address(i);
47            let s15_prime = main_trace.stack_element(15, i + 1);
48            let b1_prime = main_trace.parent_overflow_address(i + 1);
49
50            OverflowTableRow::new(b1, s15_prime, b1_prime).to_value(challenges)
51        } else if is_dyncall && is_non_empty_overflow {
52            let b1 = main_trace.parent_overflow_address(i);
53            let s15_prime = main_trace.stack_element(15, i + 1);
54            let b1_prime = main_trace.decoder_hasher_state_element(5, i);
55
56            OverflowTableRow::new(b1, s15_prime, b1_prime).to_value(challenges)
57        } else {
58            E::ONE
59        }
60    }
61
62    /// Adds a row to the stack overflow table.
63    fn get_responses_at(
64        &self,
65        main_trace: &MainTrace,
66        challenges: &Challenges<E>,
67        i: RowIndex,
68        _debugger: &mut BusDebugger<E>,
69    ) -> E {
70        let is_right_shift = main_trace.is_right_shift(i);
71
72        if is_right_shift {
73            let k0 = main_trace.clk(i);
74            let s15 = main_trace.stack_element(15, i);
75            let b1 = main_trace.parent_overflow_address(i);
76
77            let row = OverflowTableRow::new(k0, s15, b1);
78            row.to_value(challenges)
79        } else {
80            E::ONE
81        }
82    }
83
84    #[cfg(any(test, feature = "bus-debugger"))]
85    fn enforce_bus_balance(&self) -> bool {
86        true
87    }
88}

90// OVERFLOW STACK ROW
91// ================================================================================================
92
93/// A single row in the stack overflow table. Each row contains the following values:
94/// - The value of the stack item pushed into the overflow stack.
95/// - The clock cycle at which the stack item was pushed into the overflow stack.
96/// - The clock cycle of the value which was at the top of the overflow stack when this value was
97///   pushed onto it.
98#[derive(Debug, Clone, PartialEq, Eq)]
99pub struct OverflowTableRow {
100    val: Felt,
101    clk: Felt,
102    prev: Felt,
103}
104
105impl OverflowTableRow {
106    pub fn new(clk: Felt, val: Felt, prev: Felt) -> Self {
107        Self { val, clk, prev }
108    }
109}
110
111impl OverflowTableRow {
112    /// Reduces this row to a single field element in the field specified by E. This requires
113    /// at least 4 alpha values.
114    pub fn to_value<E: ExtensionField<Felt>>(&self, challenges: &Challenges<E>) -> E {
115        challenges.encode([self.clk, self.val, self.prev])
116    }
117}
