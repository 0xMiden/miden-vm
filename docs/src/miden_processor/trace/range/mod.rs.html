29/// Range checker for the VM.
30///
31/// This component is responsible for building an execution trace for all 16-bit range checks
32/// performed by the VM. Thus, the [RangeChecker] doesn't actually check if a given value fits
33/// into 16-bits, but rather keeps track of all 16-bit range checks performed by the VM.
34///
35/// ## Execution trace
36/// The execution trace generated by the range checker consists of 2 columns. Conceptually, the
37/// table starts with value 0 and ends with value 65535.
38///
39/// The layout is illustrated below.
40///
41///    m     v
42/// ├─────┴─────┤
43///
44/// In the above, the meaning of the columns is as follows:
45/// - Column `v` contains the value being range-checked where `v` must be a 16-bit value. The values
46///   must be in increasing order and the jump allowed between two values should be a power of 3
47///   less than or equal to 3^7, and duplicates are allowed.
48/// - Column `m` specifies the lookup multiplicity, which is how many lookups are to be included for
49///   a given value.
50///
51/// Thus, for example, if a value was range-checked just once, we'll need to add a single row to
52/// the table with (m, v) set to (1, v), where v is the value. If the value was range-checked 5
53/// times, we'll need to specify the row (5, v).
54pub struct RangeChecker {
55    /// Tracks lookup count for each checked value.
56    lookups: BTreeMap<u16, usize>,
57    /// Range check lookups performed by all user operations, grouped and sorted by clock cycle.
58    /// Each cycle is mapped to a vector of the range checks requested at that cycle, which can
59    /// come from the stack, memory, or both.
60    cycle_lookups: BTreeMap<RowIndex, Vec<u16>>,
61}
62
63impl RangeChecker {
64    // CONSTRUCTOR
65    // --------------------------------------------------------------------------------------------
66    /// Returns a new [RangeChecker] instantiated with an empty lookup table.
67    pub fn new() -> Self {
68        let mut lookups = BTreeMap::new();
69        // we need to make sure that the first row after the padded section and the last row of the
70        // range checker table are initialized. this simplifies trace table building later on.
71        lookups.insert(0, 0);
72        lookups.insert(u16::MAX, 0);
73        Self { lookups, cycle_lookups: BTreeMap::new() }
74    }
75
76    // TRACE MUTATORS
77    // --------------------------------------------------------------------------------------------
78
79    /// Adds the specified value to the trace of this range checker's lookups.
80    pub fn add_value(&mut self, value: u16) {
81        self.lookups.entry(value).and_modify(|v| *v += 1).or_insert(1);
82    }
83
84    /// Adds range check lookups from the stack or memory to this [RangeChecker] instance.
85    pub fn add_range_checks(&mut self, clk: RowIndex, values: &[u16]) {
86        // range checks requests only come from memory or from the stack, which always request 2 or
87        // 4 lookups respectively.
88        debug_assert!(values.len() == 2 || values.len() == 4);
89
90        for value in values.iter() {
91            // add the specified value to the trace of this range checker's lookups.
92            self.add_value(*value);
93        }
94
95        // track the range check requests at each cycle
96        // TODO: optimize this to use a struct instead of vectors, e.g. (#2793):
97        // struct MemoryLookupValues {
98        //   num_lookups: u8,
99        //   lookup_values: [u16; 6],
100        // }
101        self.cycle_lookups
102            .entry(clk)
103            .and_modify(|entry| entry.append(&mut values.to_vec()))
104            .or_insert_with(|| values.to_vec());
105    }
106
107    // EXECUTION TRACE GENERATION (INTERNAL)
108    // --------------------------------------------------------------------------------------------
109
110    /// Converts this [RangeChecker] into an execution trace with 2 columns and the number of rows
111    /// specified by the `target_len` parameter.
112    ///
113    /// If the number of rows needed to represent execution trace of this range checker is smaller
114    /// than `target_len` parameter, the trace is padded with extra rows.
115    ///
116    /// # Panics
117    /// Panics if `target_len` is not a power of two or is smaller than the trace length needed
118    /// to represent all lookups in this range checker.
119    pub fn into_trace_with_table(self, trace_len: usize, target_len: usize) -> RangeCheckTrace {
120        assert!(target_len.is_power_of_two(), "target trace length is not a power of two");
121
122        // determine the length of the trace required to support all the lookups in this range
123        // checker, and make sure this length is smaller than or equal to the target trace length.
124        assert!(trace_len <= target_len, "target trace length too small");
125
126        // allocate memory for the trace
127        let mut trace = [uninit_vector(target_len), uninit_vector(target_len)];
128
129        // determine the number of padding rows needed to get to target trace length and pad the
130        // table with the required number of rows.
131        let num_padding_rows = target_len - trace_len;
132        trace[0][..num_padding_rows].fill(MaybeUninit::new(ZERO));
133        trace[1][..num_padding_rows].fill(MaybeUninit::new(ZERO));
134
135        // build the trace table
136        let mut i = num_padding_rows;
137        let mut prev_value = 0u16;
138        for (&value, &num_lookups) in self.lookups.iter() {
139            write_rows(&mut trace, &mut i, num_lookups, value, prev_value);
140            prev_value = value;
141        }
142
143        // pad the trace with an extra row of 0 lookups for u16::MAX so that when b_range is built
144        // there is space for the inclusion of u16::MAX range check lookups before the trace ends.
145        // (When there is data at the end of the main trace, auxiliary bus columns always need to be
146        // one row longer than the main trace, since values in the bus column are based on data from
147        // the "current" row of the main trace but placed into the "next" row of the bus column.)
148        write_trace_row(&mut trace, &mut i, 0, (u16::MAX).into());
149
150        assert_eq!(i, target_len, "range checker trace not fully initialized; trace_len mismatch");
151
152        // all elements are now initialized
153        let [t0, t1] = trace;
154        let trace = unsafe { [assume_init_vec(t0), assume_init_vec(t1)] };
155
156        RangeCheckTrace {
157            trace,
158            aux_builder: AuxTraceBuilder::new(
159                self.lookups.keys().cloned().collect(),
160                self.cycle_lookups,
161                num_padding_rows,
162            ),
163        }
164    }
165
166    // PUBLIC ACCESSORS
167    // --------------------------------------------------------------------------------------------
168
169    /// Returns the number of rows needed to support all 16-bit lookups requested by the VM.
170    pub fn get_number_range_checker_rows(&self) -> usize {
171        // pad the trace length by one, to account for an extra row of the u16::MAX value at the end
172        // of the trace, required for building the `b_range` column.
173        let mut num_rows = 1;
174
175        let mut prev_value = 0u16;
176        for value in self.lookups.keys() {
177            // add one row for each value in the range checker table
178            num_rows += 1;
179            // determine the delta between this and the previous value. we need to know this delta
180            // to determine if we need to insert any "bridge" rows to the  table, this is needed
181            // since the gap between two values in the range checker can only be a power of 3 less
182            // than or equal to 3^7.
183            let delta = value - prev_value;
184            num_rows += get_num_bridge_rows(delta);
185            prev_value = *value;
186        }
187        num_rows
188    }
189
190    // TEST HELPERS
191    // --------------------------------------------------------------------------------------------
192
193    /// Returns length of execution trace required to describe all 16-bit range checks performed
194    /// by the VM.
195    #[cfg(test)]
196    pub fn trace_len(&self) -> usize {
197        self.get_number_range_checker_rows()
198    }
199
200    /// Converts this [RangeChecker] into an execution trace with 2 columns and the number of rows
201    /// specified by the `target_len` parameter.
202    ///
203    /// Wrapper for [`RangeChecker::into_trace_with_table`].
204    #[cfg(test)]
205    pub fn into_trace(self, target_len: usize) -> RangeCheckTrace {
206        let table_len = self.get_number_range_checker_rows();
207        self.into_trace_with_table(table_len, target_len)
208    }
209}

220/// Calculates the number of bridge rows that are need to be added to the trace between two values
221/// to be range checked.
222pub fn get_num_bridge_rows(delta: u16) -> usize {
223    let mut gap = delta;
224    let mut bridge_rows = 0_usize;
225    let mut stride = 3_u16.pow(7);
226    while gap != stride {
227        if gap > stride {
228            bridge_rows += 1;
229            gap -= stride;
230        } else {
231            stride /= 3;
232        }
233    }
234    bridge_rows
235}
236
237/// Adds a row for the values to be range checked. In case the difference between the current and
238/// next value is not a power of 3, this function will add additional bridge rows to the trace.
239fn write_rows(
240    trace: &mut [Vec<MaybeUninit<Felt>>],
241    step: &mut usize,
242    num_lookups: usize,
243    value: u16,
244    prev_value: u16,
245) {
246    let mut gap = value - prev_value;
247    let mut prev_val = prev_value;
248    let mut stride = 3_u16.pow(7);
249    while gap != stride {
250        if gap > stride {
251            gap -= stride;
252            prev_val += stride;
253            write_trace_row(trace, step, 0, prev_val as u64);
254        } else {
255            stride /= 3;
256        }
257    }
258    write_trace_row(trace, step, num_lookups, value as u64);
259}
260
261/// Populates a single row at the specified step in the trace table.
262fn write_trace_row(
263    trace: &mut [Vec<MaybeUninit<Felt>>],
264    step: &mut usize,
265    num_lookups: usize,
266    value: u64,
267) {
268    trace[0][*step].write(Felt::new(num_lookups as u64));
269    trace[1][*step].write(Felt::new(value));
270    *step += 1;
271}
