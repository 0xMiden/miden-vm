15// AUXILIARY TRACE BUILDER
16// ================================================================================================
17
18/// Describes how to construct the execution trace of columns related to the range checker in the
19/// auxiliary segment of the trace. These are used in multiset checks.
20#[derive(Debug, Clone)]
21pub struct AuxTraceBuilder {
22    /// A list of the unique values for which range checks are performed.
23    lookup_values: Vec<u16>,
24    /// Range check lookups performed by all user operations, grouped and sorted by the clock cycle
25    /// at which they are requested.
26    cycle_lookups: BTreeMap<RowIndex, Vec<u16>>,
27    // The index of the first row of Range Checker's trace when the padded rows end and values to
28    // be range checked start.
29    values_start: usize,
30}
31
32impl AuxTraceBuilder {
33    // CONSTRUCTOR
34    // --------------------------------------------------------------------------------------------
35    pub fn new(
36        lookup_values: Vec<u16>,
37        cycle_lookups: BTreeMap<RowIndex, Vec<u16>>,
38        values_start: usize,
39    ) -> Self {
40        Self {
41            lookup_values,
42            cycle_lookups,
43            values_start,
44        }
45    }
46
47    // AUX COLUMN BUILDERS
48    // --------------------------------------------------------------------------------------------
49
50    /// Builds and returns range checker auxiliary trace columns. Currently this consists of one
51    /// column:
52    /// - `b_range`: ensures that the range checks performed by the Range Checker match those
53    ///   requested by the Stack and Memory processors.
54    pub fn build_aux_columns<E: ExtensionField<Felt>>(
55        &self,
56        main_trace: &MainTrace,
57        challenges: &Challenges<E>,
58    ) -> Vec<Vec<E>> {
59        let b_range = self.build_aux_col_b_range(main_trace, challenges);
60        vec![b_range]
61    }
62
63    /// Builds the execution trace of the range check `b_range` column which ensure that the range
64    /// check lookups performed by user operations match those executed by the Range Checker.
65    fn build_aux_col_b_range<E: ExtensionField<Felt>>(
66        &self,
67        main_trace: &MainTrace,
68        challenges: &Challenges<E>,
69    ) -> Vec<E> {
70        // run batch inversion on the lookup values
71        let divisors = get_divisors(&self.lookup_values, challenges.alpha);
72
73        // allocate memory for the running sum column and set the initial value to ZERO
74        let mut b_range: Vec<MaybeUninit<E>> = uninit_vector(main_trace.num_rows());
75        b_range[0].write(E::ZERO);
76
77        // keep track of the last updated row in the `b_range` running sum column. `b_range` is
78        // filled with result values that are added to the next row after the operation's execution.
79        let mut b_range_idx = 0_usize;
80        // track the current running sum value to avoid reading from MaybeUninit
81        let mut current_value = E::ZERO;
82
83        // the first half of the trace only includes values from the operations.
84        for (clk, range_checks) in
85            self.cycle_lookups.range(RowIndex::from(0)..RowIndex::from(self.values_start))
86        {
87            let clk: usize = (*clk).into();
88
89            // if we skipped some cycles since the last update was processed, values in the last
90            // updated row should be copied over until the current cycle.
91            if b_range_idx < clk {
92                b_range[(b_range_idx + 1)..=clk].fill(MaybeUninit::new(current_value));
93            }
94
95            // move the column pointer to the next row.
96            b_range_idx = clk + 1;
97
98            let mut new_value = current_value;
99            // include the operation lookups
100            for lookup in range_checks.iter() {
101                let value = divisors.get(lookup).expect("invalid lookup value");
102                new_value -= *value;
103            }
104            b_range[b_range_idx].write(new_value);
105            current_value = new_value;
106        }
107
108        // if we skipped some cycles since the last update was processed, values in the last
109        // updated row should by copied over until the current cycle.
110        if b_range_idx < self.values_start {
111            b_range[(b_range_idx + 1)..=self.values_start].fill(MaybeUninit::new(current_value));
112        }
113
114        // after the padded section of the range checker table, include the lookup value specified
115        // by the range checker into the running sum at each step, and remove lookups from user ops
116        // at any step where user ops were executed.
117        //
118        // Note: we take `num_rows - 1` because the loop writes to `b_range[row_idx + 1]`, so we
119        // need to stop one row early to avoid writing past the end of the array.
120        for (row_idx, (multiplicity, lookup)) in main_trace
121            .get_column(M_COL_IDX)
122            .iter()
123            .zip(main_trace.get_column(V_COL_IDX).iter())
124            .enumerate()
125            .take(main_trace.num_rows() - 1)
126            .skip(self.values_start)
127        {
128            b_range_idx = row_idx + 1;
129
130            let mut new_value = current_value;
131            if *multiplicity != ZERO {
132                // add the value in the range checker: multiplicity / (alpha + lookup)
133                let value = divisors
134                    .get(&(lookup.as_canonical_u64() as u16))
135                    .expect("invalid lookup value");
136                new_value = current_value + *value * *multiplicity;
137            }
138
139            // subtract the range checks requested by operations
140            if let Some(range_checks) = self.cycle_lookups.get(&(row_idx as u32).into()) {
141                for lookup in range_checks.iter() {
142                    let value = divisors.get(lookup).expect("invalid lookup value");
143                    new_value -= *value;
144                }
145            }
146
147            b_range[b_range_idx].write(new_value);
148            current_value = new_value;
149        }
150
151        // at this point, all range checks from user operations and the range checker should be
152        // matched - so, the last value must be ZERO;
153        assert_eq!(current_value, E::ZERO);
154
155        if b_range_idx < b_range.len() - 1 {
156            b_range[(b_range_idx + 1)..].fill(MaybeUninit::new(E::ZERO));
157        }
158
159        // all elements are now initialized
160        unsafe { assume_init_vec(b_range) }
161    }
162}

164/// Runs batch inversion on all range check lookup values and returns a map which maps each value
165/// to the divisor used for including it in the LogUp lookup. In other words, the map contains
166/// mappings of x to 1/(alpha + x).
167fn get_divisors<E: ExtensionField<Felt>>(lookup_values: &[u16], alpha: E) -> BTreeMap<u16, E> {
168    // run batch inversion on the lookup values
169    let mut values: Vec<MaybeUninit<E>> = uninit_vector(lookup_values.len());
170    let mut inv_values: Vec<MaybeUninit<E>> = uninit_vector(lookup_values.len());
171
172    let mut acc = E::ONE;
173    for (i, (value, inv_value)) in values.iter_mut().zip(inv_values.iter_mut()).enumerate() {
174        inv_value.write(acc);
175        let v = alpha + E::from_u16(lookup_values[i]);
176        value.write(v);
177        acc *= v;
178    }
179
180    // all elements are now initialized
181    let values = unsafe { assume_init_vec(values) };
182    let mut inv_values = unsafe { assume_init_vec(inv_values) };
183
184    // invert the accumulated product
185    acc = acc.inverse();
186
187    // multiply the accumulated product by the original values to compute the inverses, then
188    // build a map of inverses for the lookup values
189    let mut log_values = BTreeMap::new();
190    for i in (0..lookup_values.len()).rev() {
191        inv_values[i] *= acc;
192        acc *= values[i];
193        log_values.insert(lookup_values[i], inv_values[i]);
194    }
195
196    log_values
197}
