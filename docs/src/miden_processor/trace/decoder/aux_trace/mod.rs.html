19// AUXILIARY TRACE BUILDER
20// ================================================================================================
21
22/// Constructs the execution traces of stack-related auxiliary trace segment columns
23/// (used in multiset checks).
24#[derive(Debug, Default, Clone, Copy)]
25pub struct AuxTraceBuilder {}
26
27impl AuxTraceBuilder {
28    /// Builds and returns decoder auxiliary trace columns p1, p2, and p3 describing states of block
29    /// stack, block hash, and op group tables respectively.
30    pub fn build_aux_columns<E: ExtensionField<Felt>>(
31        &self,
32        main_trace: &MainTrace,
33        challenges: &Challenges<E>,
34    ) -> Vec<Vec<E>> {
35        let block_stack_column_builder = BlockStackColumnBuilder::default();
36        let block_hash_column_builder = BlockHashTableColumnBuilder::default();
37        let op_group_table_column_builder = OpGroupTableColumnBuilder::default();
38
39        let p1 = block_stack_column_builder.build_aux_column(main_trace, challenges);
40        let p2 = block_hash_column_builder.build_aux_column(main_trace, challenges);
41        let p3 = op_group_table_column_builder.build_aux_column(main_trace, challenges);
42
43        debug_assert_eq!(
44            *p1.last().unwrap(),
45            E::ONE,
46            "block stack table is not empty at the end of program execution"
47        );
48        // p2 (block hash table) does NOT end at 1: the program hash boundary response
49        // is unmatched, so p2 ends at 1/program_hash_msg. This is verified by the
50        // verifier in reduced_aux_values.
51        debug_assert_eq!(
52            *p3.last().unwrap(),
53            E::ONE,
54            "op group table is not empty at the end of program execution"
55        );
56
57        vec![p1, p2, p3]
58    }
59}
