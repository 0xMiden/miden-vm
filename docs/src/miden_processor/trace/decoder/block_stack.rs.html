11/// Keeps track of code blocks which are currently being executed by the VM.
12#[derive(Debug, Default, Clone)]
13pub struct BlockStack {
14    blocks: Vec<BlockInfo>,
15}
16
17impl BlockStack {
18    // STATE ACCESSORS AND MUTATORS
19    // --------------------------------------------------------------------------------------------
20
21    /// Pushes a new code block onto the block stack and returns the address of the block's parent.
22    ///
23    /// The block is identified by its address, and we also need to know what type of a block this
24    /// is. Additionally, for CALL, SYSCALL and DYNCALL blocks, execution context info must be
25    /// provided. Other information (i.e., the block's parent, whether the block is a body of a loop
26    /// or a first child of a JOIN block) is determined from the information already on the stack.
27    pub fn push(
28        &mut self,
29        addr: Felt,
30        block_type: BlockType,
31        ctx_info: Option<ExecutionContextInfo>,
32    ) -> Felt {
33        // make sure execution context was provided for CALL, SYSCALL and DYNCALL blocks
34        if block_type == BlockType::Call
35            || block_type == BlockType::SysCall
36            || block_type == BlockType::Dyncall
37        {
38            debug_assert!(ctx_info.is_some(), "no execution context provided for a CALL block");
39        } else {
40            debug_assert!(ctx_info.is_none(), "execution context provided for a non-CALL block");
41        }
42
43        // determine additional info about the new block based on its parent
44        let (parent_addr, is_loop_body, _is_first_child) = match self.blocks.last() {
45            Some(parent) => match parent.block_type {
46                // if the parent block is a LOOP block, the new block must be a loop body
47                BlockType::Loop(loop_entered) => {
48                    debug_assert!(loop_entered, "parent is un-entered loop");
49                    (parent.addr, true, false)
50                },
51                // if the parent block is a JOIN block, figure out if the new block is the first
52                // or the second child
53                BlockType::Join(first_child_executed) => {
54                    (parent.addr, false, !first_child_executed)
55                },
56                _ => (parent.addr, false, false),
57            },
58            // if the block stack is empty, a new block is neither a body of a loop nor the first
59            // child of a JOIN block; also, we set the parent address to ZERO.
60            None => (ZERO, false, false),
61        };
62
63        self.blocks.push(BlockInfo {
64            addr,
65            block_type,
66            parent_addr,
67            ctx_info,
68            is_loop_body,
69        });
70        parent_addr
71    }
72
73    /// Removes a block from the top of the stack and returns it.
74    pub fn pop(&mut self) -> BlockInfo {
75        let block = self.blocks.pop().expect("block stack is empty");
76        // if the parent block is a JOIN block (i.e., we just finished executing a child of a JOIN
77        // block) and if the first_child_executed hasn't been set to true yet, set it to true
78        if let Some(parent) = self.blocks.last_mut()
79            && let BlockType::Join(first_child_executed) = parent.block_type
80            && !first_child_executed
81        {
82            parent.block_type = BlockType::Join(true);
83        }
84        block
85    }
86
87    /// Returns true if the block stack is empty.
88    pub fn is_empty(&self) -> bool {
89        self.blocks.is_empty()
90    }
91
92    /// Returns a reference to a block at the top of the stack.
93    pub fn peek(&self) -> &BlockInfo {
94        self.blocks.last().expect("block stack is empty")
95    }
96
97    /// Returns a mutable reference to a block at the top of the stack.
98    pub fn peek_mut(&mut self) -> &mut BlockInfo {
99        self.blocks.last_mut().expect("block stack is empty")
100    }
101}

106/// Contains basic information about a code block.
107#[derive(Debug, Clone)]
108pub struct BlockInfo {
109    pub addr: Felt,
110    block_type: BlockType,
111    pub parent_addr: Felt,
112    pub ctx_info: Option<ExecutionContextInfo>,
113    pub is_loop_body: bool,
114}
115
116impl BlockInfo {
117    /// Returns ONE if the this block is a LOOP block and the body of the loop was executed at
118    /// least once; otherwise, returns ZERO.
119    pub fn is_entered_loop(&self) -> Felt {
120        if self.block_type == BlockType::Loop(true) {
121            ONE
122        } else {
123            ZERO
124        }
125    }
126
127    /// Returns ONE if this block is a body of a LOOP block; otherwise returns ZERO.
128    pub const fn is_loop_body(&self) -> Felt {
129        if self.is_loop_body { ONE } else { ZERO }
130    }
131
132    /// Returns ONE if this block is a CALL or DYNCALL block; otherwise returns ZERO.
133    pub const fn is_call(&self) -> Felt {
134        match self.block_type {
135            BlockType::Call | BlockType::Dyncall => ONE,
136            _ => ZERO,
137        }
138    }
139
140    /// Returns ONE if this block is a SYSCALL block; otherwise returns ZERO.
141    pub const fn is_syscall(&self) -> Felt {
142        match self.block_type {
143            BlockType::SysCall => ONE,
144            _ => ZERO,
145        }
146    }
147}

149// EXECUTION CONTEXT INFO
150// ================================================================================================
151
152/// Contains information about an execution context. Execution contexts are relevant only for CALL
153/// and SYSCALL blocks.
154#[derive(Debug, Default, Clone, Copy)]
155pub struct ExecutionContextInfo {
156    /// Context ID of the block's parent.
157    pub parent_ctx: ContextId,
158    /// Hash of the function which initiated execution of the block's parent. If the parent is a
159    /// root context, this will be set to [ZERO; 4].
160    pub parent_fn_hash: Word,
161    /// Depth of the operand stack right before a CALL operation is executed.
162    pub parent_stack_depth: u32,
163    /// Address of the top row in the overflow table right before a CALL operations is executed.
164    pub parent_next_overflow_addr: Felt,
165}
166
167impl ExecutionContextInfo {
168    /// Returns an new [ExecutionContextInfo] instantiated with the specified parameters.
169    pub fn new(
170        parent_ctx: ContextId,
171        parent_fn_hash: Word,
172        parent_stack_depth: u32,
173        parent_next_overflow_addr: Felt,
174    ) -> Self {
175        Self {
176            parent_fn_hash,
177            parent_ctx,
178            parent_stack_depth,
179            parent_next_overflow_addr,
180        }
181    }
182}

187/// Specifies type of a code block with additional info for some block types.
188#[derive(Debug, Clone, Copy, PartialEq, Eq)]
189pub enum BlockType {
190    Join(bool), // internal value set to true when the first child is fully executed
191    Split,
192    Loop(bool), // internal value set to false if the loop is never entered
193    Call,
194    Dyn,
195    Dyncall,
196    SysCall,
197    BasicBlock,
198}
