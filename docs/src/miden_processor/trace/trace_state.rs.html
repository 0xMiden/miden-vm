25/// Information required to build a core trace fragment (i.e. the system, decoder and stack
26/// columns).
27///
28/// This struct is meant to be built by the processor, and consumed mutably by a core trace fragment
29/// builder. That is, as core trace generation progresses, this struct can be mutated to represent
30/// the generation context at any clock cycle within the fragment.
31///
32/// This struct is conceptually divided into 4 components:
33/// 1. core trace state: the state of the processor at any clock cycle in the fragment, initialized
34///    to the state at the first clock cycle in the fragment,
35/// 2. execution replay: information needed to replay the execution of the processor for the
36///    remainder of the fragment,
37/// 3. continuation: a stack of continuations for the processor representing the nodes in the MAST
38///    forest to execute when the current node is done executing,
39/// 4. initial MAST forest: the MAST forest being executed at the start of the fragment (which can
40///    change during execution when encountering an [`miden_core::mast::ExternalNode`] or
41///    [`miden_core::mast::DynNode`]).
42#[derive(Debug)]
43pub struct CoreTraceFragmentContext {
44    pub state: CoreTraceState,
45    pub replay: ExecutionReplay,
46    pub continuation: ContinuationStack,
47    pub initial_mast_forest: Arc<MastForest>,
48}

65/// The `SystemState` represents all the information needed to build one row of the System trace.
66///
67/// This struct captures the complete state of the system at a specific clock cycle, allowing for
68/// reconstruction of the system trace during concurrent execution.
69#[derive(Debug, Clone, PartialEq, Eq)]
70pub struct SystemState {
71    /// Current clock cycle (row index in the trace)
72    pub clk: RowIndex,
73
74    /// Execution context ID - starts at 0 (root context), changes on CALL/SYSCALL operations
75    pub ctx: ContextId,
76
77    /// Hash of the function that initiated the current execution context
78    /// - For root context: [ZERO; 4]
79    /// - For CALL/DYNCALL contexts: hash of the called function
80    /// - For SYSCALL contexts: hash remains from the calling function
81    pub fn_hash: Word,
82
83    /// Precompile transcript state (sponge capacity) used for recording `log_precompile` calls
84    /// - Initially [ZERO; 4]
85    /// - Updated with each `log_precompile` invocation
86    pub pc_transcript_state: PrecompileTranscriptState,
87}
88
89impl SystemState {
90    /// Convenience constructor that creates a new `SystemState` from a `Processor`.
91    pub(crate) fn from_processor<P: Processor>(processor: &P) -> Self {
92        Self {
93            clk: processor.system().clock(),
94            ctx: processor.system().ctx(),
95            fn_hash: processor.system().caller_hash(),
96            pc_transcript_state: processor.precompile_transcript_state(),
97        }
98    }
99}

104/// The subset of the decoder state required to build the trace.
105#[derive(Debug)]
106pub struct DecoderState {
107    /// The value of the [miden_air::trace::decoder::ADDR_COL_IDX] column
108    pub current_addr: Felt,
109    /// The address of the current MAST node's parent.
110    pub parent_addr: Felt,
111}
112
113impl DecoderState {
114    /// This function is called when start executing a node (e.g. `JOIN`, `SPLIT`, etc). It emulates
115    /// pushing a new node onto the block stack, and updates the decoder state to point to the
116    /// current node in the block stack. Hence, the `current_addr` is set to the (replayed) address
117    /// of the current node, and the `parent_addr` is set to the (replayed) address of the parent
118    /// node (i.e. the node previously on top of the block stack).
119    pub fn replay_node_start(
120        &mut self,
121        block_address_replay: &mut BlockAddressReplay,
122        block_stack_replay: &mut BlockStackReplay,
123    ) -> Result<(), ExecutionError> {
124        self.current_addr = block_address_replay.replay_block_address()?;
125        self.parent_addr = block_stack_replay.replay_node_start_parent_addr()?;
126        Ok(())
127    }
128
129    /// This function is called when we hit an `END` operation, signaling the end of execution for a
130    /// node. It updates the decoder state to point to the previous node in the block stack (which
131    /// could be renamed to "node stack"), and returns the address of the node that just ended.
132    pub fn replay_node_end(
133        &mut self,
134        block_stack_replay: &mut BlockStackReplay,
135    ) -> Result<Felt, ExecutionError> {
136        let node_end_data = block_stack_replay.replay_node_end()?;
137
138        self.current_addr = node_end_data.prev_addr;
139        self.parent_addr = node_end_data.prev_parent_addr;
140
141        Ok(node_end_data.ended_node_addr)
142    }
143}

148/// This struct captures the state of the top 16 elements of the stack at a specific clock cycle;
149/// that is, those elements that are written directly into the trace.
150///
151/// The stack trace consists of 19 columns total: 16 stack columns + 3 helper columns. The helper
152/// columns (stack_depth, overflow_addr, and overflow_helper) are computed from the stack_depth and
153/// last_overflow_addr fields.
154#[derive(Debug)]
155pub struct StackState {
156    /// Top 16 stack slots (s0 to s15). These represent the top elements of the stack that are
157    /// directly accessible.
158    pub stack_top: [Felt; MIN_STACK_DEPTH], // 16 columns
159
160    /// Current number of elements on the stack. It is guaranteed to be >= 16.
161    stack_depth: usize,
162
163    /// The last recorded overflow address for the stack - which is the clock cycle at which the
164    /// last item was pushed to the overflow
165    last_overflow_addr: Felt,
166}
167
168impl StackState {
169    /// Creates a new StackState with the provided parameters.
170    ///
171    /// `stack_top` should be the top 16 elements of the stack stored in reverse order, i.e.,
172    /// `stack_top[15]` is the topmost element (s0), and `stack_top[0]` is the bottommost element
173    /// (s15).
174    pub fn new(
175        stack_top: [Felt; MIN_STACK_DEPTH],
176        stack_depth: usize,
177        last_overflow_addr: Felt,
178    ) -> Self {
179        Self {
180            stack_top,
181            stack_depth,
182            last_overflow_addr,
183        }
184    }
185
186    /// Returns the value at the specified index in the stack top.
187    ///
188    /// # Panics
189    /// - if the index is greater than or equal to [MIN_STACK_DEPTH].
190    pub fn get(&self, index: usize) -> Felt {
191        self.stack_top[MIN_STACK_DEPTH - index - 1]
192    }
193
194    /// Returns the stack depth (b0 helper column)
195    pub fn stack_depth(&self) -> usize {
196        self.stack_depth
197    }
198
199    /// Returns the overflow address (b1 helper column) using the stack overflow replay
200    pub fn overflow_addr(&self) -> Felt {
201        self.last_overflow_addr
202    }
203
204    /// Returns the number of elements in the current context's overflow stack.
205    pub fn num_overflow_elements_in_current_ctx(&self) -> usize {
206        debug_assert!(self.stack_depth >= MIN_STACK_DEPTH);
207        self.stack_depth - MIN_STACK_DEPTH
208    }
209
210    /// Pushes the given element onto the overflow stack at the provided clock cycle.
211    pub fn push_overflow(&mut self, _element: Felt, clk: RowIndex) {
212        self.stack_depth += 1;
213        self.last_overflow_addr = clk.into();
214    }
215
216    /// Pops the top element from the overflow stack at the provided clock cycle, if any.
217    ///
218    /// If the overflow table is empty (i.e. stack depth is 16), the stack depth is unchanged, and
219    /// None is returned.
220    pub fn pop_overflow(
221        &mut self,
222        stack_overflow_replay: &mut StackOverflowReplay,
223    ) -> Result<Option<Felt>, OperationError> {
224        debug_assert!(self.stack_depth >= MIN_STACK_DEPTH);
225
226        if self.stack_depth > MIN_STACK_DEPTH {
227            let (stack_value, new_overflow_addr) = stack_overflow_replay.replay_pop_overflow()?;
228            self.stack_depth -= 1;
229            self.last_overflow_addr = new_overflow_addr;
230            Ok(Some(stack_value))
231        } else {
232            self.last_overflow_addr = ZERO;
233            Ok(None)
234        }
235    }
236
237    /// Derives the denominator of the overflow helper (h0 helper column) from the current stack
238    /// depth.
239    ///
240    /// It is expected that this values gets later inverted via batch inversion.
241    pub fn overflow_helper(&self) -> Felt {
242        let denominator = self.stack_depth() - MIN_STACK_DEPTH;
243        Felt::new(denominator as u64)
244    }
245
246    /// Starts a new execution context for this stack, resetting the stack depth to its minimum
247    /// value, and last overflow address to 0.
248    ///
249    /// This has the effect of hiding the contents of the overflow table such that it appears as if
250    /// the overflow table in the new context is empty.
251    pub fn start_context(&mut self) {
252        self.stack_depth = MIN_STACK_DEPTH;
253        self.last_overflow_addr = ZERO;
254    }
255
256    /// Restores the prior context for this stack.
257    ///
258    /// This has the effect bringing back items previously hidden from the overflow table.
259    pub fn restore_context(
260        &mut self,
261        stack_overflow_replay: &mut StackOverflowReplay,
262    ) -> Result<(), OperationError> {
263        let (stack_depth, last_overflow_addr) =
264            stack_overflow_replay.replay_restore_context_overflow_addr()?;
265        // Restore stack depth to the value from before the context switch (parallel to Process
266        // Stack behavior)
267        self.stack_depth = stack_depth;
268        self.last_overflow_addr = last_overflow_addr;
269        Ok(())
270    }
271}

479/// Records and replays all the reads made to memory, in which all elements and words read from
480/// memory during a given fragment are recorded by the fast processor, and replayed by the main
481/// trace fragment generators.
482///
483/// This is used to simulate memory reads in parallel trace generation without needing to actually
484/// access the memory chiplet.
485///
486/// Elements/words read are stored with their addresses and are assumed to be read from the same
487/// addresses that they were recorded at. This works naturally since the fast processor has exactly
488/// the same access patterns as the main trace generators (which re-executes part of the program).
489/// The read methods include debug assertions to verify address consistency.
490#[derive(Debug, Default)]
491pub struct MemoryReadsReplay {
492    elements_read: VecDeque<(Felt, Felt, ContextId, RowIndex)>,
493    words_read: VecDeque<(Word, Felt, ContextId, RowIndex)>,
494}
495
496impl MemoryReadsReplay {
497    // MUTATIONS (populated by the fast processor)
498    // --------------------------------------------------------------------------------
499
500    /// Records a read element from memory
501    pub fn record_read_element(
502        &mut self,
503        element: Felt,
504        addr: Felt,
505        ctx: ContextId,
506        clk: RowIndex,
507    ) {
508        self.elements_read.push_back((element, addr, ctx, clk));
509    }
510
511    /// Records a read word from memory
512    pub fn record_read_word(&mut self, word: Word, addr: Felt, ctx: ContextId, clk: RowIndex) {
513        self.words_read.push_back((word, addr, ctx, clk));
514    }
515
516    // ACCESSORS
517    // --------------------------------------------------------------------------------
518
519    pub fn replay_read_element(&mut self, addr: Felt) -> Result<Felt, MemoryError> {
520        let (element, stored_addr, _ctx, _clk) = self
521            .elements_read
522            .pop_front()
523            .ok_or(MemoryError::MemoryReadFailed("memory elements replay is empty".to_string()))?;
524        debug_assert_eq!(stored_addr, addr, "Address mismatch: expected {addr}, got {stored_addr}");
525        Ok(element)
526    }
527
528    pub fn replay_read_word(&mut self, addr: Felt) -> Result<Word, MemoryError> {
529        let (word, stored_addr, _ctx, _clk) = self
530            .words_read
531            .pop_front()
532            .ok_or(MemoryError::MemoryReadFailed("memory words replay is empty".to_string()))?;
533        debug_assert_eq!(stored_addr, addr, "Address mismatch: expected {addr}, got {stored_addr}");
534        Ok(word)
535    }
536
537    /// Returns an iterator over all recorded memory element reads, yielding tuples of
538    /// (element, address, context ID, clock cycle).
539    pub fn iter_read_elements(&self) -> impl Iterator<Item = (Felt, Felt, ContextId, RowIndex)> {
540        self.elements_read.iter().copied()
541    }
542
543    /// Returns an iterator over all recorded memory word reads, yielding tuples of
544    /// (word, address, context ID, clock cycle).
545    pub fn iter_read_words(&self) -> impl Iterator<Item = (Word, Felt, ContextId, RowIndex)> {
546        self.words_read.iter().copied()
547    }
548}

640/// The read methods include debug assertions to verify parameter consistency.
641#[derive(Debug, Default)]
642pub struct AdviceReplay {
643    // Stack operations
644    stack_pops: VecDeque<Felt>,
645    stack_word_pops: VecDeque<Word>,
646    stack_dword_pops: VecDeque<[Word; 2]>,

887/// Records and replays the response of requests made to the hasher chiplet during the execution of
888/// a program.
889///
890/// The hasher responses are recorded during fast processor execution and then replayed during core
891/// trace generation.
892#[derive(Debug, Default)]
893pub struct HasherResponseReplay {
894    /// Recorded hasher operations from permutation operations (HPerm).
895    ///
896    /// Each entry contains (address, output_state)
897    permutation_operations: VecDeque<(Felt, [Felt; 12])>,
898
899    /// Recorded hasher operations from Merkle path verification operations.
900    ///

915    /// permutation)

922        self.build_merkle_root_operations.push_back((addr, computed_root));
923    }

926    pub fn record_update_merkle_root(&mut self, addr: Felt, old_root: Word, new_root: Word) {
927        self.mrupdate_operations.push_back((addr, old_root, new_root));
928    }

931    // --------------------------------------------------------------------------------------------

933    /// Replays a `Hasher::permute` operation, returning its address and result
934    pub fn replay_permute(&mut self) -> Result<(Felt, [Felt; 12]), OperationError> {
935        self.permutation_operations
936            .pop_front()
937            .ok_or(OperationError::Internal("no permutation operations recorded"))
938    }
939

941    pub fn replay_build_merkle_root(&mut self) -> Result<(Felt, Word), OperationError> {
942        self.build_merkle_root_operations

945    }

954
955impl HasherInterface for HasherResponseReplay {
956    fn permute(&mut self, _state: HasherState) -> Result<(Felt, HasherState), OperationError> {
957        self.replay_permute()
958    }
959
960    fn verify_merkle_root(
961        &mut self,
962        claimed_root: Word,
963        _value: Word,
964        _path: Option<&MerklePath>,
965        _index: Felt,
966        on_err: impl FnOnce() -> OperationError,
967    ) -> Result<Felt, OperationError> {
968        let (addr, computed_root) = self.replay_build_merkle_root()?;
969        if claimed_root == computed_root {
970            Ok(addr)
971        } else {
972            // If the hasher doesn't compute the same root (using the same path),
973            // then it means that `node` is not the value currently in the tree at `index`
974            Err(on_err())
975        }
976    }
977
978    fn update_merkle_root(
979        &mut self,
980        claimed_old_root: Word,
981        _old_value: Word,
982        _new_value: Word,
983        _path: Option<&MerklePath>,
984        _index: Felt,
985        on_err: impl FnOnce() -> OperationError,
986    ) -> Result<(Felt, Word), OperationError> {
987        let (address, old_root, new_root) = self.replay_update_merkle_root()?;
988
989        if claimed_old_root == old_root {
990            Ok((address, new_root))
991        } else {
992            Err(on_err())
993        }
994    }
995}

1074// STACK OVERFLOW REPLAY
1075// ================================================================================================
1076
1077/// Implements a shim for stack overflow operations, in which all overflow values and addresses
1078/// during a given fragment are pre-recorded by the fast processor and replayed by the main trace
1079/// fragment generators.
1080///
1081/// This is used to simulate stack overflow functionality in parallel trace generation without
1082/// needing to maintain the actual overflow table. All overflow operations are recorded during
1083/// fast execution and then replayed during parallel trace generation.
1084///
1085/// The shim records overflow values (from pop operations) and overflow addresses (representing
1086/// the clock cycle of the last overflow update) and provides replay methods that return the
1087/// pre-recorded values. This works naturally since the fast processor has exactly the same
1088/// access patterns as the main trace generators.
1089#[derive(Debug)]
1090pub struct StackOverflowReplay {
1091    /// Recorded overflow values and overflow addresses from pop_overflow operations. Each entry
1092    /// represents a value that was popped from the overflow stack, and the overflow address of the
1093    /// entry at the top of the overflow stack *after* the pop operation.
1094    ///
1095    /// For example, given the following table:
1096    ///
1097    /// | Overflow Value | Overflow Address |
1098    /// |----------------|------------------|
1099    /// |      8         |         14       |
1100    /// |      2         |         16       |
1101    /// |      7         |         18       |
1102    ///
1103    /// a `pop_overflow()` operation would return (popped_value, prev_addr) = (7, 16).
1104    overflow_values: VecDeque<(Felt, Felt)>,
1105
1106    /// Recorded (stack depth, overflow address) returned when restoring a context
1107    restore_context_info: VecDeque<(usize, Felt)>,
1108}
1109
1110impl Default for StackOverflowReplay {
1111    fn default() -> Self {
1112        Self::new()
1113    }
1114}
1115
1116impl StackOverflowReplay {
1117    /// Creates a new StackOverflowReplay with empty operation vectors
1118    pub fn new() -> Self {
1119        Self {
1120            overflow_values: VecDeque::new(),
1121            restore_context_info: VecDeque::new(),
1122        }
1123    }
1124
1125    // MUTATORS
1126    // --------------------------------------------------------------------------------
1127
1128    /// Records the value returned by a pop_overflow operation, along with the overflow address
1129    /// stored in the overflow table *after* the pop. That is, `new_overflow_addr` represents the
1130    /// clock cycle at which the value *before* `value` was added to the overflow table. See the
1131    /// docstring for the `overflow_values` field for more information.
1132    ///
1133    /// This *must* only be called if there is an actual value in the overflow table to pop; that
1134    /// is, don't call if the stack depth is 16.
1135    pub fn record_pop_overflow(&mut self, value: Felt, new_overflow_addr: Felt) {
1136        self.overflow_values.push_back((value, new_overflow_addr));
1137    }
1138
1139    /// Records the overflow address when restoring a context
1140    pub fn record_restore_context_overflow_addr(&mut self, stack_depth: usize, addr: Felt) {
1141        self.restore_context_info.push_back((stack_depth, addr));
1142    }
1143
1144    // ACCESSORS
1145    // --------------------------------------------------------------------------------
1146
1147    /// Peeks at the next recorded pop_overflow operation, returning the previously recorded value
1148    /// and `new_overflow_addr` without removing it from the queue.
1149    pub fn peek_replay_pop_overflow(&self) -> Option<&(Felt, Felt)> {
1150        self.overflow_values.front()
1151    }
1152
1153    /// Replays a pop_overflow operation, returning the previously recorded value and
1154    /// `new_overflow_addr`.
1155    ///
1156    /// This *must* only be called if there is an actual value in the overflow table to pop; that
1157    /// is, don't call if the stack depth is 16.
1158    ///
1159    /// See [Self::record_pop_overflow] for more details.
1160    pub fn replay_pop_overflow(&mut self) -> Result<(Felt, Felt), OperationError> {
1161        self.overflow_values
1162            .pop_front()
1163            .ok_or(OperationError::Internal("no overflow pop operations recorded"))
1164    }
1165
1166    /// Replays the overflow address when restoring a context
1167    pub fn replay_restore_context_overflow_addr(
1168        &mut self,
1169    ) -> Result<(usize, Felt), OperationError> {
1170        self.restore_context_info
1171            .pop_front()
1172            .ok_or(OperationError::Internal("no overflow address operations recorded"))
1173    }
1174}
