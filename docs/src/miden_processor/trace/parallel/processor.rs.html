31// REPLAY PROCESSOR
32// ================================================================================================

34/// A processor implementation used in conjunction with the [`CoreTraceGenerationTracer`] in
35/// [`super::build_trace`] to replay the execution of a fragment of execution for a predetermined
36/// number of clock cycles.
37///
38/// This processor uses various "replay" structures to provide the necessary state during execution,
39/// such as stack overflows, memory reads, advice provider values, hasher responses, execution
40/// contexts, and MAST forest resolutions. The processor executes until it reaches the specified
41/// maximum clock cycle, at which point it stops execution (due to the [`ReplayStopper`]).
42///
43/// The replay structures and initial system and stack state are built by the
44/// [`crate::execution_tracer::ExecutionTracer`] in conjunction with
45/// [`crate::FastProcessor::execute_for_trace`].
46#[derive(Debug)]
47pub(crate) struct ReplayProcessor {
48    pub system: SystemState,
49    pub stack: StackState,
50    pub stack_overflow_replay: StackOverflowReplay,
51    pub execution_context_replay: ExecutionContextReplay,
52    pub advice_replay: AdviceReplay,
53    pub memory_reads_replay: MemoryReadsReplay,
54    pub hasher_response_replay: HasherResponseReplay,
55    pub mast_forest_resolution_replay: MastForestResolutionReplay,
56
57    /// The maximum clock cycle at which this processor should stop execution.
58    pub maximum_clock: RowIndex,
59}
60
61impl ReplayProcessor {
62    /// Creates a new instance of the [`ReplayProcessor`].
63    ///
64    /// The parameters are expected to be built by the
65    /// [`crate::execution_tracer::ExecutionTracer`] when used in conjunction with
66    /// [`crate::FastProcessor::execute_for_trace`].
67    pub fn new(
68        initial_system: SystemState,
69        initial_stack: StackState,
70        stack_overflow_replay: StackOverflowReplay,
71        execution_context_replay: ExecutionContextReplay,
72        advice_replay: AdviceReplay,
73        memory_reads_replay: MemoryReadsReplay,
74        hasher_response_replay: HasherResponseReplay,
75        mast_forest_resolution_replay: MastForestResolutionReplay,
76        num_clocks_to_execute: RowIndex,
77    ) -> Self {
78        let maximum_clock = initial_system.clk + num_clocks_to_execute.as_usize();
79
80        Self {
81            system: initial_system,
82            stack: initial_stack,
83            stack_overflow_replay,
84            execution_context_replay,
85            advice_replay,
86            memory_reads_replay,
87            hasher_response_replay,
88            mast_forest_resolution_replay,
89            maximum_clock,
90        }
91    }
92
93    /// Executes the processor until it reaches the end of the fragment, or until an error occurs.
94    pub fn execute<T>(
95        &mut self,
96        continuation_stack: &mut ContinuationStack,
97        current_forest: &mut Arc<MastForest>,
98        kernel: &Kernel,
99        tracer: &mut T,
100    ) -> Result<(), ExecutionError>
101    where
102        T: Tracer<Processor = Self>,
103    {
104        match self.execute_impl(continuation_stack, current_forest, kernel, tracer) {
105            ControlFlow::Continue(_) => {
106                // End of program reached - i.e. the execution loop exited without the end of
107                // fragment being reached (and thus the stopper breaking).
108                Ok(())
109            },
110            ControlFlow::Break(break_reason) => match break_reason {
111                BreakReason::Err(err) => Err(err),
112                BreakReason::Stopped(_continuation) => {
113                    // Our ReplayStopper stopped us because we reached the end of the fragment.
114                    // Hence, this is expected, and we can just return Ok.
115                    Ok(())
116                },
117            },
118        }
119    }
120
121    /// Core execution loop implementation for the replay processor.
122    ///
123    /// This method uses the [`crate::execution::execute_impl`] function to perform the core
124    /// execution loop. See its documentation for more details.
125    fn execute_impl<T>(
126        &mut self,
127        continuation_stack: &mut ContinuationStack,
128        current_forest: &mut Arc<MastForest>,
129        kernel: &Kernel,
130        tracer: &mut T,
131    ) -> ControlFlow<BreakReason>
132    where
133        T: Tracer<Processor = Self>,
134    {
135        let host = &mut NoopHost;
136        let stopper = &ReplayStopper;
137
138        while let ControlFlow::Break(internal_break_reason) =
139            execute_impl(self, continuation_stack, current_forest, kernel, host, tracer, stopper)
140        {
141            match internal_break_reason {
142                InternalBreakReason::User(break_reason) => return ControlFlow::Break(break_reason),
143                InternalBreakReason::Emit {
144                    basic_block_node_id: _,
145                    op_idx: _,
146                    continuation,
147                } => {
148                    // do nothing - in replay processor we don't need to emit anything
149
150                    // Call `finish_emit_op_execution()`, as per the sans-IO contract.
151                    finish_emit_op_execution(
152                        continuation,
153                        self,
154                        continuation_stack,
155                        current_forest,
156                        tracer,
157                        stopper,
158                    )?;
159                },
160                InternalBreakReason::LoadMastForestFromDyn { .. } => {
161                    // load mast forest from replay
162                    let (root_id, new_forest) =
163                        match self.mast_forest_resolution_replay.replay_resolution() {
164                            Ok(v) => v,
165                            Err(err) => {
166                                return ControlFlow::Break(BreakReason::Err(err));
167                            },
168                        };
169
170                    // Finish loading the MAST forest from the Dyn node, as per the sans-IO
171                    // contract.
172                    finish_load_mast_forest_from_dyn_start(
173                        root_id,
174                        new_forest,
175                        self,
176                        current_forest,
177                        continuation_stack,
178                        tracer,
179                        stopper,
180                    )?;
181                },
182                InternalBreakReason::LoadMastForestFromExternal {
183                    external_node_id,
184                    procedure_hash: _,
185                } => {
186                    // load mast forest from replay
187                    let (root_id, new_forest) =
188                        match self.mast_forest_resolution_replay.replay_resolution() {
189                            Ok(v) => v,
190                            Err(err) => {
191                                return ControlFlow::Break(BreakReason::Err(err));
192                            },
193                        };
194
195                    // Finish loading the MAST forest from the External node, as per the sans-IO
196                    // contract.
197                    finish_load_mast_forest_from_external(
198                        root_id,
199                        new_forest,
200                        external_node_id,
201                        current_forest,
202                        continuation_stack,
203                        host,
204                        tracer,
205                    )?;
206                },
207            }
208        }
209
210        // End of program reached (since loop exited without the stopper breaking)
211        ControlFlow::Continue(())
212    }
213}

241impl StackInterface for ReplayProcessor {
242    fn top(&self) -> &[Felt] {
243        &self.stack.stack_top
244    }
245
246    fn get(&self, idx: usize) -> Felt {
247        debug_assert!(idx < MIN_STACK_DEPTH);
248        self.stack.stack_top[MIN_STACK_DEPTH - idx - 1]
249    }
250
251    fn get_mut(&mut self, idx: usize) -> &mut Felt {
252        debug_assert!(idx < MIN_STACK_DEPTH);
253
254        &mut self.stack.stack_top[MIN_STACK_DEPTH - idx - 1]
255    }
256
257    fn get_word(&self, start_idx: usize) -> Word {
258        debug_assert!(start_idx < MIN_STACK_DEPTH - 4);
259
260        let word_start_idx = MIN_STACK_DEPTH - start_idx - 4;
261        let mut result: [Felt; WORD_SIZE] =
262            self.top()[range(word_start_idx, WORD_SIZE)].try_into().unwrap();
263        // Reverse so top of stack (idx 0) goes to word[0]
264        result.reverse();
265        result.into()
266    }
267
268    fn depth(&self) -> u32 {
269        (MIN_STACK_DEPTH + self.stack.num_overflow_elements_in_current_ctx()) as u32
270    }
271
272    fn set(&mut self, idx: usize, element: Felt) {
273        *self.get_mut(idx) = element;
274    }
275
276    fn set_word(&mut self, start_idx: usize, word: &Word) {
277        debug_assert!(start_idx < MIN_STACK_DEPTH - 4);
278        let word_start_idx = MIN_STACK_DEPTH - start_idx - 4;
279
280        // Reverse so word[0] ends up at the top of stack (highest internal index)
281        let mut source: [Felt; WORD_SIZE] = (*word).into();
282        source.reverse();
283
284        let word_on_stack = &mut self.stack.stack_top[range(word_start_idx, WORD_SIZE)];
285        word_on_stack.copy_from_slice(&source);
286    }
287
288    fn swap(&mut self, idx1: usize, idx2: usize) {
289        let a = self.get(idx1);
290        let b = self.get(idx2);
291        self.set(idx1, b);
292        self.set(idx2, a);
293    }
294
295    fn swapw_nth(&mut self, n: usize) {
296        // For example, for n=3, the stack words and variables look like:
297        //    3     2     1     0
298        // | ... | ... | ... | ... |
299        // ^                 ^
300        // nth_word       top_word
301        let (rest_of_stack, top_word) =
302            self.stack.stack_top.split_at_mut(MIN_STACK_DEPTH - WORD_SIZE);
303        let (_, nth_word) = rest_of_stack.split_at_mut(rest_of_stack.len() - n * WORD_SIZE);
304
305        nth_word[0..WORD_SIZE].swap_with_slice(&mut top_word[0..WORD_SIZE]);
306    }
307
308    fn rotate_left(&mut self, n: usize) {
309        let rotation_bot_index = MIN_STACK_DEPTH - n;
310        let new_stack_top_element = self.stack.stack_top[rotation_bot_index];
311
312        // shift the top n elements down by 1, starting from the bottom of the rotation.
313        for i in 0..n - 1 {
314            self.stack.stack_top[rotation_bot_index + i] =
315                self.stack.stack_top[rotation_bot_index + i + 1];
316        }
317
318        // Set the top element (which comes from the bottom of the rotation).
319        self.set(0, new_stack_top_element);
320    }
321
322    fn rotate_right(&mut self, n: usize) {
323        let rotation_bot_index = MIN_STACK_DEPTH - n;
324        let new_stack_bot_element = self.stack.stack_top[MIN_STACK_DEPTH - 1];
325
326        // shift the top n elements up by 1, starting from the top of the rotation.
327        for i in 1..n {
328            self.stack.stack_top[MIN_STACK_DEPTH - i] =
329                self.stack.stack_top[MIN_STACK_DEPTH - i - 1];
330        }
331
332        // Set the bot element (which comes from the top of the rotation).
333        self.stack.stack_top[rotation_bot_index] = new_stack_bot_element;
334    }
335
336    fn increment_size(&mut self) -> Result<(), ExecutionError> {
337        const SENTINEL_VALUE: Felt = Felt::new(Felt::ORDER_U64 - 1);
338
339        // push the last element on the overflow table
340        {
341            let last_element = self.get(MIN_STACK_DEPTH - 1);
342            self.stack.push_overflow(last_element, self.clock());
343        }
344
345        // Shift all other elements down
346        for write_idx in (1..MIN_STACK_DEPTH).rev() {
347            let read_idx = write_idx - 1;
348            self.set(write_idx, self.get(read_idx));
349        }
350
351        // Set the top element to SENTINEL_VALUE to help in debugging. Per the method docs, this
352        // value will be overwritten
353        self.set(0, SENTINEL_VALUE);
354
355        Ok(())
356    }
357
358    fn decrement_size(&mut self) -> Result<(), OperationError> {
359        // Shift all other elements up
360        for write_idx in 0..(MIN_STACK_DEPTH - 1) {
361            let read_idx = write_idx + 1;
362            self.set(write_idx, self.get(read_idx));
363        }
364
365        // Pop the last element from the overflow table
366        if let Some(last_element) = self.stack.pop_overflow(&mut self.stack_overflow_replay)? {
367            // Write the last element to the bottom of the stack
368            self.set(MIN_STACK_DEPTH - 1, last_element);
369        } else {
370            // If overflow table is empty, set the bottom element to zero
371            self.set(MIN_STACK_DEPTH - 1, ZERO);
372        }
373
374        Ok(())
375    }
376}

485// REPLAY STOPPER
486// ================================================================================================
487
488/// A stopper implementation used with the [`ReplayProcessor`] to stop execution when the end of the
489/// fragment is reached.
490#[derive(Debug)]
491pub(crate) struct ReplayStopper;
492
493impl Stopper for ReplayStopper {
494    type Processor = ReplayProcessor;
495
496    fn should_stop(
497        &self,
498        processor: &ReplayProcessor,
499        _continuation_stack: &ContinuationStack,
500        continuation_after_stop: impl FnOnce() -> Option<Continuation>,
501    ) -> ControlFlow<BreakReason> {
502        if processor.system().clock() >= processor.maximum_clock {
503            ControlFlow::Break(BreakReason::Stopped(continuation_after_stop()))
504        } else {
505            ControlFlow::Continue(())
506        }
507    }
