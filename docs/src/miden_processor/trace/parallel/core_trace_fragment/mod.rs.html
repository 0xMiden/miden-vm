13/// The columns of the main trace fragment. These consist of the system, decoder, and stack columns.
14///
15/// A fragment is a collection of columns of length `fragment_size` or less. Only the last fragment
16/// is allowed to be shorter than `fragment_size`.
17#[derive(Debug)]
18pub struct CoreTraceFragment<'a> {
19    pub columns: [&'a mut [Felt]; CORE_TRACE_WIDTH],
20}

22// BASIC BLOCK CONTEXT
23// ================================================================================================
24
25/// Keeps track of the info needed to decode a currently executing BASIC BLOCK. The info includes:
26/// - Operations which still need to be executed in the current group. The operations are encoded as
27///   opcodes (7 bits) appended one after another into a single field element, with the next
28///   operation to be executed located at the least significant position.
29/// - Number of operation groups left to be executed in the entire BASIC BLOCK.
30#[derive(Debug, Default)]
31pub struct BasicBlockContext {
32    pub current_op_group: Felt,
33    pub group_count_in_block: Felt,
34}
35
36impl BasicBlockContext {
37    /// Initializes a `BasicBlockContext` for the case where execution starts at the beginning of an
38    /// operation batch (i.e. at a SPAN or RESPAN row).
39    pub(crate) fn new_at_batch_start(
40        basic_block_node: &BasicBlockNode,
41        batch_index: usize,
42    ) -> Result<Self, OperationError> {
43        let current_batch = basic_block_node
44            .op_batches()
45            .get(batch_index)
46            .ok_or(OperationError::Internal("batch index out of bounds"))?;
47
48        Ok(Self {
49            current_op_group: current_batch.groups()[0],
50            group_count_in_block: Felt::new(
51                basic_block_node
52                    .op_batches()
53                    .iter()
54                    .skip(batch_index)
55                    .map(|batch| batch.num_groups())
56                    .sum::<usize>() as u64,
57            ),
58        })
59    }
60
61    /// Given that a trace fragment can start executing from the middle of a basic block, we need to
62    /// initialize the `BasicBlockContext` correctly to reflect the state of the decoder at that
63    /// point. This function does that initialization.
64    ///
65    /// Recall that `BasicBlockContext` keeps track of the state needed to correctly fill in the
66    /// decoder columns associated with a SPAN of operations (i.e. a basic block). This function
67    /// takes in a basic block node, the index of the current operation batch within that block,
68    /// and the index of the current operation within that batch, and initializes the
69    /// `BasicBlockContext` accordingly. In other words, it figures out how many operations are
70    /// left in the current operation group, and how many operation groups are left in the basic
71    /// block, given that we are starting execution from the specified operation.
72    pub(crate) fn new_at_op(
73        basic_block_node: &BasicBlockNode,
74        batch_index: usize,
75        op_idx_in_batch: usize,
76    ) -> Result<Self, OperationError> {
77        let op_batches = basic_block_node.op_batches();
78        let current_batch = op_batches
79            .get(batch_index)
80            .ok_or(OperationError::Internal("batch index out of bounds"))?;
81        let (current_op_group_idx, op_idx_in_group) = current_batch
82            .op_idx_in_batch_to_group(op_idx_in_batch)
83            .ok_or(OperationError::Internal("invalid op index in batch"))?;
84
85        let current_op_group = {
86            // Note: this here relies on NOOP's opcode to be 0, since `current_op_group_idx` could
87            // point to an op group that contains a NOOP inserted at runtime (i.e.
88            // padding at the end of the batch), and hence not encoded in the basic
89            // block directly. But since NOOP's opcode is 0, this works out correctly
90            // (since empty groups are also represented by 0).
91            let current_op_group = *current_batch
92                .groups()
93                .get(current_op_group_idx)
94                .ok_or(OperationError::Internal("op group index out of bounds"))?;
95
96            // Shift out all operations that are already executed in this group.
97            //
98            // Note: `group_ops_left` encodes the bits of the operations left to be executed after
99            // the current one, and so we would expect to shift `NUM_OP_BITS` by
100            // `op_idx_in_group + 1`. However, we will apply that shift right before
101            // writing to the trace, so we only shift by `op_idx_in_group` here.
102            Felt::new(current_op_group.as_canonical_u64() >> (NUM_OP_BITS * op_idx_in_group))
103        };
104
105        let group_count_in_block = {
106            let total_groups = basic_block_node.num_op_groups();
107
108            // Count groups consumed by completed batches (all batches before current one).
109            let mut groups_consumed = 0;
110            for op_batch in op_batches.iter().take(batch_index) {
111                groups_consumed += op_batch.num_groups().next_power_of_two();
112            }
113
114            // We run through previous operations of our current op group, and increment the number
115            // of groups consumed for each operation that has an immediate value
116            {
117                // Note: This is a hacky way of doing this because `OpBatch` doesn't store the
118                // information of which operation belongs to which group.
119                let mut current_op_group = current_batch
120                    .groups()
121                    .get(current_op_group_idx)
122                    .ok_or(OperationError::Internal("op group index out of bounds"))?
123                    .as_canonical_u64();
124                for _ in 0..op_idx_in_group {
125                    let current_op = (current_op_group & 0b1111111) as u8;
126                    if current_op == opcodes::PUSH {
127                        groups_consumed += 1;
128                    }
129
130                    current_op_group >>= NUM_OP_BITS; // Shift to the next operation in the group
131                }
132            }
133
134            // Add the number of complete groups before the current group in this batch. Add 1 to
135            // account for the current group (since `num_groups_left` is the number of groups left
136            // *after* being done with the current group)
137            groups_consumed += current_op_group_idx + 1;
138
139            let groups_left = total_groups
140                .checked_sub(groups_consumed)
141                .ok_or(OperationError::Internal("groups consumed exceeds total groups"))?;
142            Felt::from_u32(groups_left as u32)
143        };
144
145        Ok(Self { current_op_group, group_count_in_block })
146    }
147
148    /// Removes the operation that was just executed from the current operation group.
149    pub(crate) fn remove_operation_from_current_op_group(&mut self) {
150        let prev_op_group = self.current_op_group.as_canonical_u64();
151        self.current_op_group = Felt::new(prev_op_group >> NUM_OP_BITS);
152
153        debug_assert!(
154            prev_op_group >= self.current_op_group.as_canonical_u64(),
155            "op group underflow"
156        );
157    }
158}
