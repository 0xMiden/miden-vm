72/// Builds the main trace from the provided trace states in parallel.
73///
74/// # Example
75/// ```
76/// use miden_assembly::Assembler;
77/// use miden_processor::{DefaultHost, FastProcessor, StackInputs};
78///
79/// let program = Assembler::default().assemble_program("begin push.1 drop end").unwrap();
80/// let mut host = DefaultHost::default();
81///
82/// let trace_inputs = FastProcessor::new(StackInputs::default())
83///     .execute_trace_inputs_sync(&program, &mut host)
84///     .unwrap();
85/// let trace = miden_processor::trace::build_trace(trace_inputs).unwrap();
86///
87/// assert_eq!(*trace.program_hash(), program.hash());
88/// ```
89#[instrument(name = "build_trace", skip_all)]
90pub fn build_trace(inputs: TraceBuildInputs) -> Result<ExecutionTrace, ExecutionError> {
91    build_trace_with_max_len(inputs, MAX_TRACE_LEN)
92}

94/// Same as [`build_trace`], but with a custom hard cap.
95///
96/// When the trace would go over `max_trace_len`, this returns
97/// [`ExecutionError::TraceLenExceeded`].
98pub fn build_trace_with_max_len(
99    inputs: TraceBuildInputs,
100    max_trace_len: usize,
101) -> Result<ExecutionTrace, ExecutionError> {
102    let TraceBuildInputs {
103        execution_output,
104        trace_generation_context,
105        program_info,
106        execution_binding,
107    } = inputs;
108
109    if execution_binding.program_info() != &program_info {
110        return Err(ExecutionError::Internal("trace inputs do not match program info"));
111    }
112
113    if execution_binding.stack_outputs() != &execution_output.stack
114        || execution_binding.final_pc_transcript_state()
115            != execution_output.final_pc_transcript.state()
116        || execution_binding.advice_provider_fingerprint() != &execution_output.advice.fingerprint()
117    {
118        return Err(ExecutionError::Internal("trace inputs do not match execution output"));
119    }
120
121    let TraceGenerationContext {
122        core_trace_contexts,
123        range_checker_replay,
124        memory_writes,
125        bitwise_replay: bitwise,
126        kernel_replay,
127        hasher_for_chiplet,
128        ace_replay,
129        fragment_size,
130    } = trace_generation_context;
131
132    // Before any trace generation, check that the number of core trace rows doesn't exceed the
133    // maximum trace length. This is a necessary check to avoid OOM panics during trace generation,
134    // which can occur if the execution produces an extremely large number of steps.
135    //
136    // Note that we add 1 to the total core trace rows to account for the additional HALT opcode row
137    // that is pushed at the end of the last fragment.
138    let total_core_trace_rows = core_trace_contexts
139        .len()
140        .checked_mul(fragment_size)
141        .and_then(|n| n.checked_add(1))
142        .ok_or(ExecutionError::TraceLenExceeded(max_trace_len))?;
143    if total_core_trace_rows > max_trace_len {
144        return Err(ExecutionError::TraceLenExceeded(max_trace_len));
145    }
146
147    if core_trace_contexts.is_empty() {
148        return Err(ExecutionError::Internal(
149            "no trace fragments provided in the trace generation context",
150        ));
151    }
152
153    let chiplets = initialize_chiplets(
154        program_info.kernel().clone(),
155        &core_trace_contexts,
156        memory_writes,
157        bitwise,
158        kernel_replay,
159        hasher_for_chiplet,
160        ace_replay,
161        max_trace_len,
162    )?;
163
164    let range_checker = initialize_range_checker(range_checker_replay, &chiplets);
165
166    let mut core_trace_columns = generate_core_trace_columns(
167        core_trace_contexts,
168        program_info.kernel().clone(),
169        fragment_size,
170    )?;
171
172    // Calculate trace length
173    let core_trace_len = core_trace_columns[0].len();
174
175    // Get the number of rows for the range checker
176    let range_table_len = range_checker.get_number_range_checker_rows();
177
178    let trace_len_summary =
179        TraceLenSummary::new(core_trace_len, range_table_len, ChipletsLengths::new(&chiplets));
180
181    // Compute the final main trace length
182    let main_trace_len =
183        compute_main_trace_length(core_trace_len, range_table_len, chiplets.trace_len());
184
185    let ((), (range_checker_trace, chiplets_trace)) = rayon::join(
186        || pad_trace_columns(&mut core_trace_columns, main_trace_len),
187        || {
188            rayon::join(
189                || range_checker.into_trace_with_table(range_table_len, main_trace_len),
190                || chiplets.into_trace(main_trace_len),
191            )
192        },
193    );
194
195    // Padding to make the number of columns a multiple of 8 i.e., the Poseidon2 permutation rate
196    let padding_columns = vec![vec![ZERO; main_trace_len]; PADDED_TRACE_WIDTH - TRACE_WIDTH];
197
198    // Chain all trace columns together
199    let trace_columns: Vec<Vec<Felt>> = core_trace_columns
200        .into_iter()
201        .chain(range_checker_trace.trace)
202        .chain(chiplets_trace.trace)
203        .chain(padding_columns)
204        .collect();
205
206    // Create the MainTrace
207    let main_trace = {
208        let last_program_row = RowIndex::from((core_trace_len as u32).saturating_sub(1));
209        let col_matrix = ColMatrix::new(trace_columns);
210        MainTrace::new(col_matrix, last_program_row)
211    };
212
213    // Create aux trace builders
214    let aux_trace_builders = AuxTraceBuilders {
215        decoder: DecoderAuxTraceBuilder::default(),
216        range: range_checker_trace.aux_builder,
217        chiplets: chiplets_trace.aux_builder,
218        stack: StackAuxTraceBuilder,
219    };
220
221    Ok(ExecutionTrace::new_from_parts(
222        program_info,
223        execution_output,
224        main_trace,
225        aux_trace_builders,
226        trace_len_summary,
227    ))
228}

233fn compute_main_trace_length(
234    core_trace_len: usize,
235    range_table_len: usize,
236    chiplets_trace_len: usize,
237) -> usize {
238    // Get the trace length required to hold all execution trace steps
239    let max_len = range_table_len.max(core_trace_len).max(chiplets_trace_len);
240
241    // Pad the trace length to the next power of two
242    let trace_len = max_len.next_power_of_two();
243    core::cmp::max(trace_len, MIN_TRACE_LEN)
244}

246/// Generates core trace fragments in parallel from the provided trace fragment contexts.
247fn generate_core_trace_columns(
248    core_trace_contexts: Vec<CoreTraceFragmentContext>,
249    kernel: Kernel,
250    fragment_size: usize,
251) -> Result<Vec<Vec<Felt>>, ExecutionError> {
252    let mut core_trace_columns: Vec<Vec<Felt>> =
253        vec![vec![ZERO; core_trace_contexts.len() * fragment_size]; CORE_TRACE_WIDTH];
254
255    // Save the first stack top for initialization
256    let first_stack_top = if let Some(first_context) = core_trace_contexts.first() {
257        first_context.state.stack.stack_top.to_vec()
258    } else {
259        vec![ZERO; MIN_STACK_DEPTH]
260    };
261
262    let mut fragments = create_fragments_from_trace_columns(&mut core_trace_columns, fragment_size);
263
264    // Build the core trace fragments in parallel
265    let fragment_results: Result<Vec<_>, ExecutionError> = core_trace_contexts
266        .into_par_iter()
267        .zip(fragments.par_iter_mut())
268        .map(|(trace_state, fragment)| {
269            let (mut processor, mut tracer, mut continuation_stack, mut current_forest) =
270                split_trace_fragment_context(trace_state, fragment, fragment_size);
271
272            processor.execute(
273                &mut continuation_stack,
274                &mut current_forest,
275                &kernel,
276                &mut tracer,
277            )?;
278
279            tracer.into_final_state()
280        })
281        .collect();
282    let fragment_results = fragment_results?;
283
284    // Separate fragments, stack_rows, and system_rows
285    let mut stack_rows = Vec::new();
286    let mut system_rows = Vec::new();
287    let mut total_core_trace_rows = 0;
288
289    for final_state in fragment_results {
290        stack_rows.push(final_state.last_stack_cols);
291        system_rows.push(final_state.last_system_cols);
292        total_core_trace_rows += final_state.num_rows_written;
293    }
294
295    // Fix up stack and system rows
296    fixup_stack_and_system_rows(
297        &mut core_trace_columns,
298        fragment_size,
299        &stack_rows,
300        &system_rows,
301        &first_stack_top,
302    );
303
304    // Run batch inversion on stack's H0 helper column, processing each fragment in parallel.
305    // This must be done after fixup_stack_and_system_rows since that function overwrites the first
306    // row of each fragment with non-inverted values.
307    {
308        let h0_column = &mut core_trace_columns[STACK_TRACE_OFFSET + H0_COL_IDX];
309        h0_column[..total_core_trace_rows]
310            .par_chunks_mut(fragment_size)
311            .for_each(batch_inversion_allow_zeros);
312    }
313
314    // Truncate the core trace columns to the actual number of rows written.
315    for col in core_trace_columns.iter_mut() {
316        col.truncate(total_core_trace_rows);
317    }
318
319    push_halt_opcode_row(
320        &mut core_trace_columns,
321        system_rows.last().ok_or(ExecutionError::Internal(
322            "no trace fragments provided in the trace generation context",
323        ))?,
324        stack_rows.last().ok_or(ExecutionError::Internal(
325            "no trace fragments provided in the trace generation context",
326        ))?,
327    );
328
329    Ok(core_trace_columns)
330}

332/// Splits the core trace columns into fragments of the specified size, returning a vector of
333/// `CoreTraceFragment`s that each borrow from the original columns.
334fn create_fragments_from_trace_columns(
335    core_trace_columns: &mut [Vec<Felt>],
336    fragment_size: usize,
337) -> Vec<CoreTraceFragment<'_>> {
338    let mut column_chunks: Vec<_> = core_trace_columns
339        .iter_mut()
340        .map(|col| col.chunks_exact_mut(fragment_size))
341        .collect();
342    let mut core_trace_fragments = Vec::new();
343
344    loop {
345        let fragment_cols: Vec<&mut [Felt]> =
346            column_chunks.iter_mut().filter_map(|col_chunk| col_chunk.next()).collect();
347        assert!(
348            fragment_cols.is_empty() || fragment_cols.len() == CORE_TRACE_WIDTH,
349            "column chunks don't all have the same size"
350        );
351
352        if fragment_cols.is_empty() {
353            return core_trace_fragments;
354        } else {
355            core_trace_fragments.push(CoreTraceFragment {
356                columns: fragment_cols.try_into().expect("fragment has CORE_TRACE_WIDTH columns"),
357            });
358        }
359    }
360}

362/// Initializing the first row of each fragment with the appropriate stack and system state.
363///
364/// This needs to be done as a separate pass after all fragments have been generated, because the
365/// system and stack rows write the state at clk `i` to the row at index `i+1`. Hence, the state of
366/// the last row of any given fragment cannot be written in parallel, since any given fragment
367/// filler doesn't have access to the next fragment's first row.
368fn fixup_stack_and_system_rows(
369    core_trace_columns: &mut [Vec<Felt>],
370    fragment_size: usize,
371    stack_rows: &[[Felt; STACK_TRACE_WIDTH]],
372    system_rows: &[[Felt; SYS_TRACE_WIDTH]],
373    first_stack_top: &[Felt],
374) {
375    const MIN_STACK_DEPTH_FELT: Felt = Felt::new(MIN_STACK_DEPTH as u64);
376
377    let system_state_first_row = [
378        ZERO, // clk starts at 0
379        ZERO, // ctx starts at 0 (root context)
380        ZERO, // fn_hash[0] starts as 0
381        ZERO, // fn_hash[1] starts as 0
382        ZERO, // fn_hash[2] starts as 0
383        ZERO, // fn_hash[3] starts as 0
384    ];
385
386    // Initialize the first fragment with first_stack_top + [16, 0, 0] and first_system_state
387    {
388        // Set system state
389        for (col_idx, &value) in system_state_first_row.iter().enumerate() {
390            core_trace_columns[col_idx][0] = value;
391        }
392
393        // Set stack top (16 elements)
394        // Note: we call `rev()` here because the stack order is reversed in the trace.
395        // trace: [top, ..., bottom] vs stack: [bottom, ..., top]
396        for (stack_col_idx, &value) in first_stack_top.iter().rev().enumerate() {
397            core_trace_columns[STACK_TRACE_OFFSET + STACK_TOP_OFFSET + stack_col_idx][0] = value;
398        }
399
400        // Set stack helpers: [16, 0, 0]
401        core_trace_columns[STACK_TRACE_OFFSET + B0_COL_IDX][0] = MIN_STACK_DEPTH_FELT;
402        core_trace_columns[STACK_TRACE_OFFSET + B1_COL_IDX][0] = ZERO;
403        core_trace_columns[STACK_TRACE_OFFSET + H0_COL_IDX][0] = ZERO;
404    }
405
406    // Determine the starting row indices for each fragment after the first.
407    // We skip the first due to it already being initialized above.
408    let fragment_start_row_indices = {
409        let num_fragments = core_trace_columns[0].len() / fragment_size;
410
411        (0..).step_by(fragment_size).take(num_fragments).skip(1)
412    };
413
414    // Initialize subsequent fragments with their corresponding rows from the previous fragment
415    for (row_idx, (system_row, stack_row)) in
416        fragment_start_row_indices.zip(system_rows.iter().zip(stack_rows.iter()))
417    {
418        // Copy the system_row to the first row of this fragment
419        for (col_idx, &value) in system_row.iter().enumerate() {
420            core_trace_columns[col_idx][row_idx] = value;
421        }
422
423        // Copy the stack_row to the first row of this fragment
424        for (col_idx, &value) in stack_row.iter().enumerate() {
425            core_trace_columns[STACK_TRACE_OFFSET + col_idx][row_idx] = value;
426        }
427    }
428}

430/// Appends a row with the HALT opcode to the end of the last fragment.
431///
432/// This ensures that the trace ends with at least one HALT operation, which is necessary to satisfy
433/// the constraints.
434fn push_halt_opcode_row(
435    core_trace_columns: &mut [Vec<Felt>],
436    last_system_state: &[Felt; SYS_TRACE_WIDTH],
437    last_stack_state: &[Felt; STACK_TRACE_WIDTH],
438) {
439    // system columns
440    // ---------------------------------------------------------------------------------------
441    for (col_idx, &value) in last_system_state.iter().enumerate() {
442        core_trace_columns[col_idx].push(value);
443    }
444
445    // stack columns
446    // ---------------------------------------------------------------------------------------
447    for (col_idx, &value) in last_stack_state.iter().enumerate() {
448        core_trace_columns[STACK_TRACE_OFFSET + col_idx].push(value);
449    }
450
451    // decoder columns: padding with final decoder state
452    // ---------------------------------------------------------------------------------------
453    // Pad addr trace (decoder block address column) with ZEROs
454    core_trace_columns[DECODER_TRACE_OFFSET + ADDR_COL_IDX].push(ZERO);
455
456    // Pad op_bits columns with HALT opcode bits
457    let halt_opcode = opcodes::HALT;
458    for bit_idx in 0..NUM_OP_BITS {
459        let bit_value = Felt::from_u8((halt_opcode >> bit_idx) & 1);
460        core_trace_columns[DECODER_TRACE_OFFSET + OP_BITS_OFFSET + bit_idx].push(bit_value);
461    }
462
463    // Pad hasher state columns (8 columns)
464    // - First 4 columns: copy the last value (to propagate program hash)
465    // - Remaining 4 columns: fill with ZEROs
466    for hasher_col_idx in 0..NUM_HASHER_COLUMNS {
467        let col_idx = DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + hasher_col_idx;
468        if hasher_col_idx < 4 {
469            // For first 4 hasher columns, copy the last value to propagate program hash
470            let last_row_idx = core_trace_columns[col_idx].len() - 1;
471            let last_hasher_value = core_trace_columns[col_idx][last_row_idx];
472            core_trace_columns[col_idx].push(last_hasher_value);
473        } else {
474            // For remaining 4 hasher columns, fill with ZEROs
475            core_trace_columns[col_idx].push(ZERO);
476        }
477    }
478
479    // Pad in_span column with ZEROs
480    core_trace_columns[DECODER_TRACE_OFFSET + IN_SPAN_COL_IDX].push(ZERO);
481
482    // Pad group_count column with ZEROs
483    core_trace_columns[DECODER_TRACE_OFFSET + GROUP_COUNT_COL_IDX].push(ZERO);
484
485    // Pad op_idx column with ZEROs
486    core_trace_columns[DECODER_TRACE_OFFSET + OP_INDEX_COL_IDX].push(ZERO);
487
488    // Pad op_batch_flags columns (3 columns) with ZEROs
489    for batch_flag_idx in 0..NUM_OP_BATCH_FLAGS {
490        let col_idx = DECODER_TRACE_OFFSET + OP_BATCH_FLAGS_OFFSET + batch_flag_idx;
491        core_trace_columns[col_idx].push(ZERO);
492    }
493
494    // Pad op_bit_extra columns (2 columns)
495    // - First column: fill with ZEROs (HALT doesn't use this)
496    // - Second column: fill with ONEs (product of two most significant HALT bits, both are 1)
497    core_trace_columns[DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET].push(ZERO);
498    core_trace_columns[DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET + 1].push(ONE);
499}

501/// Initializes the ranger checker from the recorded range checks during execution and returns it.
502///
503/// Note that the maximum number of rows that the range checker can produce is 2^16, which is less
504/// than the maximum trace length (2^29). Hence, we can safely generate the entire range checker
505/// trace and then pad it to the final trace length, without worrying about hitting memory limits.
506fn initialize_range_checker(
507    range_checker_replay: RangeCheckerReplay,
508    chiplets: &Chiplets,
509) -> RangeChecker {
510    let mut range_checker = RangeChecker::new();
511
512    // Add all u32 range checks recorded during execution
513    for (clk, values) in range_checker_replay.into_iter() {
514        range_checker.add_range_checks(clk, &values);
515    }
516
517    // Add all memory-related range checks
518    chiplets.append_range_checks(&mut range_checker);
519
520    range_checker
521}

523/// Replays recorded operations to populate chiplet traces. Results were already used during
524/// execution; this pass only needs the trace-recording side effects.
525fn initialize_chiplets(
526    kernel: Kernel,
527    core_trace_contexts: &[CoreTraceFragmentContext],
528    memory_writes: MemoryWritesReplay,
529    bitwise: BitwiseReplay,
530    kernel_replay: KernelReplay,
531    hasher_for_chiplet: HasherRequestReplay,
532    ace_replay: AceReplay,
533    max_trace_len: usize,
534) -> Result<Chiplets, ExecutionError> {
535    let check_chiplets_trace_len = |chiplets: &Chiplets| -> Result<(), ExecutionError> {
536        if chiplets.trace_len() > max_trace_len {
537            return Err(ExecutionError::TraceLenExceeded(max_trace_len));
538        }
539        Ok(())
540    };
541
542    let mut chiplets = Chiplets::new(kernel);
543
544    // populate hasher chiplet
545    for hasher_op in hasher_for_chiplet.into_iter() {
546        match hasher_op {
547            HasherOp::Permute(input_state) => {
548                let _ = chiplets.hasher.permute(input_state);
549                check_chiplets_trace_len(&chiplets)?;
550            },
551            HasherOp::HashControlBlock((h1, h2, domain, expected_hash)) => {
552                let _ = chiplets.hasher.hash_control_block(h1, h2, domain, expected_hash);
553                check_chiplets_trace_len(&chiplets)?;
554            },
555            HasherOp::HashBasicBlock((forest, node_id, expected_hash)) => {
556                let node = forest
557                    .get_node_by_id(node_id)
558                    .ok_or(ExecutionError::Internal("invalid node ID in hasher replay"))?;
559                let MastNode::Block(basic_block_node) = node else {
560                    return Err(ExecutionError::Internal(
561                        "expected basic block node in hasher replay",
562                    ));
563                };
564                let op_batches = basic_block_node.op_batches();
565                let _ = chiplets.hasher.hash_basic_block(op_batches, expected_hash);
566                check_chiplets_trace_len(&chiplets)?;
567            },
568            HasherOp::BuildMerkleRoot((value, path, index)) => {
569                let _ = chiplets.hasher.build_merkle_root(value, &path, index);
570                check_chiplets_trace_len(&chiplets)?;
571            },
572            HasherOp::UpdateMerkleRoot((old_value, new_value, path, index)) => {
573                chiplets.hasher.update_merkle_root(old_value, new_value, &path, index);
574                check_chiplets_trace_len(&chiplets)?;
575            },
576        }
577    }
578
579    // populate bitwise chiplet
580    for (bitwise_op, a, b) in bitwise {
581        match bitwise_op {
582            BitwiseOp::U32And => {
583                chiplets.bitwise.u32and(a, b).map_exec_err_no_ctx()?;
584                check_chiplets_trace_len(&chiplets)?;
585            },
586            BitwiseOp::U32Xor => {
587                chiplets.bitwise.u32xor(a, b).map_exec_err_no_ctx()?;
588                check_chiplets_trace_len(&chiplets)?;
589            },
590        }
591    }
592
593    // populate memory chiplet
594    //
595    // Note: care is taken to order all the accesses by clock cycle, since the memory chiplet
596    // currently assumes that all memory accesses are issued in the same order as they appear in
597    // the trace.
598    {
599        let elements_written: Box<dyn Iterator<Item = MemoryAccess>> =
600            Box::new(memory_writes.iter_elements_written().map(|(element, addr, ctx, clk)| {
601                MemoryAccess::WriteElement(*addr, *element, *ctx, *clk)
602            }));
603        let words_written: Box<dyn Iterator<Item = MemoryAccess>> = Box::new(
604            memory_writes
605                .iter_words_written()
606                .map(|(word, addr, ctx, clk)| MemoryAccess::WriteWord(*addr, *word, *ctx, *clk)),
607        );
608        let elements_read: Box<dyn Iterator<Item = MemoryAccess>> =
609            Box::new(core_trace_contexts.iter().flat_map(|ctx| {
610                ctx.replay
611                    .memory_reads
612                    .iter_read_elements()
613                    .map(|(_, addr, ctx, clk)| MemoryAccess::ReadElement(addr, ctx, clk))
614            }));
615        let words_read: Box<dyn Iterator<Item = MemoryAccess>> =
616            Box::new(core_trace_contexts.iter().flat_map(|ctx| {
617                ctx.replay
618                    .memory_reads
619                    .iter_read_words()
620                    .map(|(_, addr, ctx, clk)| MemoryAccess::ReadWord(addr, ctx, clk))
621            }));
622
623        [elements_written, words_written, elements_read, words_read]
624            .into_iter()
625            .kmerge_by(|a, b| a.clk() < b.clk())
626            .try_for_each(|mem_access| {
627                match mem_access {
628                    MemoryAccess::ReadElement(addr, ctx, clk) => chiplets
629                        .memory
630                        .read(ctx, addr, clk)
631                        .map(|_| ())
632                        .map_err(ExecutionError::MemoryErrorNoCtx)?,
633                    MemoryAccess::WriteElement(addr, element, ctx, clk) => chiplets
634                        .memory
635                        .write(ctx, addr, clk, element)
636                        .map_err(ExecutionError::MemoryErrorNoCtx)?,
637                    MemoryAccess::ReadWord(addr, ctx, clk) => chiplets
638                        .memory
639                        .read_word(ctx, addr, clk)
640                        .map(|_| ())
641                        .map_err(ExecutionError::MemoryErrorNoCtx)?,
642                    MemoryAccess::WriteWord(addr, word, ctx, clk) => chiplets
643                        .memory
644                        .write_word(ctx, addr, clk, word)
645                        .map_err(ExecutionError::MemoryErrorNoCtx)?,
646                }
647                check_chiplets_trace_len(&chiplets)
648            })?;
649
650        enum MemoryAccess {
651            ReadElement(Felt, ContextId, RowIndex),
652            WriteElement(Felt, Felt, ContextId, RowIndex),
653            ReadWord(Felt, ContextId, RowIndex),
654            WriteWord(Felt, Word, ContextId, RowIndex),
655        }
656
657        impl MemoryAccess {
658            fn clk(&self) -> RowIndex {
659                match self {
660                    MemoryAccess::ReadElement(_, _, clk) => *clk,
661                    MemoryAccess::WriteElement(_, _, _, clk) => *clk,
662                    MemoryAccess::ReadWord(_, _, clk) => *clk,
663                    MemoryAccess::WriteWord(_, _, _, clk) => *clk,
664                }
665            }
666        }
667    }
668
669    // populate ACE chiplet
670    for (clk, circuit_eval) in ace_replay.into_iter() {
671        chiplets.ace.add_circuit_evaluation(clk, circuit_eval);
672        check_chiplets_trace_len(&chiplets)?;
673    }
674
675    // populate kernel ROM
676    for proc_hash in kernel_replay.into_iter() {
677        chiplets.kernel_rom.access_proc(proc_hash).map_exec_err_no_ctx()?;
678        check_chiplets_trace_len(&chiplets)?;
679    }
680
681    Ok(chiplets)
682}

684fn pad_trace_columns(trace_columns: &mut [Vec<Felt>], main_trace_len: usize) {
685    let total_program_rows = trace_columns[0].len();
686    assert!(total_program_rows <= main_trace_len);
687
688    let num_padding_rows = main_trace_len - total_program_rows;
689
690    // System columns
691    // ------------------------
692
693    // Pad CLK trace - fill with index values
694    for padding_row_idx in 0..num_padding_rows {
695        trace_columns[CLK_COL_IDX]
696            .push(Felt::from_u32((total_program_rows + padding_row_idx) as u32));
697    }
698
699    // Pad CTX trace - fill with ZEROs (root context)
700    trace_columns[CTX_COL_IDX].resize(main_trace_len, ZERO);
701
702    // Pad FN_HASH traces (4 columns) - fill with ZEROs as program execution must always end in the
703    // root context.
704    for fn_hash_col_idx in FN_HASH_RANGE {
705        trace_columns[fn_hash_col_idx].resize(main_trace_len, ZERO);
706    }
707
708    // Decoder columns
709    // ------------------------
710
711    // Pad addr trace (decoder block address column) with ZEROs
712    trace_columns[DECODER_TRACE_OFFSET + ADDR_COL_IDX].resize(main_trace_len, ZERO);
713
714    // Pad op_bits columns with HALT opcode bits
715    let halt_opcode = opcodes::HALT;
716    for i in 0..NUM_OP_BITS {
717        let bit_value = Felt::from_u8((halt_opcode >> i) & 1);
718        trace_columns[DECODER_TRACE_OFFSET + OP_BITS_OFFSET + i].resize(main_trace_len, bit_value);
719    }
720
721    // Pad hasher state columns (8 columns)
722    // - First 4 columns: copy the last value (to propagate program hash)
723    // - Remaining 4 columns: fill with ZEROs
724    for i in 0..NUM_HASHER_COLUMNS {
725        let col_idx = DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + i;
726        if i < 4 {
727            // For first 4 hasher columns, copy the last value to propagate program hash
728            // Safety: per our documented safety guarantees, we know that `total_program_rows > 0`,
729            // and row `total_program_rows - 1` is initialized.
730            let last_hasher_value = trace_columns[col_idx][total_program_rows - 1];
731            trace_columns[col_idx].resize(main_trace_len, last_hasher_value);
732        } else {
733            // For remaining 4 hasher columns, fill with ZEROs
734            trace_columns[col_idx].resize(main_trace_len, ZERO);
735        }
736    }
737
738    // Pad in_span column with ZEROs
739    trace_columns[DECODER_TRACE_OFFSET + IN_SPAN_COL_IDX].resize(main_trace_len, ZERO);
740
741    // Pad group_count column with ZEROs
742    trace_columns[DECODER_TRACE_OFFSET + GROUP_COUNT_COL_IDX].resize(main_trace_len, ZERO);
743
744    // Pad op_idx column with ZEROs
745    trace_columns[DECODER_TRACE_OFFSET + OP_INDEX_COL_IDX].resize(main_trace_len, ZERO);
746
747    // Pad op_batch_flags columns (3 columns) with ZEROs
748    for i in 0..NUM_OP_BATCH_FLAGS {
749        trace_columns[DECODER_TRACE_OFFSET + OP_BATCH_FLAGS_OFFSET + i]
750            .resize(main_trace_len, ZERO);
751    }
752
753    // Pad op_bit_extra columns (2 columns)
754    // - First column: fill with ZEROs (HALT doesn't use this)
755    // - Second column: fill with ONEs (product of two most significant HALT bits, both are 1)
756    trace_columns[DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET].resize(main_trace_len, ZERO);
757    trace_columns[DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET + 1].resize(main_trace_len, ONE);
758
759    // Stack columns
760    // ------------------------
761
762    // Pad stack columns with the last value in each column (analogous to Stack::into_trace())
763    for i in 0..STACK_TRACE_WIDTH {
764        let col_idx = STACK_TRACE_OFFSET + i;
765        // Safety: per our documented safety guarantees, we know that `total_program_rows > 0`,
766        // and row `total_program_rows - 1` is initialized.
767        let last_stack_value = trace_columns[col_idx][total_program_rows - 1];
768        trace_columns[col_idx].resize(main_trace_len, last_stack_value);
769    }
770}

772/// Uses the provided `CoreTraceFragmentContext` to build and return a `ReplayProcessor` and
773/// `CoreTraceGenerationTracer` that can be used to execute the fragment.
774fn split_trace_fragment_context<'a>(
775    fragment_context: CoreTraceFragmentContext,
776    fragment: &'a mut CoreTraceFragment<'a>,
777    fragment_size: usize,
778) -> (
779    ReplayProcessor,
780    CoreTraceGenerationTracer<'a>,
781    ContinuationStack,
782    Arc<MastForest>,
783) {
784    let CoreTraceFragmentContext {
785        state: CoreTraceState { system, decoder, stack },
786        replay:
787            ExecutionReplay {
788                block_stack: block_stack_replay,
789                execution_context: execution_context_replay,
790                stack_overflow: stack_overflow_replay,
791                memory_reads: memory_reads_replay,
792                advice: advice_replay,
793                hasher: hasher_response_replay,
794                block_address: block_address_replay,
795                mast_forest_resolution: mast_forest_resolution_replay,
796            },
797        continuation,
798        initial_mast_forest,
799    } = fragment_context;
800
801    let processor = ReplayProcessor::new(
802        system,
803        stack,
804        stack_overflow_replay,
805        execution_context_replay,
806        advice_replay,
807        memory_reads_replay,
808        hasher_response_replay,
809        mast_forest_resolution_replay,
810        fragment_size.into(),
811    );
812    let tracer =
813        CoreTraceGenerationTracer::new(fragment, decoder, block_address_replay, block_stack_replay);
814
815    (processor, tracer, continuation, initial_mast_forest)
816}
