69/// Builds the main trace from the provided trace states in parallel.
70///
71/// # Example
72/// ```
73/// use miden_assembly::Assembler;
74/// use miden_processor::{DefaultHost, FastProcessor, StackInputs};
75///
76/// let program = Assembler::default().assemble_program("begin push.1 drop end").unwrap();
77/// let mut host = DefaultHost::default();
78///
79/// let trace_inputs = FastProcessor::new(StackInputs::default())
80///     .execute_trace_inputs_sync(&program, &mut host)
81///     .unwrap();
82/// let trace = miden_processor::trace::build_trace(trace_inputs).unwrap();
83///
84/// assert_eq!(*trace.program_hash(), program.hash());
85/// ```
86#[instrument(name = "build_trace", skip_all)]
87pub fn build_trace(inputs: TraceBuildInputs) -> Result<ExecutionTrace, ExecutionError> {
88    build_trace_with_max_len(inputs, MAX_TRACE_LEN)
89}

91/// Same as [`build_trace`], but with a custom hard cap.
92///
93/// When the trace would go over `max_trace_len`, this returns
94/// [`ExecutionError::TraceLenExceeded`].
95pub fn build_trace_with_max_len(
96    inputs: TraceBuildInputs,
97    max_trace_len: usize,
98) -> Result<ExecutionTrace, ExecutionError> {
99    let TraceBuildInputs {
100        execution_output,
101        trace_generation_context,
102        program_info,
103        execution_binding,
104    } = inputs;
105
106    if execution_binding.program_info() != &program_info {
107        return Err(ExecutionError::Internal("trace inputs do not match program info"));
108    }
109
110    if execution_binding.stack_outputs() != &execution_output.stack
111        || execution_binding.final_pc_transcript_state()
112            != execution_output.final_pc_transcript.state()
113        || execution_binding.advice_provider_fingerprint() != &execution_output.advice.fingerprint()
114    {
115        return Err(ExecutionError::Internal("trace inputs do not match execution output"));
116    }
117
118    let TraceGenerationContext {
119        core_trace_contexts,
120        range_checker_replay,
121        memory_writes,
122        bitwise_replay: bitwise,
123        kernel_replay,
124        hasher_for_chiplet,
125        ace_replay,
126        fragment_size,
127    } = trace_generation_context;
128
129    // Before any trace generation, check that the number of core trace rows doesn't exceed the
130    // maximum trace length. This is a necessary check to avoid OOM panics during trace generation,
131    // which can occur if the execution produces an extremely large number of steps.
132    //
133    // Note that we add 1 to the total core trace rows to account for the additional HALT opcode row
134    // that is pushed at the end of the last fragment.
135    let total_core_trace_rows = core_trace_contexts
136        .len()
137        .checked_mul(fragment_size)
138        .and_then(|n| n.checked_add(1))
139        .ok_or(ExecutionError::TraceLenExceeded(max_trace_len))?;
140    if total_core_trace_rows > max_trace_len {
141        return Err(ExecutionError::TraceLenExceeded(max_trace_len));
142    }
143
144    if core_trace_contexts.is_empty() {
145        return Err(ExecutionError::Internal(
146            "no trace fragments provided in the trace generation context",
147        ));
148    }
149
150    let chiplets = initialize_chiplets(
151        program_info.kernel().clone(),
152        &core_trace_contexts,
153        memory_writes,
154        bitwise,
155        kernel_replay,
156        hasher_for_chiplet,
157        ace_replay,
158        max_trace_len,
159    )?;
160
161    let range_checker = initialize_range_checker(range_checker_replay, &chiplets);
162
163    let mut core_trace_data = generate_core_trace_row_major(
164        core_trace_contexts,
165        program_info.kernel().clone(),
166        fragment_size,
167    )?;
168
169    let core_trace_len = core_trace_data.len() / CORE_TRACE_WIDTH;
170
171    // Get the number of rows for the range checker
172    let range_table_len = range_checker.get_number_range_checker_rows();
173
174    let trace_len_summary =
175        TraceLenSummary::new(core_trace_len, range_table_len, ChipletsLengths::new(&chiplets));
176
177    // Compute the final main trace length
178    let main_trace_len =
179        compute_main_trace_length(core_trace_len, range_table_len, chiplets.trace_len());
180
181    let ((range_checker_trace, chiplets_trace), ()) = rayon::join(
182        || {
183            rayon::join(
184                || range_checker.into_trace_with_table(range_table_len, main_trace_len),
185                || chiplets.into_trace(main_trace_len),
186            )
187        },
188        || pad_core_row_major(&mut core_trace_data, main_trace_len),
189    );
190
191    // Create the MainTrace
192    let main_trace = {
193        let last_program_row = RowIndex::from((core_trace_len as u32).saturating_sub(1));
194        MainTrace::from_parts(
195            core_trace_data,
196            chiplets_trace.trace,
197            range_checker_trace.trace,
198            main_trace_len,
199            last_program_row,
200        )
201    };
202
203    // Create aux trace builders
204    let aux_trace_builders = AuxTraceBuilders {
205        decoder: DecoderAuxTraceBuilder::default(),
206        range: range_checker_trace.aux_builder,
207        chiplets: chiplets_trace.aux_builder,
208        stack: StackAuxTraceBuilder,
209    };
210
211    Ok(ExecutionTrace::new_from_parts(
212        program_info,
213        execution_output,
214        main_trace,
215        aux_trace_builders,
216        trace_len_summary,
217    ))
218}

223fn compute_main_trace_length(
224    core_trace_len: usize,
225    range_table_len: usize,
226    chiplets_trace_len: usize,
227) -> usize {
228    // Get the trace length required to hold all execution trace steps
229    let max_len = range_table_len.max(core_trace_len).max(chiplets_trace_len);
230
231    // Pad the trace length to the next power of two
232    let trace_len = max_len.next_power_of_two();
233    core::cmp::max(trace_len, MIN_TRACE_LEN)
234}

236/// Generates row-major core trace in parallel from the provided trace fragment contexts.
237fn generate_core_trace_row_major(
238    core_trace_contexts: Vec<CoreTraceFragmentContext>,
239    kernel: Kernel,
240    fragment_size: usize,
241) -> Result<Vec<Felt>, ExecutionError> {
242    let num_fragments = core_trace_contexts.len();
243    let total_allocated_rows = num_fragments * fragment_size;
244
245    let mut core_trace_data: Vec<Felt> = vec![ZERO; total_allocated_rows * CORE_TRACE_WIDTH];
246
247    // Save the first stack top for initialization
248    let first_stack_top = if let Some(first_context) = core_trace_contexts.first() {
249        first_context.state.stack.stack_top.to_vec()
250    } else {
251        vec![ZERO; MIN_STACK_DEPTH]
252    };
253
254    let writers: Vec<RowMajorTraceWriter<'_, Felt>> = core_trace_data
255        .chunks_exact_mut(fragment_size * CORE_TRACE_WIDTH)
256        .map(|chunk| RowMajorTraceWriter::new(chunk, CORE_TRACE_WIDTH))
257        .collect();
258
259    // Build the core trace fragments in parallel
260    let fragment_results: Result<Vec<_>, ExecutionError> = core_trace_contexts
261        .into_par_iter()
262        .zip(writers.into_par_iter())
263        .map(|(trace_state, writer)| {
264            let (mut processor, mut tracer, mut continuation_stack, mut current_forest) =
265                split_trace_fragment_context(trace_state, writer, fragment_size);
266
267            processor.execute(
268                &mut continuation_stack,
269                &mut current_forest,
270                &kernel,
271                &mut tracer,
272            )?;
273
274            tracer.into_final_state()
275        })
276        .collect();
277    let fragment_results = fragment_results?;
278
279    let mut stack_rows = Vec::new();
280    let mut system_rows = Vec::new();
281    let mut total_core_trace_rows = 0;
282
283    for final_state in fragment_results {
284        stack_rows.push(final_state.last_stack_cols);
285        system_rows.push(final_state.last_system_cols);
286        total_core_trace_rows += final_state.num_rows_written;
287    }
288
289    // Fix up stack and system rows
290    fixup_stack_and_system_rows(
291        &mut core_trace_data,
292        fragment_size,
293        &stack_rows,
294        &system_rows,
295        &first_stack_top,
296    );
297
298    // Run batch inversion on stack's H0 helper column, processing each fragment in parallel.
299    // This must be done after fixup_stack_and_system_rows since that function overwrites the first
300    // row of each fragment with non-inverted values.
301    {
302        let h0_col_offset = STACK_TRACE_OFFSET + H0_COL_IDX;
303        let w = CORE_TRACE_WIDTH;
304        core_trace_data[..total_core_trace_rows * w]
305            .par_chunks_mut(fragment_size * w)
306            .for_each(|fragment_chunk| {
307                let num_rows = fragment_chunk.len() / w;
308                let mut h0_vals: Vec<Felt> =
309                    (0..num_rows).map(|r| fragment_chunk[r * w + h0_col_offset]).collect();
310                batch_inversion_allow_zeros(&mut h0_vals);
311                for (r, &val) in h0_vals.iter().enumerate() {
312                    fragment_chunk[r * w + h0_col_offset] = val;
313                }
314            });
315    }
316
317    // Truncate the core trace columns to the actual number of rows written.
318    core_trace_data.truncate(total_core_trace_rows * CORE_TRACE_WIDTH);
319
320    push_halt_opcode_row(
321        &mut core_trace_data,
322        total_core_trace_rows,
323        system_rows.last().ok_or(ExecutionError::Internal(
324            "no trace fragments provided in the trace generation context",
325        ))?,
326        stack_rows.last().ok_or(ExecutionError::Internal(
327            "no trace fragments provided in the trace generation context",
328        ))?,
329    );
330
331    Ok(core_trace_data)
332}

334/// Initializing the first row of each fragment with the appropriate stack and system state.
335///
336/// This needs to be done as a separate pass after all fragments have been generated, because the
337/// system and stack rows write the state at clk `i` to the row at index `i+1`. Hence, the state of
338/// the last row of any given fragment cannot be written in parallel, since any given fragment
339/// filler doesn't have access to the next fragment's first row.
340fn fixup_stack_and_system_rows(
341    core_trace_data: &mut [Felt],
342    fragment_size: usize,
343    stack_rows: &[[Felt; STACK_TRACE_WIDTH]],
344    system_rows: &[[Felt; SYS_TRACE_WIDTH]],
345    first_stack_top: &[Felt],
346) {
347    const MIN_STACK_DEPTH_FELT: Felt = Felt::new(MIN_STACK_DEPTH as u64);
348    let w = CORE_TRACE_WIDTH;
349
350    {
351        let row = &mut core_trace_data[..w];
352
353        // Stack order in the trace is reversed vs `first_stack_top`.
354        for (stack_col_idx, &value) in first_stack_top.iter().rev().enumerate() {
355            row[STACK_TRACE_OFFSET + STACK_TOP_OFFSET + stack_col_idx] = value;
356        }
357
358        row[STACK_TRACE_OFFSET + B0_COL_IDX] = MIN_STACK_DEPTH_FELT;
359        row[STACK_TRACE_OFFSET + B1_COL_IDX] = ZERO;
360        row[STACK_TRACE_OFFSET + H0_COL_IDX] = ZERO;
361    }
362
363    let total_rows = core_trace_data.len() / w;
364    let num_fragments = total_rows / fragment_size;
365
366    for frag_idx in 1..num_fragments {
367        let row_idx = frag_idx * fragment_size;
368        let row_start = row_idx * w;
369        let system_row = &system_rows[frag_idx - 1];
370        let stack_row = &stack_rows[frag_idx - 1];
371
372        core_trace_data[row_start..row_start + SYS_TRACE_WIDTH].copy_from_slice(system_row);
373
374        let stack_start = row_start + STACK_TRACE_OFFSET;
375        core_trace_data[stack_start..stack_start + STACK_TRACE_WIDTH].copy_from_slice(stack_row);
376    }
377}

379/// Appends a HALT row (`num_rows_before` is the row count before append).
380///
381/// This ensures that the trace ends with at least one HALT operation, which is necessary to satisfy
382/// the constraints.
383fn push_halt_opcode_row(
384    core_trace_data: &mut Vec<Felt>,
385    num_rows_before: usize,
386    last_system_state: &[Felt; SYS_TRACE_WIDTH],
387    last_stack_state: &[Felt; STACK_TRACE_WIDTH],
388) {
389    let w = CORE_TRACE_WIDTH;
390    let mut row = [ZERO; CORE_TRACE_WIDTH];
391
392    // system columns
393    // ---------------------------------------------------------------------------------------
394    row[..SYS_TRACE_WIDTH].copy_from_slice(last_system_state);
395
396    // stack columns
397    // ---------------------------------------------------------------------------------------
398    row[STACK_TRACE_OFFSET..STACK_TRACE_OFFSET + STACK_TRACE_WIDTH]
399        .copy_from_slice(last_stack_state);
400
401    // Pad op_bits columns with HALT opcode bits
402    let halt_opcode = opcodes::HALT;
403    for bit_idx in 0..NUM_OP_BITS {
404        row[DECODER_TRACE_OFFSET + OP_BITS_OFFSET + bit_idx] =
405            Felt::from_u8((halt_opcode >> bit_idx) & 1);
406    }
407
408    // Pad hasher state columns (8 columns)
409    // - First 4 columns: copy the last value (to propagate program hash)
410    // - Remaining 4 columns: fill with ZEROs
411    if num_rows_before > 0 {
412        let last_row_start = (num_rows_before - 1) * w;
413        // For first 4 hasher columns, copy the last value to propagate program hash
414        for hasher_col_idx in 0..4 {
415            let col_idx = DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + hasher_col_idx;
416            row[col_idx] = core_trace_data[last_row_start + col_idx];
417        }
418    }
419
420    // Pad op_bit_extra columns (2 columns)
421    // - First column: do nothing (pre-filled with ZEROs, HALT doesn't use this)
422    // - Second column: fill with ONEs (product of two most significant HALT bits, both are 1)
423    row[DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET + 1] = ONE;
424
425    core_trace_data.extend_from_slice(&row);
426}

428/// Initializes the ranger checker from the recorded range checks during execution and returns it.
429///
430/// Note that the maximum number of rows that the range checker can produce is 2^16, which is less
431/// than the maximum trace length (2^29). Hence, we can safely generate the entire range checker
432/// trace and then pad it to the final trace length, without worrying about hitting memory limits.
433fn initialize_range_checker(
434    range_checker_replay: RangeCheckerReplay,
435    chiplets: &Chiplets,
436) -> RangeChecker {
437    let mut range_checker = RangeChecker::new();
438
439    // Add all u32 range checks recorded during execution
440    for (clk, values) in range_checker_replay.into_iter() {
441        range_checker.add_range_checks(clk, &values);
442    }
443
444    // Add all memory-related range checks
445    chiplets.append_range_checks(&mut range_checker);
446
447    range_checker
448}

450/// Replays recorded operations to populate chiplet traces. Results were already used during
451/// execution; this pass only needs the trace-recording side effects.
452fn initialize_chiplets(
453    kernel: Kernel,
454    core_trace_contexts: &[CoreTraceFragmentContext],
455    memory_writes: MemoryWritesReplay,
456    bitwise: BitwiseReplay,
457    kernel_replay: KernelReplay,
458    hasher_for_chiplet: HasherRequestReplay,
459    ace_replay: AceReplay,
460    max_trace_len: usize,
461) -> Result<Chiplets, ExecutionError> {
462    let check_chiplets_trace_len = |chiplets: &Chiplets| -> Result<(), ExecutionError> {
463        if chiplets.trace_len() > max_trace_len {
464            return Err(ExecutionError::TraceLenExceeded(max_trace_len));
465        }
466        Ok(())
467    };
468
469    let mut chiplets = Chiplets::new(kernel);
470
471    // populate hasher chiplet
472    for hasher_op in hasher_for_chiplet.into_iter() {
473        match hasher_op {
474            HasherOp::Permute(input_state) => {
475                let _ = chiplets.hasher.permute(input_state);
476                check_chiplets_trace_len(&chiplets)?;
477            },
478            HasherOp::HashControlBlock((h1, h2, domain, expected_hash)) => {
479                let _ = chiplets.hasher.hash_control_block(h1, h2, domain, expected_hash);
480                check_chiplets_trace_len(&chiplets)?;
481            },
482            HasherOp::HashBasicBlock((forest, node_id, expected_hash)) => {
483                let node = forest
484                    .get_node_by_id(node_id)
485                    .ok_or(ExecutionError::Internal("invalid node ID in hasher replay"))?;
486                let MastNode::Block(basic_block_node) = node else {
487                    return Err(ExecutionError::Internal(
488                        "expected basic block node in hasher replay",
489                    ));
490                };
491                let op_batches = basic_block_node.op_batches();
492                let _ = chiplets.hasher.hash_basic_block(op_batches, expected_hash);
493                check_chiplets_trace_len(&chiplets)?;
494            },
495            HasherOp::BuildMerkleRoot((value, path, index)) => {
496                let _ = chiplets.hasher.build_merkle_root(value, &path, index);
497                check_chiplets_trace_len(&chiplets)?;
498            },
499            HasherOp::UpdateMerkleRoot((old_value, new_value, path, index)) => {
500                chiplets.hasher.update_merkle_root(old_value, new_value, &path, index);
501                check_chiplets_trace_len(&chiplets)?;
502            },
503        }
504    }
505
506    // populate bitwise chiplet
507    for (bitwise_op, a, b) in bitwise {
508        match bitwise_op {
509            BitwiseOp::U32And => {
510                chiplets.bitwise.u32and(a, b).map_exec_err_no_ctx()?;
511                check_chiplets_trace_len(&chiplets)?;
512            },
513            BitwiseOp::U32Xor => {
514                chiplets.bitwise.u32xor(a, b).map_exec_err_no_ctx()?;
515                check_chiplets_trace_len(&chiplets)?;
516            },
517        }
518    }
519
520    // populate memory chiplet
521    //
522    // Note: care is taken to order all the accesses by clock cycle, since the memory chiplet
523    // currently assumes that all memory accesses are issued in the same order as they appear in
524    // the trace.
525    {
526        let elements_written: Box<dyn Iterator<Item = MemoryAccess>> =
527            Box::new(memory_writes.iter_elements_written().map(|(element, addr, ctx, clk)| {
528                MemoryAccess::WriteElement(*addr, *element, *ctx, *clk)
529            }));
530        let words_written: Box<dyn Iterator<Item = MemoryAccess>> = Box::new(
531            memory_writes
532                .iter_words_written()
533                .map(|(word, addr, ctx, clk)| MemoryAccess::WriteWord(*addr, *word, *ctx, *clk)),
534        );
535        let elements_read: Box<dyn Iterator<Item = MemoryAccess>> =
536            Box::new(core_trace_contexts.iter().flat_map(|ctx| {
537                ctx.replay
538                    .memory_reads
539                    .iter_read_elements()
540                    .map(|(_, addr, ctx, clk)| MemoryAccess::ReadElement(addr, ctx, clk))
541            }));
542        let words_read: Box<dyn Iterator<Item = MemoryAccess>> =
543            Box::new(core_trace_contexts.iter().flat_map(|ctx| {
544                ctx.replay
545                    .memory_reads
546                    .iter_read_words()
547                    .map(|(_, addr, ctx, clk)| MemoryAccess::ReadWord(addr, ctx, clk))
548            }));
549
550        [elements_written, words_written, elements_read, words_read]
551            .into_iter()
552            .kmerge_by(|a, b| a.clk() < b.clk())
553            .try_for_each(|mem_access| {
554                match mem_access {
555                    MemoryAccess::ReadElement(addr, ctx, clk) => chiplets
556                        .memory
557                        .read(ctx, addr, clk)
558                        .map(|_| ())
559                        .map_err(ExecutionError::MemoryErrorNoCtx)?,
560                    MemoryAccess::WriteElement(addr, element, ctx, clk) => chiplets
561                        .memory
562                        .write(ctx, addr, clk, element)
563                        .map_err(ExecutionError::MemoryErrorNoCtx)?,
564                    MemoryAccess::ReadWord(addr, ctx, clk) => chiplets
565                        .memory
566                        .read_word(ctx, addr, clk)
567                        .map(|_| ())
568                        .map_err(ExecutionError::MemoryErrorNoCtx)?,
569                    MemoryAccess::WriteWord(addr, word, ctx, clk) => chiplets
570                        .memory
571                        .write_word(ctx, addr, clk, word)
572                        .map_err(ExecutionError::MemoryErrorNoCtx)?,
573                }
574                check_chiplets_trace_len(&chiplets)
575            })?;
576
577        enum MemoryAccess {
578            ReadElement(Felt, ContextId, RowIndex),
579            WriteElement(Felt, Felt, ContextId, RowIndex),
580            ReadWord(Felt, ContextId, RowIndex),
581            WriteWord(Felt, Word, ContextId, RowIndex),
582        }
583
584        impl MemoryAccess {
585            fn clk(&self) -> RowIndex {
586                match self {
587                    MemoryAccess::ReadElement(_, _, clk) => *clk,
588                    MemoryAccess::WriteElement(_, _, _, clk) => *clk,
589                    MemoryAccess::ReadWord(_, _, clk) => *clk,
590                    MemoryAccess::WriteWord(_, _, _, clk) => *clk,
591                }
592            }
593        }
594    }
595
596    // populate ACE chiplet
597    for (clk, circuit_eval) in ace_replay.into_iter() {
598        chiplets.ace.add_circuit_evaluation(clk, circuit_eval);
599        check_chiplets_trace_len(&chiplets)?;
600    }
601
602    // populate kernel ROM
603    for proc_hash in kernel_replay.into_iter() {
604        chiplets.kernel_rom.access_proc(proc_hash).map_exec_err_no_ctx()?;
605        check_chiplets_trace_len(&chiplets)?;
606    }
607
608    Ok(chiplets)
609}

611/// Pads the core trace to `main_trace_len` rows (HALT template, CLK incremented per row).
612fn pad_core_row_major(core_trace_data: &mut Vec<Felt>, main_trace_len: usize) {
613    let w = CORE_TRACE_WIDTH;
614    let total_program_rows = core_trace_data.len() / w;
615    assert!(total_program_rows <= main_trace_len);
616    assert!(total_program_rows > 0);
617
618    let num_padding_rows = main_trace_len - total_program_rows;
619    if num_padding_rows == 0 {
620        return;
621    }
622    let last_row_start = (total_program_rows - 1) * w;
623
624    // Decoder columns
625    // ------------------------
626
627    let mut template = [ZERO; CORE_TRACE_WIDTH];
628    // Pad op_bits columns with HALT opcode bits
629    let halt_opcode = opcodes::HALT;
630    for i in 0..NUM_OP_BITS {
631        let bit_value = Felt::from_u8((halt_opcode >> i) & 1);
632        template[DECODER_TRACE_OFFSET + OP_BITS_OFFSET + i] = bit_value;
633    }
634    // Pad hasher state columns (8 columns)
635    // - First 4 columns: copy the last value (to propagate program hash)
636    // - Remaining 4 columns: fill with ZEROs
637    for i in 0..NUM_HASHER_COLUMNS {
638        let col_idx = DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + i;
639        template[col_idx] = if i < 4 {
640            // For first 4 hasher columns, copy the last value to propagate program hash
641            // Safety: per our documented safety guarantees, we know that `total_program_rows > 0`,
642            // and row `total_program_rows - 1` is initialized.
643            core_trace_data[last_row_start + col_idx]
644        } else {
645            ZERO
646        };
647    }
648
649    // Pad op_bit_extra columns (2 columns)
650    // - First column: do nothing (filled with ZEROs, HALT doesn't use this)
651    // - Second column: fill with ONEs (product of two most significant HALT bits, both are 1)
652    template[DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET + 1] = ONE;
653
654    // Stack columns
655    // ------------------------
656
657    // Pad stack columns with the last value in each column (analogous to Stack::into_trace())
658    for i in 0..STACK_TRACE_WIDTH {
659        let col_idx = STACK_TRACE_OFFSET + i;
660        // Safety: per our documented safety guarantees, we know that `total_program_rows > 0`,
661        // and row `total_program_rows - 1` is initialized.
662        template[col_idx] = core_trace_data[last_row_start + col_idx];
663    }
664
665    // System columns
666    // ------------------------
667
668    // Pad CLK trace - fill with index values
669
670    core_trace_data.reserve(num_padding_rows * w);
671    for idx in 0..num_padding_rows {
672        template[CLK_COL_IDX] = Felt::from_u32((total_program_rows + idx) as u32);
673        core_trace_data.extend_from_slice(&template);
674    }
675}

677/// Uses the provided `CoreTraceFragmentContext` to build and return a `ReplayProcessor` and
678/// `CoreTraceGenerationTracer` that can be used to execute the fragment.
679fn split_trace_fragment_context<'a>(
680    fragment_context: CoreTraceFragmentContext,
681    writer: RowMajorTraceWriter<'a, Felt>,
682    fragment_size: usize,
683) -> (
684    ReplayProcessor,
685    CoreTraceGenerationTracer<'a>,
686    ContinuationStack,
687    Arc<MastForest>,
688) {
689    let CoreTraceFragmentContext {
690        state: CoreTraceState { system, decoder, stack },
691        replay:
692            ExecutionReplay {
693                block_stack: block_stack_replay,
694                execution_context: execution_context_replay,
695                stack_overflow: stack_overflow_replay,
696                memory_reads: memory_reads_replay,
697                advice: advice_replay,
698                hasher: hasher_response_replay,
699                block_address: block_address_replay,
700                mast_forest_resolution: mast_forest_resolution_replay,
701            },
702        continuation,
703        initial_mast_forest,
704    } = fragment_context;
705
706    let processor = ReplayProcessor::new(
707        system,
708        stack,
709        stack_overflow_replay,
710        execution_context_replay,
711        advice_replay,
712        memory_reads_replay,
713        hasher_response_replay,
714        mast_forest_resolution_replay,
715        fragment_size.into(),
716    );
717    let tracer =
718        CoreTraceGenerationTracer::new(writer, decoder, block_address_replay, block_stack_replay);
719
720    (processor, tracer, continuation, initial_mast_forest)
721}
