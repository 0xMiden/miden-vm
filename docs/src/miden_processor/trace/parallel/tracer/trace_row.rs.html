33/// The data necessary to build the decoder part of a trace row.
34#[derive(Debug)]
35struct DecoderRow {
36    /// The address field to write into trace
37    pub addr: Felt,
38    /// The operation code for start operations
39    pub opcode: u8,
40    /// The two child hashes for start operations (first hash, second hash)
41    pub hasher_state: (Word, Word),
42    /// Whether this row is an operation within a basic block
43    pub in_basic_block: bool,
44    /// The group count for this operation
45    pub group_count: Felt,
46    /// The index of the operation within its operation group, or 0 if this is not a row containing
47    /// an operation in a basic block.
48    pub op_index: Felt,
49    /// The operation batch flags, encoding the number of groups present in the current operation
50    /// batch.
51    pub op_batch_flags: [Felt; NUM_OP_BATCH_FLAGS],
52}
53
54impl DecoderRow {
55    /// Creates a new `DecoderRow` for control flow operations (JOIN/SPLIT start or end).
56    ///
57    /// Control flow operations do not occur within basic blocks, so the relevant fields are set
58    /// to their default values.
59    pub fn new_control_flow(opcode: u8, hasher_state: (Word, Word), addr: Felt) -> Self {
60        Self {
61            opcode,
62            hasher_state,
63            addr,
64            in_basic_block: false,
65            group_count: ZERO,
66            op_index: ZERO,
67            op_batch_flags: [ZERO; NUM_OP_BATCH_FLAGS],
68        }
69    }
70
71    /// Creates a new `DecoderRow` for the start of a new batch in a basic block.
72    ///
73    /// This corresponds either to the SPAN or RESPAN operations.
74    pub fn new_basic_block_batch(
75        start_op: BasicBlockStartOperation,
76        op_batch: &OpBatch,
77        addr: Felt,
78        group_count: Felt,
79    ) -> Result<Self, ExecutionError> {
80        let opcode = match start_op {
81            BasicBlockStartOperation::Span => opcodes::SPAN,
82            BasicBlockStartOperation::Respan => opcodes::RESPAN,
83        };
84
85        let hasher_state = (
86            op_batch.groups()[0..4].try_into().expect("slice with incorrect length"),
87            op_batch.groups()[4..8].try_into().expect("slice with incorrect length"),
88        );
89
90        Ok(Self {
91            opcode,
92            hasher_state,
93            addr,
94            in_basic_block: false,
95            group_count,
96            op_index: ZERO,
97            op_batch_flags: get_op_batch_flags(group_count)?,
98        })
99    }
100
101    /// Creates a new `DecoderRow` for an operation within a basic block.
102    pub fn new_operation(
103        operation: Operation,
104        current_addr: Felt,
105        parent_addr: Felt,
106        op_idx_in_group: usize,
107        basic_block_ctx: &BasicBlockContext,
108        user_op_helpers: [Felt; NUM_USER_OP_HELPERS],
109    ) -> Self {
110        let hasher_state: (Word, Word) = {
111            let word1 = [
112                basic_block_ctx.current_op_group,
113                parent_addr,
114                user_op_helpers[0],
115                user_op_helpers[1],
116            ];
117            let word2 =
118                [user_op_helpers[2], user_op_helpers[3], user_op_helpers[4], user_op_helpers[5]];
119
120            (word1.into(), word2.into())
121        };
122
123        Self {
124            opcode: operation.op_code(),
125            hasher_state,
126            addr: current_addr,
127            in_basic_block: true,
128            group_count: basic_block_ctx.group_count_in_block,
129            op_index: Felt::from_u32(op_idx_in_group as u32),
130            op_batch_flags: [ZERO; NUM_OP_BATCH_FLAGS],
131        }
132    }
133}
134
135/// Enum representing the type of operation that starts a basic block.
136#[derive(Debug)]
137enum BasicBlockStartOperation {
138    Span,
139    Respan,
140}

145impl<'a> CoreTraceGenerationTracer<'a> {
146    /// Fills a trace row for SPAN start operation to the main trace fragment.
147    ///
148    /// This method creates a trace row that corresponds to the SPAN operation that starts
149    /// a basic block execution.
150    pub fn fill_basic_block_start_trace_row(
151        &mut self,
152        system: &SystemState,
153        stack: &StackState,
154        basic_block_node: &BasicBlockNode,
155    ) -> Result<(), ExecutionError> {
156        let group_count_for_block = Felt::from_u32(basic_block_node.num_op_groups() as u32);
157        let first_op_batch = basic_block_node
158            .op_batches()
159            .first()
160            .ok_or(ExecutionError::Internal("basic block should have at least one op batch"))?;
161
162        let decoder_row = DecoderRow::new_basic_block_batch(
163            BasicBlockStartOperation::Span,
164            first_op_batch,
165            self.decoder_state.parent_addr,
166            group_count_for_block,
167        )?;
168        self.fill_trace_row(system, stack, decoder_row);
169        Ok(())
170    }
171
172    /// Fills a trace row for SPAN end operation to the main trace fragment.
173    ///
174    /// This method creates a trace row that corresponds to the END operation that completes
175    /// a basic block execution.
176    pub fn fill_basic_block_end_trace_row(
177        &mut self,
178        system: &SystemState,
179        stack: &StackState,
180        basic_block_node: &BasicBlockNode,
181        hasher_state_second_word: Word,
182    ) -> Result<(), ExecutionError> {
183        let ended_node_addr = self.decoder_state.replay_node_end(&mut self.block_stack_replay)?;
184
185        let decoder_row = DecoderRow::new_control_flow(
186            opcodes::END,
187            (basic_block_node.digest(), hasher_state_second_word),
188            ended_node_addr,
189        );
190
191        self.fill_trace_row(system, stack, decoder_row);
192        Ok(())
193    }
194
195    // RESPAN
196    // -------------------------------------------------------------------------------------------
197
198    /// Processes a RESPAN operation that starts processing of a new operation batch within
199    /// the same basic block.
200    ///
201    /// This method updates the processor state and adds a corresponding trace row
202    /// to the main trace fragment.
203    pub fn fill_respan_trace_row(
204        &mut self,
205        system: &SystemState,
206        stack: &StackState,
207        op_batch: &OpBatch,
208        basic_block_context: &mut BasicBlockContext,
209    ) -> Result<(), ExecutionError> {
210        // Add RESPAN trace row
211        {
212            let decoder_row = DecoderRow::new_basic_block_batch(
213                BasicBlockStartOperation::Respan,
214                op_batch,
215                self.decoder_state.current_addr,
216                basic_block_context.group_count_in_block,
217            )?;
218            self.fill_trace_row(system, stack, decoder_row);
219        }
220
221        // Update block address for the upcoming block
222        self.decoder_state.current_addr += HASH_CYCLE_LEN_FELT;
223
224        // Update basic block context
225        basic_block_context.group_count_in_block -= ONE;
226        basic_block_context.current_op_group = op_batch.groups()[0];
227
228        Ok(())
229    }
230
231    /// Writes a trace row for an operation within a basic block.
232    ///
233    /// This must be called *after* the operation has been executed and the
234    /// stack has been updated.
235    pub fn fill_operation_trace_row(
236        &mut self,
237        system: &SystemState,
238        stack: &StackState,
239        operation: Operation,
240        op_idx_in_group: usize,
241        user_op_helpers: [Felt; NUM_USER_OP_HELPERS],
242        basic_block_context: &mut BasicBlockContext,
243    ) {
244        // update operations left to be executed in the group
245        basic_block_context.remove_operation_from_current_op_group();
246
247        // Add trace row
248        let decoder_row = DecoderRow::new_operation(
249            operation,
250            self.decoder_state.current_addr,
251            self.decoder_state.parent_addr,
252            op_idx_in_group,
253            basic_block_context,
254            user_op_helpers,
255        );
256        self.fill_trace_row(system, stack, decoder_row);
257    }
258}

263impl<'a> CoreTraceGenerationTracer<'a> {
264    // CALL operations
265    // -------------------------------------------------------------------------------------------
266
267    /// Fills a trace row for the start of a CALL/SYSCALL operation.
268    pub fn fill_call_start_trace_row(
269        &mut self,
270        system: &SystemState,
271        stack: &StackState,
272        call_node: &CallNode,
273        current_forest: &MastForest,
274    ) -> Result<(), ExecutionError> {
275        // For CALL/SYSCALL operations, the hasher state in start operations contains the callee
276        // hash in the first half, and zeros in the second half (since CALL only has one
277        // child)
278        let callee_hash: Word = get_node_in_forest(current_forest, call_node.callee())?.digest();
279        let zero_hash = Word::default();
280
281        let decoder_row = DecoderRow::new_control_flow(
282            if call_node.is_syscall() {
283                opcodes::SYSCALL
284            } else {
285                opcodes::CALL
286            },
287            (callee_hash, zero_hash),
288            self.decoder_state.parent_addr,
289        );
290
291        self.fill_trace_row(system, stack, decoder_row);
292        Ok(())
293    }
294
295    // DYN operations
296    // -------------------------------------------------------------------------------------------
297
298    /// Fills a trace row for the start of a DYN operation.
299    pub fn fill_dyn_start_trace_row(
300        &mut self,
301        system: &SystemState,
302        stack: &StackState,
303        callee_hash: Word,
304    ) {
305        let decoder_row = DecoderRow::new_control_flow(
306            opcodes::DYN,
307            (callee_hash, Word::default()),
308            self.decoder_state.parent_addr,
309        );
310        self.fill_trace_row(system, stack, decoder_row)
311    }
312
313    /// Fills a trace row for the start of a DYNCALL operation.
314    ///
315    /// The decoder hasher trace columns are populated with the callee hash, as well as the stack
316    /// helper registers (specifically their state after shifting the stack left). We need to store
317    /// those in the decoder trace so that the block stack table can access them (since in the next
318    /// row, we start a new context, and hence the stack registers are reset to their default
319    /// values).
320    pub fn fill_dyncall_start_trace_row(
321        &mut self,
322        system: &SystemState,
323        stack: &StackState,
324        callee_hash: Word,
325        ctx_info: ExecutionContextInfo,
326    ) {
327        let second_hasher_state: Word = [
328            Felt::from_u32(ctx_info.parent_stack_depth),
329            ctx_info.parent_next_overflow_addr,
330            ZERO,
331            ZERO,
332        ]
333        .into();
334
335        let decoder_row = DecoderRow::new_control_flow(
336            opcodes::DYNCALL,
337            (callee_hash, second_hasher_state),
338            self.decoder_state.parent_addr,
339        );
340        self.fill_trace_row(system, stack, decoder_row)
341    }

359            opcodes::JOIN,

362        );
363
364        self.fill_trace_row(system, stack, decoder_row);
365        Ok(())
366    }

395    pub fn fill_loop_repeat_trace_row(
396        &mut self,
397        system: &SystemState,
398        stack: &StackState,
399        loop_node: &LoopNode,
400        current_forest: &MastForest,
401        current_addr: Felt,
402    ) -> Result<(), ExecutionError> {
403        // For REPEAT operations, the hasher state in start operations contains the loop body hash
404        // in the first half.

406
407        let decoder_row = DecoderRow::new_control_flow(
408            opcodes::REPEAT,
409            // We set hasher[4] (is_loop_body) to 1

411            current_addr,
412        );
413
414        self.fill_trace_row(system, stack, decoder_row);
415        Ok(())
416    }

462
463        self.fill_trace_row(system, stack, decoder_row);
464        Ok(())
465    }

471impl<'a> CoreTraceGenerationTracer<'a> {
472    /// Fills a trace row for a control flow operation (JOIN/SPLIT start or end) to the main trace
473    /// fragment.
474    ///
475    /// This is a shared implementation that handles the common trace row generation logic
476    /// for both JOIN and SPLIT operations. The operation-specific details are provided
477    /// through the `config` parameter.
478    fn fill_trace_row(
479        &mut self,
480        system: &SystemState,
481        stack: &StackState,
482        decoder_row: DecoderRow,
483    ) {
484        let mut row = [ZERO; CORE_TRACE_WIDTH];
485
486        // System trace columns (identical for all control flow operations)
487        if let Some(ref system_cols) = self.system_cols {
488            row[..SYS_TRACE_WIDTH].copy_from_slice(system_cols);
489        }
490
491        // Decoder trace columns
492        Self::write_decoder_to_row(&mut row, &decoder_row);
493
494        // Stack trace columns (identical for all control flow operations)
495        if let Some(ref stack_cols) = self.stack_cols {
496            row[STACK_TRACE_OFFSET..STACK_TRACE_OFFSET + STACK_TRACE_WIDTH]
497                .copy_from_slice(stack_cols);
498        }
499
500        self.writer.write_row(self.row_write_index, &row);
501
502        // Store the buffer for the next call
503        self.system_cols = Some(Self::build_system_buffer(system));
504        self.stack_cols = Some(Self::build_stack_buffer(stack));
505
506        // Increment the row write index
507        self.row_write_index += 1;
508    }
509
510    fn write_decoder_to_row(row: &mut [Felt; CORE_TRACE_WIDTH], decoder_row: &DecoderRow) {
511        // Block address
512        row[DECODER_TRACE_OFFSET + ADDR_COL_IDX] = decoder_row.addr;
513
514        // Decompose operation into bits
515        let opcode = decoder_row.opcode;
516        for i in 0..NUM_OP_BITS {
517            let bit = Felt::from_u8((opcode >> i) & 1);
518            row[DECODER_TRACE_OFFSET + OP_BITS_OFFSET + i] = bit;
519        }
520
521        // Hasher state
522        let (first_hash, second_hash) = decoder_row.hasher_state;
523        row[DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET] = first_hash[0];
524        row[DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + 1] = first_hash[1];
525        row[DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + 2] = first_hash[2];
526        row[DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + 3] = first_hash[3];
527        row[DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + 4] = second_hash[0];
528        row[DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + 5] = second_hash[1];
529        row[DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + 6] = second_hash[2];
530        row[DECODER_TRACE_OFFSET + HASHER_STATE_OFFSET + 7] = second_hash[3];
531
532        // Remaining decoder trace columns (identical for all control flow operations)
533        row[DECODER_TRACE_OFFSET + OP_INDEX_COL_IDX] = decoder_row.op_index;
534        row[DECODER_TRACE_OFFSET + GROUP_COUNT_COL_IDX] = decoder_row.group_count;
535        row[DECODER_TRACE_OFFSET + IN_SPAN_COL_IDX] =
536            if decoder_row.in_basic_block { ONE } else { ZERO };
537
538        // Batch flag columns - all 0 for control flow operations
539        for i in 0..NUM_OP_BATCH_FLAGS {
540            row[DECODER_TRACE_OFFSET + OP_BATCH_FLAGS_OFFSET + i] = decoder_row.op_batch_flags[i];
541        }
542
543        // Extra bit columns
544        let bit6 = Felt::from_u8((opcode >> 6) & 1);
545        let bit5 = Felt::from_u8((opcode >> 5) & 1);
546        let bit4 = Felt::from_u8((opcode >> 4) & 1);
547        row[DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET] = bit6 * (ONE - bit5) * bit4;
548        row[DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET + 1] = bit6 * bit5;
549    }
550
551    fn build_system_buffer(system: &SystemState) -> [Felt; SYS_TRACE_WIDTH] {
552        let mut buf = [ZERO; SYS_TRACE_WIDTH];
553        buf[CLK_COL_IDX] = (system.clk + 1).into();
554        buf[CTX_COL_IDX] = system.ctx.into();
555        buf[FN_HASH_OFFSET] = system.fn_hash[0];
556        buf[FN_HASH_OFFSET + 1] = system.fn_hash[1];
557        buf[FN_HASH_OFFSET + 2] = system.fn_hash[2];
558        buf[FN_HASH_OFFSET + 3] = system.fn_hash[3];
559        buf
560    }
561
562    fn build_stack_buffer(stack: &StackState) -> [Felt; STACK_TRACE_WIDTH] {
563        let mut buf = [ZERO; STACK_TRACE_WIDTH];
564        for i in STACK_TOP_RANGE {
565            buf[STACK_TOP_OFFSET + i] = stack.get(i);
566        }
567
568        // Stack helpers (b0, b1, h0)
569        // Note: H0 will be inverted using batch inversion later
570        buf[B0_COL_IDX] = Felt::new(stack.stack_depth() as u64);
571        buf[B1_COL_IDX] = stack.overflow_addr();
572        buf[H0_COL_IDX] = stack.overflow_helper();
573        buf
574    }
575}

580/// Returns op batch flags for the specified group count.
581fn get_op_batch_flags(num_groups_left: Felt) -> Result<[Felt; 3], ExecutionError> {
582    use miden_air::trace::decoder::{
583        OP_BATCH_1_GROUPS, OP_BATCH_2_GROUPS, OP_BATCH_4_GROUPS, OP_BATCH_8_GROUPS,
584    };
585    use miden_core::mast::OP_BATCH_SIZE;
586
587    let num_groups = core::cmp::min(num_groups_left.as_canonical_u64() as usize, OP_BATCH_SIZE);
588    match num_groups {
589        8 => Ok(OP_BATCH_8_GROUPS),
590        4 => Ok(OP_BATCH_4_GROUPS),
591        2 => Ok(OP_BATCH_2_GROUPS),
592        1 => Ok(OP_BATCH_1_GROUPS),
593        _ => Err(ExecutionError::Internal(
594            "invalid number of groups in a batch, must be 1, 2, 4, or 8",
595        )),
596    }
