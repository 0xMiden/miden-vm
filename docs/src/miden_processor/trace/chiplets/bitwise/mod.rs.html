22/// Helper for the VM that computes AND and XOR bitwise operations on 32-bit values.
23/// It also builds an execution trace of these operations.
24///
25/// ## Bitwise operation execution trace (AND and XOR)
26/// The execution trace for each operation consists of 8 rows and 14 columns. At a high level,
27/// we break input values into 4-bit limbs, apply the bitwise operation to these limbs at every
28/// row starting with the most significant limb, and accumulate the result in the result column.
29///
30/// The layout of the table is illustrated below.
31///
32///    s     a     b      a0     a1     a2     a3     b0     b1     b2     b3    zp     z
33/// ├─────┴─────┴─────┴───────┴──────┴──────┴──────┴──────┴──────┴──────┴──────┴─────┴─────┤
34///
35/// In the above, the meaning of the columns is as follows:
36/// - Selector column s is used to specify the bitwise operator for each row.
37/// - Columns `a` and `b` contain accumulated 4-bit limbs of input values. Specifically, at the
38///   first row, the values of columns `a` and `b` are set to the most significant 4-bit limb of
39///   each input value. With all subsequent rows, the next most significant limb is appended to each
40///   column for the corresponding value. Thus, by the 8th row, columns `a` and `b` contain full
41///   input values for the bitwise operation.
42/// - Columns `a0` through `a3` and `b0` through `b3` contain bits of the least significant 4-bit
43///   limb of the values in `a` and `b` columns respectively.
44/// - Column `zp` contains the accumulated result of applying the bitwise operation to 4-bit limbs,
45///   but for the previous row. In the first row, it is 0.
46/// - Column `z` contains the accumulated result of applying the bitwise operation to 4-bit limbs.
47///   At the first row, column `z` contains the result of bitwise operation applied to the most
48///   significant 4-bit limbs of the input values. With every subsequent row, the next most
49///   significant 4-bit limb of the result is appended to it. Thus, by the 8th row, column `z`
50///   contains the full result of the bitwise operation.
51#[derive(Debug)]
52pub struct Bitwise {
53    trace: [Vec<Felt>; TRACE_WIDTH],
54}
55
56impl Bitwise {
57    // CONSTRUCTOR
58    // --------------------------------------------------------------------------------------------
59    /// Returns a new [Bitwise] initialized with an empty trace.
60    pub fn new() -> Self {
61        let trace = (0..TRACE_WIDTH)
62            .map(|_| Vec::with_capacity(INIT_TRACE_CAPACITY))
63            .collect::<Vec<_>>()
64            .try_into()
65            .expect("failed to convert vector to array");
66        Self { trace }
67    }
68
69    // PUBLIC ACCESSORS
70    // --------------------------------------------------------------------------------------------
71
72    /// Returns length of execution trace required to describe bitwise operations executed on the
73    /// VM.
74    pub fn trace_len(&self) -> usize {
75        self.trace[0].len()
76    }
77
78    // TRACE MUTATORS
79    // --------------------------------------------------------------------------------------------
80
81    /// Computes a bitwise AND of `a` and `b` and returns the result. We assume that `a` and `b`
82    /// are 32-bit values. If that's not the case, the result of the computation is undefined.
83    ///
84    /// This also adds 8 rows to the internal execution trace table required for computing the
85    /// operation.
86    pub fn u32and(&mut self, a: Felt, b: Felt) -> Result<Felt, OperationError> {
87        let a = assert_u32(a)? as u64;
88        let b = assert_u32(b)? as u64;
89        let mut result = 0u64;
90
91        // append 8 rows to the trace, each row computing bitwise AND in 4 bit limbs starting with
92        // the most significant limb.
93        for bit_offset in (0..32).step_by(4).rev() {
94            // append the previous row's result to the column for previous output values
95            self.trace[PREV_OUTPUT_COL_IDX].push(Felt::new(result));
96            // shift a and b so that the next 4-bit limb is in the least significant position
97            let a = a >> bit_offset;
98            let b = b >> bit_offset;
99
100            // add a new row to the trace table and populate it with binary decomposition of the 4
101            // least significant bits of a and b.
102            self.add_bitwise_trace_row(BITWISE_AND, a, b);
103
104            // compute bitwise AND of the 4 least significant bits of a and b
105            let result_4_bit = (a & b) & 0xf;
106
107            // append the 4 bit result to the result accumulator, and save the current result into
108            // the output column in the trace.
109            result = (result << 4) | result_4_bit;
110            self.trace[OUTPUT_COL_IDX].push(Felt::new(result));
111        }
112
113        Ok(Felt::new(result))
114    }
115
116    /// Computes a bitwise XOR of `a` and `b` and returns the result. We assume that `a` and `b`
117    /// are 32-bit values. If that's not the case, the result of the computation is undefined.
118    ///
119    /// This also adds 8 rows to the internal execution trace table required for computing the
120    /// operation.
121    pub fn u32xor(&mut self, a: Felt, b: Felt) -> Result<Felt, OperationError> {
122        let a = assert_u32(a)? as u64;
123        let b = assert_u32(b)? as u64;
124        let mut result = 0u64;
125
126        // append 8 rows to the trace, each row computing bitwise XOR in 4 bit limbs starting with
127        // the most significant limb.
128        for bit_offset in (0..32).step_by(4).rev() {
129            // append the previous row's result to the column for previous output values
130            self.trace[PREV_OUTPUT_COL_IDX].push(Felt::new(result));
131            // shift a and b so that the next 4-bit limb is in the least significant position
132            let a = a >> bit_offset;
133            let b = b >> bit_offset;
134
135            // add a new row to the trace table and populate it with binary decomposition of the 4
136            // least significant bits of a and b.
137            self.add_bitwise_trace_row(BITWISE_XOR, a, b);
138
139            // compute bitwise XOR of the 4 least significant bits of a and b
140            let result_4_bit = (a ^ b) & 0xf;
141
142            // append the 4 bit result to the result accumulator, and save the current result into
143            // the output column in the trace.
144            result = (result << 4) | result_4_bit;
145            self.trace[OUTPUT_COL_IDX].push(Felt::new(result));
146        }
147
148        Ok(Felt::new(result))
149    }
150
151    // EXECUTION TRACE GENERATION
152    // --------------------------------------------------------------------------------------------
153
154    /// Fills the provided trace fragment with trace data from this bitwise helper instance.
155    pub fn fill_trace(self, trace: &mut TraceFragment) {
156        // make sure fragment dimensions are consistent with the dimensions of this trace
157        debug_assert_eq!(self.trace_len(), trace.len(), "inconsistent trace lengths");
158        debug_assert_eq!(TRACE_WIDTH, trace.width(), "inconsistent trace widths");
159
160        // copy trace into the fragment column-by-column
161        // TODO: this can be parallelized to copy columns in multiple threads (#2163)
162        for (out_column, column) in trace.columns().zip(self.trace) {
163            out_column.copy_from_slice(&column);
164        }
165    }
166
167    // HELPER METHODS
168    // --------------------------------------------------------------------------------------------
169
170    /// Appends a new row to the trace table and populates the first 14 columns of trace as follows:
171    /// - Column 0 is set to the selector value for the bitwise operation being executed.
172    /// - Column 1 is set to the current value of `a`.
173    /// - Column 2 is set to the current value of `b`.
174    /// - Columns 3 to 6 are set to the 4 least-significant bits of `a`.
175    /// - Columns 7 to 10 are set to the 4 least-significant bits of `b`.
176    /// - Columns 11 and 12 are left for the output value and that of the previous row, which are
177    ///   set elsewhere.
178    fn add_bitwise_trace_row(&mut self, selector: Felt, a: u64, b: u64) {
179        self.trace[0].push(selector);
180        self.trace[A_COL_IDX].push(Felt::new(a));
181        self.trace[B_COL_IDX].push(Felt::new(b));
182
183        self.trace[A_COL_RANGE.start].push(Felt::new(a & 1));
184        self.trace[A_COL_RANGE.start + 1].push(Felt::new((a >> 1) & 1));
185        self.trace[A_COL_RANGE.start + 2].push(Felt::new((a >> 2) & 1));
186        self.trace[A_COL_RANGE.start + 3].push(Felt::new((a >> 3) & 1));
187
188        self.trace[B_COL_RANGE.start].push(Felt::new(b & 1));
189        self.trace[B_COL_RANGE.start + 1].push(Felt::new((b >> 1) & 1));
190        self.trace[B_COL_RANGE.start + 2].push(Felt::new((b >> 2) & 1));
191        self.trace[B_COL_RANGE.start + 3].push(Felt::new((b >> 3) & 1));
192    }
193}

204pub fn assert_u32(value: Felt) -> Result<u32, OperationError> {
205    u32::try_from(value.as_canonical_u64())
206        .map_err(|_| OperationError::NotU32Values { values: vec![value] })
