14/// Builds requests made to the bitwise chiplet. This can be either a request for the computation
15/// of a `XOR` or an `AND` operation.
16pub(super) fn build_bitwise_request<E: ExtensionField<Felt>>(
17    main_trace: &MainTrace,
18    is_xor: Felt,
19    challenges: &Challenges<E>,
20    row: RowIndex,
21    _debugger: &mut BusDebugger<E>,
22) -> E {
23    let bitwise_request_message = BitwiseMessage {
24        op_label: get_op_label(ONE, ZERO, is_xor, ZERO),
25        a: main_trace.stack_element(0, row),
26        b: main_trace.stack_element(1, row),
27        z: main_trace.stack_element(0, row + 1),
28        source: if is_xor == ONE { "u32xor" } else { "u32and" },
29    };
30
31    let value = bitwise_request_message.value(challenges);
32
33    #[cfg(any(test, feature = "bus-debugger"))]
34    _debugger.add_request(alloc::boxed::Box::new(bitwise_request_message), challenges);
35
36    value
37}

42/// Builds the response from the bitwise chiplet at `row`.
43pub(super) fn build_bitwise_chiplet_responses<E>(
44    main_trace: &MainTrace,
45    row: RowIndex,
46    challenges: &Challenges<E>,
47    _debugger: &mut BusDebugger<E>,
48) -> E
49where
50    E: ExtensionField<Felt>,
51{
52    let is_xor = main_trace.chiplet_selector_2(row);
53    if row.as_usize() % BITWISE_OP_CYCLE_LEN == BITWISE_OP_CYCLE_LEN - 1 {
54        let bitwise_message = BitwiseMessage {
55            op_label: get_op_label(ONE, ZERO, is_xor, ZERO),
56            a: main_trace.chiplet_bitwise_a(row),
57            b: main_trace.chiplet_bitwise_b(row),
58            z: main_trace.chiplet_bitwise_z(row),
59            source: "bitwise chiplet",
60        };
61
62        let value = bitwise_message.value(challenges);
63
64        #[cfg(any(test, feature = "bus-debugger"))]
65        _debugger.add_response(alloc::boxed::Box::new(bitwise_message), challenges);
66
67        value
68    } else {
69        E::ONE
70    }
71}

73// MESSAGE
74// ===============================================================================================

76pub struct BitwiseMessage {
77    pub op_label: Felt,
78    pub a: Felt,
79    pub b: Felt,
80    pub z: Felt,
81    pub source: &'static str,
82}
83
84impl<E> BusMessage<E> for BitwiseMessage
85where
86    E: ExtensionField<Felt>,
87{
88    fn value(&self, challenges: &Challenges<E>) -> E {
89        challenges.encode([self.op_label, self.a, self.b, self.z])
90    }
91
92    fn source(&self) -> &str {
93        self.source
94    }
95}
96
97impl Display for BitwiseMessage {
98    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult {
99        write!(
100            f,
101            "{{ op_label: {}, a: {}, b: {}, z: {} }}",
102            self.op_label, self.a, self.b, self.z
103        )
104    }
105}
