38/// Memory controller for the VM.
39///
40/// This component is responsible for tracking current memory state of the VM, as well as for
41/// building an execution trace of all memory accesses.
42///
43/// The memory is comprised of one or more segments, each segment accessible from a specific
44/// execution context. The root (kernel) context has context ID 0, and all additional contexts have
45/// increasing IDs. Within each segment, the memory is element-addressable, even though the trace
46/// tracks words for optimization purposes. That is, a single field element is located at each
47/// memory address, and we can read and write elements to/from memory either individually or in
48/// groups of four.
49///
50/// Memory for a given address is always initialized to zero. That is, reading from an address
51/// before writing to it will return ZERO.
52///
53/// ## Execution trace
54/// The layout of the memory access trace is shown below.
55///
56///   rw   ew   ctx  word_addr   idx0   idx1  clk   v0   v1   v2   v3   d0   d1   d_inv   f_scw
57/// ├────┴────┴────┴───────────┴──────┴──────┴────┴────┴────┴────┴────┴────┴────┴───────┴───────┤
58///
59/// In the above, the meaning of the columns is as follows:
60/// - `rw` is a selector column used to identify whether the memory operation is a read or a write
61///   (1 indicates a read).
62/// - `ew` is a selector column used to identify whether the memory operation is over an element or
63///   a word (1 indicates a word).
64/// - `ctx` contains execution context ID. Values in this column must increase monotonically but
65///   there can be gaps between two consecutive context IDs of up to 2^32. Also, two consecutive
66///   values can be the same.
67/// - `word_addr` contains the address of the first element in the word. For example, the value of
68///   `word_addr` for the group of addresses 40, 41, 42, 43 is 40. Note then that `word_addr` *must*
69///   be divisible by 4. Values in this column must increase monotonically for a given context but
70///   there can be gaps between two consecutive values of up to 2^32. Also, two consecutive values
71///   can be the same.
72/// - `idx0` and `idx1` are selector columns used to identify which element in the word is being
73///   accessed. Specifically, the index within the word is computed as `idx1 * 2 + idx0`.
74/// - `clk` contains the clock cycle at which a memory operation happened. Values in this column
75///   must increase monotonically for a given context and word but there can be gaps between two
76///   consecutive values of up to 2^32.
77/// - Columns `v0`, `v1`, `v2`, `v3` contain field elements stored at a given context/word/clock
78///   cycle after the memory operation.
79/// - Columns `d0` and `d1` contain lower and upper 16 bits of the delta between two consecutive
80///   context IDs, words, or clock cycles. Specifically:
81///   - When the context changes, these columns contain (`new_ctx` - `old_ctx`).
82///   - When the context remains the same but the word changes, these columns contain (`new_word`
83///     - `old_word`).
84///   - When both the context and the word remain the same, these columns contain (`new_clk` -
85///     `old_clk` - 1).
86/// - `d_inv` contains the inverse of the delta between two consecutive context IDs, words, or clock
87///   cycles computed as described above. It is the field inverse of `(d_1 * 2^16) + d_0`
88/// - `f_scw` is a flag indicating whether the context and the word of the current row are the same
89///   as in the next row.
90///
91/// For the first row of the trace, values in `d0`, `d1`, and `d_inv` are set to zeros.
92#[derive(Debug, Default)]
93pub struct Memory {
94    /// Memory segment traces sorted by their execution context ID.
95    trace: BTreeMap<ContextId, MemorySegmentTrace>,
96
97    /// Total number of entries in the trace (across all contexts); tracked separately so that we
98    /// don't have to sum up lengths of all address trace vectors for all contexts all the time.
99    num_trace_rows: usize,
100}

106    /// Returns length of execution trace required to describe all memory access operations
107    /// executed on the VM.
108    pub fn trace_len(&self) -> usize {
109        self.num_trace_rows
110    }
111
112    /// Returns the element located at the specified context/address, or None if the address hasn't
113    /// been accessed previously.
114    ///
115    /// Unlike read() which modifies the memory access trace, this method returns the value at the
116    /// specified address (if one exists) without altering the memory access trace.
117    pub fn get_value(&self, ctx: ContextId, addr: u32) -> Option<Felt> {
118        match self.trace.get(&ctx) {
119            Some(segment) => segment.get_value(addr),
120            None => None,
121        }
122    }
123
124    /// Returns the word located in memory starting at the specified address, which must be word
125    /// aligned.
126    ///
127    /// # Errors
128    /// - Returns an error if `addr` is not word aligned.
129    pub fn get_word(&self, ctx: ContextId, addr: u32) -> Result<Option<Word>, MemoryError> {
130        match self.trace.get(&ctx) {
131            Some(segment) => segment
132                .get_word(addr)
133                .map_err(|_| MemoryError::UnalignedWordAccess { addr, ctx }),
134            None => Ok(None),
135        }
136    }

138    /// Returns the entire memory state for the specified execution context at the specified cycle.
139    /// The state is returned as a vector of (address, value) tuples, and includes addresses which
140    /// have been accessed at least once.
141    pub fn get_state_at(&self, ctx: ContextId, clk: RowIndex) -> Vec<(MemoryAddress, Felt)> {
142        if clk == 0 {
143            return vec![];
144        }
145
146        match self.trace.get(&ctx) {
147            Some(segment) => segment.get_state_at(clk),
148            None => vec![],
149        }
150    }

155    /// Returns the field element located in memory at the specified context/address.
156    ///
157    /// If the specified address hasn't been previously written to, ZERO is returned. This
158    /// effectively implies that memory is initialized to ZERO.
159    ///
160    /// # Errors
161    /// - Returns an error if the address is equal or greater than 2^32.
162    /// - Returns an error if the same address is accessed more than once in the same clock cycle.
163    pub fn read(&mut self, ctx: ContextId, addr: Felt, clk: RowIndex) -> Result<Felt, MemoryError> {
164        let addr: u32 = addr
165            .as_canonical_u64()
166            .try_into()
167            .map_err(|_| MemoryError::AddressOutOfBounds { addr: addr.as_canonical_u64() })?;
168        self.num_trace_rows += 1;
169        self.trace.entry(ctx).or_default().read(ctx, addr, Felt::from(clk))
170    }
171
172    /// Returns a word located in memory at the specified context/address.
173    ///
174    /// If the specified address hasn't been previously written to, four ZERO elements are
175    /// returned. This effectively implies that memory is initialized to ZERO.
176    ///
177    /// # Errors
178    /// - Returns an error if the address is equal or greater than 2^32.
179    /// - Returns an error if the address is not aligned to a word boundary.
180    /// - Returns an error if the same address is accessed more than once in the same clock cycle.
181    pub fn read_word(
182        &mut self,
183        ctx: ContextId,
184        addr: Felt,
185        clk: RowIndex,
186    ) -> Result<Word, MemoryError> {
187        let addr: u32 = addr
188            .as_canonical_u64()
189            .try_into()
190            .map_err(|_| MemoryError::AddressOutOfBounds { addr: addr.as_canonical_u64() })?;
191        if !addr.is_multiple_of(WORD_SIZE as u32) {
192            return Err(MemoryError::UnalignedWordAccess { addr, ctx });
193        }
194
195        self.num_trace_rows += 1;
196        self.trace.entry(ctx).or_default().read_word(ctx, addr, Felt::from(clk))
197    }
198
199    /// Writes the provided field element at the specified context/address.
200    ///
201    /// # Errors
202    /// - Returns an error if the address is equal or greater than 2^32.
203    /// - Returns an error if the same address is accessed more than once in the same clock cycle.
204    pub fn write(
205        &mut self,
206        ctx: ContextId,
207        addr: Felt,
208        clk: RowIndex,
209        value: Felt,
210    ) -> Result<(), MemoryError> {
211        let addr: u32 = addr
212            .as_canonical_u64()
213            .try_into()
214            .map_err(|_| MemoryError::AddressOutOfBounds { addr: addr.as_canonical_u64() })?;
215        self.num_trace_rows += 1;
216        self.trace.entry(ctx).or_default().write(ctx, addr, Felt::from(clk), value)
217    }
218
219    /// Writes the provided word at the specified context/address.
220    ///
221    /// # Errors
222    /// - Returns an error if the address is equal or greater than 2^32.
223    /// - Returns an error if the address is not aligned to a word boundary.
224    /// - Returns an error if the same address is accessed more than once in the same clock cycle.
225    pub fn write_word(
226        &mut self,
227        ctx: ContextId,
228        addr: Felt,
229        clk: RowIndex,
230        value: Word,
231    ) -> Result<(), MemoryError> {
232        let addr: u32 = addr

234            .try_into()
235            .map_err(|_| MemoryError::AddressOutOfBounds { addr: addr.as_canonical_u64() })?;

238        }
239
240        self.num_trace_rows += 1;
241        self.trace.entry(ctx).or_default().write_word(ctx, addr, Felt::from(clk), value)
242    }

247    /// Adds all of the range checks required by the [Memory] chiplet to the provided
248    /// [RangeChecker] chiplet instance, along with their row in the finalized execution trace.
249    pub fn append_range_checks(&self, memory_start_row: RowIndex, range: &mut RangeChecker) {
250        // set the previous address and clock cycle to the first address and clock cycle of the
251        // trace; we also adjust the clock cycle so that delta value for the first row would end
252        // up being ZERO. if the trace is empty, return without any further processing.
253        let (mut prev_ctx, mut prev_addr, mut prev_clk) = match self.get_first_row_info() {
254            Some((ctx, addr, clk)) => (ctx, addr, clk.as_canonical_u64() - 1),
255            None => return,
256        };
257
258        // op range check index
259        let mut row = memory_start_row;
260
261        for (&ctx, segment) in self.trace.iter() {
262            for (&addr, addr_trace) in segment.inner().iter() {
263                // when we start a new address, we set the previous value to all zeros. the effect
264                // of this is that memory is always initialized to zero.
265                for memory_access in addr_trace {
266                    let clk = memory_access.clk().as_canonical_u64();
267
268                    // compute delta as difference between context IDs, addresses, or clock cycles
269                    let delta = if prev_ctx != ctx {
270                        (u32::from(ctx) - u32::from(prev_ctx)).into()
271                    } else if prev_addr != addr {
272                        u64::from(addr - prev_addr)
273                    } else {
274                        clk - prev_clk
275                    };
276
277                    let (delta_hi, delta_lo) = split_u32_into_u16(delta);
278                    range.add_range_checks(row, &[delta_lo, delta_hi]);
279
280                    // update values for the next iteration of the loop
281                    prev_ctx = ctx;
282                    prev_addr = addr;
283                    prev_clk = clk;
284                    row += 1_u32;
285                }
286            }
287        }
288    }

290    /// Fills the provided trace fragment with trace data from this memory instance.
291    pub fn fill_trace(self, trace: &mut TraceFragment) {
292        debug_assert_eq!(self.trace_len(), trace.len(), "inconsistent trace lengths");
293
294        // set the pervious address and clock cycle to the first address and clock cycle of the
295        // trace; we also adjust the clock cycle so that delta value for the first row would end
296        // up being ZERO. if the trace is empty, return without any further processing.
297        let (mut prev_ctx, mut prev_addr, mut prev_clk) = match self.get_first_row_info() {
298            Some((ctx, addr, clk)) => (Felt::from(ctx), Felt::from_u32(addr), clk - ONE),
299            None => return,
300        };
301
302        // iterate through addresses in ascending order, and write trace row for each memory access
303        // into the trace. we expect the trace to be 15 columns wide.
304        let mut row: RowIndex = 0.into();
305
306        for (ctx, segment) in self.trace {
307            let ctx = Felt::from(ctx);
308            for (addr, addr_trace) in segment.into_inner() {
309                // when we start a new address, we set the previous value to all zeros. the effect
310                // of this is that memory is always initialized to zero.
311                let felt_addr = Felt::from_u32(addr);
312                for memory_access in addr_trace {
313                    let clk = memory_access.clk();
314                    let value = memory_access.word();
315
316                    match memory_access.operation() {
317                        MemoryOperation::Read => trace.set(row, IS_READ_COL_IDX, MEMORY_READ),
318                        MemoryOperation::Write => trace.set(row, IS_READ_COL_IDX, MEMORY_WRITE),
319                    }
320                    let (idx1, idx0) = match memory_access.access_type() {
321                        segment::MemoryAccessType::Element { addr_idx_in_word } => {
322                            trace.set(row, IS_WORD_ACCESS_COL_IDX, MEMORY_ACCESS_ELEMENT);
323
324                            match addr_idx_in_word {
325                                0 => (ZERO, ZERO),
326                                1 => (ZERO, ONE),
327                                2 => (ONE, ZERO),
328                                3 => (ONE, ONE),
329                                _ => panic!("invalid address index in word: {addr_idx_in_word}"),
330                            }
331                        },
332                        segment::MemoryAccessType::Word => {
333                            trace.set(row, IS_WORD_ACCESS_COL_IDX, MEMORY_ACCESS_WORD);
334                            (ZERO, ZERO)
335                        },
336                    };
337                    trace.set(row, CTX_COL_IDX, ctx);
338                    trace.set(row, WORD_COL_IDX, felt_addr);
339                    trace.set(row, IDX0_COL_IDX, idx0);
340                    trace.set(row, IDX1_COL_IDX, idx1);
341                    trace.set(row, CLK_COL_IDX, clk);
342                    for (idx, col) in V_COL_RANGE.enumerate() {
343                        trace.set(row, col, value[idx]);
344                    }
345
346                    // compute delta as difference between context IDs, addresses, or clock cycles
347                    let delta = if prev_ctx != ctx {
348                        ctx - prev_ctx
349                    } else if prev_addr != felt_addr {
350                        felt_addr - prev_addr
351                    } else {
352                        clk - prev_clk
353                    };
354
355                    let (delta_hi, delta_lo) = split_element_u32_into_u16(delta);
356                    trace.set(row, D0_COL_IDX, delta_lo);
357                    trace.set(row, D1_COL_IDX, delta_hi);
358                    trace.set(row, D_INV_COL_IDX, delta.try_inverse().unwrap_or(ZERO));
359
360                    if prev_ctx == ctx && prev_addr == felt_addr {
361                        trace.set(row, FLAG_SAME_CONTEXT_AND_WORD, ONE);
362                    } else {
363                        trace.set(row, FLAG_SAME_CONTEXT_AND_WORD, ZERO);
364                    };
365
366                    // update values for the next iteration of the loop
367                    prev_ctx = ctx;
368                    prev_addr = felt_addr;
369                    prev_clk = clk;
370                    row += 1_u32;
371                }
372            }
373        }
374    }

379    /// Returns the context, address, and clock cycle of the first trace row, or None if the trace
380    /// is empty.
381    fn get_first_row_info(&self) -> Option<(ContextId, u32, Felt)> {
382        let (ctx, segment) = match self.trace.iter().next() {
383            Some((&ctx, segment)) => (ctx, segment),
384            None => return None,
385        };
386
387        let (&addr, addr_trace) = segment.inner().iter().next().expect("empty memory segment");
388
389        Some((ctx, addr, addr_trace[0].clk()))
390    }
