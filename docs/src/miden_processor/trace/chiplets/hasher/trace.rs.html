9// HASHER TRACE
10// ================================================================================================
11
12/// Execution trace of the hasher component.
13///
14/// The trace consists of 16 columns grouped logically as follows:
15/// - 3 selector columns.
16/// - 12 columns describing hasher state.
17/// - 1 node index column used for Merkle path related computations.
18#[derive(Debug, Default)]
19pub struct HasherTrace {
20    selectors: [Vec<Felt>; 3],
21    hasher_state: [Vec<Felt>; STATE_WIDTH],
22    node_index: Vec<Felt>,
23}
24
25impl HasherTrace {
26    // PUBLIC ACCESSORS
27    // --------------------------------------------------------------------------------------------
28
29    /// Returns current length of this execution trace.
30    pub fn trace_len(&self) -> usize {
31        self.selectors[0].len()
32    }
33
34    /// Returns the next row address. The address is equal to the current trace length + 1.
35    ///
36    /// The above means that row addresses start at ONE (rather than ZERO), and are incremented by
37    /// ONE at every row. Starting at ONE is needed for the decoder so that the address of the
38    /// first code block is a non-zero value.
39    pub fn next_row_addr(&self) -> Felt {
40        Felt::new(self.trace_len() as u64 + 1)
41    }
42
43    // TRACE MUTATORS
44    // --------------------------------------------------------------------------------------------
45
46    /// Appends 32 rows to the execution trace describing a single permutation of the hash function.
47    ///
48    /// The initial state of the hasher is provided via the `state` parameter. All subsequent
49    /// states are derived by applying a single round of the hash function to the previous state.
50    ///
51    /// Selector values for the first and last rows are provided via `init_selectors` and
52    /// `final_selectors` parameters. Selector values for all other rows are derived from the
53    /// selectors of the first row.
54    ///
55    /// Node index values are provided via `init_index` and `rest_index` parameters. The former is
56    /// used for the first row, and the latter for all subsequent rows.
57    pub fn append_permutation_with_index(
58        &mut self,
59        state: &mut HasherState,
60        init_selectors: Selectors,
61        final_selectors: Selectors,
62        init_index: Felt,
63        rest_index: Felt,
64    ) {
65        // append the first row of the permutation cycle
66        self.append_row(init_selectors, state, init_index);
67
68        // append the next NUM_ROUNDS - 1 rows of the permutation cycle. for these rows:
69        // - the last two selectors are carried over from row to row; the first selector is set to
70        //   ZERO.
71        // - hasher state is updated by applying a single round of the hash function for every row.
72        let next_selectors = [ZERO, init_selectors[1], init_selectors[2]];
73        for i in 0..NUM_ROUNDS - 1 {
74            apply_round(state, i);
75            self.append_row(next_selectors, state, rest_index);
76        }
77
78        // apply the last round and append the last row to the trace
79        apply_round(state, NUM_ROUNDS - 1);
80        self.append_row(final_selectors, state, rest_index);
81    }
82
83    /// Appends 32 rows to the execution trace describing a single permutation of the hash function.
84    ///
85    /// This function is similar to the append_permutation_with_index() function above, but it sets
86    /// init_index and rest_index parameters to ZEROs.
87    #[inline(always)]
88    pub fn append_permutation(
89        &mut self,
90        state: &mut HasherState,
91        init_selectors: Selectors,
92        final_selectors: Selectors,
93    ) {
94        self.append_permutation_with_index(state, init_selectors, final_selectors, ZERO, ZERO);
95    }
96
97    /// Appends a new row to the execution trace based on the supplied parameters.
98    fn append_row(&mut self, selectors: Selectors, state: &HasherState, index: Felt) {
99        for (trace_col, selector_val) in self.selectors.iter_mut().zip(selectors) {
100            trace_col.push(selector_val);
101        }
102        for (trace_col, &state_val) in self.hasher_state.iter_mut().zip(state) {
103            trace_col.push(state_val);
104        }
105        self.node_index.push(index);
106    }
107
108    /// Copies section of trace from the given range of start and end rows at the end of the trace.
109    /// The hasher state of the last row is copied to the provided state input.
110    pub fn copy_trace(&mut self, state: &mut [Felt; STATE_WIDTH], range: Range<usize>) {
111        for selector in self.selectors.iter_mut() {
112            selector.extend_from_within(range.clone());
113        }
114
115        for hasher in self.hasher_state.iter_mut() {
116            hasher.extend_from_within(range.clone());
117        }
118
119        self.node_index.extend_from_within(range.clone());
120
121        // copy the latest hasher state to the provided state slice
122        for (col, hasher) in self.hasher_state.iter().enumerate() {
123            state[col] = hasher[range.end - 1];
124        }
125    }
126
127    // EXECUTION TRACE GENERATION
128    // --------------------------------------------------------------------------------------------
129
130    /// Fills the provided trace fragment with trace data from this hasher trace instance.
131    pub fn fill_trace(self, trace: &mut TraceFragment) {
132        // make sure fragment dimensions are consistent with the dimensions of this trace
133        debug_assert_eq!(self.trace_len(), trace.len(), "inconsistent trace lengths");
134        debug_assert_eq!(TRACE_WIDTH, trace.width(), "inconsistent trace widths");
135
136        // collect all trace columns into a single vector
137        let mut columns = Vec::new();
138        self.selectors.into_iter().for_each(|c| columns.push(c));
139
140        self.hasher_state.into_iter().for_each(|c| columns.push(c));
141        columns.push(self.node_index);
142
143        // copy trace into the fragment column-by-column
144        // TODO: this can be parallelized to copy columns in multiple threads (#2163)
145        for (out_column, column) in trace.columns().zip(columns) {
146            out_column.copy_from_slice(&column);
147        }
148    }
149}
