22/// Hash chiplet for the VM.
23///
24/// This component is responsible for performing all hash-related computations for the VM, as well
25/// as building an execution trace for these computations. These computations include:
26/// * Linear hashes, including simple 2-to-1 hashes, single and multiple permutations.
27/// * Merkle path verification.
28/// * Merkle root updates.
29///
30/// ## Execution trace
31/// Hasher execution trace consists of 16 columns as illustrated below:
32///
33///   s0   s1   s2   h0   h1   h2   h3   h4   h5   h6   h7   h8   h9   h10   h11   idx
34/// ├────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴─────┴─────┴─────┤
35///
36/// In the above, the meaning of the columns is as follows:
37/// * Selector columns s0, s1, and s2 used to help select transition function for a given row.
38/// * Hasher state columns h0 through h11 used to hold the hasher state for each round of hash
39///   computation. The state is laid out as `[RATE0, RATE1, CAPACITY]`:
40///   - The first eight columns (h0-h7) represent the rate elements of the state. These are used to
41///     absorb the values to be hashed. Once a permutation is complete, hash output is located in
42///     the first four rate columns (h0, h1, h2, h3).
43///   - The last four columns (h8-h11) represent the capacity state of the sponge function.
44/// * Node index column idx used to help with Merkle path verification and Merkle root update
45///   computations. For all other computations the values in this column are set to 0s.
46///
47/// Each permutation of the hash function adds 8 rows to the execution trace. Thus, for Merkle
48/// path verification, number of rows added to the trace is 8 * path.len(), and for Merkle root
49/// update it is 16 * path.len(), since we need to perform two path verifications for each update.
50///
51/// In addition to the execution trace, the hash chiplet also maintains:
52/// - an auxiliary trace builder, which can be used to construct a running product column describing
53///   the state of the sibling table (used in Merkle root update operations).
54/// - a map of memoized execution trace, which keeps track of start and end rows of the sections of
55///   the trace of a control or span block that can be copied to be used later for program blocks
56///   encountered with the same digest instead of building it from scratch everytime. The hash of
57///   the block is used as the key here after converting it to a bytes array.
58#[derive(Debug, Default)]
59pub struct Hasher {
60    trace: HasherTrace,
61    memoized_trace_map: BTreeMap<[u8; 32], (usize, usize)>,
62}

76    /// Applies a single permutation of the hash function to the provided state and records the
77    /// execution trace of this computation.
78    ///
79    /// The returned tuple contains the hasher state after the permutation and the row address of
80    /// the execution trace at which the permutation started.
81    pub fn permute(&mut self, mut state: HasherState) -> (Felt, HasherState) {
82        let addr = self.trace.next_row_addr();
83
84        // perform the hash.
85        self.trace.append_permutation(&mut state, LINEAR_HASH, RETURN_STATE);
86
87        (addr, state)
88    }

90    /// Computes the hash of the control block by computing hash(h1, h2) and returns the result.
91    /// It also records the execution trace of this computation.
92    ///
93    /// The returned tuple also contains the row address of the execution trace at which the hash
94    /// computation started.
95    pub fn hash_control_block(
96        &mut self,
97        h1: Digest,
98        h2: Digest,
99        domain: Felt,
100        expected_hash: Digest,
101    ) -> (Felt, Digest) {
102        let addr = self.trace.next_row_addr();
103        let mut state = init_state_from_words_with_domain(&h1, &h2, domain);
104
105        if let Some((start_row, end_row)) = self.get_memoized_trace(expected_hash) {
106            // copy the trace of a block with same hash instead of building it again.
107            self.trace.copy_trace(&mut state, *start_row..*end_row);
108        } else {
109            // perform the hash.
110            self.trace.append_permutation(&mut state, LINEAR_HASH, RETURN_HASH);
111
112            self.insert_to_memoized_trace_map(addr, expected_hash);
113        };
114
115        let result = get_digest(&state);
116
117        (addr, result)
118    }

120    /// Computes a sequential hash of all operation batches in the list and returns the result. It
121    /// also records the execution trace of this computation.
122    ///
123    /// The returned tuple also contains the row address of the execution trace at which the hash
124    /// computation started.
125    pub fn hash_basic_block(
126        &mut self,
127        op_batches: &[OpBatch],
128        expected_hash: Digest,
129    ) -> (Felt, Digest) {
130        const START: Selectors = LINEAR_HASH;
131        const RETURN: Selectors = RETURN_HASH;
132        // absorb selectors are the same as linear hash selectors, but absorb selectors are
133        // applied on the last row of a permutation cycle, while linear hash selectors are
134        // applied on the first row of a permutation cycle.
135        const ABSORB: Selectors = LINEAR_HASH;
136        // to continue linear hash we need retain the 2nd and 3rd selector flags and set the
137        // 1st flag to ZERO.
138        const CONTINUE: Selectors = [ZERO, LINEAR_HASH[1], LINEAR_HASH[2]];
139
140        let addr = self.trace.next_row_addr();
141
142        // initialize the state and absorb the first operation batch into it
143        let mut state = init_state(op_batches[0].groups(), ZERO);
144
145        // check if a span block with same hash has been encountered before in which case we can
146        // directly copy it's trace.
147        let (start_row, end_row, is_memoized) =
148            if let Some((start_row, end_row)) = self.get_memoized_trace(expected_hash) {
149                (*start_row, *end_row, true)
150            } else {
151                (0, 0, false)
152            };
153
154        let num_batches = op_batches.len();
155
156        // if the span block is encountered for the first time and it's trace is not memoized,
157        // we need to build the trace from scratch.
158        if !is_memoized {
159            if num_batches == 1 {
160                // if there is only one batch to hash, we need only one permutation
161                self.trace.append_permutation(&mut state, START, RETURN);
162            } else {
163                // if there is more than one batch, we need to process the first, the last, and the
164                // middle permutations a bit differently. Specifically, selector flags for the
165                // permutations need to be set as follows:
166                // - first permutation: init linear hash on the first row, and absorb the next
167                //   operation batch on the last row.
168                // - middle permutations: continue hashing on the first row, and absorb the next
169                //   operation batch on the last row.
170                // - last permutation: continue hashing on the first row, and return the result on
171                //   the last row.
172                self.trace.append_permutation(&mut state, START, ABSORB);
173
174                for batch in op_batches.iter().take(num_batches - 1).skip(1) {
175                    absorb_into_state(&mut state, batch.groups());
176
177                    self.trace.append_permutation(&mut state, CONTINUE, ABSORB);
178                }
179
180                absorb_into_state(&mut state, op_batches[num_batches - 1].groups());
181
182                self.trace.append_permutation(&mut state, CONTINUE, RETURN);
183            }
184            self.insert_to_memoized_trace_map(addr, expected_hash);
185        } else {
186            self.trace.copy_trace(&mut state, start_row..end_row);
187        }
188
189        let result = get_digest(&state);
190
191        (addr, result)
192    }

194    /// Performs Merkle path verification computation and records its execution trace.
195    ///
196    /// The computation consists of computing a Merkle root of the specified path for a node with
197    /// the specified value, located at the specified index.
198    ///
199    /// The returned tuple contains the root of the Merkle path and the row address of the
200    /// execution trace at which the computation started.
201    ///
202    /// # Panics
203    /// Panics if:
204    /// - The provided path does not contain any nodes.
205    /// - The provided index is out of range for the specified path.
206    pub fn build_merkle_root(
207        &mut self,
208        value: Digest,
209        path: &MerklePath,
210        index: Felt,
211    ) -> (Felt, Digest) {
212        let addr = self.trace.next_row_addr();
213
214        let root = self.verify_merkle_path(
215            value,
216            path,
217            index.as_canonical_u64(),
218            MerklePathContext::MpVerify,
219        );
220
221        (addr, root)
222    }
223
224    /// Performs Merkle root update computation and records its execution trace.
225    ///
226    /// The computation consists of two Merkle path verifications, one for the old value of the
227    /// node (value before the update), and another for the new value (value after the update).
228    ///
229    /// # Panics
230    /// Panics if:
231    /// - The provided path does not contain any nodes.
232    /// - The provided index is out of range for the specified path.
233    pub fn update_merkle_root(
234        &mut self,
235        old_value: Digest,
236        new_value: Digest,
237        path: &MerklePath,
238        index: Felt,
239    ) -> MerkleRootUpdate {
240        let address = self.trace.next_row_addr();
241        let index = index.as_canonical_u64();
242
243        let old_root =
244            self.verify_merkle_path(old_value, path, index, MerklePathContext::MrUpdateOld);
245        let new_root =
246            self.verify_merkle_path(new_value, path, index, MerklePathContext::MrUpdateNew);
247
248        MerkleRootUpdate { address, old_root, new_root }
249    }

263    /// Computes a root of the provided Merkle path in the specified context. The path is assumed
264    /// to be for a node with the specified value at the specified index.
265    ///
266    /// This also records the execution trace of the Merkle path computation.
267    ///
268    /// # Panics
269    /// Panics if:
270    /// - The provided path does not contain any nodes.
271    /// - The provided index is out of range for the specified path.
272    fn verify_merkle_path(
273        &mut self,
274        value: Digest,
275        path: &MerklePath,
276        mut index: u64,
277        context: MerklePathContext,
278    ) -> Digest {
279        assert!(!path.is_empty(), "path is empty");
280        assert!(
281            index.checked_shr(path.len() as u32).unwrap_or(0) == 0,
282            "invalid index for the path"
283        );
284        let mut root = value;
285
286        // determine selectors for the specified context
287        let main_selectors = context.main_selectors();
288        let part_selectors = context.part_selectors();
289
290        if path.len() == 1 {
291            // handle path of length 1 separately because pattern for init and final selectors
292            // is different from other cases
293            self.verify_mp_leg(root, path[0], &mut index, main_selectors, RETURN_HASH)
294        } else {
295            // process the first node of the path; for this node, init and final selectors are
296            // the same
297            let sibling = path[0];
298            root = self.verify_mp_leg(root, sibling, &mut index, main_selectors, main_selectors);
299
300            // process all other nodes, except for the last one
301            for &sibling in &path[1..path.len() - 1] {
302                root =
303                    self.verify_mp_leg(root, sibling, &mut index, part_selectors, main_selectors);
304            }
305
306            // process the last node
307            let sibling = path[path.len() - 1];
308            self.verify_mp_leg(root, sibling, &mut index, part_selectors, RETURN_HASH)
309        }
310    }

312    /// Verifies a single leg of a Merkle path.
313    ///
314    /// This function does the following:
315    /// - Builds the initial hasher state based on the least significant bit of the index.
316    /// - Applies a permutation to this state and records the resulting trace.
317    /// - Returns the result of the permutation and updates the index by removing its least
318    ///   significant bit.
319    fn verify_mp_leg(
320        &mut self,
321        root: Digest,
322        sibling: Digest,
323        index: &mut u64,
324        init_selectors: Selectors,
325        final_selectors: Selectors,
326    ) -> Digest {
327        // build the hasher state based on the value of the least significant bit of the index
328        let index_bit = *index & 1;
329        let mut state = build_merge_state(&root, &sibling, index_bit);
330
331        // determine values for the node index column for this permutation. if the first selector
332        // of init_selectors is not ZERO (i.e., we are processing the first leg of the Merkle
333        // path), the index for the first row is different from the index for the other rows;
334        // otherwise, indexes are the same.
335        let (init_index, rest_index) = if init_selectors[0] == ZERO {
336            (Felt::new(*index >> 1), Felt::new(*index >> 1))
337        } else {
338            (Felt::new(*index), Felt::new(*index >> 1))
339        };
340
341        // apply the permutation to the state and record its trace
342        self.trace.append_permutation_with_index(
343            &mut state,
344            init_selectors,
345            final_selectors,
346            init_index,
347            rest_index,
348        );
349
350        // remove the least significant bit from the index and return hash result
351        *index >>= 1;
352
353        get_digest(&state)
354    }
355
356    /// Checks if a trace for a program block already exists and returns the start and end rows
357    /// of the memoized trace. Returns None otherwise.
358    fn get_memoized_trace(&self, hash: Digest) -> Option<&(usize, usize)> {
359        let key: [u8; 32] = hash.into();
360        self.memoized_trace_map.get(&key)
361    }
362
363    /// Inserts start and end rows of trace for a program block to the memoized_trace_map.
364    fn insert_to_memoized_trace_map(&mut self, addr: Felt, hash: Digest) {
365        let key: [u8; 32] = hash.into();
366        let start_row = addr.as_canonical_u64() as usize - 1;
367        let end_row = self.trace.next_row_addr().as_canonical_u64() as usize - 1;
368        self.memoized_trace_map.insert(key, (start_row, end_row));
369    }
370}

375/// Specifies the context of a Merkle path computation.
376#[derive(Debug, Clone, Copy)]
377enum MerklePathContext {
378    /// The computation is for verifying a Merkle path (MPVERIFY).
379    MpVerify,
380    /// The computation is for verifying a Merkle path to an old node during Merkle root update
381    /// procedure (MRUPDATE).
382    MrUpdateOld,
383    /// The computation is for verifying a Merkle path to a new node during Merkle root update
384    /// procedure (MRUPDATE).
385    MrUpdateNew,
386}
387
388impl MerklePathContext {
389    /// Returns selector values for this context.
390    pub fn main_selectors(&self) -> Selectors {
391        match self {
392            Self::MpVerify => MP_VERIFY,
393            Self::MrUpdateOld => MR_UPDATE_OLD,
394            Self::MrUpdateNew => MR_UPDATE_NEW,
395        }
396    }
397
398    /// Returns partial selector values for this context. Partial selector values are derived
399    /// from selector values by replacing the first selector with ZERO.
400    pub fn part_selectors(&self) -> Selectors {
401        let selectors = self.main_selectors();
402        [ZERO, selectors[1], selectors[2]]
403    }
404}

409/// Combines two words into a hasher state for Merkle path computation.
410///
411/// If index_bit = 0, the words are combined in the order (a, b), if index_bit = 1, the words are
412/// combined in the order (b, a), otherwise, the function panics.
413#[inline(always)]
414fn build_merge_state(a: &Digest, b: &Digest, index_bit: u64) -> HasherState {
415    match index_bit {
416        0 => init_state_from_words(a, b),
417        1 => init_state_from_words(b, a),
418        _ => panic!("index bit is not a binary value"),
419    }
420}

427/// Initializes hasher state with the first 8 elements to be absorbed. In accordance with the
428/// Poseidon2 padding rule, the first capacity element is set with the provided padding flag, which
429/// is assumed to be ZERO or ONE, depending on whether the number of elements to be absorbed is a
430/// multiple of the rate or not. The remaining elements in the capacity portion of the state are set
431/// to ZERO.
432///
433/// State layout: [R1, R2, CAP] where:
434/// - state[0..8] = init_values (rate)
435/// - state[8..12] = [padding_flag, ZERO, ZERO, ZERO] (capacity)
436#[inline(always)]
437pub fn init_state(init_values: &[Felt; RATE_LEN], padding_flag: Felt) -> [Felt; STATE_WIDTH] {
438    debug_assert!(
439        padding_flag == ZERO || padding_flag == ONE,
440        "first capacity element must be 0 or 1"
441    );
442    [
443        init_values[0],
444        init_values[1],
445        init_values[2],
446        init_values[3],
447        init_values[4],
448        init_values[5],
449        init_values[6],
450        init_values[7],
451        padding_flag,
452        ZERO,
453        ZERO,
454        ZERO,
455    ]
456}
457
458/// Initializes hasher state with the elements from the provided words. Because the length of the
459/// input is a multiple of the rate, all capacity elements are initialized to zero, as specified by
460/// the Poseidon2 padding rule.
461///
462/// State layout: [RATE0, RATE1, CAP] where:
463/// - state[0..4] = w1 (first rate word, also digest location)
464/// - state[4..8] = w2 (second rate word)
465/// - state[8..12] = capacity
466#[inline(always)]
467pub fn init_state_from_words(w1: &Digest, w2: &Digest) -> [Felt; STATE_WIDTH] {
468    init_state_from_words_with_domain(w1, w2, ZERO)
469}
470
471/// Initializes hasher state with elements from the provided words. Sets the second element of the
472/// capacity register to the provided domain. All other elements of the capacity register are set
473/// to 0.
474///
475/// State layout: [RATE0, RATE1, CAP] where:
476/// - state[0..4] = w1 (first rate word, also digest location)
477/// - state[4..8] = w2 (second rate word)
478/// - state[8..12] = [ZERO, domain, ZERO, ZERO] (capacity)
479#[inline(always)]
480pub fn init_state_from_words_with_domain(
481    w1: &Digest,
482    w2: &Digest,
483    domain: Felt,
484) -> [Felt; STATE_WIDTH] {
485    [w1[0], w1[1], w1[2], w1[3], w2[0], w2[1], w2[2], w2[3], ZERO, domain, ZERO, ZERO]
486}
487
488/// Absorbs the specified values into the provided state by overwriting the corresponding elements
489/// in the rate portion of the state.
490///
491/// State layout: rate is at state[0..8]
492#[inline(always)]
493pub fn absorb_into_state(state: &mut [Felt; STATE_WIDTH], values: &[Felt; RATE_LEN]) {
494    state[0] = values[0];
495    state[1] = values[1];
496    state[2] = values[2];
497    state[3] = values[3];
498    state[4] = values[4];
499    state[5] = values[5];
500    state[6] = values[6];
501    state[7] = values[7];
502}
503
504/// Returns elements representing the digest portion of the provided hasher's state.
505pub fn get_digest(state: &[Felt; STATE_WIDTH]) -> Digest {
506    state[DIGEST_RANGE].try_into().expect("failed to get digest from hasher state")
507}
