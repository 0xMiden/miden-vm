12/// If the given error is an error generated when trying to resolve an External node, and there is a
13/// caller context available in the continuation stack, use the caller node ID to build the error
14/// context.
15///
16/// In practice, `ExternalNode`s are executed via a `CallNode` or `DynNode`. Thus, if we fail to
17/// resolve an `ExternalNode`, we can look at the top of the continuation stack to find the caller
18/// node ID (that we expect to be a `CallNode` or `DynNode`), and build the diagnostic from that
19/// node.
20///
21/// For example, in MASM, the user would see an error like:
22/// ```masm
23/// x procedure with root digest <digest> could not be found
24///     ,-[::\$exec:5:13]
25///   4 |         begin
26///   5 |             call.bar::dummy_proc
27///     :             ^^^^^^^^^^^^^^^^^^^^
28///   6 |         end
29///     `----
30/// ```
31///
32/// The carets and line numbers point to the `call` instruction that triggered the error because of
33/// the remapping we do in this function.
34pub(super) fn maybe_use_caller_error_context(
35    original_err: ExecutionError,
36    current_forest: &MastForest,
37    continuation_stack: &ContinuationStack,
38    host: &mut impl BaseHost,
39) -> ExecutionError {
40    // We only care about procedure-not-found errors or malformed MAST forest errors.
41    let root_digest = match &original_err {
42        ExecutionError::ProcedureNotFound { root_digest, .. } => *root_digest,
43        ExecutionError::OperationError {
44            err: OperationError::MalformedMastForestInHost { root_digest },
45            ..
46        } => *root_digest,
47        _ => return original_err,
48    };
49
50    // Look for caller context in the continuation stack
51    let Some(top_continuation) = continuation_stack.peek_continuation() else {
52        return original_err;
53    };
54
55    // Extract parent node ID from all continuations that can lead to an external node execution.
56    let parent_node_id = match top_continuation {
57        Continuation::FinishCall(parent_node_id)
58        | Continuation::FinishJoin(parent_node_id)
59        | Continuation::FinishSplit(parent_node_id)
60        | Continuation::FinishLoop { node_id: parent_node_id, .. } => parent_node_id,
61        _ => return original_err,
62    };
63
64    // We were able to get the parent node ID, so rebuild the error with the caller's context.
65    // For ProcedureNotFound, we reconstruct with the parent's source location.
66    // For MalformedMastForestInHost, we wrap in OperationError with the parent's context.
67    match &original_err {
68        ExecutionError::ProcedureNotFound { .. } => {
69            procedure_not_found_with_context(root_digest, current_forest, *parent_node_id, host)
70        },
71        ExecutionError::OperationError { err, .. } => {
72            err.clone().with_context(current_forest, *parent_node_id, host)
73        },
74        _ => original_err,
75    }
76}
