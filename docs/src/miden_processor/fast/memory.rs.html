8/// The memory for the processor.
9///
10/// Allows to read/write elements or words to memory. Internally, it is implemented as a map from
11///(context_id, word_address) to the word stored starting at that memory location.
12///
13/// # Invariants
14/// The memory submodule assumes that the following invariants hold:
15/// - Multiple reads in the same clock cycle to the same address are allowed, but
16/// - Multiple writes in the same clock cycle to the same address are *not* allowed
17///
18/// These invariants are not enforced by [`Memory`] explicitly, but are expected to be upheld by all
19/// processor operations (i.e. all variants of the [`miden_core::operations::Operation`] enum). This
20/// is a consequence of the design of the memory chiplet constraints, which allow for multiple reads
21/// but not multiple writes in the same clock cycle to the same address.
22#[derive(Debug, Default)]
23pub struct Memory {
24    memory: BTreeMap<(ContextId, u32), Word>,
25}
26
27impl Memory {
28    /// Creates a new memory instance.
29    pub fn new() -> Self {
30        Self::default()
31    }

33    /// Reads an element from memory at the provided address in the provided context.
34    ///
35    /// # Errors
36    /// - Returns an error if the provided address is out-of-bounds.
37    #[inline(always)]
38    pub fn read_element(&self, ctx: ContextId, addr: Felt) -> Result<Felt, MemoryError> {
39        let element = self.read_element_impl(ctx, clean_addr(addr)?).unwrap_or(ZERO);
40
41        Ok(element)
42    }
43
44    /// Reads a word from memory starting at the provided address in the provided context.
45    ///
46    /// # Errors
47    /// - Returns an error if the provided address is out-of-bounds or not word-aligned.
48    #[inline(always)]
49    pub fn read_word(
50        &self,
51        ctx: ContextId,
52        addr: Felt,
53        _clk: RowIndex,
54    ) -> Result<Word, MemoryError> {
55        let addr = clean_addr(addr)?;
56        let word = self.read_word_impl(ctx, addr)?.unwrap_or(EMPTY_WORD);
57
58        Ok(word)
59    }
60
61    /// Writes an element to memory at the provided address in the provided context.
62    ///
63    /// # Errors
64    /// - Returns an error if the provided address is out-of-bounds.
65    #[inline(always)]
66    pub fn write_element(
67        &mut self,
68        ctx: ContextId,
69        addr: Felt,
70        element: Felt,
71    ) -> Result<(), MemoryError> {
72        let (word_addr, idx) = split_addr(clean_addr(addr)?);
73
74        self.memory
75            .entry((ctx, word_addr))
76            .and_modify(|word| {
77                let mut result: [Felt; WORD_SIZE];
78                result = (*word).into();
79                result[idx as usize] = element;
80                *word = result.into();
81            })
82            .or_insert_with(|| {
83                let mut word = [ZERO; WORD_SIZE];
84                word[idx as usize] = element;
85                word.into()
86            });
87
88        Ok(())
89    }
90
91    /// Writes a word to memory starting at the provided address in the provided context.
92    ///
93    /// # Errors
94    /// - Returns an error if the provided address is out-of-bounds or not word-aligned.
95    #[inline(always)]
96    pub fn write_word(
97        &mut self,
98        ctx: ContextId,
99        addr: Felt,
100        _clk: RowIndex,
101        word: Word,
102    ) -> Result<(), MemoryError> {
103        let addr = enforce_word_aligned_addr(ctx, clean_addr(addr)?)?;
104        self.memory.insert((ctx, addr), word);
105
106        Ok(())
107    }
108
109    /// Returns the entire memory state for the specified execution context.
110    ///
111    /// The state is returned as a vector of (address, value) tuples, and includes addresses which
112    /// have been accessed at least once.
113    pub fn get_memory_state(&self, ctx: ContextId) -> Vec<(MemoryAddress, Felt)> {
114        self.memory
115            .iter()
116            .filter(|((c, _), _)| *c == ctx)
117            .flat_map(|(&(_c, addr), word)| {
118                let addr: MemoryAddress = addr.into();
119                [
120                    (addr, word[0]),
121                    (addr + 1_u32, word[1]),
122                    (addr + 2_u32, word[2]),
123                    (addr + 3_u32, word[3]),
124                ]
125            })
126            .collect()
127    }

132    /// Reads an element from memory at the provided address in the provided context.
133    ///
134    /// # Returns
135    /// - The element at the provided address, if it was written previously.
136    /// - `None` if the memory was not written previously.
137    pub(crate) fn read_element_impl(&self, ctx: ContextId, addr: u32) -> Option<Felt> {
138        let (word_addr, idx) = split_addr(addr);
139
140        self.memory.get(&(ctx, word_addr)).copied().map(|word| word[idx as usize])
141    }
142
143    /// Reads a word from memory starting at the provided address in the provided context.
144    ///
145    /// # Returns
146    /// - The word starting at the provided address, if it was written previously.
147    /// - `None` if the memory was not written previously.
148    #[inline(always)]
149    pub(crate) fn read_word_impl(
150        &self,
151        ctx: ContextId,
152        addr: u32,
153    ) -> Result<Option<Word>, MemoryError> {
154        let addr = enforce_word_aligned_addr(ctx, addr)?;
155        let word = self.memory.get(&(ctx, addr)).copied();
156
157        Ok(word)
158    }

173/// Converts the provided address to a `u32` if possible.
174///
175/// # Errors
176/// - Returns an error if the provided address is out-of-bounds.
177#[inline(always)]
178fn clean_addr(addr: Felt) -> Result<u32, MemoryError> {
179    let addr = addr.as_canonical_u64();
180    addr.try_into().map_err(|_| MemoryError::AddressOutOfBounds { addr })
181}
182
183/// Splits the provided address into the word address and the index within the word.
184///
185/// Returns a tuple of the word address and the index within the word.
186fn split_addr(addr: u32) -> (u32, u32) {
187    let idx = addr % WORD_SIZE as u32;
188    (addr - idx, idx)
189}
190
191/// Enforces that the provided address is word-aligned; that is, that it be divisible by 4 (in
192/// the integer sense).
193///
194/// Returns the address as a `u32` if it is word-aligned.
195///
196/// # Errors
197/// - Returns an error if the provided address is not word-aligned.
198/// - Returns an error if the provided address is out-of-bounds.
199#[inline(always)]
200fn enforce_word_aligned_addr(ctx: ContextId, addr: u32) -> Result<u32, MemoryError> {
201    if !addr.is_multiple_of(WORD_SIZE as u32) {
202        return Err(MemoryError::UnalignedWordAccess { addr, ctx });
203    }
204
205    Ok(addr)
206}
