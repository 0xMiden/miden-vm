345fn inst_with_imm(name: &'static str, imm: &dyn PrettyPrint) -> Document {
346    use crate::prettier::*;
347
348    let imm = imm.render();
349
350    flatten(const_text(name) + const_text(".") + imm)
351}
352
353fn inst_with_felt_imm(name: &'static str, imm: &Immediate<crate::Felt>) -> Document {
354    use crate::prettier::*;
355
356    let value = match imm {
357        Immediate::Value(value) => display(*value),
358        Immediate::Constant(name) => text(name),
359    };
360
361    flatten(const_text(name) + const_text(".") + value)
362}
363
364fn inst_with_pretty_felt_params(inst: &'static str, params: &[crate::Felt]) -> Document {
365    use crate::prettier::*;
366
367    let single_line = text(inst)
368        + const_text(".")
369        + params
370            .iter()
371            .copied()
372            .map(display)
373            .reduce(|acc, doc| acc + const_text(".") + doc)
374            .unwrap_or_default();
375
376    let multi_line = params
377        .iter()
378        .copied()
379        .map(|v| text(inst) + const_text(".") + display(v))
380        .reduce(|acc, doc| acc + nl() + doc)
381        .unwrap_or_default();
382    single_line | multi_line
383}

388#[cfg(test)]
389mod tests {
390    use miden_core::crypto::hash::Poseidon2;
391    use miden_debug_types::Span;
392
393    use crate::{Felt, ast::*};
394
395    #[test]
396    fn test_instruction_display() {
397        let instruction = format!("{}", Instruction::Assert);
398        assert_eq!("assert", instruction);
399
400        let instruction = format!("{}", Instruction::Add);
401        assert_eq!("add", instruction);
402
403        let instruction = format!("{}", Instruction::AddImm(Felt::new(5).into()));
404        assert_eq!("add.5", instruction);
405
406        let instruction = format!("{}", Instruction::ExpBitLength(32));
407        assert_eq!("exp.u32", instruction);
408
409        let instruction = format!(
410            "{}",
411            Instruction::PushFeltList(vec![Felt::new(3), Felt::new(4), Felt::new(8), Felt::new(9)])
412        );
413        assert_eq!("push.3.4.8.9", instruction);
414        let instruction = format!(
415            "{}",
416            Instruction::Push(Immediate::Value(miden_debug_types::Span::unknown(
417                crate::parser::PushValue::Int(crate::parser::IntValue::U8(3))
418            )))
419        );
420        assert_eq!("push.3", instruction);
421
422        let digest = Poseidon2::hash(b"miden::core::math::u64::add");
423        let target = InvocationTarget::MastRoot(Span::unknown(digest));
424        let instruction = format!("{}", Instruction::Exec(target));
425        assert_eq!(
426            "exec.0x065c394c00227acff3545da5493cf1b79d9a9f5628db553d240edf8ef0cca04a",
427            instruction
428        );
429    }
