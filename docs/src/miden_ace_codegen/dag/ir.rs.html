19/// Node kinds in the DAG.
20///
21/// These nodes mirror the verifier expression tree after lowering:
22/// inputs are read via `InputKey`, constants are lifted into the DAG, and
23/// arithmetic nodes capture the evaluation order.
24#[derive(Debug, Clone, PartialEq, Eq, Hash)]
25pub enum NodeKind<EF> {
26    /// Layout-addressable input (public, OOD, aux, etc.).
27    Input(InputKey),
28    /// Constant extension-field value.
29    Constant(EF),
30    /// Addition node.
31    Add(NodeId, NodeId),
32    /// Subtraction node.
33    Sub(NodeId, NodeId),
34    /// Multiplication node.
35    Mul(NodeId, NodeId),
36    /// Negation node (modeled as 0 - x when emitting ops).
37    Neg(NodeId),
38}

40/// Precomputed periodic column data for DAG construction.
41#[derive(Debug, Clone)]
42pub struct PeriodicColumnData<EF> {
43    /// Per-column coefficient vectors (highest-degree first).
44    coeffs: Vec<Vec<EF>>,
45}
46
47impl<EF> PeriodicColumnData<EF> {
48    /// Convert periodic columns (evaluations) into coefficient form for DAG building.
49    ///
50    /// Applies an inverse DFT so the DAG can evaluate them at `z_k` inside the circuit.
51    pub fn from_periodic_columns<F>(periodic_columns: Vec<Vec<F>>) -> Self
52    where
53        F: TwoAdicField,
54        EF: From<F>,
55    {
56        if periodic_columns.is_empty() {
57            return Self { coeffs: Vec::new() };
58        }
59
60        let dft = NaiveDft;
61        let mut coeffs = Vec::with_capacity(periodic_columns.len());
62        for col in periodic_columns {
63            assert!(!col.is_empty(), "periodic column must not be empty");
64            assert!(col.len().is_power_of_two(), "periodic column length must be a power of two");
65            let values = dft.idft(col);
66            let coeff_row = values.into_iter().map(EF::from).collect();
67            coeffs.push(coeff_row);
68        }
69
70        Self { coeffs }
71    }
72
73    /// Number of periodic columns.
74    pub fn num_columns(&self) -> usize {
75        self.coeffs.len()
76    }
77
78    /// Maximum periodic column length (used to align powers).
79    pub fn max_period(&self) -> usize {
80        self.coeffs.iter().map(|c| c.len()).max().unwrap_or(0)
81    }
82
83    /// Iterate over the per-column coefficient vectors.
84    pub fn columns(&self) -> &[Vec<EF>] {
85        &self.coeffs
86    }
87}
