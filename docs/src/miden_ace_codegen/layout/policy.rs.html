5enum Alignment {

13struct LayoutPolicy {
14    public_values: Alignment,
15    vlpi: Alignment,
16    vlpi_stride: usize,
17    randomness: Alignment,
18    main: Alignment,
19    aux: Alignment,

26impl LayoutPolicy {
27    fn native() -> Self {
28        Self {
29            public_values: Alignment::Unaligned,
30            vlpi: Alignment::Unaligned,
31            vlpi_stride: 1,
32            randomness: Alignment::Unaligned,
33            main: Alignment::Unaligned,
34            aux: Alignment::Unaligned,
35            quotient: Alignment::Unaligned,
36            aux_bus_boundary: Alignment::Unaligned,
37            stark_vars: Alignment::Unaligned,
38            end_align: None,
39        }
40    }
41
42    fn masm() -> Self {
43        Self {
44            public_values: Alignment::QuadWord,
45            vlpi: Alignment::Word,
46            vlpi_stride: 2,
47            randomness: Alignment::Word,
48            main: Alignment::DoubleWord,
49            aux: Alignment::DoubleWord,
50            quotient: Alignment::DoubleWord,
51            aux_bus_boundary: Alignment::Word,
52            stark_vars: Alignment::Word,
53            end_align: Some(Alignment::Word),
54        }
55    }

58struct LayoutBuilder {
59    offset: usize,
60}
61
62impl LayoutBuilder {
63    fn new() -> Self {
64        Self { offset: 0 }
65    }
66
67    fn align(&mut self, alignment: Alignment) {
68        self.offset = self.offset.next_multiple_of(alignment as usize);
69    }
70
71    fn alloc(&mut self, width: usize, alignment: Alignment) -> InputRegion {
72        self.align(alignment);
73        let region = InputRegion { offset: self.offset, width };
74        self.offset += width;
75        region
76    }
77}

79impl InputLayout {
80    /// Build a native layout (no alignment/padding).
81    pub(crate) fn new(counts: InputCounts) -> Self {
82        Self::build_with_policy(counts, LayoutPolicy::native())
83    }
84
85    /// Build a MASM-compatible layout (alignment/padding enforced).
86    pub(crate) fn new_masm(counts: InputCounts) -> Self {
87        Self::build_with_policy(counts, LayoutPolicy::masm())
88    }
89
90    fn build_with_policy(counts: InputCounts, policy: LayoutPolicy) -> Self {
91        // Number of EF slots in the stark-vars block. Every ACE input slot is an
92        // extension-field element (QuadFelt). Some stark vars are base-field values
93        // embedded as (val, 0); see the slot table below for which is which.
94        const NUM_STARK_VARS: usize = 10;
95
96        let mut builder = LayoutBuilder::new();
97
98        let public_values = builder.alloc(counts.num_public, policy.public_values);
99        let vlpi_reductions = builder.alloc(counts.num_vlpi, policy.vlpi);
100        /// Number of randomness inputs (alpha + beta).
101        const NUM_RANDOMNESS_INPUTS: usize = 2;
102        let randomness = builder.alloc(NUM_RANDOMNESS_INPUTS, policy.randomness);
103        let (aux_rand_alpha, aux_rand_beta) = randomness::aux_rand_indices(randomness);
104        let main_curr = builder.alloc(counts.width, policy.main);
105        let aux_coord_width = counts.aux_width * EXT_DEGREE;
106        let aux_curr = builder.alloc(aux_coord_width, policy.aux);
107        let quotient_curr = builder.alloc(counts.num_quotient_chunks * EXT_DEGREE, policy.quotient);
108        let main_next = builder.alloc(counts.width, policy.main);
109        let aux_next = builder.alloc(aux_coord_width, policy.aux);
110        let quotient_next = builder.alloc(counts.num_quotient_chunks * EXT_DEGREE, policy.quotient);
111        let aux_bus_boundary = builder.alloc(counts.aux_width, policy.aux_bus_boundary);
112
113        let stark_vars = builder.alloc(NUM_STARK_VARS, policy.stark_vars);
114
115        // Matches utils::set_up_auxiliary_inputs_ace layout (EF slots).
116        //
117        // Extension-field values are grouped first (slots 0-6), then base-field
118        // values stored as (val, 0) in EF slots (slots 7-9).
119        //
120        //  Slot  Value               Field
121        //  ----  ------------------  -----
122        //   0    alpha               EF      Composition challenge (Horner multiplier)
123        //   1    z^N                 EF      Trace-length power (quotient deltas + vanishing
124        //   2    z_k                 EF      Periodic column eval point
125        //   3    is_first            EF      Precomputed: (z^N - 1) / (z - 1)
126        //   4    is_last             EF      Precomputed: (z^N - 1) / (z - g^{-1})
127        //   5    is_transition       EF      Precomputed: z - g^{-1}
128        //   6    gamma               EF      Batching challenge
129        //   7    weight0             base    First barycentric weight
130        //   8    f                   base    Chunk shift ratio h^N
131        //   9    s0                  base    First coset shift offset^N
132        let b = stark_vars.offset;
133        let alpha = b;
134        let z_pow_n = b + 1;
135        let z_k = b + 2;
136        let is_first = b + 3;
137        let is_last = b + 4;
138        let is_transition = b + 5;
139        let gamma = b + 6;
140        let weight0 = b + 7;
141        let f = b + 8;
142        let s0 = b + 9;
143
144        if let Some(end_align) = policy.end_align {
145            builder.align(end_align);
146        }
147
148        Self {
149            regions: LayoutRegions {
150                public_values,
151                vlpi_reductions,
152                randomness,
153                main_curr,
154                aux_curr,
155                quotient_curr,
156                main_next,
157                aux_next,
158                quotient_next,
159                aux_bus_boundary,
160                stark_vars,
161            },
162            aux_rand_alpha,
163            aux_rand_beta,
164            vlpi_stride: policy.vlpi_stride,
165            stark: StarkVarIndices {
166                alpha,
167                z_pow_n,
168                z_k,
169                is_first,
170                is_last,
171                is_transition,
172                gamma,
173                weight0,
174                f,
175                s0,
176            },
177            total_inputs: builder.offset,
178            counts,
179        }
180    }
181}

183#[cfg(test)]
184mod tests {
185    use super::super::{InputCounts, InputKey, InputLayout};
186
187    #[test]
188    fn masm_layout_vlpi_groups_use_word_stride() {
189        let counts = InputCounts {
190            width: 1,
191            aux_width: 1,
192            num_public: 8,
193            // Two logical VLPI groups in MASM occupy four EF slots total:
194            // [group0, pad0, group1, pad1].
195            num_vlpi: 4,
196            num_randomness: 2,
197            num_periodic: 0,
198            num_quotient_chunks: 1,
199        };
200        let layout = InputLayout::new_masm(counts);
201
202        let vlpi_base = layout.index(InputKey::VlpiReduction(0)).unwrap();
203        assert_eq!(layout.index(InputKey::VlpiReduction(0)), Some(vlpi_base));
204        assert_eq!(
205            layout.index(InputKey::VlpiReduction(1)),
206            Some(vlpi_base + 2),
207            "MASM VLPI groups should advance by a word-aligned stride"
208        );
209    }
210
211    #[test]
212    fn native_layout_vlpi_groups_use_unit_stride() {
213        let counts = InputCounts {
214            width: 1,
215            aux_width: 1,
216            num_public: 8,
217            num_vlpi: 2,
218            num_randomness: 2,
219            num_periodic: 0,
220            num_quotient_chunks: 1,
221        };
222        let layout = InputLayout::new(counts);
223
224        let vlpi_base = layout.index(InputKey::VlpiReduction(0)).unwrap();
225        assert_eq!(
226            layout.index(InputKey::VlpiReduction(1)),
227            Some(vlpi_base + 1),
228            "Native VLPI groups should advance by unit stride"
