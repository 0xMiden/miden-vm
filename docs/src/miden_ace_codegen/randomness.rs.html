1//! Randomness helpers for ACE layout building and DAG lowering.
2//!
3//! The ACE circuit expands (alpha, beta) into the full coefficient
4//! vector `[alpha, 1, beta, beta^2, ...]` used by the constraint system.
5
6use miden_crypto::field::Field;
7
8use crate::{
9    dag::{DagBuilder, NodeId},
10    layout::{InputKey, InputLayout, InputRegion},
11};
12
13/// Derive alpha/beta input indices from a randomness region.
14///
15/// Returns `(alpha_idx, beta_idx)` matching the memory-word layout `[beta, alpha]`.
16pub(crate) fn aux_rand_indices(randomness: InputRegion) -> (usize, usize) {
17    // Stored as [beta, alpha] in memory words.
18    let beta_idx = randomness.index(0).expect("randomness region must have at least 2 slots");
19    let alpha_idx = randomness.index(1).expect("randomness region must have at least 2 slots");
20    (alpha_idx, beta_idx)
21}
22
23/// Lower a challenge index into DAG nodes.
24///
25/// Challenge indices map to:
26/// - 0 → alpha
27/// - 1 → 1 (beta^0)
28/// - n → beta^(n-1)
29pub(crate) fn lower_challenge<EF>(
30    builder: &mut DagBuilder<EF>,
31    layout: &InputLayout,
32    index: usize,
33) -> NodeId
34where
35    EF: Field,
36{
37    let num = layout.counts.num_randomness;
38    assert!(index < num, "challenge index {index} out of range (num={num})");
39
40    if index == 0 {
41        return builder.input(InputKey::AuxRandAlpha);
42    }
43    if index == 1 {
44        return builder.constant(EF::ONE);
45    }
46    let beta_node = builder.input(InputKey::AuxRandBeta);
47    let mut power = beta_node;
48    for _ in 2..index {
49        power = builder.mul(power, beta_node);
50    }
51    power
52}
