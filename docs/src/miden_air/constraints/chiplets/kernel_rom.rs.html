1//! Kernel ROM chiplet constraints.
2//!
3//! The kernel ROM chiplet exposes the digest table used by kernel (system) calls.
4//! This module only enforces shape constraints (binary selectors, digest contiguity, and the
5//! start-of-block marker). Validity of syscall selection is enforced by decoder and chiplets' bus.
6//!
7//! ## Column layout (5 columns within chiplet)
8//!
9//! | Column  | Purpose                                           |
10//! |---------|---------------------------------------------------|
11//! | sfirst  | 1 = first row of digest block, 0 = continuation   |
12//! | r0-r3   | 4-element kernel procedure digest                 |
13//!
14//! ## Operations
15//!
16//! - **KERNELPROCINIT** (sfirst=1): first row of a digest block (public input digests)
17//! - **KERNELPROCCALL** (sfirst=0): continuation rows used by SYSCALL
18//!
19//! ## Constraints
20//!
21//! 1. sfirst must be binary
22//! 2. Digest contiguity: when sfirst'=0 and s4'=0, digest values stay the same
23//! 3. First row: when entering kernel ROM, sfirst' must be 1
24

42const SFIRST_IDX: usize = 0;

44const R1_IDX: usize = 2;
45const R2_IDX: usize = 3;
46const R3_IDX: usize = 4;

54const KERNEL_ROM_DIGEST_BASE_ID: usize = KERNEL_ROM_BASE_ID + 1;
55const KERNEL_ROM_FIRST_ROW_ID: usize = KERNEL_ROM_BASE_ID + 5;

59const KERNEL_ROM_FIRST_ROW_NAMESPACE: &str = "chiplets.kernel_rom.first_row.start";

63const KERNEL_ROM_FIRST_ROW_NAMES: [&str; 1] = [KERNEL_ROM_FIRST_ROW_NAMESPACE; 1];

81/// Enforce kernel ROM chiplet constraints.
82pub fn enforce_kernel_rom_constraints<AB>(
83    builder: &mut AB,
84    local: &MainTraceRow<AB::Var>,
85    next: &MainTraceRow<AB::Var>,
86) where
87    AB: TaggingAirBuilderExt<F = Felt>,
88{
89    // Chiplet selector columns; kernel ROM rows are selected by (s0..s4).
90    let s0: AB::Expr = local.chiplets[0].clone().into();
91    let s1: AB::Expr = local.chiplets[1].clone().into();
92    let s2: AB::Expr = local.chiplets[2].clone().into();
93    let s3: AB::Expr = local.chiplets[3].clone().into();
94    let s4: AB::Expr = local.chiplets[4].clone().into();
95    let s3_next: AB::Expr = next.chiplets[3].clone().into();
96    let s4_next: AB::Expr = next.chiplets[4].clone().into();
97
98    let kernel_rom_flag =
99        kernel_rom_chiplet_flag(s0.clone(), s1.clone(), s2.clone(), s3.clone(), s4.clone());
100    let ace_flag = ace_chiplet_flag(s0.clone(), s1.clone(), s2.clone(), s3.clone());
101
102    // Load kernel ROM columns (sfirst + 4-word digest).
103    let sfirst: AB::Expr = load_kernel_rom_col::<AB>(local, SFIRST_IDX);
104    let sfirst_next: AB::Expr = load_kernel_rom_col::<AB>(next, SFIRST_IDX);
105    let r0: AB::Expr = load_kernel_rom_col::<AB>(local, R0_IDX);
106    let r0_next: AB::Expr = load_kernel_rom_col::<AB>(next, R0_IDX);
107    let r1: AB::Expr = load_kernel_rom_col::<AB>(local, R1_IDX);
108    let r1_next: AB::Expr = load_kernel_rom_col::<AB>(next, R1_IDX);
109    let r2: AB::Expr = load_kernel_rom_col::<AB>(local, R2_IDX);
110    let r2_next: AB::Expr = load_kernel_rom_col::<AB>(next, R2_IDX);
111    let r3: AB::Expr = load_kernel_rom_col::<AB>(local, R3_IDX);
112    let r3_next: AB::Expr = load_kernel_rom_col::<AB>(next, R3_IDX);
113
114    let one: AB::Expr = AB::Expr::ONE;
115
116    // Gate transition constraints by is_transition() to avoid last-row access.
117    // ==========================================================================
118    // SELECTOR CONSTRAINT
119    // ==========================================================================
120
121    // sfirst must be binary
122    let mut idx = 0;
123    tagged_assert_zero_integrity(
124        builder,
125        &KERNEL_ROM_SFIRST_TAGS,
126        &mut idx,
127        kernel_rom_flag.clone() * sfirst.clone() * (sfirst.clone() - one.clone()),
128    );
129
130    // ==========================================================================
131    // DIGEST CONTIGUITY CONSTRAINTS
132    // ==========================================================================
133
134    // When sfirst' = 0 (not the start of a new digest block) and s4' = 0 (not exiting kernel ROM),
135    // the digest values must remain unchanged.
136    let not_exiting = one.clone() - s4_next.clone();
137    let not_new_block = one.clone() - sfirst_next.clone();
138    let contiguity_condition = not_exiting * not_new_block;
139
140    // Use a combined gate to share `kernel_rom_flag * contiguity_condition` across all 4 lanes.
141    let gate = kernel_rom_flag * contiguity_condition;
142    let mut idx = 0;
143    tagged_assert_zero(builder, &KERNEL_ROM_DIGEST_TAGS, &mut idx, gate.clone() * (r0_next - r0));
144    tagged_assert_zero(builder, &KERNEL_ROM_DIGEST_TAGS, &mut idx, gate.clone() * (r1_next - r1));
145    tagged_assert_zero(builder, &KERNEL_ROM_DIGEST_TAGS, &mut idx, gate.clone() * (r2_next - r2));
146    tagged_assert_zero(builder, &KERNEL_ROM_DIGEST_TAGS, &mut idx, gate * (r3_next - r3));
147
148    // ==========================================================================
149    // FIRST ROW CONSTRAINT
150    // ==========================================================================
151
152    // s0..s2 are stable once 1 (selector constraints), so ACE -> kernel ROM transition is
153    // determined by s3' = 1 and s4' = 0.
154    let kernel_rom_next = s3_next * (one.clone() - s4_next.clone());
155    let flag_next_row_first_kernel_rom = ace_flag * kernel_rom_next;
156
157    // First row of kernel ROM must have sfirst' = 1.
158    let mut idx = 0;
159    tagged_assert_zero(
160        builder,
161        &KERNEL_ROM_FIRST_ROW_TAGS,
162        &mut idx,
163        flag_next_row_first_kernel_rom * (sfirst_next - one),
164    );
165}

170/// Load a column from the kernel ROM section of chiplets.
171fn load_kernel_rom_col<AB>(row: &MainTraceRow<AB::Var>, col_idx: usize) -> AB::Expr
172where
173    AB: TaggingAirBuilderExt<F = Felt>,
174{
175    // Kernel ROM columns start after s0, s1, s2, s3, s4 (5 selectors)
176    let local_idx = KERNEL_ROM_OFFSET + col_idx;
177    row.chiplets[local_idx].clone().into()
178}
