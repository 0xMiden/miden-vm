1//! Hash kernel virtual table bus constraint (`b_hash_kernel`).
2//!
3//! This module enforces a single running-product column (aux index 5) which aggregates three
4//! logically separate tables:
5//!
6//! 1. **Sibling table** for Merkle root updates (hasher chiplet).
7//! 2. **ACE memory reads** (ACE chiplet requests; memory chiplet responses on `b_chiplets`).
8//! 3. **Log-precompile transcript** (capacity state transitions for LOGPRECOMPILE).
9//!
10//! Rows contribute either a request term, a response term, or the identity (when no flag is set).
11//! The request/response values use the standard message format:
12//! `alpha + sum_i beta^i * element[i]`.
13

47// Column offsets relative to chiplets array.
48const S_START: usize = HASHER_SELECTOR_COL_RANGE.start - CHIPLETS_OFFSET;
49const H_START: usize = HASHER_STATE_COL_RANGE.start - CHIPLETS_OFFSET;
50const IDX_COL: usize = HASHER_NODE_INDEX_COL_IDX - CHIPLETS_OFFSET;

52/// Tag ID and namespace for the hash-kernel (virtual table) bus transition constraint.
53const HASH_KERNEL_BUS_ID: usize = TAG_HASH_KERNEL_BUS_BASE;
54const HASH_KERNEL_BUS_NAMESPACE: &str = "chiplets.bus.hash_kernel.transition";
55const HASH_KERNEL_BUS_NAMES: [&str; 1] = [HASH_KERNEL_BUS_NAMESPACE; 1];
56const HASH_KERNEL_BUS_TAGS: TagGroup = TagGroup {
57    base: HASH_KERNEL_BUS_ID,
58    names: &HASH_KERNEL_BUS_NAMES,

64/// Enforces the hash kernel virtual table (b_hash_kernel) bus constraint.
65///
66/// This constraint combines:
67/// 1. Sibling table for Merkle root updates
68/// 2. ACE memory read requests
69/// 3. Log precompile transcript tracking
70pub fn enforce_hash_kernel_constraint<AB>(
71    builder: &mut AB,
72    local: &MainTraceRow<AB::Var>,
73    next: &MainTraceRow<AB::Var>,
74    op_flags: &OpFlags<AB::Expr>,
75    challenges: &Challenges<AB::ExprEF>,
76) where
77    AB: TaggingAirBuilderExt<F = Felt>,
78{
79    // =========================================================================
80    // AUXILIARY TRACE ACCESS
81    // =========================================================================
82
83    let (p_local, p_next) = {
84        let aux = builder.permutation();
85        let aux_local = aux.current_slice();
86        let aux_next = aux.next_slice();
87        (aux_local[B_HASH_KERNEL], aux_next[B_HASH_KERNEL])
88    };
89
90    let one = AB::Expr::ONE;
91    let one_ef = AB::ExprEF::ONE;
92
93    // =========================================================================
94    // PERIODIC VALUES
95    // =========================================================================
96
97    let (cycle_row_0, cycle_row_31) = {
98        // Clone only the periodic values we need (avoids per-eval `to_vec()` allocation).
99        let p = builder.periodic_values();
100        let cycle_row_0: AB::Expr = p[periodic::P_CYCLE_ROW_0].into();
101        let cycle_row_31: AB::Expr = p[periodic::P_CYCLE_ROW_31].into();
102        (cycle_row_0, cycle_row_31)
103    };
104
105    // =========================================================================
106    // COMMON VALUES
107    // =========================================================================
108
109    // Hasher chiplet rows have s0 = 0 (chiplet selector).
110    let chiplet_selector: AB::Expr = local.chiplets[0].clone().into();
111    let is_hasher: AB::Expr = one.clone() - chiplet_selector.clone();
112
113    // Hasher operation selectors (only meaningful within hasher chiplet)
114    let s0: AB::Expr = local.chiplets[S_START].clone().into();
115    let s1: AB::Expr = local.chiplets[S_START + 1].clone().into();
116    let s2: AB::Expr = local.chiplets[S_START + 2].clone().into();
117
118    // Node index for sibling table
119    let node_index: AB::Expr = local.chiplets[IDX_COL].clone().into();
120    let node_index_next: AB::Expr = next.chiplets[IDX_COL].clone().into();
121
122    // Hasher state for sibling values
123    let h: [AB::Expr; 12] = core::array::from_fn(|i| local.chiplets[H_START + i].clone().into());
124    let h_next: [AB::Expr; 12] =
125        core::array::from_fn(|i| next.chiplets[H_START + i].clone().into());
126
127    // =========================================================================
128    // SIBLING TABLE FLAGS AND VALUES
129    // =========================================================================
130
131    // MU/MUA flags (requests - remove siblings during new path).
132    let f_mu: AB::Expr =
133        is_hasher.clone() * flags::f_mu(cycle_row_0.clone(), s0.clone(), s1.clone(), s2.clone());
134    let f_mua: AB::Expr =
135        is_hasher.clone() * flags::f_mua(cycle_row_31.clone(), s0.clone(), s1.clone(), s2.clone());
136
137    // MV/MVA flags (responses - add siblings during old path).
138    let f_mv: AB::Expr =
139        is_hasher.clone() * flags::f_mv(cycle_row_0.clone(), s0.clone(), s1.clone(), s2.clone());
140    let f_mva: AB::Expr = is_hasher.clone() * flags::f_mva(cycle_row_31.clone(), s0, s1, s2);
141
142    // Compute sibling values based on bit b (LSB of node index).
143    // The hasher constraints enforce that b is binary on shift rows.
144    let b: AB::Expr = node_index.clone() - node_index_next.clone().double();
145    let is_b_zero = one.clone() - b.clone();
146    let is_b_one = b;
147
148    // Sibling value for current row (uses current hasher state).
149    // b selects which half of the rate holds the sibling.
150    let v_sibling_curr = compute_sibling_b0::<AB>(challenges, &node_index, &h) * is_b_zero.clone()
151        + compute_sibling_b1::<AB>(challenges, &node_index, &h) * is_b_one.clone();
152
153    // Sibling value for next row (used by MVA/MUA on the transition row).
154    let v_sibling_next = compute_sibling_b0::<AB>(challenges, &node_index, &h_next) * is_b_zero
155        + compute_sibling_b1::<AB>(challenges, &node_index, &h_next) * is_b_one;
156
157    // =========================================================================
158    // ACE MEMORY FLAGS AND VALUES
159    // =========================================================================
160
161    // ACE chiplet selector: s0=1, s1=1, s2=1, s3=0
162    let s3: AB::Expr = local.chiplets[3].clone().into();
163    let chiplet_s1: AB::Expr = local.chiplets[1].clone().into();
164    let chiplet_s2: AB::Expr = local.chiplets[2].clone().into();
165
166    let is_ace_row: AB::Expr =
167        chiplet_selector.clone() * chiplet_s1.clone() * chiplet_s2.clone() * (one.clone() - s3);
168
169    // Block selector determines read (0) vs eval (1)
170    let block_selector: AB::Expr =
171        local.chiplets[NUM_ACE_SELECTORS + SELECTOR_BLOCK_IDX].clone().into();
172
173    let f_ace_read: AB::Expr = is_ace_row.clone() * (one.clone() - block_selector.clone());
174    let f_ace_eval: AB::Expr = is_ace_row * block_selector;
175
176    // ACE columns for memory messages
177    let ace_clk: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + CLK_IDX].clone().into();
178    let ace_ctx: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + CTX_IDX].clone().into();
179    let ace_ptr: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + PTR_IDX].clone().into();
180
181    // Word read value: label + ctx + ptr + clk + 4-lane value.
182    let v_ace_word = {
183        let v0_0: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + V_0_0_IDX].clone().into();
184        let v0_1: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + V_0_1_IDX].clone().into();
185        let v1_0: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + V_1_0_IDX].clone().into();
186        let v1_1: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + V_1_1_IDX].clone().into();
187        let label: AB::Expr = AB::Expr::from(Felt::from_u8(MEMORY_READ_WORD_LABEL));
188
189        challenges.encode([
190            label,
191            ace_ctx.clone(),
192            ace_ptr.clone(),
193            ace_clk.clone(),
194            v0_0,
195            v0_1,
196            v1_0,
197            v1_1,
198        ])
199    };
200
201    // Element read value: label + ctx + ptr + clk + element.
202    let v_ace_element = {
203        let id_1: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + ID_1_IDX].clone().into();
204        let id_2: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + ID_2_IDX].clone().into();
205        let eval_op: AB::Expr = local.chiplets[NUM_ACE_SELECTORS + EVAL_OP_IDX].clone().into();
206
207        let offset1: AB::Expr = AB::Expr::from(ACE_INSTRUCTION_ID1_OFFSET);
208        let offset2: AB::Expr = AB::Expr::from(ACE_INSTRUCTION_ID2_OFFSET);
209        let element = id_1 + id_2 * offset1 + (eval_op + one.clone()) * offset2;
210        let label: AB::Expr = AB::Expr::from(Felt::from_u8(MEMORY_READ_ELEMENT_LABEL));
211
212        challenges.encode([label, ace_ctx, ace_ptr, ace_clk, element])
213    };
214
215    // =========================================================================
216    // LOG PRECOMPILE FLAGS AND VALUES
217    // =========================================================================
218
219    let f_logprecompile: AB::Expr = op_flags.log_precompile();
220
221    // CAP_PREV from helper registers (provided and constrained by the decoder logic).
222    let cap_prev: [AB::Expr; 4] = core::array::from_fn(|i| {
223        local.decoder[USER_OP_HELPERS_OFFSET + HELPER_CAP_PREV_RANGE.start + i]
224            .clone()
225            .into()
226    });
227
228    // CAP_NEXT from next-row stack.
229    let cap_next: [AB::Expr; 4] =
230        core::array::from_fn(|i| next.stack[STACK_CAP_NEXT_RANGE.start + i].clone().into());
231
232    let log_label: AB::Expr = AB::Expr::from(Felt::from_u8(LOG_PRECOMPILE_LABEL));
233
234    // CAP_PREV value (request - removed).
235    let v_cap_prev = challenges.encode([
236        log_label.clone(),
237        cap_prev[0].clone(),
238        cap_prev[1].clone(),
239        cap_prev[2].clone(),
240        cap_prev[3].clone(),
241    ]);
242
243    // CAP_NEXT value (response - inserted).
244    let v_cap_next = challenges.encode([
245        log_label,
246        cap_next[0].clone(),
247        cap_next[1].clone(),
248        cap_next[2].clone(),
249        cap_next[3].clone(),
250    ]);
251
252    // =========================================================================
253    // RUNNING PRODUCT CONSTRAINT
254    // =========================================================================
255
256    // Include the identity term when no request/response flag is set on a row.
257    // Flags are mutually exclusive by construction (chiplet selectors + op flags).
258    let request_flag_sum = f_mu.clone()
259        + f_mua.clone()
260        + f_ace_read.clone()
261        + f_ace_eval.clone()
262        + f_logprecompile.clone();
263    let requests: AB::ExprEF = v_sibling_curr.clone() * f_mu.clone()
264        + v_sibling_next.clone() * f_mua.clone()
265        + v_ace_word * f_ace_read
266        + v_ace_element * f_ace_eval
267        + v_cap_prev * f_logprecompile.clone()
268        + (one_ef.clone() - request_flag_sum);
269
270    let response_flag_sum = f_mv.clone() + f_mva.clone() + f_logprecompile.clone();
271    let responses: AB::ExprEF = v_sibling_curr * f_mv
272        + v_sibling_next * f_mva
273        + v_cap_next * f_logprecompile
274        + (one_ef - response_flag_sum);
275
276    // Running product constraint: p' * requests = p * responses
277    let p_local_ef: AB::ExprEF = p_local.into();
278    let p_next_ef: AB::ExprEF = p_next.into();
279
280    let mut idx = 0;
281    tagged_assert_zero_ext(
282        builder,
283        &HASH_KERNEL_BUS_TAGS,
284        &mut idx,
285        p_next_ef * requests - p_local_ef * responses,
286    );
287}

292/// Sibling at h[4..7]: positions [2, 7, 8, 9, 10].
293const SIBLING_B0_LAYOUT: [usize; 5] = [2, 7, 8, 9, 10];
294/// Sibling at h[0..3]: positions [2, 3, 4, 5, 6].
295const SIBLING_B1_LAYOUT: [usize; 5] = [2, 3, 4, 5, 6];
296
297fn compute_sibling_b0<AB>(
298    challenges: &Challenges<AB::ExprEF>,
299    node_index: &AB::Expr,
300    h: &[AB::Expr; 12],
301) -> AB::ExprEF
302where
303    AB: LiftedAirBuilder<F = Felt>,
304{
305    challenges.encode_sparse(
306        SIBLING_B0_LAYOUT,
307        [node_index.clone(), h[4].clone(), h[5].clone(), h[6].clone(), h[7].clone()],
308    )
309}
310
311/// Compute sibling value when b=1 (sibling at h[0..3]).
312///
313/// Message layout: alpha[0] (constant) + alpha[3] * node_index + alpha[4..7] * h[0..3].
314fn compute_sibling_b1<AB>(
315    challenges: &Challenges<AB::ExprEF>,
316    node_index: &AB::Expr,
317    h: &[AB::Expr; 12],
318) -> AB::ExprEF
319where
320    AB: LiftedAirBuilder<F = Felt>,
321{
322    challenges.encode_sparse(
323        SIBLING_B1_LAYOUT,
324        [node_index.clone(), h[0].clone(), h[1].clone(), h[2].clone(), h[3].clone()],
325    )
