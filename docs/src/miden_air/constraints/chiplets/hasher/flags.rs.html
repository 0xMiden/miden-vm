1//! Hasher chiplet flag computation functions.
2//!
3//! This module provides functions to compute operation flags for the hasher chiplet.
4//! Each flag identifies when a specific operation is active based on selector values
5//! and cycle position.
6//!
7//! ## Unused Flags
8//!
9//! Some flags are defined but unused (`#[allow(dead_code)]`):
10//!
11//! - **`f_bp`**: BP (Begin Permutation) needs no special constraints - the round function
12//!   constraints apply identically regardless of which operation started it.
13//! - **`f_hout`, `f_sout`**: The combined `f_out` flag suffices for hasher constraints.
14//!
15//! ## Selector Encoding
16//!
17//! The hasher uses 3 selector columns `s[0..2]` to encode operations:
18//!
19//! | Operation | s0 | s1 | s2 | Cycle Position     | Description |
20//! |-----------|----|----|----|--------------------|-------------|
21//! | BP        | 1  | 0  | 0  | row 0              | Begin permutation |
22//! | MP        | 1  | 0  | 1  | row 0              | Merkle path verify |
23//! | MV        | 1  | 1  | 0  | row 0              | Merkle verify (old root) |
24//! | MU        | 1  | 1  | 1  | row 0              | Merkle update (new root) |
25//! | ABP       | 1  | 0  | 0  | row 31             | Absorb for linear hash |
26//! | MPA       | 1  | 0  | 1  | row 31             | Merkle path absorb |
27//! | MVA       | 1  | 1  | 0  | row 31             | Merkle verify absorb |
28//! | MUA       | 1  | 1  | 1  | row 31             | Merkle update absorb |
29//! | HOUT      | 0  | 0  | 0  | row 31             | Hash output (digest) |
30//! | SOUT      | 0  | 0  | 1  | row 31             | State output (full) |
31use miden_core::field::PrimeCharacteristicRing;
32
33// INTERNAL HELPERS
34// ================================================================================================
35
36// INITIALIZATION FLAGS (row 0 of 32-row cycle)
37// ================================================================================================

39/// BP: Begin Permutation flag `(1,0,0)` on cycle row 0.
40///
41/// Initiates single permutation, 2-to-1 hash, or linear hash computation.
42///
43/// # Degree
44/// - Periodic: 1 (cycle_row_0)
45/// - Selectors: 3 (s0 * !s1 * !s2)
46/// - Total: 4
47#[inline]
48#[allow(dead_code)]
49pub fn f_bp<E>(cycle_row_0: E, s0: E, s1: E, s2: E) -> E
50where
51    E: PrimeCharacteristicRing,
52{
53    cycle_row_0 * s0 * (E::ONE - s1) * (E::ONE - s2)
54}
55
56/// MP: Merkle Path verification flag `(1,0,1)` on cycle row 0.
57///
58/// Initiates standard Merkle path verification computation.
59///
60/// # Degree
61/// - Periodic: 1 (cycle_row_0)
62/// - Selectors: 3 (s0 * !s1 * s2)
63/// - Total: 4
64#[inline]
65pub fn f_mp<E>(cycle_row_0: E, s0: E, s1: E, s2: E) -> E
66where
67    E: PrimeCharacteristicRing,
68{
69    cycle_row_0 * s0 * (E::ONE - s1) * s2
70}
71
72/// MV: Merkle Verify (old root) flag `(1,1,0)` on cycle row 0.
73///
74/// Begins verification for old leaf value during Merkle root update.
75///
76/// # Degree
77/// - Periodic: 1 (cycle_row_0)
78/// - Selectors: 3 (s0 * s1 * !s2)
79/// - Total: 4
80#[inline]
81pub fn f_mv<E>(cycle_row_0: E, s0: E, s1: E, s2: E) -> E
82where
83    E: PrimeCharacteristicRing,
84{
85    cycle_row_0 * s0 * s1 * (E::ONE - s2)
86}
87
88/// MU: Merkle Update (new root) flag `(1,1,1)` on cycle row 0.
89///
90/// Starts verification for new leaf value during Merkle root update.
91///
92/// # Degree
93/// - Periodic: 1 (cycle_row_0)
94/// - Selectors: 3 (s0 * s1 * s2)
95/// - Total: 4
96#[inline]
97pub fn f_mu<E>(cycle_row_0: E, s0: E, s1: E, s2: E) -> E
98where
99    E: PrimeCharacteristicRing,
100{
101    cycle_row_0 * s0 * s1 * s2
102}
103
104// ================================================================================================
105// ABSORPTION FLAGS (row 31 of 32-row cycle)
106// ================================================================================================
107
108/// ABP: Absorb for linear hash flag `(1,0,0)` on cycle row 31.
109///
110/// Absorbs next set of elements into hasher state during linear hash computation.
111///
112/// # Degree
113/// - Periodic: 1 (cycle_row_31)
114/// - Selectors: 3 (s0 * !s1 * !s2)
115/// - Total: 4
116#[inline]
117pub fn f_abp<E>(cycle_row_31: E, s0: E, s1: E, s2: E) -> E
118where
119    E: PrimeCharacteristicRing,
120{
121    cycle_row_31 * s0 * (E::ONE - s1) * (E::ONE - s2)
122}
123
124/// MPA: Merkle Path Absorb flag `(1,0,1)` on cycle row 31.
125///
126/// Absorbs next Merkle path node during standard verification.
127///
128/// # Degree
129/// - Periodic: 1 (cycle_row_31)
130/// - Selectors: 3 (s0 * !s1 * s2)
131/// - Total: 4
132#[inline]
133pub fn f_mpa<E>(cycle_row_31: E, s0: E, s1: E, s2: E) -> E
134where
135    E: PrimeCharacteristicRing,
136{
137    cycle_row_31 * s0 * (E::ONE - s1) * s2
138}
139
140/// MVA: Merkle Verify Absorb flag `(1,1,0)` on cycle row 31.
141///
142/// Absorbs next node during "old" leaf verification (Merkle root update).
143///
144/// # Degree
145/// - Periodic: 1 (cycle_row_31)
146/// - Selectors: 3 (s0 * s1 * !s2)
147/// - Total: 4
148#[inline]
149pub fn f_mva<E>(cycle_row_31: E, s0: E, s1: E, s2: E) -> E
150where
151    E: PrimeCharacteristicRing,
152{
153    cycle_row_31 * s0 * s1 * (E::ONE - s2)
154}
155
156/// MUA: Merkle Update Absorb flag `(1,1,1)` on cycle row 31.
157///
158/// Absorbs next node during "new" leaf verification (Merkle root update).
159///
160/// # Degree
161/// - Periodic: 1 (cycle_row_31)
162/// - Selectors: 3 (s0 * s1 * s2)
163/// - Total: 4
164#[inline]
165pub fn f_mua<E>(cycle_row_31: E, s0: E, s1: E, s2: E) -> E
166where
167    E: PrimeCharacteristicRing,
168{
169    cycle_row_31 * s0 * s1 * s2
170}
171
172// ================================================================================================
173// OUTPUT FLAGS (row 31 of 32-row cycle)
174// ================================================================================================
175
176/// HOUT: Hash Output flag `(0,0,0)` on cycle row 31.
177///
178/// Returns the 4-element hash result (digest).
179///
180/// # Degree
181/// - Periodic: 1 (cycle_row_31)
182/// - Selectors: 3 (!s0 * !s1 * !s2)
183/// - Total: 4
184#[inline]
185#[allow(dead_code)]
186pub fn f_hout<E>(cycle_row_31: E, s0: E, s1: E, s2: E) -> E
187where
188    E: PrimeCharacteristicRing,
189{
190    cycle_row_31 * (E::ONE - s0) * (E::ONE - s1) * (E::ONE - s2)
191}
192
193/// SOUT: State Output flag `(0,0,1)` on cycle row 31.
194///
195/// Returns the entire 12-element hasher state.
196///
197/// # Degree
198/// - Periodic: 1 (cycle_row_31)
199/// - Selectors: 3 (!s0 * !s1 * s2)
200/// - Total: 4
201#[inline]
202#[allow(dead_code)]
203pub fn f_sout<E>(cycle_row_31: E, s0: E, s1: E, s2: E) -> E
204where
205    E: PrimeCharacteristicRing,
206{
207    cycle_row_31 * (E::ONE - s0) * (E::ONE - s1) * s2
208}
209
210/// Combined output flag: `f_hout | f_sout` = `(0,0,*)` on cycle row 31.
211///
212/// True when any output operation is active (HOUT or SOUT).
213///
214/// # Degree
215/// - Periodic: 1 (cycle_row_31)
216/// - Selectors: 2 (!s0 * !s1)
217/// - Total: 3
218#[inline]
219pub fn f_out<E>(cycle_row_31: E, s0: E, s1: E) -> E
220where
221    E: PrimeCharacteristicRing,
222{
223    cycle_row_31 * (E::ONE - s0) * (E::ONE - s1)
224}
225
226/// Lookahead output flag on cycle row 30.
227///
228/// True when the *next* row will be an output operation.
229/// Used for selector stability constraints.
230///
231/// # Degree
232/// - Periodic: 1 (cycle_row_30)
233/// - Next selectors: 2 (!s0' * !s1')
234/// - Total: 3
235#[inline]
236pub fn f_out_next<E>(cycle_row_30: E, s0_next: E, s1_next: E) -> E
237where
238    E: PrimeCharacteristicRing,
239{
240    cycle_row_30 * (E::ONE - s0_next) * (E::ONE - s1_next)
241}
242
243// ================================================================================================
244// COMPOSITE FLAGS
245// ================================================================================================
246
247/// Merkle operation active flag.
248///
249/// True when any Merkle operation (MP, MV, MU, MPA, MVA, MUA) is active.
250/// Used for gating index shift constraints.
251///
252/// # Degree
253/// - Depends on constituent flags, typically 4
254#[inline]
255pub fn f_merkle_active<E>(f_mp: E, f_mv: E, f_mu: E, f_mpa: E, f_mva: E, f_mua: E) -> E
256where
257    E: PrimeCharacteristicRing,
258{
259    f_mp + f_mv + f_mu + f_mpa + f_mva + f_mua
260}
261
262/// Merkle absorb flag (row 31 only).
263///
264/// True when absorbing the next node during Merkle path computation.
265///
266/// # Degree
267/// - Depends on constituent flags, typically 4
268#[inline]
269pub fn f_merkle_absorb<E>(f_mpa: E, f_mva: E, f_mua: E) -> E
270where
271    E: PrimeCharacteristicRing,
272{
273    f_mpa + f_mva + f_mua
274}
275
276/// Continuation flag for hashing operations.
277///
278/// True when operation continues to next cycle (ABP, MPA, MVA, MUA).
279/// Constrains s0' = 0 to ensure proper sequencing.
280///
281/// # Degree
282/// - Depends on constituent flags, typically 4
283#[inline]
284pub fn f_continuation<E>(f_abp: E, f_mpa: E, f_mva: E, f_mua: E) -> E
285where
286    E: PrimeCharacteristicRing,
287{
288    f_abp + f_mpa + f_mva + f_mua
289}
