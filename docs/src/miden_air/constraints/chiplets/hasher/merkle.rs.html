1//! Hasher chiplet Merkle path constraints.
2//!
3//! This module enforces constraints specific to Merkle tree operations:
4//!
5//! - **Index shifting**: Node index shifts right by 1 bit at absorb points
6//! - **Index stability**: Index unchanged outside of Merkle operations
7//! - **Capacity reset**: Capacity lanes reset to zero on Merkle absorb
8//! - **Digest placement**: Current digest placed in rate0 or rate1 based on direction bit
9//!
10//! ## Merkle Operations
11//!
12//! | Flag | Operation | Description |
13//! |------|-----------|-------------|
14//! | MP   | Merkle Path | Standard path verification |
15//! | MV   | Merkle Verify | Old root verification (for updates) |
16//! | MU   | Merkle Update | New root computation (for updates) |
17//! | MPA  | Merkle Path Absorb | Absorb next sibling (standard) |
18//! | MVA  | Merkle Verify Absorb | Absorb next sibling (old path) |
19//! | MUA  | Merkle Update Absorb | Absorb next sibling (new path) |
20

75/// Enforces node index constraints for Merkle operations.
76///
77/// ## Index Shift Constraint
78///
79/// On Merkle start (row 0) and absorb (row 31) operations, the index shifts right by 1 bit:
80/// `i' = floor(i/2)`. The discarded bit `b = i - 2*i'` must be binary.
81///
82/// This encodes the tree traversal from leaf to root.
83///
84/// ## Index Stability Constraint
85///
86/// Outside of shift rows (and output rows), the index must remain unchanged.
87pub(super) fn enforce_node_index_constraints<AB>(
88    builder: &mut AB,
89    hasher_flag: AB::Expr,
90    cols: &HasherColumns<AB::Expr>,
91    cols_next: &HasherColumns<AB::Expr>,
92    flags: &HasherFlags<AB::Expr>,
93) where
94    AB: TaggingAirBuilderExt<F = Felt>,
95{
96    let one: AB::Expr = AB::Expr::ONE;
97
98    // -------------------------------------------------------------------------
99    // Output Index Constraint
100    // -------------------------------------------------------------------------
101
102    // Constraint 1: Index must be 0 on output rows.
103    let mut idx = 0;
104    tagged_assert_zero_integrity(
105        builder,
106        &OUTPUT_INDEX_TAGS,
107        &mut idx,
108        hasher_flag.clone() * flags.f_out.clone() * cols.node_index.clone(),
109    );
110
111    // -------------------------------------------------------------------------
112    // Index Shift Constraint
113    // -------------------------------------------------------------------------
114
115    let f_shift = flags.f_merkle_active();
116    let f_out = flags.f_out.clone();
117
118    // Direction bit: b = i - 2*i'
119    // This is the bit discarded when shifting index right by 1.
120    let b = cols.node_index.clone() - AB::Expr::TWO * cols_next.node_index.clone();
121
122    // Constraint 2: b must be binary when shifting (b^2 - b = 0)
123    let gate = hasher_flag.clone() * f_shift.clone();
124    let mut idx = 0;
125    tagged_assert_zero(builder, &MERKLE_INDEX_TAGS, &mut idx, gate * (b.square() - b.clone()));
126
127    // -------------------------------------------------------------------------
128    // Index Stability Constraint
129    // -------------------------------------------------------------------------
130
131    // Constraint 3: Index unchanged when not shifting or outputting
132    // keep = 1 - f_out - f_shift
133    let keep = one.clone() - f_out - f_shift;
134    let gate = hasher_flag.clone() * keep;
135    tagged_assert_zero(
136        builder,
137        &MERKLE_INDEX_TAGS,
138        &mut idx,
139        gate * (cols_next.node_index.clone() - cols.node_index.clone()),
140    );
141}

143/// Enforces state constraints for Merkle absorb operations (MPA/MVA/MUA on row 31).
144///
145/// ## Capacity Reset
146///
147/// The capacity lanes `h[8..12]` are reset to zero for the next 2-to-1 compression.
148///
149/// ## Digest Placement
150///
151/// The current digest `h[0..4]` is copied to either rate0 or rate1 based on direction bit `b`:
152/// - If `b=0`: digest goes to rate0 (`h'[0..4] = h[0..4]`), sibling to rate1 (witness)
153/// - If `b=1`: digest goes to rate1 (`h'[4..8] = h[0..4]`), sibling to rate0 (witness)
154pub(super) fn enforce_merkle_absorb_state<AB>(
155    builder: &mut AB,
156    hasher_flag: AB::Expr,
157    cols: &HasherColumns<AB::Expr>,
158    cols_next: &HasherColumns<AB::Expr>,
159    flags: &HasherFlags<AB::Expr>,
160) where
161    AB: TaggingAirBuilderExt<F = Felt>,
162{
163    let one: AB::Expr = AB::Expr::ONE;
164    let f_absorb = flags.f_merkle_absorb();
165
166    let digest = cols.digest();
167    let rate0_next = cols_next.rate0();
168    let rate1_next = cols_next.rate1();
169    let cap_next = cols_next.capacity();
170
171    // Direction bit: b = i - 2*i'
172    let b = cols.node_index.clone() - AB::Expr::TWO * cols_next.node_index.clone();
173
174    // -------------------------------------------------------------------------
175    // Capacity Reset Constraint
176    // -------------------------------------------------------------------------
177
178    // Constraint 1: Capacity reset to zero (batched).
179    // Use a combined gate to share `hasher_flag * f_absorb` across all 4 lanes.
180    let gate_absorb = hasher_flag.clone() * f_absorb.clone();
181    let mut idx = 0;
182    tagged_assert_zeros(
183        builder,
184        &MERKLE_ABSORB_TAGS,
185        &mut idx,
186        MERKLE_CAP_NAMESPACE,
187        core::array::from_fn::<_, 4, _>(|i| gate_absorb.clone() * cap_next[i].clone()),
188    );
189
190    // -------------------------------------------------------------------------
191    // Digest Placement Constraints
192    // -------------------------------------------------------------------------
193
194    // Constraint 2: If b=0, digest goes to rate0 (h'[0..4] = h[0..4])
195    let f_b0 = f_absorb.clone() * (one.clone() - b.clone());
196    let gate_b0 = hasher_flag.clone() * f_b0;
197    tagged_assert_zeros(
198        builder,
199        &MERKLE_ABSORB_TAGS,
200        &mut idx,
201        MERKLE_RATE0_NAMESPACE,
202        core::array::from_fn::<_, 4, _>(|i| {
203            gate_b0.clone() * (rate0_next[i].clone() - digest[i].clone())
204        }),
205    );
206
207    // Constraint 3: If b=1, digest goes to rate1 (h'[4..8] = h[0..4])
208    let f_b1 = f_absorb * b;
209    let gate_b1 = hasher_flag * f_b1;
210    tagged_assert_zeros(
211        builder,
212        &MERKLE_ABSORB_TAGS,
213        &mut idx,
214        MERKLE_RATE1_NAMESPACE,
215        core::array::from_fn::<_, 4, _>(|i| {
216            gate_b1.clone() * (rate1_next[i].clone() - digest[i].clone())
217        }),
218    );
