1//! Hasher chiplet selector consistency constraints.
2//!
3//! This module enforces constraints on the selector columns `s[0..2]` that control
4//! hasher operation modes. These constraints ensure:
5//!
6//! 1. **Booleanity**: Selector values are binary (0 or 1)
7//! 2. **Stability**: Selectors remain unchanged except at cycle boundaries
8//! 3. **Sequencing**: After absorb operations, computation continues properly
9//! 4. **Validity**: Invalid selector combinations are rejected
10//!
11//! ## Selector Encodings on Row 31
12//!
13//! | Operation | s0 | s1 | s2 | Description |
14//! |-----------|----|----|----|--------------------|
15//! | ABP       | 1  | 0  | 0  | Absorb for linear hash |
16//! | HOUT      | 0  | 0  | 0  | Output digest |
17//! | SOUT      | 0  | 0  | 1  | Output full state |
18//! | MPA       | 1  | 0  | 1  | Merkle path absorb |
19//! | MVA       | 1  | 1  | 0  | Merkle verify absorb |
20//! | MUA       | 1  | 1  | 1  | Merkle update absorb |
21

61/// Enforces selector consistency constraints for the hasher chiplet.
62///
63/// ## Constraints
64///
65/// 1. **Selector stability**: `s[1]` and `s[2]` must remain unchanged across rows, except:
66///    - On output rows (HOUT/SOUT on row 31), where selectors can change for the next cycle.
67///    - On lookahead rows (row 30 when next row is output), to prepare output selectors.
68///
69/// 2. **Continuation sequencing**: After ABP/MPA/MVA/MUA (absorb operations on row 31), the next
70///    cycle must continue hashing, so `s[0]' = 0`.
71///
72/// 3. **Invalid combination rejection**: On row 31, if `s[0]=0` then `s[1]` must also be 0. This
73///    prevents invalid selector states like `(0,1,0)` or `(0,1,1)`.
74pub(super) fn enforce_selector_consistency<AB>(
75    builder: &mut AB,
76    hasher_flag: AB::Expr,
77    cols: &HasherColumns<AB::Expr>,
78    cols_next: &HasherColumns<AB::Expr>,
79    flags: &HasherFlags<AB::Expr>,
80) where
81    AB: TaggingAirBuilderExt<F = Felt>,
82{
83    let one: AB::Expr = AB::Expr::ONE;
84
85    // -------------------------------------------------------------------------
86    // Constraint 1: Selector stability
87    // -------------------------------------------------------------------------
88    // s[1] and s[2] unchanged unless f_out or f_out_next.
89    // Constraint: (1 - f_out - f_out_next) * (s[i]' - s[i]) = 0
90    // Note: f_out and f_out_next are mutually exclusive (row30 vs row31), so no overlap.
91    let stability_gate = one.clone() - flags.f_out.clone() - flags.f_out_next.clone();
92
93    // Use a combined gate to share `hasher_flag * stability_gate` across both stability
94    // constraints.
95    let gate = hasher_flag.clone() * stability_gate;
96    let mut idx = 0;
97    tagged_assert_zeros(
98        builder,
99        &SELECTOR_CONSIST_TAGS,
100        &mut idx,
101        SELECTOR_STABILITY_NAMESPACE,
102        [
103            gate.clone() * (cols_next.s1.clone() - cols.s1.clone()),
104            gate * (cols_next.s2.clone() - cols.s2.clone()),
105        ],
106    );
107
108    // Continuation constraint: hasher_flag * flag_cont * s0' = 0.
109    // (Single constraint, so no batching benefit beyond using `.when(gate)`.)
110    let gate = hasher_flag.clone() * flags.f_continuation();
111    tagged_assert_zero(builder, &SELECTOR_CONSIST_TAGS, &mut idx, gate * cols_next.s0.clone());
112
113    // -------------------------------------------------------------------------
114    // Constraint 3: Invalid selector combinations rejection
115    // -------------------------------------------------------------------------
116    // On row31, if s0 = 0 then s1 must be 0. This prevents (0,1,*) combinations.
117    // Constraint: row31 * (1 - s0) * s1 = 0
118    tagged_assert_zero_integrity(
119        builder,
120        &SELECTOR_CONSIST_TAGS,
121        &mut idx,
122        hasher_flag
123            * flags.cycle_row_31.clone()
124            * (one.clone() - cols.s0.clone())
125            * cols.s1.clone(),
126    );
127}

129/// Enforces that selector columns are binary.
130///
131/// This is called from the main permutation constraint with the step gate.
132pub fn enforce_selector_booleanity<AB>(
133    builder: &mut AB,
134    hasher_flag: AB::Expr,
135    s0: AB::Var,
136    s1: AB::Var,
137    s2: AB::Var,
138) where
139    AB: TaggingAirBuilderExt<F = Felt>,
140{
141    let s0: AB::Expr = s0.into();
142    let s1: AB::Expr = s1.into();
143    let s2: AB::Expr = s2.into();
144    let mut idx = 0;
145    tagged_assert_zeros_integrity(
146        builder,
147        &SELECTOR_BOOL_TAGS,
148        &mut idx,
149        SELECTOR_BOOL_NAMESPACE,
150        [
151            hasher_flag.clone() * s0.clone() * (s0 - AB::Expr::ONE),
152            hasher_flag.clone() * s1.clone() * (s1 - AB::Expr::ONE),
153            hasher_flag * s2.clone() * (s2 - AB::Expr::ONE),
154        ],
155    );
