1//! Chiplet selector system constraints.
2//!
3//! This module implements the hierarchical chiplet selector system that determines
4//! which chiplet is active at any given row.
5//!
6//! ## Selector Hierarchy
7//!
8//! The chiplet system uses 5 selector columns `s[0..4]` to identify active chiplets:
9//!
10//! | Chiplet     | Active when                    | Selector pattern |
11//! |-------------|--------------------------------|------------------|
12//! | Hasher      | `!s0`                          | `(0, *, *, *, *)` |
13//! | Bitwise     | `s0 * !s1`                     | `(1, 0, *, *, *)` |
14//! | Memory      | `s0 * s1 * !s2`                | `(1, 1, 0, *, *)` |
15//! | ACE         | `s0 * s1 * s2 * !s3`           | `(1, 1, 1, 0, *)` |
16//! | Kernel ROM  | `s0 * s1 * s2 * s3 * !s4`      | `(1, 1, 1, 1, 0)` |
17//!
18//! ## Constraints
19//!
20//! 1. **Binary constraints**: Each selector is binary when it could be active
21//! 2. **Stability constraints**: Once a selector becomes 1, it stays 1 (no 1→0 transitions)
22
23use miden_core::field::PrimeCharacteristicRing;
24use miden_crypto::stark::air::{AirBuilder, LiftedAirBuilder};
25
26use crate::{
27    Felt, MainTraceRow,
28    constraints::tagging::{TaggingAirBuilderExt, ids::TAG_CHIPLETS_BASE},
29};

31// TAGGING IDS
32// ================================================================================================
33
34/// Base ID for chiplet selector constraints.
35const CHIPLET_SELECTORS_BASE_ID: usize = TAG_CHIPLETS_BASE;
36
37/// Constraint namespaces in assertion order.
38const CHIPLET_SELECTORS_NAMES: [&str; 10] = [
39    "chiplets.selectors.s0.binary",
40    "chiplets.selectors.s1.binary",
41    "chiplets.selectors.s2.binary",
42    "chiplets.selectors.s3.binary",
43    "chiplets.selectors.s4.binary",
44    "chiplets.selectors.s0.stability",
45    "chiplets.selectors.s1.stability",
46    "chiplets.selectors.s2.stability",
47    "chiplets.selectors.s3.stability",
48    "chiplets.selectors.s4.stability",
49];

52// ================================================================================================

54/// Enforce chiplet selector constraints.
55///
56/// This enforces:
57/// 1. Binary constraints for each selector level
58/// 2. Stability constraints (no 1→0 transitions)
59pub fn enforce_chiplet_selectors<AB>(
60    builder: &mut AB,
61    local: &MainTraceRow<AB::Var>,
62    next: &MainTraceRow<AB::Var>,
63) where
64    AB: LiftedAirBuilder<F = Felt>,
65{
66    // Load selector columns (chiplets[0..5] are the selectors)
67    let s0: AB::Expr = local.chiplets[0].clone().into();
68    let s1: AB::Expr = local.chiplets[1].clone().into();
69    let s2: AB::Expr = local.chiplets[2].clone().into();
70    let s3: AB::Expr = local.chiplets[3].clone().into();
71    let s4: AB::Expr = local.chiplets[4].clone().into();
72
73    let s0_next: AB::Expr = next.chiplets[0].clone().into();
74    let s1_next: AB::Expr = next.chiplets[1].clone().into();
75    let s2_next: AB::Expr = next.chiplets[2].clone().into();
76    let s3_next: AB::Expr = next.chiplets[3].clone().into();
77    let s4_next: AB::Expr = next.chiplets[4].clone().into();
78
79    let one: AB::Expr = AB::Expr::ONE;
80
81    // ==========================================================================
82    // BINARY CONSTRAINTS
83    // ==========================================================================
84    // Each selector is binary when it could be active
85
86    // s0 is always binary
87    builder.tagged(CHIPLET_SELECTORS_BASE_ID, CHIPLET_SELECTORS_NAMES[0], |builder| {
88        builder.assert_zero(s0.clone() * (s0.clone() - one.clone()));
89    });
90
91    // s1 is binary when s0 = 1 (bitwise, memory, ACE, or kernel ROM could be active)
92    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 1, CHIPLET_SELECTORS_NAMES[1], |builder| {
93        builder.when(s0.clone()).assert_zero(s1.clone() * (s1.clone() - one.clone()));
94    });
95
96    // s2 is binary when s0 = 1 and s1 = 1 (memory, ACE, or kernel ROM could be active)
97    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 2, CHIPLET_SELECTORS_NAMES[2], |builder| {
98        builder
99            .when(s0.clone())
100            .when(s1.clone())
101            .assert_zero(s2.clone() * (s2.clone() - one.clone()));
102    });
103
104    // s3 is binary when s0 = s1 = s2 = 1 (ACE or kernel ROM could be active)
105    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 3, CHIPLET_SELECTORS_NAMES[3], |builder| {
106        builder
107            .when(s0.clone())
108            .when(s1.clone())
109            .when(s2.clone())
110            .assert_zero(s3.clone() * (s3.clone() - one.clone()));
111    });
112
113    // s4 is binary when s0 = s1 = s2 = s3 = 1 (kernel ROM could be active)
114    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 4, CHIPLET_SELECTORS_NAMES[4], |builder| {
115        builder
116            .when(s0.clone())
117            .when(s1.clone())
118            .when(s2.clone())
119            .when(s3.clone())
120            .assert_zero(s4.clone() * (s4.clone() - one.clone()));
121    });
122
123    // ==========================================================================
124    // STABILITY CONSTRAINTS (transition only)
125    // ==========================================================================
126    // Once a selector becomes 1, it must stay 1 (forbids 1→0 transitions)
127
128    // s0' = s0 when s0 = 1
129    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 5, CHIPLET_SELECTORS_NAMES[5], |builder| {
130        builder
131            .when_transition()
132            .when(s0.clone())
133            .assert_zero(s0_next.clone() - s0.clone());
134    });
135
136    // s1' = s1 when s0 = 1 and s1 = 1
137    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 6, CHIPLET_SELECTORS_NAMES[6], |builder| {
138        builder
139            .when_transition()
140            .when(s0.clone())
141            .when(s1.clone())
142            .assert_zero(s1_next.clone() - s1.clone());
143    });
144
145    // s2' = s2 when s0 = s1 = s2 = 1
146    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 7, CHIPLET_SELECTORS_NAMES[7], |builder| {
147        builder
148            .when_transition()
149            .when(s0.clone())
150            .when(s1.clone())
151            .when(s2.clone())
152            .assert_zero(s2_next.clone() - s2.clone());
153    });
154
155    // s3' = s3 when s0 = s1 = s2 = s3 = 1
156    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 8, CHIPLET_SELECTORS_NAMES[8], |builder| {
157        builder
158            .when_transition()
159            .when(s0.clone())
160            .when(s1.clone())
161            .when(s2.clone())
162            .when(s3.clone())
163            .assert_zero(s3_next.clone() - s3.clone());
164    });
165
166    // s4' = s4 when s0 = s1 = s2 = s3 = s4 = 1
167    builder.tagged(CHIPLET_SELECTORS_BASE_ID + 9, CHIPLET_SELECTORS_NAMES[9], |builder| {
168        builder
169            .when_transition()
170            .when(s0)
171            .when(s1)
172            .when(s2)
173            .when(s3)
174            .when(s4.clone())
175            .assert_zero(s4_next - s4);
176    });
177}

182/// Bitwise chiplet active flag: `s0 * !s1`.
183#[inline]
184pub fn bitwise_chiplet_flag<E: PrimeCharacteristicRing>(s0: E, s1: E) -> E {
185    s0 * (E::ONE - s1)
186}
187
188/// Memory chiplet active flag: `s0 * s1 * !s2`.
189#[inline]
190pub fn memory_chiplet_flag<E: PrimeCharacteristicRing>(s0: E, s1: E, s2: E) -> E {
191    s0 * s1 * (E::ONE - s2)
192}
193
194/// ACE chiplet active flag: `s0 * s1 * s2 * !s3`.
195#[inline]
196pub fn ace_chiplet_flag<E: PrimeCharacteristicRing>(s0: E, s1: E, s2: E, s3: E) -> E {
197    s0 * s1 * s2 * (E::ONE - s3)
198}
199
200/// Kernel ROM chiplet active flag: `s0 * s1 * s2 * s3 * !s4`.
201#[inline]
202pub fn kernel_rom_chiplet_flag<E: PrimeCharacteristicRing>(s0: E, s1: E, s2: E, s3: E, s4: E) -> E {
203    s0 * s1 * s2 * s3 * (E::ONE - s4)
