1//! Decoder constraints module.
2//!
3//! This module contains constraints for the decoder component of the Miden VM.
4//! The decoder handles instruction decoding, control flow, and basic block management.
5//!
6//! ## Constraint Categories
7//!
8//! 1. **Op Bit Constraints**: Ensure operation bits are binary
9//! 2. **Op Bits Extra Columns**: Ensure degree reduction columns are correctly computed
10//! 3. **Batch Flag Constraints**: Ensure batch flags are binary and properly set
11//! 4. **In-Span Constraints**: Ensure in-span flag transitions correctly
12//! 5. **Group Count Constraints**: Ensure group count transitions correctly
13//!
14//! ## Mental Model
15//!
16//! The decoder trace is the control-flow spine of the VM. Each row is either:
17//! - **inside a basic block** (sp = 1) where ops execute and counters advance, or
18//! - **a control-flow row** (sp = 0) that starts/ends/reshapes a block.
19//!
20//! The constraints below enforce three linked ideas:
21//! 1. **Opcode decoding is well-formed** (op bits and degree-reduction columns are consistent).
22//! 2. **Span state is coherent** (sp, group_count, op_index evolve exactly as control-flow allows).
23//! 3. **Hasher lanes match batch semantics** (batch flags and h0..h7 encode the pending groups).
24//!
25//! Read the sections in that order: first the binary/format checks, then the span state machine,
26//! then the counters and packing rules that make group decoding deterministic.
27//!
28//! ## Decoder Trace Layout
29//!
30//! The decoder trace consists of the following columns:
31//! - `addr`: Block address (row address in hasher table)
32//! - `b0-b6`: 7 operation bits encoding the opcode
33//! - `h0-h7`: 8 hasher state columns (shared between decoding and program hashing)
34//! - `sp`: In-span flag (1 when inside basic block, 0 otherwise)
35//! - `gc`: Group count (remaining operation groups in current span)
36//! - `ox`: Operation index (position within current operation group, 0-8)
37//! - `c0, c1, c2`: Batch flags (encode number of groups in current batch)
38//! - `e0, e1`: Extra columns for degree reduction
39

247/// Asserts a value is binary (0 or 1): `x * (x - 1) = 0`.
248fn assert_binary<AB>(builder: &mut AB, idx: usize, value: AB::Expr)
249where
250    AB: TaggingAirBuilderExt,
251{
252    assert_zero_integrity(builder, idx, value.clone() * (value - AB::Expr::ONE));
253}
254
255/// Computes the opcode value from op bits: `b0 + 2*b1 + ... + 64*b6`.
256fn op_bits_to_value<AB>(bits: &[AB::Expr; NUM_OP_BITS]) -> AB::Expr
257where
258    AB: LiftedAirBuilder,
259{
260    bits.iter().enumerate().fold(AB::Expr::ZERO, |acc, (i, bit)| {
261        acc + bit.clone() * AB::Expr::from_u16(1u16 << i)
262    })
263}

268/// Enforces decoder main-trace constraints (entry point).
269pub fn enforce_main<AB>(
270    builder: &mut AB,
271    local: &MainTraceRow<AB::Var>,
272    next: &MainTraceRow<AB::Var>,
273    op_flags: &OpFlags<AB::Expr>,
274) where
275    AB: TaggingAirBuilderExt,
276{
277    // Load decoder columns using typed struct
278    let cols: DecoderColumns<AB::Expr> = DecoderColumns::from_row::<AB>(local);
279    let cols_next: DecoderColumns<AB::Expr> = DecoderColumns::from_row::<AB>(next);
280    let op_flags_next = OpFlags::new(ExprDecoderAccess::<AB::Var, AB::Expr>::new(next));
281
282    enforce_in_span_constraints(builder, &cols, &cols_next, op_flags);
283    enforce_op_bits_binary(builder, &cols);
284    enforce_extra_columns(builder, &cols);
285    enforce_op_bit_group_constraints(builder, &cols);
286    enforce_batch_flags_binary(builder, &cols);
287    enforce_general_constraints(builder, local, next, op_flags, &op_flags_next);
288    enforce_group_count_constraints(builder, &cols, &cols_next, local, op_flags, &op_flags_next);
289    enforce_op_group_decoding_constraints(
290        builder,
291        &cols,
292        &cols_next,
293        local,
294        next,
295        op_flags,
296        &op_flags_next,
297    );
298    enforce_op_index_constraints(builder, &cols, &cols_next, op_flags);
299    enforce_batch_flags_constraints(builder, &cols, local, op_flags);
300    enforce_block_address_constraints(builder, &cols, &cols_next, op_flags);
301    enforce_control_flow_constraints(builder, &cols, op_flags);
302}

307/// Typed access to decoder columns.
308///
309/// This struct provides named access to decoder columns, eliminating error-prone
310/// index arithmetic. Created from a `MainTraceRow` reference.
311///
312/// ## Layout
313/// - `addr`: Block address (row address in hasher table)
314/// - `op_bits[0..7]`: Operation bits b0-b6 encoding the opcode
315/// - `in_span`: In-span flag (sp) - 1 when inside basic block
316/// - `group_count`: Group count (gc) - remaining operation groups
317/// - `op_index`: Operation index (ox) - position within group (0-8)
318/// - `batch_flags[0..3]`: Batch flags c0, c1, c2
319/// - `extra[0..2]`: Extra columns e0, e1 for degree reduction
320///
321/// Note: the 8 decoder hasher-state lanes live in the decoder trace but are not included here.
322/// Constraints which depend on these lanes access them directly via `MainTraceRow` accessors.
323pub struct DecoderColumns<E> {
324    /// Block address (row address in hasher table)
325    pub addr: E,
326    /// Operation bits b0-b6 (7 bits encoding the opcode)
327    pub op_bits: [E; NUM_OP_BITS],
328    /// In-span flag (1 when inside basic block, 0 otherwise)
329    pub in_span: E,
330    /// Group count (remaining operation groups in current span)
331    pub group_count: E,
332    /// Operation index (position within current operation group, 0-8)
333    pub op_index: E,
334    /// Batch flags c0, c1, c2 (encode number of groups in current batch)
335    pub batch_flags: [E; NUM_BATCH_FLAGS],
336    /// Extra columns e0, e1 for degree reduction
337    pub extra: [E; 2],
338}
339
340impl<E: Clone> DecoderColumns<E> {
341    /// Extract decoder columns from a main trace row.
342    pub fn from_row<AB>(row: &MainTraceRow<AB::Var>) -> Self
343    where
344        AB: LiftedAirBuilder,
345        AB::Var: Into<E> + Clone,
346    {
347        DecoderColumns {
348            addr: row.decoder[ADDR_OFFSET].clone().into(),
349            op_bits: core::array::from_fn(|i| row.decoder[OP_BITS_OFFSET + i].clone().into()),
350            in_span: row.decoder[IN_SPAN_OFFSET].clone().into(),
351            group_count: row.decoder[GROUP_COUNT_OFFSET].clone().into(),
352            op_index: row.decoder[OP_INDEX_OFFSET].clone().into(),
353            batch_flags: core::array::from_fn(|i| {
354                row.decoder[BATCH_FLAGS_OFFSET + i].clone().into()
355            }),
356            extra: core::array::from_fn(|i| row.decoder[EXTRA_COLS_OFFSET + i].clone().into()),
357        }
358    }
359}

364/// Enforces in-span (sp) constraints.
365///
366/// The in-span flag indicates whether we're inside a basic block:
367/// - sp = 1 when executing operations inside a basic block
368/// - sp = 0 for SPAN, RESPAN, END, and control-flow operations
369///
370/// This is the entry point to the decoder state machine. Once sp is set by SPAN/RESPAN,
371/// the rest of the decoder constraints (gc, ox, batch flags) are interpreted relative to
372/// being inside that span.
373///
374/// Constraints:
375/// 1. sp is binary: sp * (sp - 1) = 0
376/// 2. After SPAN operation, sp' = 1: span_flag * (1 - sp') = 0
377/// 3. After RESPAN operation, sp' = 1: respan_flag * (1 - sp') = 0
378fn enforce_in_span_constraints<AB>(
379    builder: &mut AB,
380    cols: &DecoderColumns<AB::Expr>,
381    cols_next: &DecoderColumns<AB::Expr>,
382    op_flags: &OpFlags<AB::Expr>,
383) where
384    AB: TaggingAirBuilderExt,
385{
386    let sp = cols.in_span.clone();
387    let sp_next = cols_next.in_span.clone();
388
389    // Boundary: execution starts outside any basic block.
390    assert_zero_first_row(builder, IN_SPAN_BASE, sp.clone());
391
392    // Constraint 1: sp is binary, so span state is well-formed.
393    let sp_binary = sp.clone() * (sp - AB::Expr::ONE);
394    assert_zero_integrity(builder, IN_SPAN_BASE + 1, sp_binary);
395
396    // Constraint 2: After SPAN, the next row must be inside a span.
397    // span_flag * (1 - sp') = 0
398    let span_flag = op_flags.span();
399    assert_zero_transition(
400        builder,
401        IN_SPAN_BASE + 2,
402        span_flag * (AB::Expr::ONE - sp_next.clone()),
403    );
404
405    // Constraint 3: After RESPAN, the next row must be inside a span.
406    // respan_flag * (1 - sp') = 0
407    let respan_flag = op_flags.respan();
408    assert_zero_transition(builder, IN_SPAN_BASE + 3, respan_flag * (AB::Expr::ONE - sp_next));
409}

411/// Enforces that all operation bits (b0-b6) are binary (0 or 1).
412///
413/// For each bit bi: bi * (bi - 1) = 0
414fn enforce_op_bits_binary<AB>(builder: &mut AB, cols: &DecoderColumns<AB::Expr>)
415where
416    AB: TaggingAirBuilderExt,
417{
418    for i in 0..NUM_OP_BITS {
419        // Each opcode bit must be 0 or 1 to make decoding deterministic.
420        assert_binary(builder, OP_BITS_BASE + i, cols.op_bits[i].clone());
421    }
422}

424/// Enforces that the extra columns (e0, e1) are correctly computed from op bits.
425///
426/// These columns are used for degree reduction in operation flag computation:
427/// - e0 = b6 * (1 - b5) * b4
428/// - e1 = b6 * b5
429fn enforce_extra_columns<AB>(builder: &mut AB, cols: &DecoderColumns<AB::Expr>)
430where
431    AB: TaggingAirBuilderExt,
432{
433    let b4 = cols.op_bits[4].clone();
434    let b5 = cols.op_bits[5].clone();
435    let b6 = cols.op_bits[6].clone();
436
437    let e0 = cols.extra[0].clone();
438    let e1 = cols.extra[1].clone();
439
440    // e0 = b6 * (1 - b5) * b4.
441    // This extra register exists to reduce the degree of op-flag selectors for the
442    // `101...` opcode group (see docs/src/design/stack/op_constraints.md).
443    let expected_e0 = b6.clone() * (AB::Expr::ONE - b5.clone()) * b4;
444    assert_zero_integrity(builder, EXTRA_BASE, e0 - expected_e0);
445
446    // e1 = b6 * b5.
447    // This extra register exists to reduce the degree of op-flag selectors for the
448    // `11...` opcode group (see docs/src/design/stack/op_constraints.md).
449    let expected_e1 = b6 * b5;
450    assert_zero_integrity(builder, EXTRA_BASE + 1, e1 - expected_e1);
451}

453/// Enforces opcode-bit constraints for grouped opcode families.
454///
455/// - U32 ops (prefix `100`) must have b0 = 0.
456/// - Very-high-degree ops (prefix `11`) must have b0 = b1 = 0.
457fn enforce_op_bit_group_constraints<AB>(builder: &mut AB, cols: &DecoderColumns<AB::Expr>)
458where
459    AB: TaggingAirBuilderExt,
460{
461    let b0 = cols.op_bits[0].clone();
462    let b1 = cols.op_bits[1].clone();
463    let b4 = cols.op_bits[4].clone();
464    let b5 = cols.op_bits[5].clone();
465    let b6 = cols.op_bits[6].clone();
466
467    // U32 prefix pattern: b6=1, b5=0, b4=0. Under this prefix, b0 must be 0 to
468    // eliminate invalid opcodes in the U32 opcode subset.
469    let u32_prefix = b6.clone() * (AB::Expr::ONE - b5.clone()) * (AB::Expr::ONE - b4);
470    assert_zero_integrity(builder, OP_BIT_GROUP_BASE, u32_prefix * b0.clone());
471
472    // Very-high prefix pattern: b6=1, b5=1. Under this prefix, b0 and b1 must be 0
473    // to eliminate invalid opcodes in the very-high opcode subset.
474    let very_high_prefix = b6 * b5;
475    assert_zero_integrity(builder, OP_BIT_GROUP_BASE + 1, very_high_prefix.clone() * b0);
476    assert_zero_integrity(builder, OP_BIT_GROUP_BASE + 2, very_high_prefix * b1);
477}

479/// Enforces that batch flags (c0, c1, c2) are binary.
480///
481/// For each flag ci: ci * (ci - 1) = 0
482fn enforce_batch_flags_binary<AB>(builder: &mut AB, cols: &DecoderColumns<AB::Expr>)
483where
484    AB: TaggingAirBuilderExt,
485{
486    for i in 0..NUM_BATCH_FLAGS {
487        // Batch flags are selectors; they must be boolean.
488        assert_binary(builder, BATCH_FLAGS_BINARY_BASE + i, cols.batch_flags[i].clone());
489    }

492/// Enforces general decoder constraints derived from opcode semantics.
493///
494/// These are the opcode-specific rules that aren't captured by the generic counters:
495/// - SPLIT/LOOP: s0 must be binary (branch selector)
496/// - DYN: upper hasher lanes are zero (callee hash lives in lower half)
497/// - REPEAT: must be inside loop body and s0 = 1
498/// - END + REPEAT': carry hasher state forward
499/// - HALT: absorbing
500fn enforce_general_constraints<AB>(
501    builder: &mut AB,
502    local: &MainTraceRow<AB::Var>,
503    next: &MainTraceRow<AB::Var>,
504    op_flags: &OpFlags<AB::Expr>,
505    op_flags_next: &OpFlags<AB::Expr>,
506) where
507    AB: TaggingAirBuilderExt,
508{
509    let s0: AB::Expr = local.stack[0].clone().into();
510
511    // SPLIT/LOOP: top stack value must be binary (branch selector).
512    let split_or_loop = op_flags.split() + op_flags.loop_op();
513    let s0_binary = s0.clone() * (s0.clone() - AB::Expr::ONE);
514    assert_zero_integrity(builder, GENERAL_BASE, split_or_loop * s0_binary);
515
516    // DYN: the first half holds the callee digest; the second half must be zero.
517    let f_dyn = op_flags.dyn_op();
518    for i in 0..4 {
519        let hi: AB::Expr = local.decoder[decoder_cols::HASHER_STATE_OFFSET + 4 + i].clone().into();
520        assert_zero_integrity(builder, GENERAL_BASE + 1 + i, f_dyn.clone() * hi);
521    }
522
523    // REPEAT: top stack must be 1 and we must be in a loop body (h4=1).
524    let f_repeat = op_flags.repeat();
525    let h4: AB::Expr = local.decoder[decoder_cols::IS_LOOP_BODY_FLAG_COL_IDX].clone().into();
526    assert_zero_integrity(
527        builder,
528        GENERAL_BASE + 5,
529        f_repeat.clone() * (AB::Expr::ONE - s0.clone()),
530    );
531    assert_zero_integrity(builder, GENERAL_BASE + 6, f_repeat * (AB::Expr::ONE - h4));
532
533    // END inside a loop: if is_loop flag is set, top stack must be 0.
534    let f_end = op_flags.end();
535    let h5: AB::Expr = local.decoder[decoder_cols::IS_LOOP_FLAG_COL_IDX].clone().into();
536    assert_zero_integrity(builder, GENERAL_BASE + 7, f_end.clone() * h5 * s0);
537
538    // END followed by REPEAT: carry h0..h4 into the next row.
539    let f_repeat_next = op_flags_next.repeat();
540    for i in 0..5 {
541        let hi: AB::Expr = local.decoder[decoder_cols::HASHER_STATE_OFFSET + i].clone().into();
542        let hi_next: AB::Expr = next.decoder[decoder_cols::HASHER_STATE_OFFSET + i].clone().into();
543        assert_zero_transition(
544            builder,
545            GENERAL_BASE + 8 + i,
546            f_end.clone() * f_repeat_next.clone() * (hi_next - hi),
547        );
548    }
549
550    // HALT is absorbing: it can only be followed by HALT.
551    let f_halt = op_flags.halt();
552    let f_halt_next = op_flags_next.halt();
553    assert_zero_transition(builder, GENERAL_BASE + 13, f_halt * (AB::Expr::ONE - f_halt_next));
554}

556/// Enforces group count (gc) constraints.
557///
558/// The group count tracks remaining operation groups in the current basic block:
559/// - gc starts at the total number of groups when SPAN/RESPAN is executed
560/// - gc decrements by 1 when processing SPAN, RESPAN, or completing a group
561/// - gc must be 0 when END is executed
562///
563/// Intuition:
564/// - `delta_gc = gc - gc'` is the number of groups consumed on this row.
565/// - Inside a span, delta_gc can only be 0 or 1.
566/// - SPAN/RESPAN/PUSH must consume a group immediately, so delta_gc must be 1.
567///
568/// Constraints:
569/// 1. Inside basic block, gc can only stay same or decrement by 1: sp * delta_gc * (delta_gc - 1) =
570///    0  (where delta_gc = gc - gc')
571/// 2. When END is executed, gc must be 0: end_flag * gc = 0
572/// 3. During SPAN/RESPAN, gc must decrement by 1: (span_flag + respan_flag) * (1 - delta_gc) = 0
573fn enforce_group_count_constraints<AB>(
574    builder: &mut AB,
575    cols: &DecoderColumns<AB::Expr>,
576    cols_next: &DecoderColumns<AB::Expr>,
577    local: &MainTraceRow<AB::Var>,
578    op_flags: &OpFlags<AB::Expr>,
579    op_flags_next: &OpFlags<AB::Expr>,
580) where
581    AB: TaggingAirBuilderExt,
582{
583    let sp = cols.in_span.clone();
584    let gc = cols.group_count.clone();
585    let gc_next = cols_next.group_count.clone();
586    let h0: AB::Expr = local.decoder[decoder_cols::HASHER_STATE_OFFSET].clone().into();
587
588    // delta_gc = gc - gc' (how much gc decrements; expected to be 0 or 1)
589    let delta_gc = gc.clone() - gc_next;
590
591    // is_push = push flag (PUSH is the only operation with immediate value)
592    let is_push = op_flags.push();
593
594    // Constraint 1: Inside a span, gc can only stay the same or decrement by 1.
595    // sp * delta_gc * (delta_gc - 1) = 0
596    // This ensures: if sp=1 and delta_gc != 0, then delta_gc must equal 1
597    assert_zero_transition(
598        builder,
599        GROUP_COUNT_BASE,
600        sp.clone() * delta_gc.clone() * (delta_gc.clone() - AB::Expr::ONE),
601    );
602
603    // Constraint 2: If gc decrements and this is not a PUSH-immediate row,
604    // then h0 must be zero (no immediate value packed into the group).
605    // sp * delta_gc * (1 - is_push) * h0 = 0
606    assert_zero_transition(
607        builder,
608        GROUP_COUNT_BASE + 1,
609        sp.clone() * delta_gc.clone() * (AB::Expr::ONE - is_push.clone()) * h0,
610    );
611
612    // Constraint 3: SPAN/RESPAN/PUSH must consume a group immediately.
613    // (span_flag + respan_flag + is_push) * (delta_gc - 1) = 0
614    let span_flag = op_flags.span();
615    let respan_flag = op_flags.respan();
616    assert_zero_transition(
617        builder,
618        GROUP_COUNT_BASE + 2,
619        (span_flag + respan_flag + is_push) * (delta_gc.clone() - AB::Expr::ONE),
620    );
621
622    // Constraint 4: If the next op is END or RESPAN, gc cannot decrement on this row.
623    // delta_gc * (end' + respan') = 0
624    let end_next = op_flags_next.end();
625    let respan_next = op_flags_next.respan();
626    assert_zero_transition(
627        builder,
628        GROUP_COUNT_BASE + 3,
629        delta_gc.clone() * (end_next + respan_next),
630    );
631
632    // Constraint 5: END closes the span, so gc must be 0.
633    // end_flag * gc = 0
634    let end_flag = op_flags.end();
635    assert_zero_integrity(builder, GROUP_COUNT_BASE + 4, end_flag * gc);
636}

638/// Enforces op group decoding constraints for the `h0` register.
639///
640/// `h0` is a packed buffer of pending op groups. When a group is started or continued,
641/// the next opcode is shifted into `h0` (base 2^7, because opcodes fit in 7 bits).
642/// When the next op is END or RESPAN, the buffer must be empty (h0 = 0).
643fn enforce_op_group_decoding_constraints<AB>(
644    builder: &mut AB,
645    cols: &DecoderColumns<AB::Expr>,
646    cols_next: &DecoderColumns<AB::Expr>,
647    local: &MainTraceRow<AB::Var>,
648    next: &MainTraceRow<AB::Var>,
649    op_flags: &OpFlags<AB::Expr>,
650    op_flags_next: &OpFlags<AB::Expr>,
651) where
652    AB: TaggingAirBuilderExt,
653{
654    let sp = cols.in_span.clone();
655    let sp_next = cols_next.in_span.clone();
656
657    let gc = cols.group_count.clone();
658    let gc_next = cols_next.group_count.clone();
659    let delta_gc = gc - gc_next;
660
661    let f_span = op_flags.span();
662    let f_respan = op_flags.respan();
663    let is_push = op_flags.push();
664
665    // f_sgc is set when gc stays the same inside a basic block.
666    let f_sgc = sp.clone() * sp_next * (AB::Expr::ONE - delta_gc.clone());
667
668    let h0: AB::Expr = local.decoder[decoder_cols::HASHER_STATE_OFFSET].clone().into();
669    let h0_next: AB::Expr = next.decoder[decoder_cols::HASHER_STATE_OFFSET].clone().into();
670
671    // Compute op' from next-row op bits (b0' + 2*b1' + ... + 64*b6').
672    let op_next = op_bits_to_value::<AB>(&cols_next.op_bits);
673
674    // When SPAN/RESPAN/PUSH or when gc doesn't change, shift h0 by op'.
675    // (h0 - h0' * 2^7 - op') = 0 under the combined flag.
676    let op_group_base = AB::Expr::from_u16(1u16 << 7);
677    let h0_shift = h0.clone() - h0_next * op_group_base - op_next;
678    assert_zero_transition(
679        builder,
680        OP_GROUP_DECODING_BASE,
681        (f_span + f_respan + is_push + f_sgc) * h0_shift,
682    );
683
684    // If the next op is END or RESPAN, the current h0 must be 0 (no pending group).
685    let end_next = op_flags_next.end();
686    let respan_next = op_flags_next.respan();
687    assert_zero_transition(builder, OP_GROUP_DECODING_BASE + 1, sp * (end_next + respan_next) * h0);
688}

690/// Enforces op index (ox) constraints.
691///
692/// The op index tracks the position of the current operation within its operation group:
693/// - ox ranges from 0 to 8 (9 operations per group)
694/// - ox resets to 0 when entering a new group (SPAN, RESPAN, or group boundary)
695/// - ox increments by 1 for each operation within a group
696///
697/// Intuition:
698/// - `ng = delta_gc - is_push` is 1 when a new group starts (excluding PUSH-immediate rows).
699/// - When a new group starts, ox' must be 0.
700/// - Otherwise, ox increments by 1 while sp stays 1.
701///
702/// Constraints:
703/// 1. After SPAN/RESPAN, ox' = 0: (span_flag + respan_flag) * ox' = 0
704/// 2. When starting new group inside basic block (gc decrements), ox' = 0: sp * delta_gc * ox' = 0
705/// 3. When inside basic block and not starting new group, ox increments by 1: sp * sp' * (1 -
706///    delta_gc) * (ox' - ox - 1) = 0
707/// 4. Op index must be in range [0, 8]: ox * (ox-1) * (ox-2) * (ox-3) * (ox-4) * (ox-5) * (ox-6) *
708///    (ox-7) * (ox-8) = 0
709fn enforce_op_index_constraints<AB>(
710    builder: &mut AB,
711    cols: &DecoderColumns<AB::Expr>,
712    cols_next: &DecoderColumns<AB::Expr>,
713    op_flags: &OpFlags<AB::Expr>,
714) where
715    AB: TaggingAirBuilderExt,
716{
717    let sp = cols.in_span.clone();
718    let sp_next = cols_next.in_span.clone();
719    let gc = cols.group_count.clone();
720    let gc_next = cols_next.group_count.clone();
721    let ox = cols.op_index.clone();
722    let ox_next = cols_next.op_index.clone();
723
724    // delta_gc = gc - gc' (how much gc decrements; 1 when entering new group)
725    let delta_gc = gc.clone() - gc_next;
726
727    // is_push = push flag (PUSH is the only operation with immediate value)
728    let is_push = op_flags.push();
729
730    // ng = delta_gc - is_push
731    // This equals 1 when we're starting a new operation group (not due to immediate op)
732    let ng = delta_gc - is_push;
733
734    let span_flag = op_flags.span();
735    let respan_flag = op_flags.respan();
736
737    // Constraint 1: SPAN/RESPAN start a fresh group, so ox' = 0.
738    // (span_flag + respan_flag) * ox' = 0
739    assert_zero_transition(builder, OP_INDEX_BASE, (span_flag + respan_flag) * ox_next.clone());
740
741    // Constraint 2: When a new group starts inside a span, ox' = 0.
742    // sp * ng * ox' = 0
743    assert_zero_transition(builder, OP_INDEX_BASE + 1, sp.clone() * ng.clone() * ox_next.clone());
744
745    // Constraint 3: When staying in the same group, ox increments by 1.
746    // sp * sp' * (1 - ng) * (ox' - ox - 1) = 0
747    let delta_ox = ox_next - ox.clone() - AB::Expr::ONE;
748    assert_zero_transition(
749        builder,
750        OP_INDEX_BASE + 2,
751        sp * sp_next * (AB::Expr::ONE - ng) * delta_ox,
752    );
753
754    // Constraint 4: ox must be in range [0, 8] (9 ops per group).
755    // ∏_{i=0}^{8}(ox - i) = 0
756    let mut range_check = ox.clone();
757    for i in 1..=8u64 {
758        range_check *= ox.clone() - AB::Expr::from_u16(i as u16);
759    }
760    assert_zero_integrity(builder, OP_INDEX_BASE + 3, range_check);
761}

763/// Enforces op batch flag constraints and associated hasher-state zeroing rules.
764///
765/// Batch flags encode how many groups were emitted by SPAN/RESPAN:
766/// - g8, g4, g2, g1 correspond to batches of 8, 4, 2, or 1 groups. The hasher lanes h1..h7 store
767///   the group values; unused lanes must be zeroed.
768fn enforce_batch_flags_constraints<AB>(
769    builder: &mut AB,
770    cols: &DecoderColumns<AB::Expr>,
771    local: &MainTraceRow<AB::Var>,
772    op_flags: &OpFlags<AB::Expr>,
773) where
774    AB: TaggingAirBuilderExt,
775{
776    let bc0 = cols.batch_flags[0].clone();
777    let bc1 = cols.batch_flags[1].clone();
778    let bc2 = cols.batch_flags[2].clone();
779
780    // Batch flag decoding matches trace::decoder batch encodings.
781    let f_g8 = bc0.clone();
782    let f_g4 = (AB::Expr::ONE - bc0.clone()) * bc1.clone() * (AB::Expr::ONE - bc2.clone());
783    let f_g2 = (AB::Expr::ONE - bc0.clone()) * (AB::Expr::ONE - bc1.clone()) * bc2.clone();
784    let f_g1 = (AB::Expr::ONE - bc0) * bc1 * bc2;
785
786    let f_span = op_flags.span();
787    let f_respan = op_flags.respan();
788    let span_or_respan = f_span + f_respan;
789
790    // When SPAN or RESPAN, exactly one batch flag must be set.
791    assert_zero_integrity(
792        builder,
793        BATCH_FLAGS_BASE,
794        span_or_respan.clone() - (f_g1.clone() + f_g2.clone() + f_g4.clone() + f_g8),
795    );
796
797    // When not SPAN/RESPAN, all batch flags must be zero.
798    assert_zero_integrity(
799        builder,
800        BATCH_FLAGS_BASE + 1,
801        (AB::Expr::ONE - span_or_respan)
802            * (cols.batch_flags[0].clone()
803                + cols.batch_flags[1].clone()
804                + cols.batch_flags[2].clone()),
805    );
806
807    // When batch has <=4 groups, h4..h7 must be zero (unused lanes).
808    let small_batch = f_g1.clone() + f_g2.clone() + f_g4.clone();
809    for i in 0..4 {
810        let hi: AB::Expr = local.decoder[decoder_cols::HASHER_STATE_OFFSET + 4 + i].clone().into();
811        assert_zero_integrity(builder, BATCH_FLAGS_BASE + 2 + i, small_batch.clone() * hi);
812    }
813
814    // When batch has <=2 groups, h2..h3 must be zero (unused lanes).
815    let tiny_batch = f_g1.clone() + f_g2.clone();
816    for i in 0..2 {
817        let hi: AB::Expr = local.decoder[decoder_cols::HASHER_STATE_OFFSET + 2 + i].clone().into();
818        assert_zero_integrity(builder, BATCH_FLAGS_BASE + 6 + i, tiny_batch.clone() * hi);
819    }
820
821    // When batch has 1 group, h1 must be zero (unused lane).
822    let h1: AB::Expr = local.decoder[decoder_cols::HASHER_STATE_OFFSET + 1].clone().into();
823    assert_zero_integrity(builder, BATCH_FLAGS_BASE + 8, f_g1 * h1);
824}

826/// Enforces block address (addr) constraints.
827///
828/// The block address identifies the current code block in the hasher table:
829/// - addr stays constant inside a basic block (sp = 1)
830/// - addr increments by HASH_CYCLE_LEN (32) after RESPAN
831/// - addr must be 0 when HALT is executed
832///
833/// This ties the span state to the hasher table position.
834///
835/// Constraints:
836/// 1. Inside basic block, address unchanged: sp * (addr' - addr) = 0
837/// 2. RESPAN increments address by HASH_CYCLE_LEN: respan_flag * (addr' - addr - HASH_CYCLE_LEN) =
838///    0
839/// 3. HALT has addr = 0: halt_flag * addr = 0
840fn enforce_block_address_constraints<AB>(
841    builder: &mut AB,
842    cols: &DecoderColumns<AB::Expr>,
843    cols_next: &DecoderColumns<AB::Expr>,
844    op_flags: &OpFlags<AB::Expr>,
845) where
846    AB: TaggingAirBuilderExt,
847{
848    let sp = cols.in_span.clone();
849    let addr = cols.addr.clone();
850    let addr_next = cols_next.addr.clone();
851
852    // Constraint 1: Inside a span, address must stay the same.
853    // sp * (addr' - addr) = 0
854    assert_zero_transition(builder, ADDR_BASE, sp * (addr_next.clone() - addr.clone()));
855
856    // Constraint 2: RESPAN moves to the next hash block (Poseidon2 = 32 rows).
857    // respan_flag * (addr' - addr - HASH_CYCLE_LEN) = 0
858    let hash_cycle_len: AB::Expr = AB::Expr::from_u16(HASH_CYCLE_LEN as u16);
859    let respan_flag = op_flags.respan();
860    assert_zero_transition(
861        builder,
862        ADDR_BASE + 1,
863        respan_flag * (addr_next - addr.clone() - hash_cycle_len),
864    );
865
866    // Constraint 3: HALT forces addr = 0.
867    // halt_flag * addr = 0
868    let halt_flag = op_flags.halt();
869    assert_zero_integrity(builder, ADDR_BASE + 2, halt_flag * addr);
870}

872/// Enforces control flow constraints.
873///
874/// When outside a basic block (sp = 0), only control flow operations can execute.
875/// This is expressed as: fctrl = 1 - sp, or equivalently: (1 - sp) * (1 - fctrl) = 0
876///
877/// Control flow operations include:
878/// - SPAN, JOIN, SPLIT, LOOP (block start operations)
879/// - END, REPEAT, RESPAN, HALT (block transition operations)
880/// - DYN, DYNCALL, CALL, SYSCALL (procedure invocations)
881///
882/// Constraints:
883/// 1. When sp = 0, control_flow must be 1: (1 - sp) * (1 - fctrl) = 0
884fn enforce_control_flow_constraints<AB>(
885    builder: &mut AB,
886    cols: &DecoderColumns<AB::Expr>,
887    op_flags: &OpFlags<AB::Expr>,
888) where
889    AB: TaggingAirBuilderExt,
890{
891    let sp = cols.in_span.clone();
892
893    // Constraint: sp and control_flow must be complementary.
894    // 1 - sp - fctrl = 0
895    let ctrl_flag = op_flags.control_flow();
896    assert_zero_integrity(builder, CONTROL_FLOW_BASE, AB::Expr::ONE - sp - ctrl_flag);
897}
