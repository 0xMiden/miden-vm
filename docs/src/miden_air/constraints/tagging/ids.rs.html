3/// Base ID for the system constraint group.

12/// Base ID for the system clock constraints.

17pub const TAG_SYSTEM_FN_HASH_BASE: usize = TAG_SYSTEM_CTX_BASE + TAG_SYSTEM_CTX_COUNT;
