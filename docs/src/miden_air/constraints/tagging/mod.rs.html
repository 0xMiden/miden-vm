27/// Tag metadata for a constraint group (base ID + ordered names).
28#[derive(Clone, Copy)]
29pub struct TagGroup {
30    pub base: usize,
31    pub names: &'static [&'static str],
32}
33
34/// Tag and assert a single constraint, advancing the per-group index.
35pub fn tagged_assert_zero<AB: TaggingAirBuilderExt>(
36    builder: &mut AB,
37    group: &TagGroup,
38    idx: &mut usize,
39    expr: AB::Expr,
40) {
41    debug_assert!(*idx < group.names.len(), "tag index out of bounds");
42    let id = group.base + *idx;
43    let name = group.names[*idx];
44    builder.tagged(id, name, |builder| {
45        builder.when_transition().assert_zero(expr);
46    });
47    *idx += 1;
48}

66/// Tag and assert a fixed list of constraints, advancing the per-group index.
67pub fn tagged_assert_zeros<AB: TaggingAirBuilderExt, const N: usize>(
68    builder: &mut AB,
69    group: &TagGroup,
70    idx: &mut usize,
71    namespace: &'static str,
72    exprs: [AB::Expr; N],
73) {
74    debug_assert!(*idx + N <= group.names.len(), "tag index out of bounds");
75    let ids: [usize; N] = core::array::from_fn(|i| group.base + *idx + i);
76    builder.tagged_list(ids, namespace, |builder| {
77        builder.when_transition().assert_zeros(exprs);
78    });
79    *idx += N;
80}
