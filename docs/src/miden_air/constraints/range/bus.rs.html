1//! Range checker bus constraint.
2//!
3//! This module enforces the LogUp protocol for the range checker bus (b_range).
4//! The range checker validates that values are within [0, 2^16) by tracking requests
5//! from stack and memory components against the range table responses.
6//!
7//! ## LogUp Protocol
8//!
9//! The bus accumulator b_range uses the LogUp protocol:
10//! - Boundary: b_range[0] = 0 and b_range[last] = 0 (enforced by the wrapper AIR)
11//! - Transition: b_range' = b_range + responses - requests
12//!
13//! Where requests come from stack (4 lookups) and memory (2 lookups), and
14//! responses come from the range table (V column with multiplicity).
15

17use miden_crypto::stark::air::{ExtensionBuilder, LiftedAirBuilder, WindowAccess};

24        RANGE_CHECK_TRACE_OFFSET, chiplets, decoder, range,
25    },
26};

32const STACK_LOOKUP_BASE: usize = decoder::USER_OP_HELPERS_OFFSET;
33const OP_BIT_4_COL_IDX: usize = decoder::OP_BITS_RANGE.start + 4;
34const OP_BIT_5_COL_IDX: usize = decoder::OP_BITS_RANGE.start + 5;
35const OP_BIT_6_COL_IDX: usize = decoder::OP_BITS_RANGE.start + 6;
36const CHIPLET_S0_IDX: usize = CHIPLET_S0_COL_IDX - CHIPLETS_OFFSET;
37const CHIPLET_S1_IDX: usize = CHIPLET_S1_COL_IDX - CHIPLETS_OFFSET;
38const CHIPLET_S2_IDX: usize = CHIPLET_S2_COL_IDX - CHIPLETS_OFFSET;
39const MEMORY_D0_IDX: usize = chiplets::MEMORY_D0_COL_IDX - CHIPLETS_OFFSET;
40const MEMORY_D1_IDX: usize = chiplets::MEMORY_D1_COL_IDX - CHIPLETS_OFFSET;
41const RANGE_M_COL_IDX: usize = range::M_COL_IDX - RANGE_CHECK_TRACE_OFFSET;
42const RANGE_V_COL_IDX: usize = range::V_COL_IDX - RANGE_CHECK_TRACE_OFFSET;

52/// Enforces the range checker bus constraint for LogUp checks.
53///
54/// This constraint tracks range check requests from other components (stack and memory)
55/// using the LogUp protocol. The bus accumulator b_range must start and end at 0,
56/// and transition according to the LogUp update rule.
57///
58/// ## Constraint Degree
59///
60/// This is a degree-9 constraint.
61///
62/// ## Lookups
63///
64/// - Stack lookups (4): decoder helper columns (USER_OP_HELPERS_OFFSET..+4)
65/// - Memory lookups (2): memory delta limbs (MEMORY_D0, MEMORY_D1)
66/// - Range response: range V column with multiplicity range M column
67pub fn enforce_bus<AB>(builder: &mut AB, local: &MainTraceRow<AB::Var>)
68where
69    AB: LiftedAirBuilder,
70{
71    // In Miden VM, auxiliary trace is always present
72
73    // Extract values needed for constraints
74    let aux = builder.permutation();
75    let aux_local = aux.current_slice();
76    let aux_next = aux.next_slice();
77    let b_local = aux_local[range::B_RANGE_COL_IDX];
78    let b_next = aux_next[range::B_RANGE_COL_IDX];
79
80    let challenges = builder.permutation_randomness();
81    let alpha = challenges[0];
82
83    // Denominators for LogUp
84    // Memory lookups: mv0 = alpha + chiplets[MEMORY_D0], mv1 = alpha + chiplets[MEMORY_D1]
85    let mv0: AB::ExprEF = alpha.into() + local.chiplets[MEMORY_D0_IDX].clone().into();
86    let mv1: AB::ExprEF = alpha.into() + local.chiplets[MEMORY_D1_IDX].clone().into();
87
88    // Stack lookups: sv0-sv3 = alpha + decoder helper columns
89    let sv0: AB::ExprEF = alpha.into() + local.decoder[STACK_LOOKUP_BASE].clone().into();
90    let sv1: AB::ExprEF = alpha.into() + local.decoder[STACK_LOOKUP_BASE + 1].clone().into();
91    let sv2: AB::ExprEF = alpha.into() + local.decoder[STACK_LOOKUP_BASE + 2].clone().into();
92    let sv3: AB::ExprEF = alpha.into() + local.decoder[STACK_LOOKUP_BASE + 3].clone().into();
93
94    // Range check value: alpha + range V column
95    let range_check: AB::ExprEF = alpha.into() + local.range[RANGE_V_COL_IDX].clone().into();
96
97    // Combined lookup denominators
98    let memory_lookups = mv0.clone() * mv1.clone();
99    let stack_lookups = sv0.clone() * sv1.clone() * sv2.clone() * sv3.clone();
100    let lookups = range_check.clone() * stack_lookups.clone() * memory_lookups.clone();
101
102    // Flags for conditional inclusion
103    // u32_rc_op = op_bit[6] * (1 - op_bit[5]) * (1 - op_bit[4])
104    let not_4: AB::Expr = AB::Expr::ONE - local.decoder[OP_BIT_4_COL_IDX].clone().into();
105    let not_5: AB::Expr = AB::Expr::ONE - local.decoder[OP_BIT_5_COL_IDX].clone().into();
106    let u32_rc_op: AB::Expr = local.decoder[OP_BIT_6_COL_IDX].clone().into() * not_5 * not_4;
107    let sflag_rc_mem = range_check.clone() * memory_lookups.clone() * u32_rc_op;
108
109    // chiplets_memory_flag = s0 * s1 * (1 - s2)
110    let s_0: AB::Expr = local.chiplets[CHIPLET_S0_IDX].clone().into();
111    let s_1: AB::Expr = local.chiplets[CHIPLET_S1_IDX].clone().into();
112    let s_2: AB::Expr = local.chiplets[CHIPLET_S2_IDX].clone().into();
113    let chiplets_memory_flag: AB::Expr = s_0 * s_1 * (AB::Expr::ONE - s_2);
114    let mflag_rc_stack = range_check * stack_lookups.clone() * chiplets_memory_flag;
115
116    // LogUp transition constraint terms
117    let b_next_term = b_next.into() * lookups.clone();
118    let b_term = b_local.into() * lookups;
119    let rc_term = stack_lookups * memory_lookups * local.range[RANGE_M_COL_IDX].clone().into();
120
121    // Stack lookup removal terms
122    let s0_term = sflag_rc_mem.clone() * sv1.clone() * sv2.clone() * sv3.clone();
123    let s1_term = sflag_rc_mem.clone() * sv0.clone() * sv2.clone() * sv3.clone();
124    let s2_term = sflag_rc_mem.clone() * sv0.clone() * sv1.clone() * sv3;
125    let s3_term = sflag_rc_mem * sv0 * sv1 * sv2;
126
127    // Memory lookup removal terms
128    let m0_term: AB::ExprEF = mflag_rc_stack.clone() * mv1;
129    let m1_term = mflag_rc_stack * mv0;
130
131    // Main constraint: b_next * lookups = b * lookups + rc_term - s0_term - s1_term - s2_term -
132    // s3_term - m0_term - m1_term
133    builder.tagged(TAG_RANGE_BUS_BASE, RANGE_BUS_NAME, |builder| {
134        builder.when_transition().assert_zero_ext(
135            b_next_term - b_term - rc_term
136                + s0_term
137                + s1_term
138                + s2_term
139                + s3_term
140                + m0_term
141                + m1_term,
142        );
143    });
144}
