1//! Operation flags for stack constraints.
2//!
3//! This module computes operation flags from decoder op bits. These flags are used
4//! throughout stack constraints to gate constraint enforcement based on which operation
5//! is currently being executed.
6//!
7//! ## Operation Degree Categories
8//!
9//! Operations are grouped by their flag computation degree:
10//! - **Degree 7**: 64 operations (opcodes 0-63) - use all 7 op bits
11//! - **Degree 6**: 8 operations (opcodes 64-79) - u32 operations
12//! - **Degree 5**: 16 operations (opcodes 80-95) - use op_bit_extra[0]
13//! - **Degree 4**: 8 operations (opcodes 96-127) - use op_bit_extra[1]
14//!
15//! ## Composite Flags
16//!
17//! The module also computes composite flags that combine multiple operations:
18//! - `no_shift_at(i)`: stack position i unchanged
19//! - `left_shift_at(i)`: stack shifts left at position i
20//! - `right_shift_at(i)`: stack shifts right at position i
21

79/// Operation flags for all stack operations.
80///
81/// Computes all operation flag expressions from decoder op bits. Only one flag will be
82/// non-zero for any given row. The flags are computed using intermediate values to
83/// minimize the number of multiplications.
84///
85/// This struct is parameterized by the expression type `E` which allows it to work
86/// with both concrete field elements (for testing) and symbolic expressions (for
87/// constraint generation).
88#[allow(dead_code)]
89pub struct OpFlags<E> {
90    degree7_op_flags: [E; NUM_DEGREE_7_OPS],
91    degree6_op_flags: [E; NUM_DEGREE_6_OPS],
92    degree5_op_flags: [E; NUM_DEGREE_5_OPS],
93    degree4_op_flags: [E; NUM_DEGREE_4_OPS],
94    no_shift_flags: [E; NUM_STACK_IMPACT_FLAGS],
95    left_shift_flags: [E; NUM_STACK_IMPACT_FLAGS],
96    right_shift_flags: [E; NUM_STACK_IMPACT_FLAGS],
97
98    left_shift: E,
99    right_shift: E,
100    control_flow: E,
101    overflow: E,
102    u32_rc_op: E,
103}

105/// Helper trait for accessing decoder columns from a trace row.
106pub trait DecoderAccess<E> {
107    /// Returns the value of op_bit[index] from the decoder.
108    fn op_bit(&self, index: usize) -> E;
109
110    /// Returns the value of op_bit_extra[index] from the decoder.
111    fn op_bit_extra(&self, index: usize) -> E;
112
113    /// Returns the h0 helper register (overflow indicator).
114    fn overflow_register(&self) -> E;
115
116    /// Returns the stack depth (b0 column).
117    fn stack_depth(&self) -> E;
118
119    /// Returns the is_loop flag from the decoder.
120    fn is_loop_end(&self) -> E;
121}
122
123/// Implement DecoderAccess for MainTraceRow references.
124impl<T> DecoderAccess<T> for &crate::MainTraceRow<T>
125where
126    T: Clone,
127{
128    #[inline]
129    fn op_bit(&self, index: usize) -> T {
130        self.decoder[OP_BITS_RANGE.start + index].clone()
131    }
132
133    #[inline]
134    fn op_bit_extra(&self, index: usize) -> T {
135        self.decoder[OP_BITS_EXTRA_COLS_RANGE.start + index].clone()
136    }
137
138    #[inline]
139    fn overflow_register(&self) -> T {
140        self.stack[H0_COL_IDX].clone()
141    }
142
143    #[inline]
144    fn stack_depth(&self) -> T {
145        self.stack[B0_COL_IDX].clone()
146    }
147
148    #[inline]
149    fn is_loop_end(&self) -> T {
150        self.decoder[IS_LOOP_FLAG_COL_IDX].clone()
151    }
152}

200/// Helper function to compute binary NOT: 1 - x
201#[inline]
202fn binary_not<E>(x: E) -> E
203where
204    E: Clone + core::ops::Sub<Output = E> + PrimeCharacteristicRing,
205{
206    E::ONE - x

227macro_rules! op_flag_getters {
228    ($array:ident, $( $(#[$meta:meta])* $name:ident => $op:expr ),* $(,)?) => {
229        $(
230            $(#[$meta])*

239#[allow(dead_code)]
240impl<E> OpFlags<E>
241where
242    E: Clone
243        + Default
244        + core::ops::Add<Output = E>
245        + core::ops::Sub<Output = E>
246        + core::ops::Mul<Output = E>
247        + PrimeCharacteristicRing,
248{
249    /// Creates a new OpFlags instance by computing all flags from the decoder columns.
250    ///
251    /// The computation uses intermediate values to minimize multiplications:
252    /// - Degree 7 flags: computed hierarchically from op bits
253    /// - Degree 6 flags: u32 operations, share common prefix `100`
254    /// - Degree 5 flags: use op_bit_extra[0] for degree reduction
255    /// - Degree 4 flags: use op_bit_extra[1] for degree reduction
256    pub fn new<D: DecoderAccess<E>>(frame: D) -> Self {
257        // Initialize arrays with default values
258        let mut degree7_op_flags: [E; NUM_DEGREE_7_OPS] = core::array::from_fn(|_| E::default());
259        let mut degree6_op_flags: [E; NUM_DEGREE_6_OPS] = core::array::from_fn(|_| E::default());
260        let mut degree5_op_flags: [E; NUM_DEGREE_5_OPS] = core::array::from_fn(|_| E::default());
261        let mut degree4_op_flags: [E; NUM_DEGREE_4_OPS] = core::array::from_fn(|_| E::default());
262        let mut no_shift_flags: [E; NUM_STACK_IMPACT_FLAGS] =
263            core::array::from_fn(|_| E::default());
264        let mut left_shift_flags: [E; NUM_STACK_IMPACT_FLAGS] =
265            core::array::from_fn(|_| E::default());
266        let mut right_shift_flags: [E; NUM_STACK_IMPACT_FLAGS] =
267            core::array::from_fn(|_| E::default());
268
269        // Get op bits and their binary negations.
270        let op: [Op<E>; 7] = core::array::from_fn(|i| {
271            let bit = frame.op_bit(i);
272            let not = binary_not(bit.clone());
273            Op { bit, not }
274        });
275
276        // --- Low-degree prefix selectors for composite flags ---
277        // These produce degree-5 left_shift and right_shift composite flags.
278        // per spec: https://0xmiden.github.io/miden-vm/design/stack/op_constraints.html#shift-left-flag
279
280        // Prefix `010` selector: (1-b6)*b5*(1-b4) - degree 3
281        // Covers all degree-7 operations with this prefix (left shift ops)
282        let prefix_010 = op[6].not() * op[5].is() * op[4].not();
283
284        // Prefix `011` selector: (1-b6)*b5*b4 - degree 3
285        // Covers all degree-7 operations with this prefix (right shift ops)
286        let prefix_011 = op[6].not() * op[5].is() * op[4].is();
287
288        // Prefix `10011` selector: b6*(1-b5)*(1-b4)*b3*b2 - degree 5
289        // Covers U32ADD3 and U32MADD (both cause left shift by 2)
290        let add3_madd_prefix = op[6].is() * op[5].not() * op[4].not() * op[3].is() * op[2].is();
291
292        // --- Computation of degree 7 operation flags ---
293
294        // Intermediate values computed from most significant bits
295        degree7_op_flags[0] = op[5].not() * op[4].not();
296        degree7_op_flags[16] = op[5].not() * op[4].is();
297        degree7_op_flags[32] = op[5].is() * op[4].not();
298        // Prefix `11` in bits [5..4] (binary 110000 = 48).
299        degree7_op_flags[0b110000] = op[5].is() * op[4].is();
300
301        // Flag of prefix `100` - all degree 6 u32 operations
302        let f100 = degree7_op_flags[0].clone() * op[6].is();
303        // Flag of prefix `1000` - u32 arithmetic operations
304        let f1000 = f100.clone() * op[3].not();
305
306        let not_6_not_3 = op[6].not() * op[3].not();
307        let not_6_yes_3 = op[6].not() * op[3].is();
308
309        // Add fourth most significant bit along with most significant bit
310        for i in (0..64).step_by(16) {
311            let base = degree7_op_flags[i].clone();
312            degree7_op_flags[i + 8] = base.clone() * not_6_yes_3.clone();
313            degree7_op_flags[i] = base * not_6_not_3.clone();
314        }
315
316        // Flag of prefix `011` - degree 7 right shift operations
317        let f011 = degree7_op_flags[48].clone() + degree7_op_flags[56].clone();
318        // Flag of prefix `010` - degree 7 left shift operations (reserved for future use)
319        let _f010 = degree7_op_flags[32].clone() + degree7_op_flags[40].clone();
320        // Flag of prefix `0000` - no shift from position 1 onwards
321        let f0000 = degree7_op_flags[0].clone();
322        // Flag of prefix `0100` - left shift from position 2 onwards
323        let f0100 = degree7_op_flags[32].clone();
324
325        // Add fifth most significant bit
326        for i in (0..64).step_by(8) {
327            let base = degree7_op_flags[i].clone();
328            degree7_op_flags[i + 4] = base.clone() * op[2].is();
329            degree7_op_flags[i] = base * op[2].not();
330        }
331
332        // Add sixth most significant bit
333        for i in (0..64).step_by(4) {
334            let base = degree7_op_flags[i].clone();
335            degree7_op_flags[i + 2] = base.clone() * op[1].is();
336            degree7_op_flags[i] = base * op[1].not();
337        }
338
339        // Cache flags for mov{up/dn}{2-8}, swapw{2-3} operations
340        let mov2_flag = degree7_op_flags[10].clone();
341        let mov3_flag = degree7_op_flags[12].clone();
342        let mov4_flag = degree7_op_flags[16].clone();
343        let mov5_flag = degree7_op_flags[18].clone();
344        let mov6_flag = degree7_op_flags[20].clone();
345        let mov7_flag = degree7_op_flags[22].clone();
346        let mov8_flag = degree7_op_flags[26].clone();
347        let swapwx_flag = degree7_op_flags[28].clone();
348        let adv_popw_expacc = degree7_op_flags[14].clone();
349
350        // Add least significant bit
351        for i in (0..64).step_by(2) {
352            let base = degree7_op_flags[i].clone();
353            degree7_op_flags[i + 1] = base.clone() * op[0].is();
354            degree7_op_flags[i] = base * op[0].not();
355        }
356
357        let ext2mul_flag = degree7_op_flags[25].clone();
358
359        // Flag when items from first position onwards are copied over (excludes NOOP)
360        let no_change_1_flag = f0000.clone() - degree7_op_flags[0].clone();
361        // Flag when items from second position onwards shift left (excludes ASSERT)
362        let left_change_1_flag = f0100 - degree7_op_flags[32].clone();
363
364        // --- Computation of degree 6 operation flags ---
365
366        // Degree 6 flag prefix is `100`
367        let degree_6_flag = op[6].is() * op[5].not() * op[4].not();
368
369        // Degree 6 flags do not use the first bit (op_bits[0])
370        let not_2_not_3 = op[2].not() * op[3].not();
371        let yes_2_not_3 = op[2].is() * op[3].not();
372        let not_2_yes_3 = op[2].not() * op[3].is();
373        let yes_2_yes_3 = op[2].is() * op[3].is();
374
375        degree6_op_flags[0] = op[1].not() * not_2_not_3.clone(); // U32ADD
376        degree6_op_flags[1] = op[1].is() * not_2_not_3.clone(); // U32SUB
377        degree6_op_flags[2] = op[1].not() * yes_2_not_3.clone(); // U32MUL
378        degree6_op_flags[3] = op[1].is() * yes_2_not_3.clone(); // U32DIV
379        degree6_op_flags[4] = op[1].not() * not_2_yes_3.clone(); // U32SPLIT
380        degree6_op_flags[5] = op[1].is() * not_2_yes_3.clone(); // U32ASSERT2
381        degree6_op_flags[6] = op[1].not() * yes_2_yes_3.clone(); // U32ADD3
382        degree6_op_flags[7] = op[1].is() * yes_2_yes_3.clone(); // U32MADD
383
384        // Multiply by degree 6 flag
385        for flag in degree6_op_flags.iter_mut() {
386            *flag = flag.clone() * degree_6_flag.clone();
387        }
388
389        // --- Computation of degree 5 operation flags ---
390
391        // Degree 5 flag uses the first degree reduction column
392        let degree_5_flag = frame.op_bit_extra(0);
393
394        let not_0_not_1 = op[0].not() * op[1].not();
395        let yes_0_not_1 = op[0].is() * op[1].not();
396        let not_0_yes_1 = op[0].not() * op[1].is();
397        let yes_0_yes_1 = op[0].is() * op[1].is();
398
399        degree5_op_flags[0] = not_0_not_1.clone() * op[2].not(); // HPERM
400        degree5_op_flags[1] = yes_0_not_1.clone() * op[2].not(); // MPVERIFY
401        degree5_op_flags[2] = not_0_yes_1.clone() * op[2].not(); // PIPE
402        degree5_op_flags[3] = yes_0_yes_1.clone() * op[2].not(); // MSTREAM
403        degree5_op_flags[4] = not_0_not_1.clone() * op[2].is(); // SPLIT
404        degree5_op_flags[5] = yes_0_not_1.clone() * op[2].is(); // LOOP
405        degree5_op_flags[6] = not_0_yes_1.clone() * op[2].is(); // SPAN
406        degree5_op_flags[7] = yes_0_yes_1.clone() * op[2].is(); // JOIN
407
408        // Second half shares same lower 3 bits
409        for i in 0..8 {
410            degree5_op_flags[i + 8] = degree5_op_flags[i].clone();
411        }
412
413        // Update with op_bit[3] and degree 5 flag
414        let deg_5_not_3 = op[3].not() * degree_5_flag.clone();
415        for flag in degree5_op_flags.iter_mut().take(8) {
416            *flag = flag.clone() * deg_5_not_3.clone();
417        }
418        let deg_5_yes_3 = op[3].is() * degree_5_flag.clone();
419        for flag in degree5_op_flags.iter_mut().skip(8) {
420            *flag = flag.clone() * deg_5_yes_3.clone();
421        }
422
423        // --- Computation of degree 4 operation flags ---
424
425        // Degree 4 flag uses the second degree reduction column
426        let degree_4_flag = frame.op_bit_extra(1);
427
428        // Degree 4 flags do not use the first two bits
429        degree4_op_flags[0] = not_2_not_3.clone(); // MRUPDATE
430        degree4_op_flags[1] = yes_2_not_3; // (unused)
431        degree4_op_flags[2] = not_2_yes_3; // SYSCALL
432        degree4_op_flags[3] = yes_2_yes_3; // CALL
433
434        // Second half shares same lower 4 bits
435        for i in 0..4 {
436            degree4_op_flags[i + 4] = degree4_op_flags[i].clone();
437        }
438
439        // Update with op_bit[4] and degree 4 flag
440        let deg_4_not_4 = op[4].not() * degree_4_flag.clone();
441        for flag in degree4_op_flags.iter_mut().take(4) {
442            *flag = flag.clone() * deg_4_not_4.clone();
443        }
444        let deg_4_yes_4 = op[4].is() * degree_4_flag.clone();
445        for flag in degree4_op_flags.iter_mut().skip(4) {
446            *flag = flag.clone() * deg_4_yes_4.clone();
447        }
448
449        // --- No shift composite flags computation ---
450
451        // Flag for END operation causing stack to shift left (depends on whether in loop)
452        let shift_left_on_end = degree4_op_flags[4].clone() * frame.is_loop_end();
453
454        no_shift_flags[0] = degree7_op_flags[0].clone() // NOOP
455            + degree6_op_flags[5].clone() // U32ASSERT2
456            + degree5_op_flags[1].clone() // MPVERIFY
457            + degree5_op_flags[6].clone() // SPAN
458            + degree5_op_flags[7].clone() // JOIN
459            + degree7_op_flags[31].clone() // EMIT
460            + degree4_op_flags[6].clone() // RESPAN
461            + degree4_op_flags[7].clone() // HALT
462            + degree4_op_flags[3].clone() // CALL
463            + degree4_op_flags[4].clone() * binary_not(frame.is_loop_end()); // END (non-loop)
464
465        no_shift_flags[1] = no_shift_flags[0].clone() + no_change_1_flag;
466        no_shift_flags[2] = no_shift_flags[1].clone()
467            + degree7_op_flags[8].clone() // SWAP
468            + f1000.clone(); // u32 arithmetic
469        no_shift_flags[3] = no_shift_flags[2].clone() + mov2_flag.clone();
470        no_shift_flags[4] = no_shift_flags[3].clone()
471            + mov3_flag.clone()
472            + adv_popw_expacc.clone()
473            + swapwx_flag.clone()
474            + ext2mul_flag.clone()
475            + degree4_op_flags[0].clone(); // MRUPDATE
476
477        no_shift_flags[5] = no_shift_flags[4].clone() + mov4_flag.clone();
478        no_shift_flags[6] = no_shift_flags[5].clone() + mov5_flag.clone();
479        no_shift_flags[7] = no_shift_flags[6].clone() + mov6_flag.clone();
480        no_shift_flags[8] =
481            no_shift_flags[7].clone() + mov7_flag.clone() + degree7_op_flags[24].clone()
482                - degree7_op_flags[28].clone();
483
484        no_shift_flags[9] = no_shift_flags[8].clone() + mov8_flag.clone();
485        no_shift_flags[10] = no_shift_flags[9].clone();
486        no_shift_flags[11] = no_shift_flags[9].clone();
487        // SWAPW3; SWAPW2; HPERM
488        no_shift_flags[12] = no_shift_flags[9].clone() - degree7_op_flags[29].clone()
489            + degree7_op_flags[28].clone()
490            + degree5_op_flags[0].clone();
491        no_shift_flags[13] = no_shift_flags[12].clone();
492        no_shift_flags[14] = no_shift_flags[12].clone();
493        no_shift_flags[15] = no_shift_flags[12].clone();
494
495        // --- Left shift composite flags computation ---
496
497        let movdnn_flag = degree7_op_flags[11].clone()
498            + degree7_op_flags[13].clone()
499            + degree7_op_flags[17].clone()
500            + degree7_op_flags[19].clone()
501            + degree7_op_flags[21].clone()
502            + degree7_op_flags[23].clone()
503            + degree7_op_flags[27].clone();
504
505        let split_loop_flag = degree5_op_flags[4].clone() + degree5_op_flags[5].clone();
506        let add3_madd_flag = degree6_op_flags[6].clone() + degree6_op_flags[7].clone();
507
508        left_shift_flags[1] = degree7_op_flags[32].clone()
509            + movdnn_flag.clone()
510            + degree7_op_flags[41].clone()
511            + degree7_op_flags[45].clone()
512            + degree7_op_flags[47].clone()
513            + degree7_op_flags[46].clone()
514            + split_loop_flag.clone()
515            + shift_left_on_end.clone()
516            + degree5_op_flags[8].clone() // DYN
517            + degree5_op_flags[12].clone(); // DYNCALL
518
519        left_shift_flags[2] = left_shift_flags[1].clone() + left_change_1_flag;
520        left_shift_flags[3] =
521            left_shift_flags[2].clone() + add3_madd_flag.clone() + degree7_op_flags[42].clone()
522                - degree7_op_flags[11].clone();
523        left_shift_flags[4] = left_shift_flags[3].clone() - degree7_op_flags[13].clone();
524        left_shift_flags[5] = left_shift_flags[4].clone() + degree7_op_flags[44].clone()
525            - degree7_op_flags[17].clone();
526        left_shift_flags[6] = left_shift_flags[5].clone() - degree7_op_flags[19].clone();
527        left_shift_flags[7] = left_shift_flags[6].clone() - degree7_op_flags[21].clone();
528        left_shift_flags[8] = left_shift_flags[7].clone() - degree7_op_flags[23].clone();
529        left_shift_flags[9] = left_shift_flags[8].clone() + degree7_op_flags[43].clone()
530            - degree7_op_flags[27].clone();
531        left_shift_flags[10] = left_shift_flags[9].clone();
532        left_shift_flags[11] = left_shift_flags[9].clone();
533        left_shift_flags[12] = left_shift_flags[9].clone();
534        left_shift_flags[13] = left_shift_flags[9].clone();
535        left_shift_flags[14] = left_shift_flags[9].clone();
536        left_shift_flags[15] = left_shift_flags[9].clone();
537
538        // --- Right shift composite flags computation ---
539
540        let movupn_flag = degree7_op_flags[10].clone()
541            + degree7_op_flags[12].clone()
542            + degree7_op_flags[16].clone()
543            + degree7_op_flags[18].clone()
544            + degree7_op_flags[20].clone()
545            + degree7_op_flags[22].clone()
546            + degree7_op_flags[26].clone();
547
548        right_shift_flags[0] = f011.clone()
549            + degree5_op_flags[11].clone() // PUSH
550            + movupn_flag.clone();
551
552        right_shift_flags[1] = right_shift_flags[0].clone() + degree6_op_flags[4].clone(); // U32SPLIT
553
554        right_shift_flags[2] = right_shift_flags[1].clone() - degree7_op_flags[10].clone();
555        right_shift_flags[3] = right_shift_flags[2].clone() - degree7_op_flags[12].clone();
556        right_shift_flags[4] = right_shift_flags[3].clone() - degree7_op_flags[16].clone();
557        right_shift_flags[5] = right_shift_flags[4].clone() - degree7_op_flags[18].clone();
558        right_shift_flags[6] = right_shift_flags[5].clone() - degree7_op_flags[20].clone();
559        right_shift_flags[7] = right_shift_flags[6].clone() - degree7_op_flags[22].clone();
560        right_shift_flags[8] = right_shift_flags[7].clone() - degree7_op_flags[26].clone();
561        right_shift_flags[9] = right_shift_flags[8].clone();
562        right_shift_flags[10] = right_shift_flags[8].clone();
563        right_shift_flags[11] = right_shift_flags[8].clone();
564        right_shift_flags[12] = right_shift_flags[8].clone();
565        right_shift_flags[13] = right_shift_flags[8].clone();
566        right_shift_flags[14] = right_shift_flags[8].clone();
567        right_shift_flags[15] = right_shift_flags[8].clone();
568
569        // --- Other composite flags ---
570
571        // Flag if stack shifted right (degree 6, dominated by U32SPLIT)
572        // Uses prefix_011 (degree 3) instead of f011 (degree 4) for lower base degree
573        let right_shift = prefix_011.clone()
574            + degree5_op_flags[11].clone() // PUSH
575            + degree6_op_flags[4].clone(); // U32SPLIT
576
577        // Flag if stack shifted left (degree 5).
578        // Uses low-degree prefixes to keep left_shift at degree 5 (avoids degree growth).
579        // Note: DYNCALL is intentionally excluded; see stack overflow depth constraints.
580        let left_shift = prefix_010.clone()
581            + add3_madd_prefix.clone()
582            + split_loop_flag
583            + degree4_op_flags[5].clone() // REPEAT
584            + shift_left_on_end
585            + degree5_op_flags[8].clone(); // DYN
586
587        // Flag if current operation is a control flow operation.
588        //
589        // Control flow operations are the only operations that can execute when outside a basic
590        // block (i.e., when in_span = 0). This is enforced by the decoder constraint:
591        //   (1 - in_span) * (1 - control_flow) = 0
592        //
593        // Control flow operations (must include ALL of these):
594        // - Block starters: SPAN, JOIN, SPLIT, LOOP
595        // - Block transitions: END, REPEAT, RESPAN, HALT
596        // - Dynamic execution: DYN, DYNCALL
597        // - Procedure calls: CALL, SYSCALL
598        //
599        // IMPORTANT: If a new control flow operation is added, it MUST be included here,
600        // otherwise the decoder constraint will fail when executing that operation.
601        let control_flow = degree_5_flag * op[3].not() * op[2].is() // SPAN, JOIN, SPLIT, LOOP
602            + degree_4_flag * op[4].is() // END, REPEAT, RESPAN, HALT
603            + degree5_op_flags[8].clone() // DYN
604            + degree5_op_flags[12].clone() // DYNCALL
605            + degree4_op_flags[2].clone() // SYSCALL
606            + degree4_op_flags[3].clone(); // CALL
607
608        // Flag if current operation is a degree 6 u32 operation
609        let u32_rc_op = f100;
610
611        // Flag if overflow table contains values
612        let overflow = (frame.stack_depth() - E::from_u64(16)) * frame.overflow_register();
613
614        Self {
615            degree7_op_flags,
616            degree6_op_flags,
617            degree5_op_flags,
618            degree4_op_flags,
619            no_shift_flags,
620            left_shift_flags,
621            right_shift_flags,
622            left_shift,
623            right_shift,
624            control_flow,
625            overflow,
626            u32_rc_op,
627        }
628    }

945/// Maps opcode of an operation to the index in its respective degree flag array.
946pub const fn get_op_index(opcode: u8) -> usize {
947    let opcode = opcode as usize;
948
949    if opcode <= DEGREE_7_OPCODE_ENDS {
950        // Index of a degree 7 operation (0-63)
951        opcode
952    } else if opcode <= DEGREE_6_OPCODE_ENDS {
953        // Index of a degree 6 operation (64-79, even opcodes only)
954        (opcode - DEGREE_6_OPCODE_STARTS) / 2
955    } else if opcode <= DEGREE_5_OPCODE_ENDS {
956        // Index of a degree 5 operation (80-95)
957        opcode - DEGREE_5_OPCODE_STARTS
958    } else {
959        // Index of a degree 4 operation (96-127, every 4th opcode)
960        (opcode - DEGREE_4_OPCODE_STARTS) / 4
961    }
962}

967/// Generates a test trace row with the op bits set for a given opcode.
968///
969/// This creates a minimal trace row where:
970/// - Op bits are set according to the opcode's binary representation
971/// - Op bits extra columns are computed for degree reduction
972/// - All other columns are zero
973#[cfg(test)]
974pub fn generate_test_row(opcode: usize) -> crate::MainTraceRow<miden_core::Felt> {
975    use miden_core::{ONE, ZERO};
976
977    use crate::trace::{
978        CHIPLETS_WIDTH, DECODER_TRACE_WIDTH, RANGE_CHECK_TRACE_WIDTH, STACK_TRACE_WIDTH,
979        decoder::OP_BITS_EXTRA_COLS_RANGE,
980    };
981
982    // Get op bits for this opcode
983    let op_bits = get_op_bits(opcode);
984
985    // Initialize decoder array with zeros
986    let mut decoder = [ZERO; DECODER_TRACE_WIDTH];
987
988    // Set op bits (indices 1-7 in decoder, after addr column at index 0)
989    for (i, &bit) in op_bits.iter().enumerate() {
990        decoder[OP_BITS_RANGE.start + i] = bit;
991    }
992
993    // Compute and set op bits extra columns for degree reduction
994    let bit_6 = op_bits[6];
995    let bit_5 = op_bits[5];
996    let bit_4 = op_bits[4];
997
998    // op_bit_extra[0] = bit_6 * (1 - bit_5) * bit_4 (degree 5 flag)
999    decoder[OP_BITS_EXTRA_COLS_RANGE.start] = bit_6 * (ONE - bit_5) * bit_4;
1000
1001    // op_bit_extra[1] = bit_6 * bit_5 (degree 4 flag)
1002    decoder[OP_BITS_EXTRA_COLS_RANGE.start + 1] = bit_6 * bit_5;
1003
1004    crate::MainTraceRow {
1005        clk: ZERO,
1006        ctx: ZERO,
1007        fn_hash: [ZERO; 4],
1008        decoder,
1009        stack: [ZERO; STACK_TRACE_WIDTH],
1010        range: [ZERO; RANGE_CHECK_TRACE_WIDTH],
1011        chiplets: [ZERO; CHIPLETS_WIDTH],
1012    }
1013}

1015/// Returns a 7-bit array representation of an opcode.
1016#[cfg(test)]
1017pub fn get_op_bits(opcode: usize) -> [miden_core::Felt; NUM_OP_BITS] {
1018    use miden_core::{Felt, ZERO};
1019
1020    let mut opcode_copy = opcode;
1021    let mut bit_array = [ZERO; NUM_OP_BITS];
1022
1023    for bit in bit_array.iter_mut() {
1024        *bit = Felt::new((opcode_copy & 1) as u64);
1025        opcode_copy >>= 1;
1026    }
1027
1028    assert_eq!(opcode_copy, 0, "Opcode must be 7 bits");
1029    bit_array
1030}
