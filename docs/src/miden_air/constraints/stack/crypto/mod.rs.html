120/// Enforces crypto operation constraints.
121pub fn enforce_main<AB>(
122    builder: &mut AB,
123    local: &MainTraceRow<AB::Var>,
124    next: &MainTraceRow<AB::Var>,
125    op_flags: &OpFlags<AB::Expr>,
126) where
127    AB: LiftedAirBuilder,
128{
129    let mut idx = 0usize;
130    enforce_cryptostream_constraints(builder, local, next, op_flags, &mut idx);
131    enforce_hornerbase_constraints(builder, local, next, op_flags, &mut idx);
132    enforce_hornerext_constraints(builder, local, next, op_flags, &mut idx);
133    enforce_frie2f4_constraints(builder, local, next, op_flags, &mut idx);
134}

139fn enforce_cryptostream_constraints<AB>(
140    builder: &mut AB,
141    local: &MainTraceRow<AB::Var>,
142    next: &MainTraceRow<AB::Var>,
143    op_flags: &OpFlags<AB::Expr>,
144    idx: &mut usize,
145) where
146    AB: LiftedAirBuilder,
147{
148    // CRYPTOSTREAM keeps the top of the stack stable except for the two counters
149    // that track the stream offset. Those counters advance by 8 (one word) per row.
150    // Everything is gated by the op flag, so the constraints are active only when
151    // CRYPTOSTREAM is executed.
152    let eight: AB::Expr = AB::Expr::from_u16(8);
153    let gate = op_flags.cryptostream();
154
155    assert_zero(
156        builder,
157        idx,
158        gate.clone() * (next.stack[8].clone().into() - local.stack[8].clone().into()),
159    );
160    assert_zero(
161        builder,
162        idx,
163        gate.clone() * (next.stack[9].clone().into() - local.stack[9].clone().into()),
164    );
165    assert_zero(
166        builder,
167        idx,
168        gate.clone() * (next.stack[10].clone().into() - local.stack[10].clone().into()),
169    );
170    assert_zero(
171        builder,
172        idx,
173        gate.clone() * (next.stack[11].clone().into() - local.stack[11].clone().into()),
174    );
175    assert_zero(
176        builder,
177        idx,
178        gate.clone()
179            * (next.stack[12].clone().into() - (local.stack[12].clone().into() + eight.clone())),
180    );
181    assert_zero(
182        builder,
183        idx,
184        gate.clone() * (next.stack[13].clone().into() - (local.stack[13].clone().into() + eight)),
185    );
186    assert_zero(
187        builder,
188        idx,
189        gate.clone() * (next.stack[14].clone().into() - local.stack[14].clone().into()),
190    );
191    assert_zero(
192        builder,
193        idx,
194        gate * (next.stack[15].clone().into() - local.stack[15].clone().into()),
195    );
196}

198fn enforce_hornerbase_constraints<AB>(
199    builder: &mut AB,
200    local: &MainTraceRow<AB::Var>,
201    next: &MainTraceRow<AB::Var>,
202    op_flags: &OpFlags<AB::Expr>,
203    idx: &mut usize,
204) where
205    AB: LiftedAirBuilder,
206{
207    // HORNERBASE evaluates a degree-7 polynomial over the quadratic extension.
208    // The accumulator lives in stack[14..16] and is updated using helper values
209    // supplied (nondeterministically) through the decoder helper registers. We
210    // enforce the algebraic relationships that must hold between the helpers, the
211    // coefficients on the stack, and the accumulator update. These constraints
212    // also implicitly bind the helper values to the required power relations.
213    let gate = op_flags.hornerbase();
214
215    // The lower 14 stack registers remain unchanged during HORNERBASE.
216    for i in 0..14 {
217        assert_zero(
218            builder,
219            idx,
220            gate.clone() * (next.stack[i].clone().into() - local.stack[i].clone().into()),
221        );
222    }
223
224    // Decoder helper columns contain alpha components and intermediate temporaries.
225    // We read them starting at USER_OP_HELPERS_OFFSET to avoid hardcoding column indices.
226    let base = USER_OP_HELPERS_OFFSET;
227    let a0: AB::Expr = local.decoder[base].clone().into();
228    let a1: AB::Expr = local.decoder[base + 1].clone().into();
229    let tmp1_0: AB::Expr = local.decoder[base + 2].clone().into();
230    let tmp1_1: AB::Expr = local.decoder[base + 3].clone().into();
231    let tmp0_0: AB::Expr = local.decoder[base + 4].clone().into();
232    let tmp0_1: AB::Expr = local.decoder[base + 5].clone().into();
233
234    let acc0: AB::Expr = local.stack[14].clone().into();
235    let acc1: AB::Expr = local.stack[15].clone().into();
236    let acc0_next: AB::Expr = next.stack[14].clone().into();
237    let acc1_next: AB::Expr = next.stack[15].clone().into();
238
239    // Coefficients are read from the bottom of the stack.
240    let c: [AB::Expr; 8] = core::array::from_fn(|i| local.stack[i].clone().into());
241
242    // Quadratic extension view (Fp2 with u^2 = 7):
243    // - alpha = (a0, a1), acc = (acc0, acc1)
244    // - tmp0 = (tmp0_0, tmp0_1), tmp1 = (tmp1_0, tmp1_1)
245    // - c[0..7] are base-field scalars
246    //
247    // Horner form:
248    //   tmp0 = acc * alpha^2 + (c0 * alpha + c1)
249    //   tmp1 = tmp0 * alpha^3 + (c2 * alpha^2 + c3 * alpha + c4)
250    //   acc' = tmp1 * alpha^3 + (c5 * alpha^2 + c6 * alpha + c7)
251    let alpha: QuadFeltExpr<AB::Expr> = QuadFeltExpr::new(&a0, &a1);
252    let alpha2 = alpha.clone().square();
253    let alpha3 = alpha2.clone() * alpha.clone();
254    let acc: QuadFeltExpr<AB::Expr> = QuadFeltExpr::new(&acc0, &acc1);
255    let acc_alpha2: QuadFeltExpr<AB::Expr> = acc * alpha2.clone();
256    let tmp0: QuadFeltExpr<AB::Expr> = QuadFeltExpr::new(&tmp0_0, &tmp0_1);
257    let tmp0_alpha3: QuadFeltExpr<AB::Expr> = tmp0.clone() * alpha3.clone();
258    let tmp1: QuadFeltExpr<AB::Expr> = QuadFeltExpr::new(&tmp1_0, &tmp1_1);
259
260    // tmp0 = acc * alpha^2 + (c0 * alpha + c1)
261    let tmp0_expected: QuadFeltExpr<AB::Expr> =
262        acc_alpha2 + alpha.clone() * c[0].clone() + c[1].clone();
263    let [tmp0_exp_0, tmp0_exp_1] = tmp0_expected.into_parts();
264
265    // tmp1 = tmp0 * alpha^3 + (c2 * alpha^2 + c3 * alpha + c4)
266    let tmp1_expected: QuadFeltExpr<AB::Expr> =
267        tmp0_alpha3 + alpha2.clone() * c[2].clone() + alpha.clone() * c[3].clone() + c[4].clone();
268    let [tmp1_exp_0, tmp1_exp_1] = tmp1_expected.into_parts();
269
270    // acc' = tmp1 * alpha^3 + (alpha^2 * c5 + alpha * c6 + c7)
271    let acc_expected: QuadFeltExpr<AB::Expr> =
272        tmp1 * alpha3 + alpha2.clone() * c[5].clone() + alpha.clone() * c[6].clone() + c[7].clone();
273    let [acc_exp_0, acc_exp_1] = acc_expected.into_parts();
274
275    assert_zero(builder, idx, gate.clone() * (tmp0_0 - tmp0_exp_0));
276    assert_zero(builder, idx, gate.clone() * (tmp0_1 - tmp0_exp_1));
277    assert_zero(builder, idx, gate.clone() * (tmp1_0 - tmp1_exp_0));
278    assert_zero(builder, idx, gate.clone() * (tmp1_1 - tmp1_exp_1));
279    assert_zero(builder, idx, gate.clone() * (acc0_next - acc_exp_0));
280    assert_zero(builder, idx, gate * (acc1_next - acc_exp_1));
281}

283fn enforce_hornerext_constraints<AB>(
284    builder: &mut AB,
285    local: &MainTraceRow<AB::Var>,
286    next: &MainTraceRow<AB::Var>,
287    op_flags: &OpFlags<AB::Expr>,
288    idx: &mut usize,
289) where
290    AB: LiftedAirBuilder,
291{
292    // HORNEREXT evaluates a degree-3 polynomial over the quadratic extension with
293    // a smaller helper set. As with HORNERBASE, helper values are supplied via
294    // decoder columns, and the constraints below bind them to the required
295    // power relations and accumulator update.
296    let gate = op_flags.hornerext();
297
298    // The lower 14 stack registers are unchanged by HORNEREXT.
299    for i in 0..14 {
300        assert_zero(
301            builder,
302            idx,
303            gate.clone() * (next.stack[i].clone().into() - local.stack[i].clone().into()),
304        );
305    }
306
307    // Helper columns and accumulator values.
308    let base = USER_OP_HELPERS_OFFSET;
309    let a0: AB::Expr = local.decoder[base].clone().into();
310    let a1: AB::Expr = local.decoder[base + 1].clone().into();
311    let tmp0: AB::Expr = local.decoder[base + 4].clone().into();
312    let tmp1: AB::Expr = local.decoder[base + 5].clone().into();
313
314    let acc0: AB::Expr = local.stack[14].clone().into();
315    let acc1: AB::Expr = local.stack[15].clone().into();
316    let acc0_next: AB::Expr = next.stack[14].clone().into();
317    let acc1_next: AB::Expr = next.stack[15].clone().into();
318
319    // Coefficients live at the bottom of the stack.
320    let s: [AB::Expr; 8] = core::array::from_fn(|i| local.stack[i].clone().into());
321
322    // Quadratic extension view (Fp2 with u^2 = 7):
323    // - alpha = (a0, a1), acc = (acc0, acc1), tmp = (tmp0, tmp1)
324    // - extension coefficients use the stack pairs: c0 = (s0, s1), c1 = (s2, s3), c2 = (s4, s5), c3
325    //   = (s6, s7)
326    //
327    // Horner form:
328    //   tmp  = acc * alpha^2 + (c0 * alpha + c1)
329    //   acc' = tmp * alpha^2 + (c2 * alpha + c3)
330    let alpha: QuadFeltExpr<AB::Expr> = QuadFeltExpr::new(&a0, &a1);
331    let alpha2 = alpha.clone().square();
332    let acc: QuadFeltExpr<AB::Expr> = QuadFeltExpr::new(&acc0, &acc1);
333    let acc_alpha2: QuadFeltExpr<AB::Expr> = acc * alpha2.clone();
334    let tmp: QuadFeltExpr<AB::Expr> = QuadFeltExpr::new(&tmp0, &tmp1);
335    let tmp_alpha2: QuadFeltExpr<AB::Expr> = tmp * alpha2;
336
337    let c0 = QuadFeltExpr::new(&s[0], &s[1]);
338    let c1 = QuadFeltExpr::new(&s[2], &s[3]);
339    let c2 = QuadFeltExpr::new(&s[4], &s[5]);
340    let c3 = QuadFeltExpr::new(&s[6], &s[7]);
341
342    // tmp  = acc * alpha^2 + (c0 * alpha + c1)
343    let tmp_expected: QuadFeltExpr<AB::Expr> = acc_alpha2 + alpha.clone() * c0 + c1;
344    let [tmp_exp_0, tmp_exp_1] = tmp_expected.into_parts();
345
346    // acc' = tmp * alpha^2 + (c2 * alpha + c3)
347    let acc_expected: QuadFeltExpr<AB::Expr> = tmp_alpha2 + alpha * c2 + c3;
348    let [acc_exp_0, acc_exp_1] = acc_expected.into_parts();
349
350    assert_zero(builder, idx, gate.clone() * (tmp0 - tmp_exp_0));
351    assert_zero(builder, idx, gate.clone() * (tmp1 - tmp_exp_1));
352    assert_zero(builder, idx, gate.clone() * (acc0_next - acc_exp_0));
353    assert_zero(builder, idx, gate * (acc1_next - acc_exp_1));
354}

356/// Enforces constraints for the FRI ext2fold4 operation.
357///
358/// This operation folds 4 query values into a single value for the FRI protocol.
359///
360/// Input stack:
361///   [v0, v1, v2, v3, v4, v5, v6, v7, f_pos, p, poe, pe0, pe1, a0, a1, cptr]
362///
363/// Output stack:
364///   [t0_0, t0_1, t1_0, t1_1, f0, f1, f2, f3, poe^2, f_tau, cptr+8, poe^4, f_pos, ne0, ne1, eptr]
365///
366/// Helper registers: [ev0, ev1, es0, es1, x, x_inv]
367fn enforce_frie2f4_constraints<AB>(
368    builder: &mut AB,
369    local: &MainTraceRow<AB::Var>,
370    next: &MainTraceRow<AB::Var>,
371    op_flags: &OpFlags<AB::Expr>,
372    idx: &mut usize,
373) where
374    AB: LiftedAirBuilder,
375{
376    let gate = op_flags.frie2f4();
377
378    // --- read current state ---
379    let v0: AB::Expr = local.stack[0].clone().into();
380    let v1: AB::Expr = local.stack[1].clone().into();
381    let v2: AB::Expr = local.stack[2].clone().into();
382    let v3: AB::Expr = local.stack[3].clone().into();
383    let v4: AB::Expr = local.stack[4].clone().into();
384    let v5: AB::Expr = local.stack[5].clone().into();
385    let v6: AB::Expr = local.stack[6].clone().into();
386    let v7: AB::Expr = local.stack[7].clone().into();
387    let f_pos: AB::Expr = local.stack[8].clone().into();
388    let p: AB::Expr = local.stack[9].clone().into();
389    let poe: AB::Expr = local.stack[10].clone().into();
390    let pe0: AB::Expr = local.stack[11].clone().into();
391    let pe1: AB::Expr = local.stack[12].clone().into();
392    let a0: AB::Expr = local.stack[13].clone().into();
393    let a1: AB::Expr = local.stack[14].clone().into();
394    let cptr: AB::Expr = local.stack[15].clone().into();
395
396    // --- read helper registers ---
397    let base = USER_OP_HELPERS_OFFSET;
398    let ev0: AB::Expr = local.decoder[base].clone().into();
399    let ev1: AB::Expr = local.decoder[base + 1].clone().into();
400    let es0: AB::Expr = local.decoder[base + 2].clone().into();
401    let es1: AB::Expr = local.decoder[base + 3].clone().into();
402    let x: AB::Expr = local.decoder[base + 4].clone().into();
403    let x_inv: AB::Expr = local.decoder[base + 5].clone().into();
404
405    // --- read next state ---
406    let t0_0: AB::Expr = next.stack[0].clone().into();
407    let t0_1: AB::Expr = next.stack[1].clone().into();
408    let t1_0: AB::Expr = next.stack[2].clone().into();
409    let t1_1: AB::Expr = next.stack[3].clone().into();
410    let f0: AB::Expr = next.stack[4].clone().into();
411    let f1: AB::Expr = next.stack[5].clone().into();
412    let f2: AB::Expr = next.stack[6].clone().into();
413    let f3: AB::Expr = next.stack[7].clone().into();
414    let poe2: AB::Expr = next.stack[8].clone().into();
415    let f_tau: AB::Expr = next.stack[9].clone().into();
416    let cptr_next: AB::Expr = next.stack[10].clone().into();
417    let poe4: AB::Expr = next.stack[11].clone().into();
418    let f_pos_next: AB::Expr = next.stack[12].clone().into();
419    let ne0: AB::Expr = next.stack[13].clone().into();
420    let ne1: AB::Expr = next.stack[14].clone().into();
421
422    // --- constants ---
423    let two: AB::Expr = AB::Expr::from_u16(2);
424    let three: AB::Expr = AB::Expr::from_u16(3);
425    let eight: AB::Expr = AB::Expr::from_u16(8);
426    // tau^{-1}, tau^{-2}, tau^{-3} where tau is the 4th root of unity
427    let tau_inv: AB::Expr = AB::Expr::from_u64(18446462594437873665);
428    let tau2_inv: AB::Expr = AB::Expr::from_u64(18446744069414584320);
429    let tau3_inv: AB::Expr = AB::Expr::from_u64(281474976710656);
430
431    // ========================================================================
432    // Group 1: Domain flags are binary and exactly one is set (5 constraints)
433    // ========================================================================
434
435    // f0^2 = f0
436    assert_zero(builder, idx, gate.clone() * (f0.clone() * f0.clone() - f0.clone()));
437    // f1^2 = f1
438    assert_zero(builder, idx, gate.clone() * (f1.clone() * f1.clone() - f1.clone()));
439    // f2^2 = f2
440    assert_zero(builder, idx, gate.clone() * (f2.clone() * f2.clone() - f2.clone()));
441    // f3^2 = f3
442    assert_zero(builder, idx, gate.clone() * (f3.clone() * f3.clone() - f3.clone()));
443    // f0 + f1 + f2 + f3 = 1
444    assert_zero(
445        builder,
446        idx,
447        gate.clone() * (f0.clone() + f1.clone() + f2.clone() + f3.clone() - AB::Expr::ONE),
448    );
449
450    // ========================================================================
451    // Group 2: p = 4 * f_pos_next + d_seg_from_flags (1 constraint)
452    // ========================================================================
453    // This single constraint binds p to both f_pos (output at next.stack[12]) and the
454    // domain segment flags. d_seg_from_flags uses the bit-reversal mapping:
455    //   f0=1 (seg 0) -> d_seg=0, f1=1 (seg 1) -> d_seg=2,
456    //   f2=1 (seg 2) -> d_seg=1, f3=1 (seg 3) -> d_seg=3
457    // So: d_seg = 2*f1 + f2 + 3*f3, and p = 4*f_pos + d_seg.
458    let four: AB::Expr = AB::Expr::from_u16(4);
459    assert_zero(
460        builder,
461        idx,
462        gate.clone()
463            * (p - (four * f_pos_next.clone()
464                + two.clone() * f1.clone()
465                + f2.clone()
466                + three * f3.clone())),
467    );
468
469    // ========================================================================
470    // Group 3: Tau factor from domain flags (1 constraint)
471    // ========================================================================
472    // f_tau = f0*1 + f1*TAU_INV + f2*TAU2_INV + f3*TAU3_INV
473    assert_zero(
474        builder,
475        idx,
476        gate.clone()
477            * (f_tau.clone()
478                - (f0.clone()
479                    + tau_inv.clone() * f1.clone()
480                    + tau2_inv * f2.clone()
481                    + tau3_inv * f3.clone())),
482    );
483
484    // ========================================================================
485    // Group 4: x = poe * f_tau, x * x_inv = 1 (2 constraints)
486    // ========================================================================
487    assert_zero(builder, idx, gate.clone() * (x.clone() - poe.clone() * f_tau));
488    assert_zero(builder, idx, gate.clone() * (x * x_inv.clone() - AB::Expr::ONE));
489
490    // ========================================================================
491    // Group 5: ev = alpha * x_inv (2 constraints)
492    // ========================================================================
493    let alpha = QuadFeltExpr(a0, a1);
494    let ev = QuadFeltExpr(ev0, ev1);
495    let [c0, c1] = (ev.clone() - alpha * x_inv.clone()).into_parts();
496    assert_zero(builder, idx, gate.clone() * c0);
497    assert_zero(builder, idx, gate.clone() * c1);
498
499    // ========================================================================
500    // Group 6: es = ev^2 in Fp2 (2 constraints)
501    // ========================================================================
502    let es = QuadFeltExpr(es0, es1);
503    let [c0, c1] = (es.clone() - ev.clone().square()).into_parts();
504    assert_zero(builder, idx, gate.clone() * c0);
505    assert_zero(builder, idx, gate.clone() * c1);
506
507    // ========================================================================
508    // Group 7: tmp0 = fold2(q0, q2, ev) (2 constraints)
509    // ========================================================================
510    // fold2(a, b, ep) = ((a+b) + (a-b)*ep) / 2
511    let q0 = QuadFeltExpr(v0.clone(), v1.clone());
512    let q2 = QuadFeltExpr(v2.clone(), v3.clone());
513    let tmp0 = QuadFeltExpr(t0_0.clone(), t0_1.clone());
514    let [c0, c1] =
515        (tmp0.clone() * two.clone() - (q0.clone() + q2.clone()) - (q0 - q2) * ev.clone())
516            .into_parts();
517    assert_zero(builder, idx, gate.clone() * c0);
518    assert_zero(builder, idx, gate.clone() * c1);
519
520    // ========================================================================
521    // Group 8: tmp1 = fold2(q1, q3, ev * TAU_INV) (2 constraints)
522    // ========================================================================
523    let q1 = QuadFeltExpr(v4.clone(), v5.clone());
524    let q3 = QuadFeltExpr(v6.clone(), v7.clone());
525    let tmp1 = QuadFeltExpr(t1_0.clone(), t1_1.clone());
526    let ev_tau = ev * tau_inv.clone();
527    let [c0, c1] =
528        (tmp1.clone() * two.clone() - (q1.clone() + q3.clone()) - (q1 - q3) * ev_tau).into_parts();
529    assert_zero(builder, idx, gate.clone() * c0);
530    assert_zero(builder, idx, gate.clone() * c1);
531
532    // ========================================================================
533    // Group 9: folded = fold2(tmp0, tmp1, es) (2 constraints)
534    // ========================================================================
535    let ne = QuadFeltExpr(ne0, ne1);
536    let [c0, c1] =
537        (ne * two.clone() - (tmp0.clone() + tmp1.clone()) - (tmp0 - tmp1) * es).into_parts();
538    assert_zero(builder, idx, gate.clone() * c0);
539    assert_zero(builder, idx, gate.clone() * c1);
540
541    // ========================================================================
542    // Group 10: Consistency check -- pe = query_values[d_seg] (2 constraints)
543    // ========================================================================
544    // d_seg is the bit-reversed coset index. The query_values are also in bit-reversed
545    // order, so indexing by d_seg directly gives the correct value.
546    //   d_seg=0 -> f0=1 -> q[0]=(v0,v1)
547    //   d_seg=1 -> f2=1 -> q[1]=(v2,v3)
548    //   d_seg=2 -> f1=1 -> q[2]=(v4,v5)
549    //   d_seg=3 -> f3=1 -> q[3]=(v6,v7)
550    assert_zero(
551        builder,
552        idx,
553        gate.clone()
554            * (pe0 - (v0 * f0.clone() + v4 * f1.clone() + v2 * f2.clone() + v6 * f3.clone())),
555    );
556    assert_zero(builder, idx, gate.clone() * (pe1 - (v1 * f0 + v5 * f1 + v3 * f2 + v7 * f3)));
557
558    // ========================================================================
559    // Group 11: poe powers (2 constraints)
560    // ========================================================================
561    // poe^2 = poe * poe
562    assert_zero(builder, idx, gate.clone() * (poe2.clone() - poe.clone() * poe));
563    // poe^4 = poe^2 * poe^2
564    assert_zero(builder, idx, gate.clone() * (poe4 - poe2.clone() * poe2));
565
566    // ========================================================================
567    // Group 12: Pointer increment and position transfer (2 constraints)
568    // ========================================================================
569    // cptr' = cptr + 8
570    assert_zero(builder, idx, gate.clone() * (cptr_next - cptr - eight));
571    // f_pos' = f_pos
572    assert_zero(builder, idx, gate * (f_pos_next - f_pos));
573}

575fn assert_zero<AB: TaggingAirBuilderExt>(builder: &mut AB, idx: &mut usize, expr: AB::Expr) {
576    tagged_assert_zero_integrity(builder, &STACK_CRYPTO_TAGS, idx, expr);
