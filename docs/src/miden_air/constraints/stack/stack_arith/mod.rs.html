100/// Enforces stack arith/u32 constraints.
101pub fn enforce_main<AB>(
102    builder: &mut AB,
103    local: &MainTraceRow<AB::Var>,
104    next: &MainTraceRow<AB::Var>,
105    op_flags: &OpFlags<AB::Expr>,
106) where
107    AB: LiftedAirBuilder,
108{
109    let mut idx = 0usize;
110
111    let s0: AB::Expr = local.stack[0].clone().into();
112    let s1: AB::Expr = local.stack[1].clone().into();
113    let s2: AB::Expr = local.stack[2].clone().into();
114    let s3: AB::Expr = local.stack[3].clone().into();
115
116    let s0_next: AB::Expr = next.stack[0].clone().into();
117    let s1_next: AB::Expr = next.stack[1].clone().into();
118    let s2_next: AB::Expr = next.stack[2].clone().into();
119    let s3_next: AB::Expr = next.stack[3].clone().into();
120
121    // Decoder helper columns: h0..h5 are stored starting at USER_OP_HELPERS_OFFSET.
122    // These helpers are op-specific and are validated by the constraints below.
123    // - h0 is used as an inverse witness (EQ/EQZ) or exp_val (EXPACC).
124    // - h1..h4 hold u32 limbs / range-check witnesses for u32 ops.
125    // - h5 is currently unused in this module.
126    let base = USER_OP_HELPERS_OFFSET;
127    let uop_h0: AB::Expr = local.decoder[base].clone().into();
128    let uop_h1: AB::Expr = local.decoder[base + 1].clone().into();
129    let uop_h2: AB::Expr = local.decoder[base + 2].clone().into();
130    let uop_h3: AB::Expr = local.decoder[base + 3].clone().into();
131    let uop_h4: AB::Expr = local.decoder[base + 4].clone().into();
132
133    // Field ops.
134    let is_add = op_flags.add();
135    let is_neg = op_flags.neg();
136    let is_mul = op_flags.mul();
137    let is_inv = op_flags.inv();
138    let is_incr = op_flags.incr();
139    let is_not = op_flags.not();
140    let is_and = op_flags.and();
141    let is_or = op_flags.or();
142    let is_eq = op_flags.eq();
143    let is_eqz = op_flags.eqz();
144    let is_expacc = op_flags.expacc();
145    let is_ext2mul = op_flags.ext2mul();
146
147    // U32 ops.
148    let is_u32add = op_flags.u32add();
149    let is_u32sub = op_flags.u32sub();
150    let is_u32mul = op_flags.u32mul();
151    let is_u32div = op_flags.u32div();
152    let is_u32split = op_flags.u32split();
153    let is_u32assert2 = op_flags.u32assert2();
154    let is_u32add3 = op_flags.u32add3();
155    let is_u32madd = op_flags.u32madd();
156
157    // -------------------------------------------------------------------------
158    // Field ops
159    // -------------------------------------------------------------------------
160    assert_zero(builder, &mut idx, is_add * (s0_next.clone() - (s0.clone() + s1.clone())));
161    assert_zero(builder, &mut idx, is_neg * (s0_next.clone() + s0.clone()));
162    assert_zero(builder, &mut idx, is_mul * (s0_next.clone() - s0.clone() * s1.clone()));
163    assert_zero(builder, &mut idx, is_inv * (s0_next.clone() * s0.clone() - AB::Expr::ONE));
164    assert_zero(builder, &mut idx, is_incr * (s0_next.clone() - s0.clone() - AB::Expr::ONE));
165
166    assert_zero_integrity(
167        builder,
168        &mut idx,
169        is_not.clone() * (s0.clone() * (s0.clone() - AB::Expr::ONE)),
170    );
171    assert_zero(builder, &mut idx, is_not * (s0.clone() + s0_next.clone() - AB::Expr::ONE));
172
173    assert_zero_integrity(
174        builder,
175        &mut idx,
176        is_and.clone() * (s0.clone() * (s0.clone() - AB::Expr::ONE)),
177    );
178    assert_zero_integrity(
179        builder,
180        &mut idx,
181        is_and.clone() * (s1.clone() * (s1.clone() - AB::Expr::ONE)),
182    );
183    assert_zero(builder, &mut idx, is_and * (s0_next.clone() - s0.clone() * s1.clone()));
184
185    assert_zero_integrity(
186        builder,
187        &mut idx,
188        is_or.clone() * (s0.clone() * (s0.clone() - AB::Expr::ONE)),
189    );
190    assert_zero_integrity(
191        builder,
192        &mut idx,
193        is_or.clone() * (s1.clone() * (s1.clone() - AB::Expr::ONE)),
194    );
195    assert_zero(
196        builder,
197        &mut idx,
198        is_or * (s0_next.clone() - (s0.clone() + s1.clone() - s0.clone() * s1.clone())),
199    );
200
201    // EQ: if s0 != s1, h0 acts as 1/(s0 - s1) and forces s0' = 0; if equal, s0' = 1.
202    let eq_diff = s0.clone() - s1.clone();
203    assert_zero(builder, &mut idx, is_eq.clone() * (eq_diff.clone() * s0_next.clone()));
204    assert_zero(
205        builder,
206        &mut idx,
207        is_eq * (s0_next.clone() - (AB::Expr::ONE - eq_diff * uop_h0.clone())),
208    );
209
210    // EQZ: if s0 != 0, h0 acts as 1/s0 and forces s0' = 0; if zero, s0' = 1.
211    assert_zero(builder, &mut idx, is_eqz.clone() * (s0.clone() * s0_next.clone()));
212    assert_zero(
213        builder,
214        &mut idx,
215        is_eqz * (s0_next.clone() - (AB::Expr::ONE - s0.clone() * uop_h0.clone())),
216    );
217
218    // EXPACC: exp_next = exp^2, exp_val = 1 + (exp - 1) * exp_bit, acc_next = acc * exp_val.
219    let exp = s1.clone();
220    let acc = s2.clone();
221    let exp_bit = s0_next.clone();
222    let exp_next = s1_next.clone();
223    let acc_next = s2_next.clone();
224    let exp_b = s3.clone();
225    let exp_b_next = s3_next.clone();
226    let exp_val = uop_h0.clone();
227    let two: AB::Expr = AB::Expr::from_u16(2);
228
229    assert_zero(builder, &mut idx, is_expacc.clone() * (exp_next - exp.clone() * exp.clone()));
230    assert_zero(
231        builder,
232        &mut idx,
233        is_expacc.clone()
234            * (exp_val.clone() - AB::Expr::ONE - (exp - AB::Expr::ONE) * exp_bit.clone()),
235    );
236    assert_zero(builder, &mut idx, is_expacc.clone() * (acc_next - acc * exp_val));
237    assert_zero(
238        builder,
239        &mut idx,
240        is_expacc.clone() * (exp_b - exp_b_next * two - exp_bit.clone()),
241    );
242    assert_zero(builder, &mut idx, is_expacc * (exp_bit.clone() * (exp_bit - AB::Expr::ONE)));
243
244    let ext_b0 = s0.clone();
245    let ext_b1 = s1.clone();
246    let ext_a0 = s2.clone();
247    let ext_a1 = s3.clone();
248    let ext_d0 = s0_next.clone();
249    let ext_d1 = s1_next.clone();
250    let ext_c0 = s2_next.clone();
251    let ext_c1 = s3_next.clone();
252    let ext_a0_b0 = ext_a0.clone() * ext_b0.clone();
253    let ext_a1_b1 = ext_a1.clone() * ext_b1.clone();
254
255    let seven: AB::Expr = AB::Expr::from_u16(7);
256    assert_zero(builder, &mut idx, is_ext2mul.clone() * (ext_d0 - ext_b0.clone()));
257    assert_zero(builder, &mut idx, is_ext2mul.clone() * (ext_d1 - ext_b1.clone()));
258    assert_zero(
259        builder,
260        &mut idx,
261        is_ext2mul.clone() * (ext_c0 - (ext_a0_b0.clone() + seven.clone() * ext_a1_b1.clone())),
262    );
263    assert_zero(
264        builder,
265        &mut idx,
266        is_ext2mul * (ext_c1 - ((ext_a0 + ext_a1) * (ext_b0 + ext_b1) - ext_a0_b0 - ext_a1_b1)),
267    );
268
269    // -------------------------------------------------------------------------
270    // U32 ops
271    // -------------------------------------------------------------------------
272    let two_pow_16: AB::Expr = AB::Expr::from_u64(1u64 << 16);
273    let two_pow_32: AB::Expr = AB::Expr::from_u64(1u64 << 32);
274    let two_pow_48: AB::Expr = AB::Expr::from_u64(1u64 << 48);
275    let two_pow_32_minus_one: AB::Expr = AB::Expr::from_u64((1u64 << 32) - 1);
276
277    // U32 limbs: v_lo = h1*2^16 + h0, v_hi = h3*2^16 + h2.
278    let u32_v_lo = uop_h1.clone() * two_pow_16.clone() + uop_h0.clone();
279    let u32_v_hi = uop_h3.clone() * two_pow_16 + uop_h2.clone();
280    let u32_v48 = uop_h2.clone() * two_pow_32.clone() + u32_v_lo.clone();
281    let u32_v64 = uop_h3.clone() * two_pow_48 + u32_v48.clone();
282
283    // Element validity check for u32split/u32mul/u32madd.
284    let u32_split_mul_madd = is_u32split.clone() + is_u32mul.clone() + is_u32madd.clone();
285    let u32_v_hi_comp = AB::Expr::ONE - uop_h4.clone() * (two_pow_32_minus_one - u32_v_hi.clone());
286    assert_zero_integrity(
287        builder,
288        &mut idx,
289        u32_split_mul_madd * (u32_v_hi_comp * u32_v_lo.clone()),
290    );
291
292    let u32_two_outputs = is_u32split.clone()
293        + is_u32add.clone()
294        + is_u32add3.clone()
295        + is_u32mul.clone()
296        + is_u32madd.clone();
297    assert_zero(
298        builder,
299        &mut idx,
300        u32_two_outputs.clone() * (s0_next.clone() - u32_v_lo.clone()),
301    );
302    assert_zero(builder, &mut idx, u32_two_outputs * (s1_next.clone() - u32_v_hi.clone()));
303
304    assert_zero_integrity(builder, &mut idx, is_u32split * (s0.clone() - u32_v64.clone()));
305    assert_zero_integrity(
306        builder,
307        &mut idx,
308        is_u32add * (s0.clone() + s1.clone() - u32_v48.clone()),
309    );
310    assert_zero_integrity(
311        builder,
312        &mut idx,
313        is_u32add3 * (s0.clone() + s1.clone() + s2.clone() - u32_v48.clone()),
314    );
315
316    assert_zero(
317        builder,
318        &mut idx,
319        is_u32sub.clone()
320            * (s1.clone() - (s0.clone() + s1_next.clone() - s0_next.clone() * two_pow_32.clone())),
321    );
322    assert_zero(
323        builder,
324        &mut idx,
325        is_u32sub.clone() * (s0_next.clone() * (s0_next.clone() - AB::Expr::ONE)),
326    );
327    assert_zero(builder, &mut idx, is_u32sub * (s1_next.clone() - u32_v_lo.clone()));
328
329    assert_zero_integrity(
330        builder,
331        &mut idx,
332        is_u32mul * (s0.clone() * s1.clone() - u32_v64.clone()),
333    );
334    assert_zero_integrity(
335        builder,
336        &mut idx,
337        is_u32madd * (s0.clone() * s1.clone() + s2 - u32_v64.clone()),
338    );
339
340    assert_zero(
341        builder,
342        &mut idx,
343        is_u32div.clone() * (s1.clone() - (s0.clone() * s1_next.clone() + s0_next.clone())),
344    );
345    assert_zero(
346        builder,
347        &mut idx,
348        is_u32div.clone() * (s1.clone() - s1_next.clone() - u32_v_lo.clone()),
349    );
350    assert_zero(
351        builder,
352        &mut idx,
353        is_u32div * (s0.clone() - s0_next.clone() - (u32_v_hi.clone() + AB::Expr::ONE)),
354    );
355
356    assert_zero(builder, &mut idx, is_u32assert2.clone() * (s0_next - u32_v_hi.clone()));
357    assert_zero(builder, &mut idx, is_u32assert2 * (s1_next - u32_v_lo));
358}

363fn assert_zero<AB: TaggingAirBuilderExt>(builder: &mut AB, idx: &mut usize, expr: AB::Expr) {
364    tagged_assert_zero(builder, &STACK_ARITH_TAGS, idx, expr);
365}

367fn assert_zero_integrity<AB: TaggingAirBuilderExt>(
368    builder: &mut AB,
369    idx: &mut usize,
370    expr: AB::Expr,
371) {
372    tagged_assert_zero_integrity(builder, &STACK_ARITH_TAGS, idx, expr);
