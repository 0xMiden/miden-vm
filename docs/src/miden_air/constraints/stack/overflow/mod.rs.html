1//! Stack overflow constraints.
2//!
3//! This module contains constraints for the stack overflow table bookkeeping columns.
4//! The stack overflow table tracks items that have "overflowed" below the accessible portion
5//! of the operand stack (the top 16 positions).
6//!
7//! ## Columns
8//!
9//! - `b0`: Stack depth (always >= 16)
10//! - `b1`: Address of the top row in the overflow table (clk value when item was pushed)
11//! - `h0`: Overflow flag helper = 1/(b0 - 16) when b0 > 16, else 0
12//!
13//! ## Constraints
14//!
15//! 1. **Stack depth transition** (degree 7):
16//!    - No shift: depth stays the same
17//!    - Right shift: depth increases by 1
18//!    - Left shift with non-empty overflow: depth decreases by 1 (we pop from the overflow table)
19//!    - CALL/SYSCALL/DYNCALL: depth resets to 16
20//!
21//! 2. **Overflow flag** (degree 3):
22//!    - When overflow table is empty (b0 = 16), h0 must be 0
23//!    - When overflow table has values (b0 > 16), h0 = 1/(b0 - 16)
24//!
25//! 3. **Overflow index** (degree 7, 8):
26//!    - On right shift: b1' = clk (record when item was pushed)
27//!    - On left shift with depth = 16: stack[15]' = 0 (no item to restore)
28

50/// Base tag ID for stack overflow constraints.
51const STACK_OVERFLOW_BASE_ID: usize = TAG_STACK_OVERFLOW_BASE;
52
53/// Tag namespaces for stack overflow constraints (boundary + transition).
54const STACK_OVERFLOW_NAMES: [&str; TAG_STACK_OVERFLOW_COUNT] = [
55    "stack.overflow.depth.first_row",
56    "stack.overflow.depth.last_row",
57    "stack.overflow.addr.first_row",
58    "stack.overflow.addr.last_row",
59    "stack.overflow.depth.transition",
60    "stack.overflow.flag.transition",
61    "stack.overflow.addr.transition",
62    "stack.overflow.zero_insert.transition",
63];

68/// Enforces all stack overflow constraints.
69///
70/// This function enforces:
71/// 1. Stack depth transitions correctly based on the operation type
72/// 2. Overflow flag h0 is set correctly
73/// 3. Overflow bookkeeping index b1 is updated correctly on shifts
74/// 4. Last stack item is zeroed on left shift when depth = 16
75pub fn enforce_main<AB>(
76    builder: &mut AB,
77    local: &MainTraceRow<AB::Var>,
78    next: &MainTraceRow<AB::Var>,
79    op_flags: &OpFlags<AB::Expr>,
80) where
81    AB: LiftedAirBuilder,
82{
83    // Boundary constraints: stack depth and overflow pointer must start/end clean.
84    let sixteen: AB::Expr = AB::Expr::from_u16(16);
85    let zero: AB::Expr = AB::Expr::ZERO;
86    builder.tagged(STACK_OVERFLOW_BASE_ID, STACK_OVERFLOW_NAMES[0], |builder| {
87        builder
88            .when_first_row()
89            .assert_zero(local.stack[B0_COL_IDX].clone().into() - sixteen.clone());
90    });
91    builder.tagged(STACK_OVERFLOW_BASE_ID + 1, STACK_OVERFLOW_NAMES[1], |builder| {
92        builder
93            .when_last_row()
94            .assert_zero(local.stack[B0_COL_IDX].clone().into() - sixteen);
95    });
96    builder.tagged(STACK_OVERFLOW_BASE_ID + 2, STACK_OVERFLOW_NAMES[2], |builder| {
97        builder
98            .when_first_row()
99            .assert_zero(local.stack[B1_COL_IDX].clone().into() - zero.clone());
100    });
101    builder.tagged(STACK_OVERFLOW_BASE_ID + 3, STACK_OVERFLOW_NAMES[3], |builder| {
102        builder
103            .when_last_row()
104            .assert_zero(local.stack[B1_COL_IDX].clone().into() - zero);
105    });
106
107    // Transition constraints: depth bookkeeping, overflow flag, and pointer updates.
108    enforce_stack_depth_constraints(builder, local, next, op_flags);
109    enforce_overflow_flag_constraints(builder, local, op_flags);
110    enforce_overflow_index_constraints(builder, local, next, op_flags);
111}

116/// Enforces stack depth transition constraints.
117///
118/// The stack depth (b0) changes based on the operation:
119/// - No shift operations: depth unchanged
120/// - Right shift operations: depth += 1
121/// - Left shift operations with non-empty overflow: depth -= 1 (we pop from the overflow table)
122/// - CALL/SYSCALL/DYNCALL: depth = 16 (reset)
123///
124/// The END operation exiting a CALL/SYSCALL block is handled separately via multiset constraints.
125fn enforce_stack_depth_constraints<AB>(
126    builder: &mut AB,
127    local: &MainTraceRow<AB::Var>,
128    next: &MainTraceRow<AB::Var>,
129    op_flags: &OpFlags<AB::Expr>,
130) where
131    AB: LiftedAirBuilder,
132{
133    let depth: AB::Expr = local.stack[B0_COL_IDX].clone().into();
134    let depth_next: AB::Expr = next.stack[B0_COL_IDX].clone().into();
135
136    // Flag for CALL, DYNCALL, or SYSCALL operations
137    let call_or_dyncall_or_syscall = op_flags.call() + op_flags.dyncall() + op_flags.syscall();
138
139    // Flag for END operation that ends a CALL/DYNCALL or SYSCALL block
140    let is_call_or_dyncall_end: AB::Expr = local.decoder[IS_CALL_FLAG_COL_IDX].clone().into();
141    let is_syscall_end: AB::Expr = local.decoder[IS_SYSCALL_FLAG_COL_IDX].clone().into();
142    let call_or_dyncall_or_syscall_end = op_flags.end() * (is_call_or_dyncall_end + is_syscall_end);
143
144    // Invariants relied on here:
145    // - Aggregate left_shift/right_shift flags are 0 on CALL/SYSCALL/END rows.
146    // - DYNCALL is excluded from the aggregate left_shift flag (its stack effect is handled via
147    //   per-position left_shift_at flags plus the call-entry depth reset).
148    //
149    // We have three regimes:
150    //
151    // 1) CALL/SYSCALL/DYNCALL entry: force b0' = 16 (handled by call_part below).
152    // 2) END-of-call: depth restoration is validated by block stack constraints; we don't enforce
153    //    the shift law here.
154    // 3) All other rows ("normal ops"): depth follows the shift law b0' - b0 + f_shl * f_ov - f_shr
155    //    = 0.
156    //
157    // Why we mask only the (b0' - b0) term:
158    //
159    // - On CALL/SYSCALL/END rows, the aggregate shift flags are 0 by construction. (CALL/SYSCALL
160    //   are explicitly no-shift ops; END sets left_shift only for loop exits; DYNCALL is
161    //   intentionally excluded from the aggregate left_shift flag.)
162    // - Therefore, the shift terms already vanish on those rows, and masking them would only
163    //   increase polynomial degree.
164    // - We still need to suppress the raw (b0' - b0) term on END-of-call rows, hence the mask.
165    let normal_mask =
166        AB::Expr::ONE - call_or_dyncall_or_syscall.clone() - call_or_dyncall_or_syscall_end;
167    let depth_delta_part = (depth_next.clone() - depth.clone()) * normal_mask;
168
169    // Left shift with non-empty overflow: when f_shl=1 and f_ov=1, depth must decrement by 1.
170    // This contributes +1 to the LHS, enforcing b0' = b0 - 1.
171    let left_shift_part = op_flags.left_shift() * op_flags.overflow();
172
173    // Right shift: when f_shr=1, depth must increment by 1.
174    // This contributes -1 to the LHS, enforcing b0' = b0 + 1.
175    let right_shift_part = op_flags.right_shift();
176
177    // CALL/SYSCALL/DYNCALL: depth resets to 16 when entering a new context.
178    let call_part = call_or_dyncall_or_syscall * (depth_next - AB::Expr::from_u16(16));
179
180    // Combined constraint: normal depth update + shift effects + call reset = 0.
181    builder.tagged(STACK_OVERFLOW_BASE_ID + 4, STACK_OVERFLOW_NAMES[4], |builder| {
182        builder
183            .when_transition()
184            .assert_zero(depth_delta_part + left_shift_part - right_shift_part + call_part);
185    });
186}

188/// Enforces overflow flag constraints.
189///
190/// The overflow flag h0 must satisfy:
191/// - (1 - overflow) * (b0 - 16) = 0
192///
193/// This ensures:
194/// - When b0 = 16 (no overflow): the constraint is satisfied for any h0
195/// - When b0 > 16 (overflow): h0 must be set such that overflow = (b0 - 16) * h0 = 1
196fn enforce_overflow_flag_constraints<AB>(
197    builder: &mut AB,
198    local: &MainTraceRow<AB::Var>,
199    op_flags: &OpFlags<AB::Expr>,
200) where
201    AB: LiftedAirBuilder,
202{
203    let depth: AB::Expr = local.stack[B0_COL_IDX].clone().into();
204
205    // (1 - overflow) * (depth - 16) = 0
206    // When depth > 16, overflow must be 1 (meaning h0 = 1/(depth - 16))
207    // When depth = 16, this constraint is satisfied regardless of overflow
208    let constraint = (AB::Expr::ONE - op_flags.overflow()) * (depth - AB::Expr::from_u16(16));
209
210    builder.tagged(STACK_OVERFLOW_BASE_ID + 5, STACK_OVERFLOW_NAMES[5], |builder| {
211        builder.assert_zero(constraint);
212    });
213}

215/// Enforces overflow bookkeeping index constraints.
216///
217/// Two constraints:
218/// 1. On right shift: b1' = clk (record the clock cycle when item was pushed to overflow)
219/// 2. On left shift with depth = 16: stack[15]' = 0 (no item to restore from overflow)
220fn enforce_overflow_index_constraints<AB>(
221    builder: &mut AB,
222    local: &MainTraceRow<AB::Var>,
223    next: &MainTraceRow<AB::Var>,
224    op_flags: &OpFlags<AB::Expr>,
225) where
226    AB: LiftedAirBuilder,
227{
228    let overflow_addr_next: AB::Expr = next.stack[B1_COL_IDX].clone().into();
229    let clk: AB::Expr = local.clk.clone().into();
230    let last_stack_item_next: AB::Expr = next.stack[15].clone().into();
231
232    // On right shift, the overflow address should be set to current clk
233    let right_shift_constraint = (overflow_addr_next - clk) * op_flags.right_shift();
234    builder.tagged(STACK_OVERFLOW_BASE_ID + 6, STACK_OVERFLOW_NAMES[6], |builder| {
235        builder.when_transition().assert_zero(right_shift_constraint);
236    });
237
238    // On left shift when depth = 16 (no overflow), last stack item should be zero
239    let left_shift_constraint =
240        (AB::Expr::ONE - op_flags.overflow()) * op_flags.left_shift() * last_stack_item_next;
241    builder.tagged(STACK_OVERFLOW_BASE_ID + 7, STACK_OVERFLOW_NAMES[7], |builder| {
242        builder.when_transition().assert_zero(left_shift_constraint);
243    });
