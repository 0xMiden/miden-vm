1//! Stack overflow table bus constraint.
2//!
3//! This module enforces the running product constraint for the stack overflow table (p1).
4//! The stack overflow table tracks values that overflow from the 16-element operand stack.
5//!
6//! The bus accumulator p1 uses a multiset check:
7//! - Boundary: p1[0] = 1 and p1[last] = 1 (enforced by the wrapper AIR)
8//! - Transition: p1' * requests = p1 * responses
9//!
10//! Where:
11//! - Responses (adding rows): When right_shift, a row is added with (clk, s15, b1)
12//! - Requests (removing rows): When (left_shift OR dyncall) AND non_empty_overflow, a row is
13//!   removed with (b1, s15', b1') or (b1, s15', hasher_state[5]) for dyncall
14//!
15//! ## Row Encoding
16//!
17//! Each row in the overflow table is encoded as:
18//! `alpha + beta^0 * clk + beta^1 * val + beta^2 * prev`
19
20use miden_core::field::PrimeCharacteristicRing;
21use miden_crypto::stark::air::{ExtensionBuilder, LiftedAirBuilder, WindowAccess};
22
23use crate::{
24    MainTraceRow,
25    constraints::{
26        bus::indices::P1_STACK,
27        op_flags::OpFlags,
28        tagging::{TaggingAirBuilderExt, ids::TAG_STACK_OVERFLOW_BUS_BASE},
29    },
30    trace::{
31        Challenges,
32        decoder::HASHER_STATE_RANGE,
33        stack::{B0_COL_IDX, B1_COL_IDX, H0_COL_IDX},
34    },
35};
36
37/// Tag ID and namespace for the stack overflow bus transition constraint.
38const STACK_OVERFLOW_BUS_ID: usize = TAG_STACK_OVERFLOW_BUS_BASE;
39const STACK_OVERFLOW_BUS_NAME: &str = "stack.overflow.bus.transition";
40
41// ENTRY POINTS
42// ================================================================================================
43
44/// Enforces the stack overflow table bus constraint.
45///
46/// This constraint tracks overflow table operations using a running product:
47/// - Adding rows when right_shift (element pushed off stack position 15)
48/// - Removing rows when (left_shift OR dyncall) AND overflow is non-empty
49pub fn enforce_bus<AB>(
50    builder: &mut AB,
51    local: &MainTraceRow<AB::Var>,
52    next: &MainTraceRow<AB::Var>,
53    op_flags: &OpFlags<AB::Expr>,
54    challenges: &Challenges<AB::ExprEF>,
55) where
56    AB: LiftedAirBuilder,
57{
58    // Auxiliary trace must be present.
59
60    // Extract auxiliary trace values.
61    let (p1_local, p1_next) = {
62        let aux = builder.permutation();
63        let aux_local = aux.current_slice();
64        let aux_next = aux.next_slice();
65        (aux_local[P1_STACK], aux_next[P1_STACK])
66    };
67
68    let one_ef = AB::ExprEF::ONE;
69
70    // ============================================================================================
71    // TRANSITION CONSTRAINT
72    // ============================================================================================
73
74    // -------------------------------------------------------------------------
75    // Stack and bookkeeping column values
76    // -------------------------------------------------------------------------
77
78    // Current row values
79    let clk: AB::Expr = local.clk.clone().into();
80    let s15: AB::Expr = local.stack[15].clone().into();
81    let b0: AB::Expr = local.stack[B0_COL_IDX].clone().into();
82    let b1: AB::Expr = local.stack[B1_COL_IDX].clone().into();
83    let h0: AB::Expr = local.stack[H0_COL_IDX].clone().into();
84
85    // Next row values (needed for removal)
86    let s15_next: AB::Expr = next.stack[15].clone().into();
87    let b1_next: AB::Expr = next.stack[B1_COL_IDX].clone().into();
88
89    // Hasher state element 5, used by DYNCALL to store the new overflow table pointer.
90    let hasher_state_5: AB::Expr = local.decoder[HASHER_STATE_RANGE.start + 5].clone().into();
91
92    // -------------------------------------------------------------------------
93    // Overflow condition: (b0 - 16) * h0 = 1 when overflow is non-empty
94    // -------------------------------------------------------------------------
95
96    let sixteen = AB::Expr::from_u16(16);
97    let is_non_empty_overflow: AB::Expr = (b0 - sixteen) * h0;
98
99    // -------------------------------------------------------------------------
100    // Operation flags
101    // -------------------------------------------------------------------------
102
103    let right_shift = op_flags.right_shift();
104    let left_shift = op_flags.left_shift();
105    let dyncall = op_flags.dyncall();
106
107    // -------------------------------------------------------------------------
108    // Row value encoding: alpha + beta^0 * clk + beta^1 * val + beta^2 * prev
109    // -------------------------------------------------------------------------
110
111    // Response row value (adding to table during right_shift):
112    let response_row = challenges.encode([clk.clone(), s15.clone(), b1.clone()]);
113
114    // Request row value for left_shift (removing from table):
115    let request_row_left = challenges.encode([b1.clone(), s15_next.clone(), b1_next.clone()]);
116
117    // Request row value for dyncall (removing from table):
118    let request_row_dyncall =
119        challenges.encode([b1.clone(), s15_next.clone(), hasher_state_5.clone()]);
120
121    // -------------------------------------------------------------------------
122    // Compute response and request terms
123    // -------------------------------------------------------------------------
124
125    // Response: right_shift * response_row + (1 - right_shift)
126    let response: AB::ExprEF = response_row * right_shift.clone() + (one_ef.clone() - right_shift);
127
128    // Request flags
129    let left_flag: AB::Expr = left_shift * is_non_empty_overflow.clone();
130    let dyncall_flag: AB::Expr = dyncall * is_non_empty_overflow;
131    let request_flag_sum: AB::Expr = left_flag.clone() + dyncall_flag.clone();
132
133    // Request: left_flag * left_value + dyncall_flag * dyncall_value + (1 - sum(flags))
134    let request: AB::ExprEF = request_row_left * left_flag.clone()
135        + request_row_dyncall * dyncall_flag.clone()
136        + (one_ef.clone() - request_flag_sum);
137
138    // -------------------------------------------------------------------------
139    // Main running product constraint
140    // -------------------------------------------------------------------------
141
142    let lhs = p1_next.into() * request;
143    let rhs = p1_local.into() * response;
144
145    builder.tagged(STACK_OVERFLOW_BUS_ID, STACK_OVERFLOW_BUS_NAME, |builder| {
146        builder.when_transition().assert_zero_ext(lhs - rhs);
147    });
148}
