1//! General stack transition constraints.
2//!
3//! This module contains the general constraints that enforce how stack items transition
4//! based on the operation type (no shift, left shift, right shift).
5//!
6//! ## Stack Transition Model
7//!
8//! The stack has 16 visible positions (0-15). For each operation, stack items can:
9//! - **Stay in place** (no shift): item stays at same position
10//! - **Shift left**: item moves to position i from position i+1
11//! - **Shift right**: item moves to position i from position i-1
12//!
13//! ## Constraints
14//!
15//! 1. **Position 0**: Can receive from position 0 (no shift) or position 1 (left shift). Right
16//!    shift doesn't apply - position 0 gets a new value pushed.
17//!
18//! 2. **Positions 1-14**: Can receive from position i (no shift), i+1 (left shift), or i-1 (right
19//!    shift).
20//!
21//! 3. **Position 15**: Can receive from position 15 (no shift) or position 14 (right shift). Left
22//!    shift at position 15 is handled by overflow constraints (zeroing).
23//!
24//! 4. **Top binary**: Enforced by the specific op constraints that require it.
25

42/// Number of general stack constraints.
43/// 16 constraints for stack item transitions.
44pub const NUM_CONSTRAINTS: usize = TAG_STACK_GENERAL_COUNT;
45
46/// Tag base ID for stack general constraints.
47/// Tag namespaces for stack general constraints.
48const STACK_GENERAL_NAMES: [&str; NUM_CONSTRAINTS] = [
49    "stack.general.transition.0",
50    "stack.general.transition.1",
51    "stack.general.transition.2",
52    "stack.general.transition.3",
53    "stack.general.transition.4",
54    "stack.general.transition.5",
55    "stack.general.transition.6",
56    "stack.general.transition.7",
57    "stack.general.transition.8",
58    "stack.general.transition.9",
59    "stack.general.transition.10",
60    "stack.general.transition.11",
61    "stack.general.transition.12",
62    "stack.general.transition.13",
63    "stack.general.transition.14",
64    "stack.general.transition.15",
65];

70/// Enforces all general stack transition constraints.
71///
72/// This includes:
73/// - 16 constraints for stack item transitions at each position
74pub fn enforce_main<AB>(
75    builder: &mut AB,
76    local: &MainTraceRow<AB::Var>,
77    next: &MainTraceRow<AB::Var>,
78    op_flags: &OpFlags<AB::Expr>,
79) where
80    AB: LiftedAirBuilder,
81{
82    // For each position i, the constraint ensures that the next value is consistent
83    // with the current value based on the shift flags:
84    //
85    // next[i] * flag_sum = no_shift[i] * current[i]
86    //                    + left_shift[i+1] * current[i+1]
87    //                    + right_shift[i-1] * current[i-1]
88    //
89    // where flag_sum is the sum of applicable flags for that position.
90
91    // Position 0: no right shift (new value pushed instead)
92    // next[0] * flag_sum = no_shift[0] * current[0] + left_shift[1] * current[1]
93    {
94        let flag_sum = op_flags.no_shift_at(0) + op_flags.left_shift_at(1);
95        let expected = op_flags.no_shift_at(0) * local.stack[0].clone().into()
96            + op_flags.left_shift_at(1) * local.stack[1].clone().into();
97        let actual: AB::Expr = next.stack[0].clone().into();
98
99        builder.tagged(TAG_STACK_GENERAL_BASE, STACK_GENERAL_NAMES[0], |builder| {
100            builder.when_transition().assert_zero(actual * flag_sum - expected);
101        });
102    }
103
104    // Positions 1-14: all three shift types possible.
105    for (i, &namespace) in STACK_GENERAL_NAMES.iter().enumerate().take(15).skip(1) {
106        let flag_sum = op_flags.no_shift_at(i)
107            + op_flags.left_shift_at(i + 1)
108            + op_flags.right_shift_at(i - 1);
109
110        let expected = op_flags.no_shift_at(i) * local.stack[i].clone().into()
111            + op_flags.left_shift_at(i + 1) * local.stack[i + 1].clone().into()
112            + op_flags.right_shift_at(i - 1) * local.stack[i - 1].clone().into();
113        let actual: AB::Expr = next.stack[i].clone().into();
114
115        let id = TAG_STACK_GENERAL_BASE + i;
116        builder.tagged(id, namespace, |builder| {
117            builder.when_transition().assert_zero(actual * flag_sum - expected);
118        });
119    }
120
121    // Position 15: no left shift (handled by overflow constraints)
122    // next[15] * flag_sum = no_shift[15] * current[15] + right_shift[14] * current[14]
123    {
124        let flag_sum = op_flags.no_shift_at(15) + op_flags.right_shift_at(14);
125        let expected = op_flags.no_shift_at(15) * local.stack[15].clone().into()
126            + op_flags.right_shift_at(14) * local.stack[14].clone().into();
127        let actual: AB::Expr = next.stack[15].clone().into();
128
129        builder.tagged(TAG_STACK_GENERAL_BASE + 15, STACK_GENERAL_NAMES[15], |builder| {
130            builder.when_transition().assert_zero(actual * flag_sum - expected);
131        });
132    }
133}
