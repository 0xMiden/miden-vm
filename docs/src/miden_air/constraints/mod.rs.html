4//! Miden VM Constraints
5//!
6//! This module contains the constraint functions for the Miden VM processor.
7//!
8//! ## Organization
9//!
10//! Constraints are separated into two categories:
11//!
12//! ### Main Trace Constraints
13//! - system: clock, ctx, fn_hash transitions
14//! - range: range checker V column transitions
15//! - stack: general stack constraints
16//!
17//! ### Bus Constraints (Auxiliary Trace)
18//! - range::bus
19//!
20//! Bus constraints access the auxiliary trace via `builder.permutation()` and use
21//! random challenges from `builder.permutation_randomness()` for multiset/LogUp verification.
22//!
23//! Additional components (decoder, chiplets) are introduced in later constraint chunks.
24

43/// Enforces all main trace constraints.
44pub fn enforce_main<AB>(
45    builder: &mut AB,
46    local: &MainTraceRow<AB::Var>,
47    next: &MainTraceRow<AB::Var>,
48) where
49    AB: LiftedAirBuilder<F = Felt>,
50{
51    system::enforce_main(builder, local, next);
52    range::enforce_main(builder, local, next);
53
54    let op_flags = op_flags::OpFlags::new(op_flags::ExprDecoderAccess::<_, AB::Expr>::new(local));
55    stack::enforce_main(builder, local, next, &op_flags);
56    decoder::enforce_main(builder, local, next, &op_flags);
57    chiplets::enforce_main(builder, local, next);
58}

60/// Enforces all auxiliary (bus) constraints: boundary + transition.
61///
62/// Bus soundness relies on three mechanisms:
63///   1. **Boundary** -- aux columns start at identity (1 or 0)
64///   2. **Transition** -- row-to-row update rules determine all subsequent values
65///   3. **`reduced_aux_values`** -- final values satisfy bus identities
66pub fn enforce_bus<AB>(
67    builder: &mut AB,
68    local: &MainTraceRow<AB::Var>,
69    next: &MainTraceRow<AB::Var>,
70) where
71    AB: LiftedAirBuilder<F = Felt>,
72{
73    let r = builder.permutation_randomness();
74    let challenges = Challenges::<AB::ExprEF>::new(r[0].into(), r[1].into());
75    let op_flags = op_flags::OpFlags::new(op_flags::ExprDecoderAccess::<_, AB::Expr>::new(local));
76
77    enforce_bus_boundary(builder);
78
79    range::bus::enforce_bus(builder, local);
80    stack::bus::enforce_bus(builder, local, next, &op_flags, &challenges);
81    decoder::bus::enforce_bus(builder, local, next, &op_flags, &challenges);
82    chiplets::bus::enforce_bus(builder, local, next, &op_flags, &challenges);
83}

85/// Enforces boundary constraints on all auxiliary columns.
86///
87/// **First row:** running-product columns start at 1, LogUp sums start at 0.
88///
89/// **Last row:** each aux column must equal its committed final value (from
90/// `permutation_values`). This binds the aux trace polynomial to the values checked
91/// by `reduced_aux_values`, preventing a malicious prover from committing arbitrary
92/// finals that satisfy the bus identity without matching the actual trace.
93fn enforce_bus_boundary<AB>(builder: &mut AB)
94where
95    AB: LiftedAirBuilder<F = Felt>,
96{
97    // First row: running products = 1, LogUp sums = 0.
98    enforce_bus_first_row(builder);
99
100    // Last row: bind aux trace to committed finals checked by `reduced_aux_values`.
101    enforce_bus_last_row(builder);
102}

104fn enforce_bus_first_row<AB>(builder: &mut AB)
105where
106    AB: LiftedAirBuilder<F = Felt>,
107{
108    use bus::indices::*;
109    use tagging::{TaggingAirBuilderExt, ids::TAG_BUS_BOUNDARY_BASE};
110
111    const N: usize = 8;
112    let ids: [usize; N] = core::array::from_fn(|i| TAG_BUS_BOUNDARY_BASE + i);
113
114    let aux = builder.permutation();
115    let aux_local = aux.current_slice();
116
117    let p1 = aux_local[P1_BLOCK_STACK];
118    let p2 = aux_local[P2_BLOCK_HASH];
119    let p3 = aux_local[P3_OP_GROUP];
120    let s_aux = aux_local[P1_STACK];
121    let b_hk = aux_local[B_HASH_KERNEL];
122    let b_ch = aux_local[B_CHIPLETS];
123    let b_rng = aux_local[B_RANGE];
124    let v_wir = aux_local[V_WIRING];
125
126    builder.tagged_list(ids, "bus.boundary.first_row", |builder| {
127        let mut first = builder.when_first_row();
128        first.assert_one_ext(p1);
129        first.assert_one_ext(p2);
130        first.assert_one_ext(p3);
131        first.assert_one_ext(s_aux);
132        first.assert_one_ext(b_hk);
133        first.assert_one_ext(b_ch);
134        first.assert_zero_ext(b_rng);
135        first.assert_zero_ext(v_wir);
136    });
137}

139fn enforce_bus_last_row<AB>(builder: &mut AB)
140where
141    AB: LiftedAirBuilder<F = Felt>,
142{
143    use tagging::{
144        TaggingAirBuilderExt,
145        ids::{TAG_BUS_BOUNDARY_BASE, TAG_BUS_BOUNDARY_FIRST_ROW_COUNT},
146    };
147
148    use crate::trace::AUX_TRACE_WIDTH;
149
150    const N: usize = AUX_TRACE_WIDTH;
151    let ids: [usize; N] =
152        core::array::from_fn(|i| TAG_BUS_BOUNDARY_BASE + TAG_BUS_BOUNDARY_FIRST_ROW_COUNT + i);
153
154    let cols: [AB::VarEF; N] = {
155        let aux = builder.permutation();
156        let s = aux.current_slice();
157        core::array::from_fn(|i| s[i])
158    };
159    let finals: [AB::ExprEF; N] = {
160        let f = builder.permutation_values();
161        core::array::from_fn(|i| f[i].clone().into())
162    };
163
164    builder.tagged_list(ids, "bus.boundary.last_row", |builder| {
165        let mut last = builder.when_last_row();
166        for (col, expected) in cols.into_iter().zip(finals) {
167            last.assert_eq_ext(col, expected);
168        }
169    });
