1//! STARK configuration factories for different hash functions.
2//!
3//! Each factory creates a [`StarkConfig`](miden_crypto::stark::StarkConfig) bundling the
4//! PCS parameters, LMCS commitment scheme, and Fiat-Shamir challenger for proving and verification.
5
6use alloc::vec;

8use miden_core::{Felt, field::QuadFelt};
9use miden_crypto::{
10    field::Field,
11    hash::{
12        blake::Blake3Hasher, keccak::Keccak256Hash, poseidon2::Poseidon2Permutation256,
13        rpo::RpoPermutation256, rpx::RpxPermutation256,
14    },
15    stark::{
16        GenericStarkConfig,
17        challenger::{DuplexChallenger, HashChallenger, SerializingChallenger64},

26// PCS PARAMETERS
27// ================================================================================================
28
29/// Log2 of the FRI blowup factor (blowup = 8).
30const LOG_BLOWUP: u8 = 3;
31/// Log2 of the FRI folding arity (arity = 4).
32const LOG_FOLDING_ARITY: u8 = 2;
33/// Log2 of the final polynomial degree (degree = 128).
34const LOG_FINAL_DEGREE: u8 = 7;
35/// Proof-of-work bits for FRI folding challenges.
36const FOLDING_POW_BITS: usize = 16;
37/// Proof-of-work bits for DEEP composition polynomial.
38const DEEP_POW_BITS: usize = 0;
39/// Number of FRI query repetitions.
40const NUM_QUERIES: usize = 27;
41/// Proof-of-work bits for query phase.
42const QUERY_POW_BITS: usize = 0;
43
44/// Default PCS parameters shared by all hash function configurations.
45pub fn pcs_params() -> PcsParams {
46    PcsParams::new(
47        LOG_BLOWUP,
48        LOG_FOLDING_ARITY,
49        LOG_FINAL_DEGREE,
50        FOLDING_POW_BITS,
51        DEEP_POW_BITS,
52        NUM_QUERIES,
53        QUERY_POW_BITS,
54    )
55    .expect("invalid PCS parameters")
56}

59// ================================================================================================
60
61// Byte-oriented hashes (Blake3, Keccak).
62
63/// Digest size in bytes for byte-oriented hashes.
64const BYTE_DIGEST_SIZE: usize = 32;
65/// Number of inputs to the Merkle compression function.
66const COMPRESSION_INPUTS: usize = 2;
67
68// Algebraic hashes (RPO, Poseidon2, RPX).
69
70/// Sponge state width in field elements.
71const SPONGE_WIDTH: usize = 12;
72/// Sponge rate (absorbable elements per permutation).
73const SPONGE_RATE: usize = 8;
74/// Sponge digest width in field elements.
75const DIGEST_WIDTH: usize = 4;

77// SHARED TYPE ALIASES
78// ================================================================================================
79
80type PackedFelt = <Felt as Field>::Packing;
81
82/// Miden VM STARK configuration with pre-filled common type parameters.
83///
84/// All Miden configurations use `Felt` as the base field, `QuadFelt` as the extension field,
85/// and `Radix2DitParallel<Felt>` as the DFT. Only the LMCS commitment scheme (`L`) and
86/// Fiat-Shamir challenger (`Ch`) vary by hash function.
87pub type MidenStarkConfig<L, Ch> =
88    GenericStarkConfig<Felt, QuadFelt, L, Radix2DitParallel<Felt>, Ch>;
89
90/// Byte-oriented LMCS (for Blake3, Keccak).
91type ByteLmcs<H> = LmcsConfig<
92    Felt,
93    u8,
94    ChainingHasher<H>,
95    CompressionFunctionFromHasher<H, COMPRESSION_INPUTS, BYTE_DIGEST_SIZE>,
96    BYTE_DIGEST_SIZE,
97    BYTE_DIGEST_SIZE,
98>;
99
100/// Byte-oriented challenger (for Blake3, Keccak).
101type ByteChallenger<H> = SerializingChallenger64<Felt, HashChallenger<u8, H, BYTE_DIGEST_SIZE>>;
102
103/// Algebraic LMCS (for RPO, Poseidon2, RPX).
104type AlgLmcs<P> = LmcsConfig<
105    PackedFelt,
106    PackedFelt,
107    StatefulSponge<P, SPONGE_WIDTH, SPONGE_RATE, DIGEST_WIDTH>,
108    TruncatedPermutation<P, COMPRESSION_INPUTS, DIGEST_WIDTH, SPONGE_WIDTH>,
109    SPONGE_WIDTH,
110    DIGEST_WIDTH,
111>;
112
113/// Algebraic duplex challenger (for RPO, Poseidon2, RPX).
114type AlgChallenger<P> = DuplexChallenger<Felt, P, SPONGE_WIDTH, SPONGE_RATE>;

119/// Creates a Blake3_256-based STARK configuration.
120pub fn blake3_256_config(
121    params: PcsParams,
122) -> MidenStarkConfig<ByteLmcs<Blake3Hasher>, ByteChallenger<Blake3Hasher>> {
123    let lmcs = LmcsConfig::new(
124        ChainingHasher::new(Blake3Hasher),
125        CompressionFunctionFromHasher::new(Blake3Hasher),
126    );
127    let challenger = SerializingChallenger64::from_hasher(vec![], Blake3Hasher);
128    GenericStarkConfig::new(params, lmcs, Radix2DitParallel::default(), challenger)
129}
130
131/// Creates a Keccak-based STARK configuration.
132pub fn keccak_config(
133    params: PcsParams,
134) -> MidenStarkConfig<ByteLmcs<Keccak256Hash>, ByteChallenger<Keccak256Hash>> {
135    let hash = Keccak256Hash {};
136    let lmcs = LmcsConfig::new(ChainingHasher::new(hash), CompressionFunctionFromHasher::new(hash));
137    let challenger = SerializingChallenger64::from_hasher(vec![], hash);
138    GenericStarkConfig::new(params, lmcs, Radix2DitParallel::default(), challenger)
139}

141/// Creates an RPO-based STARK configuration.
142pub fn rpo_config(
143    params: PcsParams,
144) -> MidenStarkConfig<AlgLmcs<RpoPermutation256>, AlgChallenger<RpoPermutation256>> {
145    let perm = RpoPermutation256;
146    let lmcs = LmcsConfig::new(StatefulSponge::new(perm), TruncatedPermutation::new(perm));
147    let challenger = DuplexChallenger::new(perm);
148    GenericStarkConfig::new(params, lmcs, Radix2DitParallel::default(), challenger)
149}
