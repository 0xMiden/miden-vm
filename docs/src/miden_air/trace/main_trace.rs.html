46/// Column layout of the main trace row.
47#[derive(Debug)]
48#[repr(C)]
49pub struct MainTraceRow<T> {
50    // System
51    pub clk: T,
52    pub ctx: T,
53    pub fn_hash: [T; WORD_SIZE],
54
55    // Decoder
56    pub decoder: [T; DECODER_TRACE_WIDTH],
57
58    // Stack
59    pub stack: [T; STACK_TRACE_WIDTH],
60
61    // Range checker
62    pub range: [T; RANGE_CHECK_TRACE_WIDTH],
63
64    // Chiplets
65    pub chiplets: [T; CHIPLETS_WIDTH],
66}
67
68impl<T> Borrow<MainTraceRow<T>> for [T] {
69    fn borrow(&self) -> &MainTraceRow<T> {
70        debug_assert_eq!(self.len(), crate::TRACE_WIDTH);
71        let (prefix, shorts, suffix) = unsafe { self.align_to::<MainTraceRow<T>>() };
72        debug_assert!(prefix.is_empty(), "Alignment should match");
73        debug_assert!(suffix.is_empty(), "Alignment should match");
74        debug_assert_eq!(shorts.len(), 1);
75        &shorts[0]
76    }
77}
78
79impl<T> BorrowMut<MainTraceRow<T>> for [T] {
80    fn borrow_mut(&mut self) -> &mut MainTraceRow<T> {
81        debug_assert_eq!(self.len(), crate::TRACE_WIDTH);
82        let (prefix, shorts, suffix) = unsafe { self.align_to_mut::<MainTraceRow<T>>() };
83        debug_assert!(prefix.is_empty(), "Alignment should match");
84        debug_assert!(suffix.is_empty(), "Alignment should match");
85        debug_assert_eq!(shorts.len(), 1);
86        &mut shorts[0]
87    }
88}

112    /// Converts this column-major `MainTrace` into a row-major matrix.
113    pub fn to_row_major(&self) -> RowMajorMatrix<Felt> {
114        let num_rows = self.columns.num_rows();
115        let mut col_major_data = Vec::with_capacity(num_rows * self.columns.num_cols());
116        for col in self.columns.columns() {
117            col_major_data.extend_from_slice(col);
118        }
119        // Treat the flat column-major data as a "row-major" matrix with width = num_rows
120        // (i.e. each "row" is one original column), then transpose to get true row-major.
121        RowMajorMatrix::new(col_major_data, num_rows).transpose()
122    }

125        self.columns.num_rows()
126    }

263    /// Constructs the i-th op code value from its individual bits.
264    pub fn get_op_code(&self, i: RowIndex) -> Felt {
265        let col_b0 = self.columns.get_column(DECODER_TRACE_OFFSET + 1);
266        let col_b1 = self.columns.get_column(DECODER_TRACE_OFFSET + 2);
267        let col_b2 = self.columns.get_column(DECODER_TRACE_OFFSET + 3);
268        let col_b3 = self.columns.get_column(DECODER_TRACE_OFFSET + 4);
269        let col_b4 = self.columns.get_column(DECODER_TRACE_OFFSET + 5);
270        let col_b5 = self.columns.get_column(DECODER_TRACE_OFFSET + 6);
271        let col_b6 = self.columns.get_column(DECODER_TRACE_OFFSET + 7);
272        let [b0, b1, b2, b3, b4, b5, b6] =
273            [col_b0[i], col_b1[i], col_b2[i], col_b3[i], col_b4[i], col_b5[i], col_b6[i]];
274        b0 + b1 * Felt::from_u64(2)
275            + b2 * Felt::from_u64(4)
276            + b3 * Felt::from_u64(8)
277            + b4 * Felt::from_u64(16)
278            + b5 * Felt::from_u64(32)
279            + b6 * Felt::from_u64(64)
280    }

287    /// Returns a flag indicating whether the current operation induces a left shift of the operand
288    /// stack.
289    pub fn is_left_shift(&self, i: RowIndex) -> bool {
290        let b0 = self.columns.get(DECODER_TRACE_OFFSET + 1, i.into());
291        let b1 = self.columns.get(DECODER_TRACE_OFFSET + 2, i.into());
292        let b2 = self.columns.get(DECODER_TRACE_OFFSET + 3, i.into());
293        let b3 = self.columns.get(DECODER_TRACE_OFFSET + 4, i.into());
294        let b4 = self.columns.get(DECODER_TRACE_OFFSET + 5, i.into());
295        let b5 = self.columns.get(DECODER_TRACE_OFFSET + 6, i.into());
296        let b6 = self.columns.get(DECODER_TRACE_OFFSET + 7, i.into());
297        let e0 = self.columns.get(DECODER_TRACE_OFFSET + OP_BITS_EXTRA_COLS_OFFSET, i.into());
298        let h5 = self.columns.get(DECODER_TRACE_OFFSET + IS_LOOP_FLAG_COL_IDX, i.into());
299
300        // group with left shift effect grouped by a common prefix
301        ([b6, b5, b4] == [ZERO, ONE, ZERO])||
302        // U32ADD3 or U32MADD
303        ([b6, b5, b4, b3, b2] == [ONE, ZERO, ZERO, ONE, ONE]) ||
304        // SPLIT or LOOP block
305        ([e0, b3, b2, b1] == [ONE, ZERO, ONE, ZERO]) ||
306        // REPEAT
307        ([b6, b5, b4, b3, b2, b1, b0] == [ONE, ONE, ONE, ZERO, ONE, ZERO, ZERO]) ||
308        // DYN
309        ([b6, b5, b4, b3, b2, b1, b0] == [ONE, ZERO, ONE, ONE, ZERO, ZERO, ZERO]) ||
310        // END of a loop
311        ([b6, b5, b4, b3, b2, b1, b0] == [ONE, ONE, ONE, ZERO, ZERO, ZERO, ZERO] && h5 == ONE)
312    }

314    /// Returns a flag indicating whether the current operation induces a right shift of the operand
315    /// stack.
316    pub fn is_right_shift(&self, i: RowIndex) -> bool {
317        let b0 = self.columns.get(DECODER_TRACE_OFFSET + 1, i.into());
318        let b1 = self.columns.get(DECODER_TRACE_OFFSET + 2, i.into());
319        let b2 = self.columns.get(DECODER_TRACE_OFFSET + 3, i.into());
320        let b3 = self.columns.get(DECODER_TRACE_OFFSET + 4, i.into());
321        let b4 = self.columns.get(DECODER_TRACE_OFFSET + 5, i.into());
322        let b5 = self.columns.get(DECODER_TRACE_OFFSET + 6, i.into());
323        let b6 = self.columns.get(DECODER_TRACE_OFFSET + 7, i.into());
324
325        // group with right shift effect grouped by a common prefix
326        [b6, b5, b4] == [ZERO, ONE, ONE]||
327        // u32SPLIT 100_1000
328        ([b6, b5, b4, b3, b2, b1, b0] == [ONE, ZERO, ZERO, ONE, ZERO, ZERO, ZERO]) ||
329        // PUSH i.e., 101_1011
330        ([b6, b5, b4, b3, b2, b1, b0] == [ONE, ZERO, ONE, ONE, ZERO, ONE, ONE])
331    }

652    /// Returns `true` if the hasher chiplet flags indicate the initialization of verifying
653    /// a Merkle path to an old node during Merkle root update procedure (MRUPDATE).
654    pub fn f_mv(&self, i: RowIndex) -> bool {
655        i.as_usize().is_multiple_of(HASH_CYCLE_LEN)
656            && self.chiplet_selector_0(i) == ZERO
657            && self.chiplet_selector_1(i) == ONE
658            && self.chiplet_selector_2(i) == ONE
659            && self.chiplet_selector_3(i) == ZERO
660    }

662    /// Returns `true` if the hasher chiplet flags indicate the continuation of verifying
663    /// a Merkle path to an old node during Merkle root update procedure (MRUPDATE).
664    pub fn f_mva(&self, i: RowIndex) -> bool {
665        (i.as_usize() % HASH_CYCLE_LEN == LAST_CYCLE_ROW)
666            && self.chiplet_selector_0(i) == ZERO
667            && self.chiplet_selector_1(i) == ONE
668            && self.chiplet_selector_2(i) == ONE
669            && self.chiplet_selector_3(i) == ZERO
670    }
