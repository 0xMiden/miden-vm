40/// Executes and proves the specified `program` and returns the result together with a STARK-based
41/// proof of the program's execution.
42///
43/// This is an async function that works on all platforms including wasm32.
44///
45/// - `stack_inputs` specifies the initial state of the stack for the VM.
46/// - `host` specifies the host environment which contain non-deterministic (secret) inputs for the
47///   prover
48/// - `options` defines parameters for STARK proof generation.
49///
50/// # Errors
51/// Returns an error if program execution or STARK proof generation fails for any reason.
52#[instrument("prove_program", skip_all)]
53pub async fn prove(
54    program: &Program,
55    stack_inputs: StackInputs,
56    advice_inputs: AdviceInputs,
57    host: &mut impl Host,
58    options: ProvingOptions,
59) -> Result<(StackOutputs, ExecutionProof), ExecutionError> {
60    // execute the program to create an execution trace using FastProcessor
61    let processor =
62        FastProcessor::new_with_options(stack_inputs, advice_inputs, *options.execution_options());
63
64    let (execution_output, trace_generation_context) =
65        processor.execute_for_trace(program, host).await?;
66
67    let trace = build_trace(execution_output, trace_generation_context, program.to_info())?;
68
69    tracing::event!(
70        tracing::Level::INFO,
71        "Generated execution trace of {} columns and {} steps (padded from {})",
72        miden_air::trace::TRACE_WIDTH,
73        trace.trace_len_summary().padded_trace_len(),
74        trace.trace_len_summary().trace_len()
75    );
76
77    let stack_outputs = *trace.stack_outputs();
78    let precompile_requests = trace.precompile_requests().to_vec();
79    let hash_fn = options.hash_fn();
80
81    // Convert trace to row-major format
82    let trace_matrix = {
83        let _span = tracing::info_span!("to_row_major_matrix").entered();
84        trace.to_row_major_matrix()
85    };
86
87    // Build public inputs and extract fixed/variable-length components
88    let (public_values, kernel_felts) = trace.public_inputs().to_air_inputs();
89    let var_len_public_inputs: &[&[Felt]] = &[&kernel_felts];
90
91    // Get aux trace builders
92    let aux_builder = trace.aux_trace_builders();
93
94    // Generate STARK proof using lifted prover
95    let params = config::pcs_params();
96    let proof_bytes = match hash_fn {
97        HashFunction::Blake3_256 => {
98            let config = config::blake3_256_config(params);
99            prove_stark(&config, &trace_matrix, &public_values, var_len_public_inputs, &aux_builder)
100        },
101        HashFunction::Keccak => {
102            let config = config::keccak_config(params);
103            prove_stark(&config, &trace_matrix, &public_values, var_len_public_inputs, &aux_builder)
104        },
105        HashFunction::Rpo256 => {
106            let config = config::rpo_config(params);
107            prove_stark(&config, &trace_matrix, &public_values, var_len_public_inputs, &aux_builder)
108        },
109        HashFunction::Poseidon2 => {
110            let config = config::poseidon2_config(params);
111            prove_stark(&config, &trace_matrix, &public_values, var_len_public_inputs, &aux_builder)
112        },
113        HashFunction::Rpx256 => {
114            let config = config::rpx_config(params);
115            prove_stark(&config, &trace_matrix, &public_values, var_len_public_inputs, &aux_builder)
116        },
117    }?;
118
119    let proof = ExecutionProof::new(proof_bytes, hash_fn, precompile_requests);
120
121    Ok((stack_outputs, proof))
122}

124/// Synchronous wrapper for the async `prove()` function.
125///
126/// This method is only available on non-wasm32 targets. On wasm32, use the
127/// async `prove()` method directly since wasm32 runs in the browser's event loop.
128///
129/// # Panics
130/// Panics if called from within an existing Tokio runtime. Use the async `prove()`
131/// method instead in async contexts.
132#[cfg(not(target_arch = "wasm32"))]
133#[instrument("prove_program_sync", skip_all)]
134pub fn prove_sync(
135    program: &Program,
136    stack_inputs: StackInputs,
137    advice_inputs: AdviceInputs,
138    host: &mut impl Host,
139    options: ProvingOptions,
140) -> Result<(StackOutputs, ExecutionProof), ExecutionError> {
141    match tokio::runtime::Handle::try_current() {
142        Ok(_handle) => {
143            // We're already inside a Tokio runtime - this is not supported
144            // because we cannot safely create a nested runtime or move the
145            // non-Send host reference to another thread
146            panic!(
147                "Cannot call prove_sync from within a Tokio runtime. \
148                 Use the async prove() method instead."
149            )
150        },
151        Err(_) => {
152            // No runtime exists - create one and use it
153            let rt = tokio::runtime::Builder::new_current_thread().build().unwrap();
154            rt.block_on(prove(program, stack_inputs, advice_inputs, host, options))
155        },
156    }
157}

162/// Generates a STARK proof for the given trace and public values.
163///
164/// Pre-seeds the challenger with `public_values`, then delegates to the lifted
165/// prover. Returns the serialized proof bytes.
166pub fn prove_stark<SC>(
167    config: &SC,
168    trace: &RowMajorMatrix<Felt>,
169    public_values: &[Felt],
170    var_len_public_inputs: VarLenPublicInputs<'_, Felt>,
171    aux_builder: &AuxTraceBuilders,
172) -> Result<Vec<u8>, ExecutionError>
173where
174    SC: StarkConfig<Felt, QuadFelt>,
175    <SC::Lmcs as Lmcs>::Commitment: Serialize,
176{
177    let log_trace_height = trace.height().ilog2() as u8;
178
179    let mut challenger = config.challenger();
180    challenger.observe_slice(public_values);
181    // TODO: observe log_trace_height in the transcript for Fiat-Shamir binding.
182    // TODO: observe var_len_public_inputs in the transcript for Fiat-Shamir binding.
183    //   This also requires updating the recursive verifier to absorb both fixed and
184    //   variable-length public inputs.
185    // TODO: observe ACE commitment once ACE verification is integrated.
186    // See https://github.com/0xMiden/miden-vm/issues/2822
187    let output: StarkOutput<Felt, QuadFelt, SC> = miden_crypto::stark::prover::prove_single(
188        config,
189        &ProcessorAir,
190        trace,
191        public_values,
192        var_len_public_inputs,
193        aux_builder,
194        challenger,
195    )
196    .map_err(|e| ExecutionError::ProvingError(e.to_string()))?;
197    // Proof serialization via bincode; see https://github.com/0xMiden/miden-vm/issues/2550
198    // We serialize `(log_trace_height, proof)` as a tuple; this is a temporary approach until
199    // the lifted STARK integrates trace height on its side.
200    let proof_bytes = bincode::serialize(&(log_trace_height, &output.proof))
201        .map_err(|e| ExecutionError::ProvingError(e.to_string()))?;
202    Ok(proof_bytes)
203}
