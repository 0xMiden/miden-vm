38/// Builder for a [`MastForest`].
39///
40/// The purpose of the builder is to ensure that the underlying MAST forest contains as little
41/// information as possible needed to adequately describe the logical MAST forest. Specifically:
42/// - The builder ensures that only one copy of nodes that have the same MAST root and decorators is
43///   added to the MAST forest (i.e., two nodes that have the same MAST root and decorators will
44///   have the same [`MastNodeId`]).
45/// - The builder tries to merge adjacent basic blocks and eliminate the source block whenever this
46///   does not have an impact on other nodes in the forest.
47#[derive(Clone, Debug, Default)]
48pub struct MastForestBuilder {
49    /// The MAST forest being built by this builder; this MAST forest is up-to-date - i.e., all
50    /// nodes added to the MAST forest builder are also immediately added to the underlying MAST
51    /// forest.
52    pub(crate) mast_forest: MastForest,
53    /// A map of all procedures added to the MAST forest indexed by their global procedure ID.
54    /// This includes all local, exported, and re-exported procedures. In case multiple procedures
55    /// with the same digest are added to the MAST forest builder, only the first procedure is
56    /// added to the map, and all subsequent insertions are ignored.
57    procedures: BTreeMap<GlobalItemIndex, Procedure>,
58    /// A map from procedure MAST root to its global procedure index. Similar to the `procedures`
59    /// map, this map contains only the first inserted procedure for procedures with the same MAST
60    /// root.
61    proc_gid_by_mast_root: BTreeMap<Word, GlobalItemIndex>,
62    /// A map of MAST node fingerprints to their corresponding positions in the MAST forest.
63    node_id_by_fingerprint: BTreeMap<MastNodeFingerprint, MastNodeId>,
64    /// The reverse mapping of `node_id_by_fingerprint`. This map caches the fingerprints of all
65    /// nodes (for performance reasons).
66    hash_by_node_id: BTreeMap<MastNodeId, MastNodeFingerprint>,
67    /// A map of decorator fingerprints to their corresponding positions in the MAST forest.
68    decorator_id_by_fingerprint: BTreeMap<DecoratorFingerprint, DecoratorId>,
69    /// A set of IDs for basic blocks which have been merged into a bigger basic blocks. This is
70    /// used as a candidate set of nodes that may be eliminated if the are not referenced by any
71    /// other node in the forest and are not a root of any procedure.
72    merged_basic_block_ids: BTreeSet<MastNodeId>,
73    /// A MastForest that contains the MAST of all statically-linked libraries, it's used to find
74    /// precompiled procedures and copy their subtrees instead of inserting external nodes.
75    statically_linked_mast: Arc<MastForest>,
76    /// Keeps track of the new ids assigned to nodes that are copied from the MAST of
77    /// statically-linked libraries.
78    statically_linked_mast_remapping: Remapping,
79    /// Keeps track of the new ids assigned to decorators that are copied from the MAST of
80    /// statically-linked libraries.
81    statically_linked_decorator_remapping: BTreeMap<DecoratorId, DecoratorId>,
82    /// Pending AssemblyOp mappings to be registered at build time.
83    ///
84    /// These are collected during assembly and registered all at once in sorted node order
85    /// when `build()` is called. This is necessary because the CSR structure requires nodes
86    /// to be added in sequential order, but nodes may be created in any order during assembly.
87    pending_asm_op_mappings: Vec<(MastNodeId, Vec<(usize, AsmOpId)>)>,
88}

90impl MastForestBuilder {
91    /// Creates a new builder which will transitively include the MAST of any procedures referenced
92    /// in the provided set of statically-linked libraries.
93    ///
94    /// In all other cases, references to procedures not present in the main MastForest are assumed
95    /// to be dynamically-linked, and are inserted as an external node. Dynamically-linked libraries
96    /// must be provided separately to the processor at runtime.
97    pub fn new<'a>(
98        static_libraries: impl IntoIterator<Item = &'a Library>,
99    ) -> Result<Self, Report> {
100        // All statically-linked libraries are merged into a single MastForest.
101        let forests = static_libraries.into_iter().map(|lib| lib.mast_forest().as_ref());
102        let (statically_linked_mast, _remapping) = MastForest::merge(forests).into_diagnostic()?;
103        // The AdviceMap of the statically-linked forest is copied to the forest being built.
104        //
105        // This might include excess advice map data in the built MastForest, but we currently do
106        // not do any analysis to determine what advice map data is actually required by parts of
107        // the library(s) that are actually linked into the output.
108        let mut mast_forest = MastForest::default();
109        *mast_forest.advice_map_mut() = statically_linked_mast.advice_map().clone();
110        Ok(MastForestBuilder {
111            mast_forest,
112            statically_linked_mast: Arc::new(statically_linked_mast),
113            ..Self::default()
114        })
115    }
116
117    /// Returns a reference to the underlying [`MastForest`].
118    pub fn mast_forest(&self) -> &MastForest {
119        &self.mast_forest
120    }
121
122    /// Removes the unused nodes that were created as part of the assembly process, and returns the
123    /// resulting MAST forest.
124    ///
125    /// It also returns the map from old node IDs to new node IDs. Any [`MastNodeId`] used in
126    /// reference to the old [`MastForest`] should be remapped using this map.
127    pub fn build(mut self) -> (MastForest, BTreeMap<MastNodeId, MastNodeId>) {
128        // Register all pending AssemblyOp mappings in sorted node order.
129        // The CSR structure requires nodes to be added sequentially.
130        // We must also merge mappings for duplicate node_ids (can happen when control flow nodes
131        // like Call are deduplicated but still have asm_ops registered).
132        let deduped_mappings =
133            deduplicate_asm_op_mappings(core::mem::take(&mut self.pending_asm_op_mappings));
134
135        for (node_id, asm_op_mappings) in deduped_mappings {
136            let (num_operations, adjusted_mappings) =
137                compute_operations_and_adjust_mappings(&self.mast_forest[node_id], asm_op_mappings);
138
139            // Errors here are programming errors since we control the ordering.
140            // Use expect to surface any issues during development.
141            self.mast_forest
142                .debug_info_mut()
143                .register_asm_ops(node_id, num_operations, adjusted_mappings)
144                .expect("failed to register AssemblyOps - internal ordering error");
145        }
146
147        let nodes_to_remove = get_nodes_to_remove(self.merged_basic_block_ids, &self.mast_forest);
148        let id_remappings = self.mast_forest.remove_nodes(&nodes_to_remove);
149
150        (self.mast_forest, id_remappings)
151    }
152}

154/// Computes the number of operations for a node and adjusts AssemblyOp indices if needed.
155///
156/// For basic block nodes, adjusts indices to account for padding NOOPs in OpBatches.
157/// For control flow nodes, computes the operation count from the maximum index.
158fn compute_operations_and_adjust_mappings(
159    node: &MastNode,
160    asm_op_mappings: Vec<(usize, AsmOpId)>,
161) -> (usize, Vec<(usize, AsmOpId)>) {
162    match node {
163        MastNode::Block(block) => (
164            block.num_operations() as usize,
165            BasicBlockNode::adjust_asm_op_indices(asm_op_mappings, block.op_batches()),
166        ),
167        _ => {
168            let num_ops = asm_op_mappings.iter().map(|(idx, _)| idx + 1).max().unwrap_or(0);
169            (num_ops, asm_op_mappings)
170        },
171    }
172}

174/// Deduplicates AssemblyOp mappings by node_id, keeping only the first registration.
175///
176/// Mappings are sorted by node_id, then deduplicated. This is necessary because control flow
177/// nodes like Call can be deduplicated, resulting in multiple registrations for the same node_id.
178fn deduplicate_asm_op_mappings(
179    mut mappings: Vec<(MastNodeId, Vec<(usize, AsmOpId)>)>,
180) -> Vec<(MastNodeId, Vec<(usize, AsmOpId)>)> {
181    mappings.sort_by_key(|(node_id, _)| *node_id);
182
183    let mut seen_node_ids = BTreeSet::new();
184    mappings
185        .into_iter()
186        .filter(|(node_id, _)| seen_node_ids.insert(*node_id))
187        .collect()
188}

190/// Takes the set of MAST node ids (all basic blocks) that were merged as part of the assembly
191/// process (i.e. they were contiguous and were merged into a single basic block), and returns the
192/// subset of nodes that can be removed from the MAST forest.
193///
194/// Specifically, MAST node ids can be reused, so merging a basic block doesn't mean it should be
195/// removed (specifically in the case where another node refers to it). Hence, we cycle through all
196/// nodes of the forest and only mark for removal those nodes that are not referenced by any node.
197/// We also ensure that procedure roots are not removed.
198fn get_nodes_to_remove(
199    merged_node_ids: BTreeSet<MastNodeId>,
200    mast_forest: &MastForest,
201) -> BTreeSet<MastNodeId> {
202    // make sure not to remove procedure roots
203    let mut nodes_to_remove: BTreeSet<MastNodeId> = merged_node_ids
204        .iter()
205        .filter(|&&mast_node_id| !mast_forest.is_procedure_root(mast_node_id))
206        .copied()
207        .collect();
208
209    for node in mast_forest.nodes() {
210        node.for_each_child(|child_id| {
211            if nodes_to_remove.contains(&child_id) {
212                nodes_to_remove.remove(&child_id);
213            }
214        });
215    }
216
217    nodes_to_remove
218}

247/// Procedure insertion
248impl MastForestBuilder {
249    /// Inserts a procedure into this MAST forest builder.
250    ///
251    /// If the procedure with the same ID already exists in this forest builder, this will have
252    /// no effect.
253    pub fn insert_procedure(
254        &mut self,
255        gid: GlobalItemIndex,
256        procedure: Procedure,
257    ) -> Result<(), Report> {
258        // Check if an entry is already in this cache slot.
259        //
260        // If there is already a cache entry, but it conflicts with what we're trying to cache,
261        // then raise an error.
262        if self.procedures.contains_key(&gid) {
263            // The global procedure index and the MAST root resolve to an already cached version of
264            // this procedure, or an alias of it, nothing to do.
265            //
266            // TODO: We should emit a warning for this, because while it is not an error per se, it
267            // does reflect that we're doing work we don't need to be doing. However, emitting a
268            // warning only makes sense if this is controllable by the user, and it isn't yet
269            // clear whether this edge case will ever happen in practice anyway.
270            return Ok(());
271        }
272
273        // We don't have a cache entry yet, but we do want to make sure we don't have a conflicting
274        // cache entry with the same MAST root:
275        if let Some(cached) = self.find_procedure_by_mast_root(&procedure.mast_root()) {
276            // Handle the case where a procedure with no locals is lowered to a MastForest
277            // consisting only of an `External` node to another procedure which has one or more
278            // locals. This will result in the calling procedure having the same digest as the
279            // callee, but the two procedures having mismatched local counts. When this occurs,
280            // we want to use the procedure with non-zero local count as the definition, and treat
281            // the other procedure as an alias, which can be referenced like any other procedure,
282            // but the MAST returned for it will be that of the "real" definition.
283            let cached_locals = cached.num_locals();
284            let procedure_locals = procedure.num_locals();
285            let mismatched_locals = cached_locals != procedure_locals;
286            let is_valid =
287                !mismatched_locals || core::cmp::min(cached_locals, procedure_locals) == 0;
288            if !is_valid {
289                let first = cached.path();
290                let second = procedure.path();
291                return Err(report!(
292                    "two procedures found with same mast root, but conflicting definitions ('{}' and '{}')",
293                    first,
294                    second
295                ));
296            }
297        }
298
299        self.mast_forest.make_root(procedure.body_node_id());
300        self.proc_gid_by_mast_root.insert(procedure.mast_root(), gid);
301        self.procedures.insert(gid, procedure);
302
303        Ok(())
304    }
305}

308/// Joining nodes
309impl MastForestBuilder {
310    /// Builds a tree of `JOIN` operations to combine the provided MAST node IDs.
311    ///
312    /// If `asm_op` is provided, each created `JoinNode` will have the given [`AssemblyOp`]
313    /// registered, enabling source-location diagnostics for errors that occur within join
314    /// contexts (e.g., when an `ExternalNode` fails to resolve).
315    pub fn join_nodes(
316        &mut self,
317        node_ids: Vec<MastNodeId>,
318        asm_op: Option<AssemblyOp>,
319    ) -> Result<MastNodeId, Report> {
320        debug_assert!(!node_ids.is_empty(), "cannot combine empty MAST node id list");
321
322        let mut node_ids = self.merge_contiguous_basic_blocks(node_ids)?;
323
324        // build a binary tree of blocks joining them using JOIN blocks
325        while node_ids.len() > 1 {
326            let last_mast_node_id = if node_ids.len().is_multiple_of(2) {
327                None
328            } else {
329                node_ids.pop()
330            };
331
332            let mut source_node_ids = Vec::new();
333            core::mem::swap(&mut node_ids, &mut source_node_ids);
334
335            let mut source_mast_node_iter = source_node_ids.drain(0..);
336            while let (Some(left), Some(right)) =
337                (source_mast_node_iter.next(), source_mast_node_iter.next())
338            {
339                let join_mast_node_id = self.ensure_join(left, right, vec![], vec![])?;
340
341                if let Some(ref asm_op) = asm_op {
342                    self.register_node_asm_op(join_mast_node_id, asm_op.clone())?;
343                }
344
345                node_ids.push(join_mast_node_id);
346            }
347            if let Some(mast_node_id) = last_mast_node_id {
348                node_ids.push(mast_node_id);
349            }
350        }
351
352        Ok(node_ids.remove(0))
353    }
354
355    /// Returns a list of [`MastNodeId`]s built from merging the contiguous basic blocks
356    /// found in the provided list of [`MastNodeId`]s.
357    fn merge_contiguous_basic_blocks(
358        &mut self,
359        node_ids: Vec<MastNodeId>,
360    ) -> Result<Vec<MastNodeId>, Report> {
361        let mut merged_node_ids = Vec::with_capacity(node_ids.len());
362        let mut contiguous_basic_block_ids: Vec<MastNodeId> = Vec::new();
363
364        for mast_node_id in node_ids {
365            if self.mast_forest[mast_node_id].is_basic_block() {
366                contiguous_basic_block_ids.push(mast_node_id);
367            } else {
368                merged_node_ids.extend(self.merge_basic_blocks(&contiguous_basic_block_ids)?);
369                contiguous_basic_block_ids.clear();
370
371                merged_node_ids.push(mast_node_id);
372            }
373        }
374
375        merged_node_ids.extend(self.merge_basic_blocks(&contiguous_basic_block_ids)?);
376
377        Ok(merged_node_ids)
378    }
379
380    /// Creates a new basic block by appending all operations and decorators in the provided list of
381    /// basic blocks (which are assumed to be contiguous).
382    ///
383    /// # Panics
384    /// - Panics if a provided [`MastNodeId`] doesn't refer to a basic block node.
385    fn merge_basic_blocks(
386        &mut self,
387        contiguous_basic_block_ids: &[MastNodeId],
388    ) -> Result<Vec<MastNodeId>, Report> {
389        if contiguous_basic_block_ids.is_empty() {
390            return Ok(Vec::new());
391        }
392        if contiguous_basic_block_ids.len() == 1 {
393            return Ok(contiguous_basic_block_ids.to_vec());
394        }
395
396        let mut operations: Vec<Operation> = Vec::new();
397        let mut decorators = DecoratorList::new();
398        // Track asm_ops being accumulated for merged blocks, with adjusted indices
399        let mut merged_asm_ops: Vec<(usize, AsmOpId)> = Vec::new();
400
401        let mut merged_basic_blocks: Vec<MastNodeId> = Vec::new();
402
403        for &basic_block_id in contiguous_basic_block_ids {
404            // check if the block should be merged with other blocks
405            if should_merge(
406                self.mast_forest.is_procedure_root(basic_block_id),
407                self.mast_forest[basic_block_id]
408                    .get_basic_block()
409                    .expect("merge_basic_blocks: expected BasicBlockNode")
410                    .num_op_batches(),
411            ) {
412                // Collect decorators and operations from the block (while still borrowing)
413                // We need owned copies so we can drop the borrow before mutating self
414                let (block_decorators, block_ops) = {
415                    let basic_block_node =
416                        self.mast_forest[basic_block_id].get_basic_block().unwrap();
417                    let block_decorators: Vec<_> =
418                        basic_block_node.raw_decorator_iter(&self.mast_forest).collect();
419                    let block_ops: Vec<Operation> = basic_block_node
420                        .op_batches()
421                        .iter()
422                        .flat_map(|b| b.raw_ops().copied())
423                        .collect();
424                    (block_decorators, block_ops)
425                };
426                let ops_offset = operations.len();
427
428                // Transfer any pending asm_ops for this block to the merged result
429                self.transfer_asm_ops_for_merge(basic_block_id, ops_offset, &mut merged_asm_ops);
430
431                // Add decorators with adjusted indices
432                for (op_idx, decorator) in block_decorators {
433                    decorators.push((op_idx + ops_offset, decorator));
434                }
435                operations.extend(block_ops);
436            } else {
437                // If we don't want to merge this block, flush the buffer of operations into a
438                // new block, and add the un-merged block after it.
439                if !operations.is_empty() {
440                    let block_ops = core::mem::take(&mut operations);
441                    let block_decorators = core::mem::take(&mut decorators);
442                    let block_asm_ops = core::mem::take(&mut merged_asm_ops);
443                    let merged_basic_block_id = self.ensure_block_with_asm_op_ids(
444                        block_ops,
445                        block_decorators,
446                        block_asm_ops,
447                        vec![],
448                        vec![],
449                    )?;
450
451                    merged_basic_blocks.push(merged_basic_block_id);
452                }
453                merged_basic_blocks.push(basic_block_id);
454            }
455        }
456
457        // Mark the removed basic blocks as merged
458        self.merged_basic_block_ids.extend(contiguous_basic_block_ids.iter());
459
460        if !operations.is_empty() || !decorators.is_empty() {
461            let merged_basic_block = self.ensure_block_with_asm_op_ids(
462                operations,
463                decorators,
464                merged_asm_ops,
465                vec![],
466                vec![],
467            )?;
468            merged_basic_blocks.push(merged_basic_block);
469        }
470
471        Ok(merged_basic_blocks)
472    }
473
474    /// Helper to transfer pending asm_ops from a block being merged.
475    /// Removes the asm_ops from pending_asm_op_mappings and adds them to the merged list
476    /// with indices adjusted by the given offset.
477    fn transfer_asm_ops_for_merge(
478        &mut self,
479        source_block_id: MastNodeId,
480        ops_offset: usize,
481        merged_asm_ops: &mut Vec<(usize, AsmOpId)>,
482    ) {
483        // Partition pending mappings into those matching source_block_id and the rest
484        let (matched, rest): (Vec<_>, Vec<_>) = core::mem::take(&mut self.pending_asm_op_mappings)
485            .into_iter()
486            .partition(|(node_id, _)| *node_id == source_block_id);
487
488        self.pending_asm_op_mappings = rest;
489
490        // Add matched asm_ops to merged list with adjusted indices
491        for (_, asm_ops) in matched {
492            merged_asm_ops.extend(
493                asm_ops.into_iter().map(|(op_idx, asm_op_id)| (op_idx + ops_offset, asm_op_id)),
494            );
495        }
496    }
497
498    /// Like ensure_block but takes pre-existing AsmOpIds instead of AssemblyOps.
499    /// Used when merging blocks that already have their asm_ops registered.
500    fn ensure_block_with_asm_op_ids(
501        &mut self,
502        operations: Vec<Operation>,
503        decorators: DecoratorList,
504        asm_op_ids: Vec<(usize, AsmOpId)>,
505        before_enter: Vec<DecoratorId>,
506        after_exit: Vec<DecoratorId>,
507    ) -> Result<MastNodeId, Report> {
508        let block = BasicBlockNodeBuilder::new(operations, decorators)
509            .with_before_enter(before_enter)
510            .with_after_exit(after_exit);
511        let (node_id, is_new) = self.ensure_node_exists(block)?;
512
513        // Only register AssemblyOps for newly created nodes.
514        if is_new && !asm_op_ids.is_empty() {
515            self.pending_asm_op_mappings.push((node_id, asm_op_ids));
516        }
517
518        Ok(node_id)
519    }
520}

523/// Node inserters
524impl MastForestBuilder {
525    /// Adds a decorator to the forest, and returns the [`Decorator`] associated with it.
526    pub fn ensure_decorator(&mut self, decorator: Decorator) -> Result<DecoratorId, Report> {
527        let decorator_hash = decorator.fingerprint();
528
529        if let Some(decorator_id) = self.decorator_id_by_fingerprint.get(&decorator_hash) {
530            // decorator already exists in the forest; return previously assigned id
531            Ok(*decorator_id)
532        } else {
533            let new_decorator_id = self
534                .mast_forest
535                .add_decorator(decorator)
536                .into_diagnostic()
537                .wrap_err("assembler failed to add new decorator")?;
538            self.decorator_id_by_fingerprint.insert(decorator_hash, new_decorator_id);
539
540            Ok(new_decorator_id)
541        }
542    }
543
544    /// Adds a debug variable to the forest, and returns the [`DebugVarId`] associated with it.
545    ///
546    /// Unlike decorators, debug variables are not deduplicated since each occurrence
547    /// represents a specific point in program execution where the variable's location
548    /// is being tracked.
549    pub fn add_debug_var(
550        &mut self,
551        debug_var: miden_core::operations::DebugVarInfo,
552    ) -> Result<miden_core::mast::DebugVarId, Report> {
553        self.mast_forest
554            .add_debug_var(debug_var)
555            .into_diagnostic()
556            .wrap_err("assembler failed to add debug variable")
557    }
558
559    /// Adds a node to the forest, and returns the [`MastNodeId`] associated with it.
560    ///
561    /// Note that only one copy of nodes that have the same MAST root and decorators is added to the
562    /// MAST forest; two nodes that have the same MAST root and decorators will have the same
563    /// [`MastNodeId`].
564    pub(crate) fn ensure_node(
565        &mut self,
566        builder: impl MastForestContributor,
567    ) -> Result<MastNodeId, Report> {
568        let (node_id, _is_new) = self.ensure_node_exists(builder)?;
569        Ok(node_id)
570    }

572    /// Adds a node to the forest if it doesn't already exist.
573    ///
574    /// Returns `(node_id, is_new)` where `is_new` is true if the node was newly added,
575    /// or false if a duplicate node already existed.
576    fn ensure_node_exists(
577        &mut self,
578        builder: impl MastForestContributor,
579    ) -> Result<(MastNodeId, bool), Report> {
580        let node_fingerprint = builder
581            .fingerprint_for_node(&self.mast_forest, &self.hash_by_node_id)
582            .expect("hash_by_node_id should contain the fingerprints of all children of `node`");
583
584        if let Some(node_id) = self.node_id_by_fingerprint.get(&node_fingerprint) {
585            // node already exists in the forest; return previously assigned id
586            Ok((*node_id, false))
587        } else {
588            let new_node_id = builder
589                .add_to_forest(&mut self.mast_forest)
590                .into_diagnostic()
591                .wrap_err("assembler failed to add new node")?;
592            self.node_id_by_fingerprint.insert(node_fingerprint, new_node_id);
593            self.hash_by_node_id.insert(new_node_id, node_fingerprint);
594
595            Ok((new_node_id, true))
596        }
597    }

599    /// Adds a basic block node to the forest, and returns the [`MastNodeId`] associated with it.
600    ///
601    /// The `asm_ops` parameter contains AssemblyOp metadata for operations in this block. Each
602    /// entry is `(op_idx, AssemblyOp)` where `op_idx` is the operation index the AssemblyOp
603    /// corresponds to.
604    ///
605    /// Note: AssemblyOps are only registered for newly created nodes. If a duplicate node already
606    /// exists in the forest (deduplication), the asm_ops are ignored since the existing node
607    /// already has its metadata registered.
608    ///
609    /// The actual registration of AssemblyOp mappings is deferred until `build()` is called,
610    /// to ensure nodes are registered in sequential order as required by the CSR structure.
611    pub fn ensure_block(
612        &mut self,
613        operations: Vec<Operation>,
614        decorators: DecoratorList,
615        asm_ops: Vec<(usize, AssemblyOp)>,
616        before_enter: Vec<DecoratorId>,
617        after_exit: Vec<DecoratorId>,
618    ) -> Result<MastNodeId, Report> {
619        let block = BasicBlockNodeBuilder::new(operations, decorators)
620            .with_before_enter(before_enter)
621            .with_after_exit(after_exit);
622        let (node_id, is_new) = self.ensure_node_exists(block)?;
623
624        // Only register AssemblyOps for newly created nodes.
625        // Deduplicated nodes already have their asm_ops registered from when they were first added.
626        if is_new && !asm_ops.is_empty() {
627            let mut asm_op_mappings = Vec::with_capacity(asm_ops.len());
628            for (op_idx, asm_op) in asm_ops {
629                let asm_op_id = self
630                    .mast_forest
631                    .debug_info_mut()
632                    .add_asm_op(asm_op)
633                    .into_diagnostic()
634                    .wrap_err("failed to add AssemblyOp")?;
635                asm_op_mappings.push((op_idx, asm_op_id));
636            }
637            // Defer registration until build() to ensure sequential node order.
638            self.pending_asm_op_mappings.push((node_id, asm_op_mappings));
639        }
640
641        Ok(node_id)
642    }

741    /// Collects all decorators from a subtree in the statically linked forest and copies them
742    /// to the target forest, populating the decorator remapping.
743    ///
744    /// This must be called before copying nodes from the subtree to ensure all decorator IDs
745    /// can be properly remapped.
746    fn collect_decorators_from_subtree(&mut self, root_id: &MastNodeId) -> Result<(), Report> {
747        // Clear the decorator remapping for this subtree
748        self.statically_linked_decorator_remapping.clear();
749
750        // Iterate through all nodes in the subtree
751        for node_id in SubtreeIterator::new(root_id, &self.statically_linked_mast.clone()) {
752            // Get all decorator IDs used by this node
753            let decorator_ids: Vec<DecoratorId> = {
754                let mut ids = Vec::new();
755
756                // Collect before_enter decorators
757                ids.extend(self.statically_linked_mast.before_enter_decorators(node_id));
758
759                // Collect after_exit decorators
760                ids.extend(self.statically_linked_mast.after_exit_decorators(node_id));
761
762                // For BasicBlockNode, also collect op-indexed decorators
763                if let MastNode::Block(block_node) = &self.statically_linked_mast[node_id] {
764                    for (_idx, decorator_id) in
765                        block_node.indexed_decorator_iter(&self.statically_linked_mast)
766                    {
767                        ids.push(decorator_id);
768                    }
769                }
770
771                ids
772            };
773
774            // Copy each decorator to the target forest if not already copied
775            for old_decorator_id in decorator_ids {
776                if !self.statically_linked_decorator_remapping.contains_key(&old_decorator_id) {
777                    let decorator = self.statically_linked_mast[old_decorator_id].clone();
778                    let new_decorator_id = self.ensure_decorator(decorator)?;
779                    self.statically_linked_decorator_remapping
780                        .insert(old_decorator_id, new_decorator_id);
781                }
782            }
783        }
784
785        Ok(())
786    }

808    /// Adds a node corresponding to the given MAST root, according to how it is linked.
809    ///
810    /// * If statically-linked, then the entire subtree is copied, and the MastNodeId of the root of
811    ///   the inserted subtree is returned.
812    /// * If dynamically-linked, then an external node is inserted, and its MastNodeId is returned
813    pub fn ensure_external_link(&mut self, mast_root: Word) -> Result<MastNodeId, Report> {
814        if let Some(root_id) = self.statically_linked_mast.find_procedure_root(mast_root) {
815            // First, collect and copy all decorators from the subtree
816            self.collect_decorators_from_subtree(&root_id)?;
817
818            // Then copy all nodes with remapped children and decorators
819            for old_id in SubtreeIterator::new(&root_id, &self.statically_linked_mast.clone()) {
820                let node = self.statically_linked_mast[old_id].clone();
821                let builder = self.build_with_remapped_ids(old_id, node)?;
822                let new_id = self.ensure_node(builder)?;
823                self.statically_linked_mast_remapping.insert(old_id, new_id);
824            }
825            Ok(root_id.remap(&self.statically_linked_mast_remapping))
826        } else {
827            self.ensure_node(ExternalNodeBuilder::new(mast_root))
828        }
829    }

831    /// Adds a list of decorators to the provided node to be executed before the node executes.
832    ///
833    /// If other decorators are already present, the new decorators are added to the end of the
834    /// list.
835    pub fn append_before_enter(
836        &mut self,
837        node_id: MastNodeId,
838        decorator_ids: Vec<DecoratorId>,
839    ) -> Result<(), MastForestError> {
840        // Extract the existing node and convert it to a builder
841        let mut decorated_builder = self.mast_forest[node_id].clone().to_builder(&self.mast_forest);
842        decorated_builder.append_before_enter(decorator_ids);
843        let new_node_fingerprint =
844            decorated_builder.fingerprint_for_node(&self.mast_forest, &self.hash_by_node_id)?;
845        self.mast_forest[node_id] = decorated_builder.build(&self.mast_forest)?;
846
847        self.hash_by_node_id.insert(node_id, new_node_fingerprint);
848        self.node_id_by_fingerprint.insert(new_node_fingerprint, node_id);
849        Ok(())
850    }

852    pub fn append_after_exit(
853        &mut self,
854        node_id: MastNodeId,
855        decorator_ids: Vec<DecoratorId>,
856    ) -> Result<(), MastForestError> {
857        // Extract the existing node and convert it to a builder
858        let mut decorated_builder = self.mast_forest[node_id].clone().to_builder(&self.mast_forest);
859        decorated_builder.append_after_exit(decorator_ids);
860        let new_node_fingerprint =
861            decorated_builder.fingerprint_for_node(&self.mast_forest, &self.hash_by_node_id)?;
862        self.mast_forest[node_id] = decorated_builder.build(&self.mast_forest)?;
863
864        self.hash_by_node_id.insert(node_id, new_node_fingerprint);
865        self.node_id_by_fingerprint.insert(new_node_fingerprint, node_id);
866        Ok(())
867    }

968/// Determines if we want to merge a block with other blocks. Currently, this works as follows:
969/// - If the block is a procedure, we merge it only if the number of operation batches is smaller
970///   then the threshold (currently set at 32). The reasoning is based on an estimate of the the
971///   runtime penalty of not inlining the procedure. We assume that this penalty is roughly 3 extra
972///   nodes in the MAST and so would require 3 additional hashes at runtime. Since hashing each
973///   operation batch requires 1 hash, this basically implies that if the runtime penalty is more
974///   than 10%, we inline the block, but if it is less than 10% we accept the penalty to make
975///   deserialization faster.
976/// - If the block is not a procedure, we always merge it because: (1) if it is a large block, it is
977///   likely to be unique and, thus, the original block will be orphaned and removed later; (2) if
978///   it is a small block, there is a large run-time benefit for inlining it.
979fn should_merge(is_procedure: bool, num_op_batches: usize) -> bool {
980    if is_procedure {
981        num_op_batches < PROCEDURE_INLINING_THRESHOLD
982    } else {
983        true
984    }
985}
