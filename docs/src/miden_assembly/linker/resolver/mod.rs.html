23/// A [Resolver] is used to perform symbol resolution in the context of a specific module.
24///
25/// It is instantiated along with a [ResolverCache] to cache frequently-referenced symbols, and a
26/// [SymbolResolver] for resolving externally-defined symbols.
27pub struct Resolver<'a, 'b: 'a> {
28    pub resolver: &'a SymbolResolver<'b>,
29    pub cache: &'a mut ResolverCache,
30    pub current_module: ModuleIndex,
31}
32
33/// A [ResolverCache] is used to cache resolutions of type and constant expressions to concrete
34/// values that contain no references to other symbols. Since these resolutions can be expensive
35/// to compute, and often represent items which are referenced multiple times, we cache them to
36/// avoid recomputing the same information over and over again.
37#[derive(Default)]
38pub struct ResolverCache {
39    pub types: BTreeMap<GlobalItemIndex, ast::types::Type>,
40    pub constants: BTreeMap<GlobalItemIndex, ast::ConstantValue>,
41}

43impl<'a, 'b: 'a> ConstEnvironment for Resolver<'a, 'b> {
44    type Error = LinkerError;
45
46    fn get_source_file_for(&self, span: SourceSpan) -> Option<Arc<SourceFile>> {
47        self.resolver.source_manager().get(span.source_id()).ok()
48    }
49
50    fn get(&self, name: &Ident) -> Result<Option<CachedConstantValue<'_>>, Self::Error> {
51        let context = SymbolResolutionContext {
52            span: name.span(),
53            module: self.current_module,
54            kind: None,
55        };
56        let gid = match self.resolver.resolve_local(&context, name)? {
57            SymbolResolution::Exact { gid, .. } => gid,
58            SymbolResolution::Local(index) => self.current_module + index.into_inner(),
59            SymbolResolution::MastRoot(_) | SymbolResolution::Module { .. } => {
60                return Err(LinkerError::InvalidConstantRef {
61                    span: context.span,
62                    source_file: self.get_source_file_for(context.span),
63                });
64            },
65            SymbolResolution::External(path) => {
66                return Err(LinkerError::UndefinedSymbol {
67                    span: context.span,
68                    source_file: self.get_source_file_for(context.span),
69                    path: path.into_inner(),
70                });
71            },
72        };
73
74        match self.cache.constants.get(&gid).map(CachedConstantValue::Hit) {
75            some @ Some(_) => Ok(some),
76            None => match self.resolver.linker()[gid].item() {
77                SymbolItem::Compiled(ItemInfo::Constant(info)) => {
78                    Ok(Some(CachedConstantValue::Hit(&info.value)))
79                },
80                SymbolItem::Constant(item) => Ok(Some(CachedConstantValue::Miss(&item.value))),
81                _ => Err(LinkerError::InvalidConstantRef {
82                    span: name.span(),
83                    source_file: self.get_source_file_for(name.span()),
84                }),
85            },
86        }
87    }
88
89    fn get_by_path(
90        &self,
91        path: Span<&Path>,
92    ) -> Result<Option<CachedConstantValue<'_>>, Self::Error> {
93        let context = SymbolResolutionContext {
94            span: path.span(),
95            module: self.current_module,
96            kind: None,
97        };
98        let gid = match self.resolver.resolve_path(&context, path)? {
99            SymbolResolution::Exact { gid, .. } => gid,
100            SymbolResolution::Local(index) => self.current_module + index.into_inner(),
101            SymbolResolution::MastRoot(_) | SymbolResolution::Module { .. } => {
102                return Err(LinkerError::InvalidConstantRef {
103                    span: context.span,
104                    source_file: self.get_source_file_for(context.span),
105                });
106            },
107            SymbolResolution::External(path) => {
108                return Err(LinkerError::UndefinedSymbol {
109                    span: context.span,
110                    source_file: self.get_source_file_for(context.span),
111                    path: path.into_inner(),
112                });
113            },
114        };
115        if let Some(cached) = self.cache.constants.get(&gid) {
116            return Ok(Some(CachedConstantValue::Hit(cached)));
117        }
118        match self.resolver.linker()[gid].item() {
119            SymbolItem::Compiled(ItemInfo::Constant(info)) => {
120                Ok(Some(CachedConstantValue::Hit(&info.value)))
121            },
122            SymbolItem::Constant(item) => Ok(Some(CachedConstantValue::Miss(&item.value))),
123            SymbolItem::Compiled(_) | SymbolItem::Procedure(_) | SymbolItem::Type(_) => {
124                Err(LinkerError::InvalidConstantRef {
125                    span: context.span,
126                    source_file: self.get_source_file_for(context.span),
127                })
128            },
129            SymbolItem::Alias { .. } => {
130                unreachable!("the resolver should have expanded all aliases")
131            },
132        }
133    }
134
135    /// Cache evaluated constants so long as they evaluated to a ConstantValue, and we can resolve
136    /// the path to a known GlobalItemIndex
137    fn on_eval_completed(&mut self, path: Span<&Path>, value: &ast::ConstantExpr) {
138        let Some(value) = value.as_value() else {
139            return;
140        };
141        let context = SymbolResolutionContext {
142            span: path.span(),
143            module: self.current_module,
144            kind: None,
145        };
146        let gid = match self.resolver.resolve_path(&context, path) {
147            Ok(SymbolResolution::Exact { gid, .. }) => gid,
148            Ok(SymbolResolution::Local(index)) => self.current_module + index.into_inner(),
149            _ => return,
150        };
151        self.cache.constants.insert(gid, value);
152    }
153}

155impl<'a, 'b: 'a> ast::TypeResolver<LinkerError> for Resolver<'a, 'b> {
156    #[inline]
157    fn source_manager(&self) -> Arc<dyn SourceManager> {
158        self.resolver.source_manager_arc()
159    }
160    #[inline]
161    fn resolve_local_failed(&self, err: SymbolResolutionError) -> LinkerError {
162        LinkerError::from(err)
163    }
164
165    fn get_type(
166        &mut self,
167        context: SourceSpan,
168        gid: GlobalItemIndex,
169    ) -> Result<types::Type, LinkerError> {
170        match self.resolver.linker()[gid].item() {
171            SymbolItem::Compiled(ItemInfo::Type(info)) => Ok(info.ty.clone()),
172            SymbolItem::Type(ast::TypeDecl::Enum(ty)) => {
173                // When resolving an EnumType, we must do three things:
174                //
175                // * Resolve the discriminant type
176                // * Resolve the discriminant value and payload type for each variant
177                // * Construct the midenc_hir_type::EnumType, and validate that the enum is valid
178                //   according to the rules it enforces
179                let mut variants = SmallVec::<[types::Variant; 4]>::new_const();
180                for variant in ty.variants() {
181                    let discriminant_value = match self.resolver.linker().const_eval(
182                        gid,
183                        &variant.discriminant,
184                        self.cache,
185                    )? {
186                        ast::ConstantValue::Int(v) => Some(v.as_canonical_u64() as u128),
187                        invalid => {
188                            return Err(LinkerError::Related {
189                                errors: vec![RelatedError::new(Report::from(diagnostic!(
190                                    severity = Severity::Error,
191                                    labels = vec![LabeledSpan::at(
192                                        invalid.span(),
193                                        "invalid enum discriminant: expected an integer"
194                                    )],
195                                    "invalid enum type"
196                                )))]
197                                .into_boxed_slice(),
198                            });
199                        },
200                    };
201                    variants.push(types::Variant {
202                        name: variant.name.clone().into_inner(),
203                        value: match variant.value_ty.as_ref() {
204                            Some(t) => t.resolve_type(self)?,
205                            None => None,
206                        },
207                        discriminant_value,
208                    });
209                }
210                types::EnumType::new(ty.name().clone().into_inner(), ty.ty().clone(), variants)
211                    .map(|t| types::Type::Enum(Arc::new(t)))
212                    .map_err(|err| LinkerError::Related {
213                        errors: vec![RelatedError::from(Report::from(diagnostic!(
214                            severity = Severity::Error,
215                            labels = vec![LabeledSpan::at(context, err.to_string())],
216                            "invalid enum type"
217                        )))]
218                        .into_boxed_slice(),
219                    })
220            },
221            SymbolItem::Type(ast::TypeDecl::Alias(ty)) => {
222                Ok(ty.ty.resolve_type(self)?.expect("unreachable"))
223            },
224            SymbolItem::Compiled(_) | SymbolItem::Constant(_) | SymbolItem::Procedure(_) => {
225                Err(LinkerError::InvalidTypeRef {
226                    span: context,
227                    source_file: self.get_source_file_for(context),
228                })
229            },
230            SymbolItem::Alias { .. } => unreachable!("resolver should have expanded all aliases"),
231        }
232    }
233
234    fn get_local_type(
235        &mut self,
236        context: SourceSpan,
237        id: ItemIndex,
238    ) -> Result<Option<types::Type>, LinkerError> {
239        self.get_type(context, self.current_module + id).map(Some)
240    }
241
242    fn resolve_type_ref(&mut self, ty: Span<&Path>) -> Result<SymbolResolution, LinkerError> {
243        let context = SymbolResolutionContext {
244            span: ty.span(),
245            module: self.current_module,
246            kind: None,
247        };
248        match self.resolver.resolve_path(&context, ty)? {
249            exact @ SymbolResolution::Exact { .. } => Ok(exact),
250            SymbolResolution::Local(index) => {
251                let (span, index) = index.into_parts();
252                let current_module = &self.resolver.linker()[self.current_module];
253                let item = current_module[index].name();
254                let path = Span::new(span, current_module.path().join(item).into());
255                Ok(SymbolResolution::Exact { gid: self.current_module + index, path })
256            },
257            SymbolResolution::MastRoot(_) | SymbolResolution::Module { .. } => {
258                Err(LinkerError::InvalidTypeRef {
259                    span: ty.span(),
260                    source_file: self.get_source_file_for(ty.span()),
261                })
262            },
263            SymbolResolution::External(path) => Err(LinkerError::UndefinedSymbol {
264                span: ty.span(),
265                source_file: self.get_source_file_for(ty.span()),
266                path: path.into_inner(),
267            }),
268        }
269    }
