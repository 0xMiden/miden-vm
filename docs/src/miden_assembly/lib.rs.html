68/// Pushes the provided value onto the stack using the most optimal sequence of operations.
69fn push_value_ops(value: Felt) -> alloc::vec::Vec<Operation> {
70    use miden_core::operations::Operation::*;
71
72    if value == ZERO {
73        vec![Pad]
74    } else if value == ONE {
75        vec![Pad, Incr]
76    } else {
77        vec![Push(value)]
78    }
79}
