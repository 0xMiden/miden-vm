18/// Appends `PUSH` operation to the basic block to push provided constant value onto the stack.
19///
20/// In cases when the immediate value is 0, `PUSH` operation is replaced with `PAD`. Also, in cases
21/// when immediate value is 1, `PUSH` operation is replaced with `PAD INCR` because in most cases
22/// this will be more efficient than doing a `PUSH`.
23pub fn push_one<T>(imm: T, block_builder: &mut BasicBlockBuilder)
24where
25    T: Into<Felt>,
26{
27    push_felt(block_builder, imm.into());
28}
29
30/// Appends `PUSH` operations to the basic block to push two or more provided constant values onto
31/// the stack, up to a maximum of 16 values.
32///
33/// In cases when the immediate value is 0, `PUSH` operation is replaced with `PAD`. Also, in cases
34/// when immediate value is 1, `PUSH` operation is replaced with `PAD INCR` because in most cases
35/// this will be more efficient than doing a `PUSH`.
36pub fn push_many<T>(imms: &[T], block_builder: &mut BasicBlockBuilder)
37where
38    T: Into<Felt> + Copy,
39{
40    imms.iter().for_each(|imm| push_felt(block_builder, (*imm).into()));
41}
42
43/// Appends `PUSH` operations to the basic block to push a Word constant onto the stack in
44/// little-endian order.
45///
46/// The elements are pushed in reverse order (Word[3], Word[2], Word[1], Word[0]) so that
47/// Word[0] ends up on top of the stack (low element at lowest address / on top).
48///
49/// After this operation, calling `get_stack_word(0)` will return the original Word.
50pub fn push_word(word: &[Felt; 4], block_builder: &mut BasicBlockBuilder) {
51    // Push in reverse order so Word[0] ends up on top
52    for felt in word.iter().rev() {
53        push_felt(block_builder, *felt);
54    }
55}
56
57/// Appends `PUSH` operations to the basic block using the [Felt]s obtained from the Word value
58/// using the provided range.
59///
60/// The elements are pushed in reverse order so that the first element of the slice ends up on
61/// top of the stack, consistent with `push_word` behavior.
62///
63/// For example, if `WORD = [1, 2, 3, 4]`, then `push.WORD[0..2]` results in stack `[1, 2, ...]`
64/// with 1 on top.
65///
66/// # Errors
67/// Returns an error if:
68/// - The provided [`IntValue`] is not a [`IntValue::Word`].
69/// - The provided range is malformed.
70pub fn push_word_slice(
71    imm: &Immediate<WordValue>,
72    range: &Range<usize>,
73    block_builder: &mut BasicBlockBuilder,
74) -> Result<(), Report> {
75    let v = imm.expect_value();
76    match v.0.get(range.clone()) {
77        // invalid range case (i.e. [8..5])
78        None => {
79            return Err(Report::new(ParsingError::InvalidRange {
80                span: imm.span(),
81                range: range.clone(),
82            }));
83        },
84        // empty range case (i.e. [2..2])
85        Some([]) => {
86            return Err(Report::new(ParsingError::EmptySlice {
87                span: imm.span(),
88                range: range.clone(),
89            }));
90        },
91        // Push in reverse order so slice[0] ends up on top, consistent with push_word
92        Some(values) => {
93            for felt in values.iter().rev() {
94                push_felt(block_builder, *felt);
95            }
96        },
97    }
98
99    Ok(())
100}

105/// Appends a sequence of operations to the span needed for executing locaddr.i instruction. This
106/// consists of putting i onto the stack and then executing LOCADDR operation.
107///
108/// # Errors
109/// Returns an error if index is greater than the number of procedure locals.
110pub fn locaddr(
111    block_builder: &mut BasicBlockBuilder,
112    index: u16,
113    proc_ctx: &ProcedureContext,
114    instr_span: SourceSpan,
115) -> Result<(), Report> {
116    local_to_absolute_addr(block_builder, proc_ctx, index, proc_ctx.num_locals(), true, instr_span)
117}

119/// Appends CALLER operation to the span which puts the hash of the function which created the
120/// latest execution context.
121pub fn caller(block_builder: &mut BasicBlockBuilder) {
122    block_builder.push_op(Caller);
