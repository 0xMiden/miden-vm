1use miden_core::WORD_SIZE;
2
3use crate::{ProcedureContext, ast::DebugOptions, diagnostics::Report};
4
5/// Compiles the AST representation of a `debug` instruction into its VM representation.
6///
7/// This function does not currently return any errors, but may in the future.
8///
9/// See [crate::Assembler] for an overview of AST compilation.
10pub fn compile_options(
11    options: &DebugOptions,
12    proc_ctx: &ProcedureContext,
13) -> Result<miden_core::operations::DebugOptions, Report> {
14    type Ast = DebugOptions;
15    type Vm = miden_core::operations::DebugOptions;
16
17    // Use word-aligned num_locals for address calculations (same alignment as frame pointer)
18    let aligned_num_locals = proc_ctx.num_locals().next_multiple_of(WORD_SIZE as u16);
19
20    // NOTE: these `ast::Immediate::expect_value()` calls *should* be safe, because by the time
21    // we're compiling debug options all immediate-constant arguments should be resolved.
22    let compiled = match options {
23        Ast::StackAll => Vm::StackAll,
24        Ast::StackTop(n) => Vm::StackTop(n.expect_value()),
25        Ast::MemAll => Vm::MemAll,
26        Ast::MemInterval(start, end) => Vm::MemInterval(start.expect_value(), end.expect_value()),
27        Ast::LocalInterval(start, end) => {
28            let (start, end) = (start.expect_value(), end.expect_value());
29            Vm::LocalInterval(start, end, aligned_num_locals)
30        },
31        Ast::LocalRangeFrom(index) => {
32            let index = index.expect_value();
33            Vm::LocalInterval(index, index, aligned_num_locals)
34        },
35        Ast::LocalAll => {
36            let end_exclusive = Ord::max(1, proc_ctx.num_locals());
37            Vm::LocalInterval(0, end_exclusive - 1, aligned_num_locals)
38        },
39        Ast::AdvStackTop(n) => Vm::AdvStackTop(n.expect_value()),
40    };
41
42    Ok(compiled)
43}
