6/// Given a stack in the following initial configuration [b0, b1, a0, a1, ...] where a = (a0, a1)
7/// and b = (b0, b1) represent elements in the extension field of degree 2, this series
8/// of operations outputs the result c = (c0, c1) where c0 = a0 + b0 and c1 = a1 + b1.
9///
10/// This operation takes 5 VM cycles.
11pub fn ext2_add(block_builder: &mut BasicBlockBuilder) {
12    #[rustfmt::skip]
13    let ops = [
14        Swap,           // [b1, b0, a0, a1, ...]
15        MovUp3,         // [a1, b1, b0, a0, ...]
16        Add,            // [a1+b1, b0, a0, ...]
17        MovDn2,         // [b0, a0, a1+b1, ...]
18        Add             // [a0+b0, a1+b1, ...] = [c0, c1, ...]
19    ];
20    block_builder.push_ops(ops);
21}

23/// Given a stack in the following initial configuration [b0, b1, a0, a1, ...] where a = (a0, a1)
24/// and b = (b0, b1) represent elements in the extension field of degree 2, this series
25/// of operations outputs the result c = (c0, c1) where c0 = a0 - b0 and c1 = a1 - b1.
26///
27/// This operation takes 7 VM cycles.
28pub fn ext2_sub(block_builder: &mut BasicBlockBuilder) {
29    #[rustfmt::skip]
30    let ops = [
31        Neg,        // [-b0, b1, a0, a1, ...] (negate low coef)
32        Swap,       // [b1, -b0, a0, a1, ...]
33        Neg,        // [-b1, -b0, a0, a1, ...]
34        MovUp3,     // [a1, -b1, -b0, a0, ...]
35        Add,        // [a1-b1, -b0, a0, ...]
36        MovDn2,     // [-b0, a0, a1-b1, ...]
37        Add         // [a0-b0, a1-b1, ...] = [c0, c1, ...]
38    ];
39    block_builder.push_ops(ops);
40}

52/// Given a stack in the following initial configuration [b0, b1, a0, a1, ...] where a = (a0, a1)
53/// and b = (b0, b1) represent elements in the extension field of degree 2, this series
54/// of operations outputs the result c = (c0, c1) where c = a * b^-1.
55///
56/// This operation takes 11 VM cycles.
57pub fn ext2_div(block_builder: &mut BasicBlockBuilder) {
58    block_builder.push_system_event(Ext2Inv);
59    #[rustfmt::skip]
60    let ops = [
61        AdvPop,         // [b1', b0, b1, a0, a1, ...] (gets high coef of inverse first)
62        AdvPop,         // [b0', b1', b0, b1, a0, a1, ...] (then low coef)
63        Ext2Mul,        // [b0', b1', 1, 0, a0, a1, ...] (result c0=1, c1=0 for identity)
64        MovUp3,         // [0, b0', b1', 1, a0, a1, ...] (move c1 to top)
65        Eqz,            // [1, b0', b1', 1, a0, a1, ...] (verify c1 == 0)
66        Assert(ZERO),   // [b0', b1', 1, a0, a1, ...]
67        MovUp2,         // [1, b0', b1', a0, a1, ...] (move c0 to top)
68        Assert(ZERO),   // [b0', b1', a0, a1, ...] (verify c0 is truthy, i.e. == 1)
69        Ext2Mul,        // [b0', b1', c0, c1, ...] (c = a * b^-1)
70        Drop,           // [b1', c0, c1, ...]
71        Drop            // [c0, c1, ...] (LE result)
72    ];
73    block_builder.push_ops(ops);
74}

88    ];
89    block_builder.push_ops(ops);

92/// Given an invertible quadratic extension field element on the stack, this routine computes
93/// multiplicative inverse of that element, using non-deterministic technique
94/// (i.e. it takes help of advice provider).
95/// To ensure that non-deterministic computation resulted in correct value, it multiplies input
96/// operand with computed output, over quadratic extension field which must produce multiplicative
97/// identity (1, 0) of quadratic extension field. In case input operand is additive identity which
98/// can't be inverted, program execution fails, as advice provider won't calculate multiplicative
99/// inverse in that case.
100///
101/// Expected input stack
102///
103/// [a0, a1, ...] | a = (a0, a1) ∈ Quadratic extension field over F_p, p = 2^64 - 2^32 + 1
104///
105/// Expected output stack
106///
107/// [a'0, a'1, ...] | a' = (a'0, a'1) ∈ Quadratic extension field over F_p, p = 2^64 - 2^32 + 1
108///
109/// Following is what is checked after reading result of computation, performed outside of VM
110///
111/// a  = (a0, a1)
112/// a' = (a'0, a'1) ( = a ^ -1 )
113///
114/// b  = a * a' ( mod Q ) | Q = irreducible polynomial x^2 - 7 over F_p, p = 2^64 - 2^32 + 1
115/// assert b  = (1, 0) | (1, 0) is the multiplicative identity of extension field.
116///
117/// This operation takes 8 VM cycles.
118pub fn ext2_inv(block_builder: &mut BasicBlockBuilder) -> Result<(), Report> {
119    block_builder.push_system_event(Ext2Inv);
120    #[rustfmt::skip]
121    let ops = [
122        AdvPop,         // [a1', a0, a1, ...] (gets high coef of inverse first)
123        AdvPop,         // [a0', a1', a0, a1, ...] (then low coef)
124        Ext2Mul,        // [a0', a1', 1, 0, ...] (result c0=1, c1=0 for identity)
125        MovUp3,         // [0, a0', a1', 1, ...] (move c1 to top)
126        Eqz,            // [1, a0', a1', 1, ...] (verify c1 == 0)
127        Assert(ZERO),   // [a0', a1', 1, ...]
128        MovUp2,         // [1, a0', a1', ...] (move c0 to top)
129        Assert(ZERO),   // [a0', a1', ...] (verify c0 is truthy, i.e. == 1)
130    ];
131    block_builder.push_ops(ops);
132
133    Ok(())
134}
