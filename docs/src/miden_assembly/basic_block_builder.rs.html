23// ================================================================================================

25/// Information about an instruction being tracked, pending cycle count computation.
26///
27/// When an instruction is encountered during assembly, we don't yet know how many VM cycles
28/// it will consume. This struct holds the instruction's metadata until the cycle count can
29/// be computed (when the next instruction begins or the block ends).
30#[derive(Debug)]
31struct PendingAsmOp {
32    /// Operation index where this instruction started (before any ops were added for it).
33    op_start: usize,
34    /// Source location in the original source file, if available.
35    location: Option<Location>,
36    /// The fully-qualified procedure path (e.g., "std::math::u64::add").
37    context_name: String,
38    /// The string representation of the instruction (e.g., "add", "push.1").
39    op: String,
40}

45/// A helper struct for constructing basic blocks while compiling procedure bodies.
46///
47/// Operations and decorators can be added to a basic block builder via various `add_*()` and
48/// `push_*()` methods, and then basic blocks can be extracted from the builder via `extract_*()`
49/// methods.
50///
51/// The same basic block builder can be used to construct many blocks. It is expected that when the
52/// last basic block in a procedure's body is constructed [`Self::try_into_basic_block`] will be
53/// used.
54#[derive(Debug)]
55pub struct BasicBlockBuilder<'a> {
56    ops: Vec<Operation>,
57    decorators: DecoratorList,
58    epilogue: Vec<Operation>,
59    /// Pending assembly operation info, waiting for cycle count to be computed.
60    pending_asm_op: Option<PendingAsmOp>,
61    /// Finalized AssemblyOps with their operation indices (op_idx, AssemblyOp).
62    asm_ops: Vec<(usize, AssemblyOp)>,
63    /// Debug variables attached to operations in this block.
64    /// Each entry is (op_index, debug_var_id) similar to decorators.
65    debug_vars: Vec<(usize, DebugVarId)>,
66    mast_forest_builder: &'a mut MastForestBuilder,
67}
68
69/// Constructors
70impl<'a> BasicBlockBuilder<'a> {
71    /// Returns a new [`BasicBlockBuilder`] instantiated with the specified optional wrapper.
72    ///
73    /// If the wrapper is provided, the prologue of the wrapper is immediately appended to the
74    /// vector of span operations. The epilogue of the wrapper is appended to the list of operations
75    /// upon consumption of the builder via the [`Self::try_into_basic_block`] method.
76    pub(super) fn new(
77        wrapper: Option<BodyWrapper>,
78        mast_forest_builder: &'a mut MastForestBuilder,
79    ) -> Self {
80        match wrapper {
81            Some(wrapper) => Self {
82                ops: wrapper.prologue,
83                decorators: Vec::new(),
84                epilogue: wrapper.epilogue,
85                pending_asm_op: None,
86                asm_ops: Vec::new(),
87                debug_vars: Vec::new(),
88                mast_forest_builder,
89            },
90            None => Self {
91                ops: Default::default(),
92                decorators: Default::default(),
93                epilogue: Default::default(),
94                pending_asm_op: None,
95                asm_ops: Vec::new(),
96                debug_vars: Default::default(),
97                mast_forest_builder,
98            },
99        }
100    }
101}

116/// Operations
117impl BasicBlockBuilder<'_> {
118    /// Adds the specified operation to the list of basic block operations.
119    pub fn push_op(&mut self, op: Operation) {
120        self.ops.push(op);
121    }
122
123    /// Adds the specified sequence of operations to the list of basic block operations.
124    pub fn push_ops<I, O>(&mut self, ops: I)
125    where
126        I: IntoIterator<Item = O>,
127        O: Borrow<Operation>,
128    {
129        self.ops.extend(ops.into_iter().map(|o| *o.borrow()));
130    }
131
132    /// Adds the specified operation n times to the list of basic block operations.
133    pub fn push_op_many(&mut self, op: Operation, n: usize) {
134        let new_len = self.ops.len() + n;
135        self.ops.resize(new_len, op);
136    }
137
138    /// Converts the system event into its corresponding event ID, and adds an `Emit` operation
139    /// to the list of basic block operations.
140    pub fn push_system_event(&mut self, sys_event: SystemEvent) {
141        let event_id = sys_event.event_id();
142        self.push_ops([Operation::Push(event_id.as_felt()), Operation::Emit, Operation::Drop]);
143    }
144}

146/// Decorators
147impl BasicBlockBuilder<'_> {
148    /// Add the specified decorator to the list of basic block decorators.
149    pub fn push_decorator(&mut self, decorator: Decorator) -> Result<(), Report> {
150        let decorator_id = self.mast_forest_builder.ensure_decorator(decorator)?;
151        self.decorators.push((self.ops.len(), decorator_id));
152
153        Ok(())
154    }
155
156    /// Tracks an instruction for AssemblyOp metadata collection.
157    ///
158    /// This stores the instruction's metadata in a pending state. The cycle count will be
159    /// computed when [`Self::set_instruction_cycle_count`] is called (typically after all
160    /// operations for the instruction have been added).
161    pub fn track_instruction(
162        &mut self,
163        instruction: &Span<Instruction>,
164        proc_ctx: &ProcedureContext,
165    ) -> Result<(), Report> {
166        let span = instruction.span();
167        self.pending_asm_op = Some(PendingAsmOp {
168            op_start: self.ops.len(),
169            location: proc_ctx.source_manager().location(span).ok(),
170            context_name: proc_ctx.path().to_string(),
171            op: instruction.to_string(),
172        });
173
174        Ok(())
175    }
176
177    /// Finalizes the pending AssemblyOp with the computed cycle count.
178    ///
179    /// Computes the number of cycles elapsed since the last invocation of
180    /// [`Self::track_instruction`] and creates an [`AssemblyOp`] with that cycle count.
181    ///
182    /// If the cycle count is 0 (instruction did not contribute any operations to the basic block,
183    /// e.g., exec, call, syscall), returns the [`AssemblyOp`] so it can be attached to a node-level
184    /// decorator. Otherwise, stores the [`AssemblyOp`] in the internal `asm_ops` list for later
185    /// registration with the node's debug info.
186    pub fn set_instruction_cycle_count(&mut self) -> Option<AssemblyOp> {
187        let pending = self.pending_asm_op.take().expect("no pending asm op to finalize");
188
189        // Compute the cycle count for the instruction
190        let cycle_count = self.ops.len() - pending.op_start;
191
192        let asm_op =
193            AssemblyOp::new(pending.location, pending.context_name, cycle_count as u8, pending.op);
194
195        match cycle_count {
196            0 => Some(asm_op),
197            _ => {
198                self.asm_ops.push((pending.op_start, asm_op));
199                None
200            },
201        }
202    }
203
204    /// Adds a debug variable to the list of debug variables for this basic block.
205    ///
206    /// Debug variables are stored in dedicated CSR storage (not as decorators) and are
207    /// only accessed by the debugger. They track source-level variable locations at
208    /// specific points in program execution.
209    pub fn push_debug_var(&mut self, debug_var: DebugVarInfo) -> Result<(), Report> {
210        let debug_var_id = self.mast_forest_builder.add_debug_var(debug_var)?;
211        self.debug_vars.push((self.ops.len(), debug_var_id));
212        Ok(())
213    }
214}

216/// Basic Block Constructors
217impl BasicBlockBuilder<'_> {
218    /// Creates and returns a new basic block node from the operations and decorators currently in
219    /// this builder.
220    ///
221    /// If there are no operations however, then no node is created, the decorators are left
222    /// untouched and `None` is returned. Use [`Self::drain_decorators`] to retrieve the decorators
223    /// in this case.
224    ///
225    /// This consumes all operations in the builder, but does not touch the operations in the
226    /// epilogue of the builder.
227    pub fn make_basic_block(&mut self) -> Result<Option<MastNodeId>, Report> {
228        if !self.ops.is_empty() {
229            let ops = self.ops.drain(..).collect();
230            let decorators = self.decorators.drain(..).collect();
231            let asm_ops = core::mem::take(&mut self.asm_ops);
232            let debug_vars: Vec<(usize, DebugVarId)> = self.debug_vars.drain(..).collect();
233
234            let basic_block_node_id =
235                self.mast_forest_builder
236                    .ensure_block(ops, decorators, asm_ops, vec![], vec![])?;
237
238            // Register debug variables for this node (stored separately from decorators)
239            if !debug_vars.is_empty() {
240                self.mast_forest_builder
241                    .register_debug_vars_for_node(basic_block_node_id, debug_vars)?;
242            }
243
244            Ok(Some(basic_block_node_id))
245        } else {
246            Ok(None)
247        }
248    }
249
250    /// Creates and returns a new basic block node from the operations and decorators currently in
251    /// this builder. If there are no operations however, we return the decorators that were
252    /// accumulated up until this point. If the builder is empty, then no node is created and
253    /// `Nothing` is returned.
254    ///
255    /// The main differences with [`Self::make_basic_block`] are:
256    /// - Operations contained in the epilogue of the builder are appended to the list of ops which
257    ///   go into the new BASIC BLOCK node.
258    /// - The builder is consumed in the process.
259    /// - Hence, any remaining decorators if no basic block was created are drained and returned.
260    pub fn try_into_basic_block(mut self) -> Result<BasicBlockOrDecorators, Report> {
261        self.ops.append(&mut self.epilogue);
262
263        if let Some(basic_block_node_id) = self.make_basic_block()? {
264            Ok(BasicBlockOrDecorators::BasicBlock(basic_block_node_id))
265        } else if let Some(decorator_ids) = self.drain_decorators() {
266            Ok(BasicBlockOrDecorators::Decorators(decorator_ids))
267        } else {
268            Ok(BasicBlockOrDecorators::Nothing)
269        }
270    }
271
272    /// Drains and returns the decorators in the builder, if any.
273    ///
274    /// This should only be called after [`Self::make_basic_block`], when no blocks were created.
275    /// In other words, there MUST NOT be any operations left in the builder when this is called.
276    ///
277    /// # Panics
278    ///
279    /// Panics if there are still operations left in the builder.
280    pub fn drain_decorators(&mut self) -> Option<Vec<DecoratorId>> {
281        assert!(self.ops.is_empty());
282        if !self.decorators.is_empty() {
283            Some(self.decorators.drain(..).map(|(_, decorator_id)| decorator_id).collect())
284        } else {
285            None
286        }
287    }
288}

290/// Holds either the node id of a basic block, or a list of decorators that are currently not
291/// attached to any node.
292pub enum BasicBlockOrDecorators {
293    BasicBlock(MastNodeId),
294    Decorators(Vec<DecoratorId>),
295    Nothing,
296}
